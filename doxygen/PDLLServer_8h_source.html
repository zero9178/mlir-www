30 /// This class implements all of the PDLL related functionality necessary for a
31 /// language server. This class allows for keeping the PDLL specific logic
32 /// separate from the logic that involves LSP server/client communication.
33 class PDLLServer {

37  const std::vector<std::string> &extraDirs)
38  : compilationDatabases(compilationDatabases), extraDirs(extraDirs) {}

40  /// The filenames for databases containing compilation commands for PDLL
41  /// files passed to the server.
42  const std::vector<std::string> &compilationDatabases;
43 
44  /// Additional list of include directories to search.
45  const std::vector<std::string> &extraDirs;
46  };
