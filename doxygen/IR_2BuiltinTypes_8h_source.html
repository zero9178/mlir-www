224  Builder(ArrayRef<int64_t> shape, Type elementType, Attribute encoding)

234  return *this;

239  return *this;

242  /// Erase a dim from shape @pos.
243  Builder &dropDim(unsigned pos) {
244  assert(pos < shape.size() && "overflow");
245  if (storage.empty())
246  storage.append(shape.begin(), shape.end());

268 /// This is a builder type that keeps local references to arguments. Arguments
269 /// that are passed into the builder must outlive the builder.
270 class VectorType::Builder {
271 public:
272  /// Build from another VectorType.
273  explicit Builder(VectorType other)
274  : shape(other.getShape()), elementType(other.getElementType()),
275  numScalableDims(other.getNumScalableDims()) {}
276 
277  /// Build from scratch.
278  Builder(ArrayRef<int64_t> shape, Type elementType,
279  unsigned numScalableDims = 0)
280  : shape(shape), elementType(elementType),
281  numScalableDims(numScalableDims) {}
282 
283  Builder &setShape(ArrayRef<int64_t> newShape,
284  unsigned newNumScalableDims = 0) {
285  numScalableDims = newNumScalableDims;
286  shape = newShape;
287  return *this;
288  }
289 
290  Builder &setElementType(Type newElementType) {
291  elementType = newElementType;
292  return *this;
293  }
294 
295  /// Erase a dim from shape @pos.
296  Builder &dropDim(unsigned pos) {
297  assert(pos < shape.size() && "overflow");
298  if (pos >= shape.size() - numScalableDims)
299  numScalableDims--;
300  if (storage.empty())
301  storage.append(shape.begin(), shape.end());
302  storage.erase(storage.begin() + pos);
303  shape = {storage.data(), storage.size()};
304  return *this;
305  }
306 
307  /// In the particular case where the vector has a single dimension that we
308  /// drop, return the scalar element type.
309  // TODO: unify once we have a VectorType that supports 0-D.
310  operator Type() {
311  if (shape.empty())
312  return elementType;
313  return VectorType::get(shape, elementType, numScalableDims);
314  }
315 
316 private:
317  ArrayRef<int64_t> shape;
318  // Owning shape data for copy-on-write operations.
319  SmallVector<int64_t> storage;
320  Type elementType;
321  unsigned numScalableDims;
322 };

406 /// Returns the strides of the MemRef if the layout map is in strided form.
407 /// MemRefs with a layout map in strided form include:
408 /// 1. empty or identity layout map, in which case the stride information is
409 /// the canonical form computed from sizes;
410 /// 2. single affine map layout of the form `K + k0 * d0 + ... kn * dn`,
411 /// where K and ki's are constants or symbols.
412 ///
413 /// A stride specification is a list of integer values that are either static
414 /// or dynamic (encoded with getDynamicStrideOrOffset()). Strides encode the
415 /// distance in the number of elements between successive entries along a
416 /// particular dimension.
417 ///
418 /// For example, `memref<42x16xf32, (64 * d0 + d1)>` specifies a view into a
419 /// non-contiguous memory region of `42` by `16` `f32` elements in which the
420 /// distance between two consecutive elements along the outer dimension is `1`
421 /// and the distance between two consecutive elements along the inner dimension
422 /// is `64`.
423 ///
424 /// The convention is that the strides for dimensions d0, .. dn appear in
425 /// order to make indexing intuitive into the result.
426 LogicalResult getStridesAndOffset(MemRefType t,
427  SmallVectorImpl<int64_t> &strides,
428  int64_t &offset);
429 LogicalResult getStridesAndOffset(MemRefType t,
430  SmallVectorImpl<AffineExpr> &strides,
431  AffineExpr &offset);
432 
433 /// Given a list of strides (in which MemRefType::getDynamicStrideOrOffset()
434 /// represents a dynamic value), return the single result AffineMap which
435 /// represents the linearized strided layout map. Dimensions correspond to the
436 /// offset followed by the strides in order. Symbols are inserted for each
437 /// dynamic dimension in order. A stride cannot take value `0`.
438 ///
439 /// Examples:
440 /// =========
441 ///
442 /// 1. For offset: 0 strides: ?, ?, 1 return
443 /// (i, j, k)[M, N]->(M * i + N * j + k)
444 ///
445 /// 2. For offset: 3 strides: 32, ?, 16 return
446 /// (i, j, k)[M]->(3 + 32 * i + M * j + 16 * k)
447 ///
448 /// 3. For offset: ? strides: ?, ?, ? return
449 /// (i, j, k)[off, M, N, P]->(off + M * i + N * j + P * k)
450 AffineMap makeStridedLinearLayoutMap(ArrayRef<int64_t> strides, int64_t offset,
451  MLIRContext *context);
452 
453 /// Return a version of `t` with identity layout if it can be determined
454 /// statically that the layout is the canonical contiguous strided layout.
455 /// Otherwise pass `t`'s layout into `simplifyAffineMap` and return a copy of
456 /// `t` with simplified layout.
457 MemRefType canonicalizeStridedLayout(MemRefType t);
458 
459 /// Return a version of `t` with a layout that has all dynamic offset and
460 /// strides. This is used to erase the static layout.
461 MemRefType eraseStridedLayout(MemRefType t);
462 
463 /// Given MemRef `sizes` that are either static or dynamic, returns the
464 /// canonical "contiguous" strides AffineExpr. Strides are multiplicative and
465 /// once a dynamic dimension is encountered, all canonical strides become
466 /// dynamic and need to be encoded with a different symbol.
467 /// For canonical strides expressions, the offset is always 0 and and fastest
468 /// varying stride is always `1`.
469 ///
470 /// Examples:
471 /// - memref<3x4x5xf32> has canonical stride expression
472 /// `20*exprs[0] + 5*exprs[1] + exprs[2]`.
473 /// - memref<3x?x5xf32> has canonical stride expression
474 /// `s0*exprs[0] + 5*exprs[1] + exprs[2]`.
475 /// - memref<3x4x?xf32> has canonical stride expression
476 /// `s1*exprs[0] + s0*exprs[1] + exprs[2]`.

478  ArrayRef<AffineExpr> exprs,
