22 /// Asserts that the operations provided as template arguments implement the
23 /// TransformOpInterface and MemoryEffectsOpInterface. This must be a dynamic
24 /// assertion since interface implementations may be registered at runtime.
25 template <typename OpTy>
26 static inline void checkImplementsTransformInterface(MLIRContext *context) {
27  // Since the operation is being inserted into the Transform dialect and the
28  // dialect does not implement the interface fallback, only check for the op
29  // itself having the interface implementation.
30  RegisteredOperationName opName =
31  *RegisteredOperationName::lookup(OpTy::getOperationName(), context);
32  assert((opName.hasInterface<TransformOpInterface>() ||
33  opName.hasTrait<OpTrait::IsTerminator>()) &&
34  "non-terminator ops injected into the transform dialect must "
35  "implement TransformOpInterface");
36  assert(opName.hasInterface<MemoryEffectOpInterface>() &&
37  "ops injected into the transform dialect must implement "
38  "MemoryEffectsOpInterface");
39 }
40 } // namespace detail
41 #endif // NDEBUG

50 /// Base class for extensions of the Transform dialect that supports injecting
51 /// operations into the Transform dialect at load time. Concrete extensions are
52 /// expected to derive this class and register operations in the constructor.
53 /// They can be registered with the DialectRegistry and automatically applied
54 /// to the Transform dialect when it is loaded.
55 template <typename DerivedTy, typename... ExtraDialects>
56 class TransformDialectExtension
57  : public DialectExtension<DerivedTy, TransformDialect, ExtraDialects...> {
58  using Initializer = std::function<void(TransformDialect *)>;
59  using DialectLoader = std::function<void(MLIRContext *)>;
60 
61 public:
62  /// Extension application hook. Actually loads the dependent dialects and
63  /// registers the additional operations. Not expected to be called directly.
64  void apply(MLIRContext *context, TransformDialect *transformDialect,
65  ExtraDialects *...) const final {
66  for (const DialectLoader &loader : dialectLoaders)
67  loader(context);
68  for (const Initializer &init : opInitializers)
69  init(transformDialect);
70  transformDialect->mergeInPDLMatchHooks(std::move(pdlMatchConstraintFns));
71  }
72 
73 protected:
74  /// Injects the operations into the Transform dialect. The operations must
75  /// implement the TransformOpInterface and MemoryEffectsOpInterface, and the
76  /// implementations must be already available when the operation is injected.
77  template <typename... OpTys>
78  void registerTransformOps() {
79  opInitializers.push_back([](TransformDialect *transformDialect) {
80  transformDialect->addOperationsChecked<OpTys...>();
81  });
82  }
83 
84  /// Declares that this Transform dialect extension depends on the dialect
85  /// provided as template parameter. When the Transform dialect is loaded,
86  /// dependent dialects will be loaded as well. This is intended for dialects
87  /// that contain attributes and types used in creation and canonicalization of
88  /// the injected operations.
89  template <typename DialectTy>
90  void declareDependentDialect() {
91  dialectLoaders.push_back(
92  [](MLIRContext *context) { context->loadDialect<DialectTy>(); });
93  }
94 
95  /// Injects the named constraint to make it available for use with the
96  /// PDLMatchOp in the transform dialect.
97  void registerPDLMatchConstraintFn(StringRef name,
98  PDLConstraintFunction &&fn) {
99  pdlMatchConstraintFns.try_emplace(name,
100  std::forward<PDLConstraintFunction>(fn));
101  }
102  template <typename ConstraintFnTy>
103  void registerPDLMatchConstraintFn(StringRef name, ConstraintFnTy &&fn) {
104  pdlMatchConstraintFns.try_emplace(
105  name, ::mlir::detail::pdl_function_builder::buildConstraintFn(
106  std::forward<ConstraintFnTy>(fn)));
107  }
108 
109 private:
110  SmallVector<Initializer> opInitializers;
111  SmallVector<DialectLoader> dialectLoaders;
112 
113  /// A list of constraints that should be made availble to PDL patterns
114  /// processed by PDLMatchOp in the Transform dialect.
115  ///
116  /// Declared as mutable so its contents can be moved in the `apply` const
117  /// method, which is only called once.
118  mutable llvm::StringMap<PDLConstraintFunction> pdlMatchConstraintFns;
119 };
