43 static Value
44 createLinalgBodyCalculationForElementwiseOp(Operation *op, ValueRange args,
45  ArrayRef<Type> resultTypes,
46  PatternRewriter &rewriter) {
47  Location loc = op->getLoc();
48  auto elementTy =
49  op->getOperand(0).getType().cast<ShapedType>().getElementType();
50 
51  // tosa::AbsOp
52  if (isa<tosa::AbsOp>(op) && elementTy.isa<FloatType>())
53  return rewriter.create<math::AbsOp>(loc, resultTypes, args);
54 
55  if (isa<tosa::AbsOp>(op) && elementTy.isa<IntegerType>()) {
56  auto zero = rewriter.create<arith::ConstantOp>(
57  loc, rewriter.getZeroAttr(elementTy));
58  auto cmp = rewriter.create<arith::CmpIOp>(loc, arith::CmpIPredicate::sgt,
59  args[0], zero);
60  auto neg = rewriter.create<arith::SubIOp>(loc, zero, args[0]);
61  return rewriter.create<arith::SelectOp>(loc, cmp, args[0], neg);
62  }
63 
64  // tosa::AddOp
65  if (isa<tosa::AddOp>(op) && elementTy.isa<FloatType>())
66  return rewriter.create<arith::AddFOp>(loc, resultTypes, args);
67 
68  if (isa<tosa::AddOp>(op) && elementTy.isa<IntegerType>())
69  return rewriter.create<arith::AddIOp>(loc, resultTypes, args);
70 
71  // tosa::SubOp
72  if (isa<tosa::SubOp>(op) && elementTy.isa<FloatType>())
73  return rewriter.create<arith::SubFOp>(loc, resultTypes, args);
74 
75  if (isa<tosa::SubOp>(op) && elementTy.isa<IntegerType>())
76  return rewriter.create<arith::SubIOp>(loc, resultTypes, args);
77 
78  // tosa::MulOp
79  if (isa<tosa::MulOp>(op) && elementTy.isa<FloatType>()) {
80  if (dyn_cast<tosa::MulOp>(op).shift() != 0) {
81  (void)rewriter.notifyMatchFailure(op,
82  "Cannot have shift value for float");
83  return nullptr;
84  }
85  return rewriter.create<arith::MulFOp>(loc, resultTypes, args);
86  }
87 
88  // tosa::DivOp
89  if (isa<tosa::DivOp>(op) && elementTy.isa<IntegerType>())
90  return rewriter.create<arith::DivSIOp>(loc, resultTypes, args);
91 
92  // tosa::ReciprocalOp
93  if (isa<tosa::ReciprocalOp>(op) && elementTy.isa<FloatType>()) {
94  auto one =
95  rewriter.create<arith::ConstantOp>(loc, FloatAttr::get(elementTy, 1));
96  return rewriter.create<arith::DivFOp>(loc, resultTypes, one, args[0]);
97  }
98 
99  if (isa<tosa::MulOp>(op) && elementTy.isa<IntegerType>()) {
100  Value a = args[0];
101  Value b = args[1];
102  auto shift =
103  op->getAttr("shift").cast<IntegerAttr>().getValue().getSExtValue();
104  if (shift > 0) {
105  auto shiftConst =
106  rewriter.create<arith::ConstantIntOp>(loc, shift, /*bitwidth=*/8);
107  if (!a.getType().isInteger(32))
108  a = rewriter.create<arith::ExtSIOp>(loc, rewriter.getI32Type(), a);
109 
110  if (!b.getType().isInteger(32))
111  b = rewriter.create<arith::ExtSIOp>(loc, rewriter.getI32Type(), b);
112 
113  auto result = rewriter.create<tosa::ApplyScaleOp>(
114  loc, rewriter.getI32Type(), a, b, shiftConst,
115  rewriter.getBoolAttr(false));
116 
117  if (elementTy.isInteger(32))
118  return result;
119 
120  return rewriter.create<arith::TruncIOp>(loc, elementTy, result);
121  }
122 
123  int aWidth = a.getType().getIntOrFloatBitWidth();
124  int bWidth = b.getType().getIntOrFloatBitWidth();
125  int cWidth = resultTypes[0].getIntOrFloatBitWidth();
126 
127  if (aWidth < cWidth)
128  a = rewriter.create<arith::ExtSIOp>(loc, resultTypes[0], a);
129  if (bWidth < cWidth)
130  b = rewriter.create<arith::ExtSIOp>(loc, resultTypes[0], b);
131 
132  return rewriter.create<arith::MulIOp>(loc, resultTypes, a, b);
133  }
134 
135  // tosa::NegateOp
136  if (isa<tosa::NegateOp>(op) && elementTy.isa<FloatType>())
137  return rewriter.create<arith::NegFOp>(loc, resultTypes, args);
138 
139  if (isa<tosa::NegateOp>(op) && elementTy.isa<IntegerType>() &&
140  !cast<tosa::NegateOp>(op).quantization_info()) {
141  auto constant =
142  rewriter.create<arith::ConstantOp>(loc, IntegerAttr::get(elementTy, 0));
143  return rewriter.create<arith::SubIOp>(loc, resultTypes, constant, args[0]);
144  }
145 
146  if (isa<tosa::NegateOp>(op) && elementTy.isa<IntegerType>() &&
147  cast<tosa::NegateOp>(op).quantization_info()) {
148  auto quantizationInfo = cast<tosa::NegateOp>(op).quantization_info();
149  int32_t inputBitWidth = elementTy.getIntOrFloatBitWidth();
150  int64_t inZp =
151  quantizationInfo.getValue().input_zp().getValue().getSExtValue();
152  int64_t outZp =
153  quantizationInfo.getValue().output_zp().getValue().getSExtValue();
154 
155  // Compute the maximum value that can occur in the intermediate buffer.
156  int64_t zpAdd = inZp + outZp;
157  int64_t maxValue = APInt::getSignedMaxValue(inputBitWidth).getSExtValue() +
158  std::abs(zpAdd) + 1;
159 
160  // Convert that maximum value into the maximum bitwidth needed to represent
161  // it. We assume 48-bit numbers may be supported further in the pipeline.
162  int intermediateBitWidth = 64;
163  if (maxValue <= APInt::getSignedMaxValue(16).getSExtValue()) {
164  intermediateBitWidth = 16;
165  } else if (maxValue <= APInt::getSignedMaxValue(32).getSExtValue()) {
166  intermediateBitWidth = 32;
167  } else if (maxValue <= APInt::getSignedMaxValue(48).getSExtValue()) {
168  intermediateBitWidth = 48;
169  }
170 
171  Type intermediateType = rewriter.getIntegerType(intermediateBitWidth);
172  Value zpAddValue = rewriter.create<arith::ConstantOp>(
173  loc, rewriter.getIntegerAttr(intermediateType, zpAdd));
174 
175  // The negation can be applied by doing:
176  // outputValue = inZp + outZp - inputValue
177  auto ext = rewriter.create<arith::ExtSIOp>(loc, intermediateType, args[0]);
178  auto sub = rewriter.create<arith::SubIOp>(loc, zpAddValue, ext);
179 
180  // Clamp to the negation range.
181  auto min = rewriter.create<arith::ConstantIntOp>(
182  loc, APInt::getSignedMinValue(inputBitWidth).getSExtValue(),
183  intermediateType);
184  auto max = rewriter.create<arith::ConstantIntOp>(
185  loc, APInt::getSignedMaxValue(inputBitWidth).getSExtValue(),
186  intermediateType);
187  auto clamp = clampHelper<arith::CmpIOp>(
188  loc, sub, min, max, arith::CmpIPredicate::slt, rewriter);
189 
190  // Truncate to the final value.
191  return rewriter.create<arith::TruncIOp>(loc, elementTy, clamp);
192  }
193 
194  // tosa::BitwiseAndOp
195  if (isa<tosa::BitwiseAndOp>(op) && elementTy.isa<IntegerType>())
196  return rewriter.create<arith::AndIOp>(loc, resultTypes, args);
197 
198  // tosa::BitwiseOrOp
199  if (isa<tosa::BitwiseOrOp>(op) && elementTy.isa<IntegerType>())
200  return rewriter.create<arith::OrIOp>(loc, resultTypes, args);
201 
202  // tosa::BitwiseNotOp
203  if (isa<tosa::BitwiseNotOp>(op) && elementTy.isa<IntegerType>()) {
204  auto allOnesAttr = rewriter.getIntegerAttr(
205  elementTy, APInt::getAllOnes(elementTy.getIntOrFloatBitWidth()));
206  auto allOnes = rewriter.create<arith::ConstantOp>(loc, allOnesAttr);
207  return rewriter.create<arith::XOrIOp>(loc, resultTypes, args[0], allOnes);
208  }
209 
210  // tosa::BitwiseXOrOp
211  if (isa<tosa::BitwiseXorOp>(op) && elementTy.isa<IntegerType>())
212  return rewriter.create<arith::XOrIOp>(loc, resultTypes, args);
213 
214  // tosa::LogicalLeftShiftOp
215  if (isa<tosa::LogicalLeftShiftOp>(op) && elementTy.isa<IntegerType>())
216  return rewriter.create<arith::ShLIOp>(loc, resultTypes, args);
217 
218  // tosa::LogicalRightShiftOp
219  if (isa<tosa::LogicalRightShiftOp>(op) && elementTy.isa<IntegerType>())
220  return rewriter.create<arith::ShRUIOp>(loc, resultTypes, args);
221 
222  // tosa::ArithmeticRightShiftOp
223  if (isa<tosa::ArithmeticRightShiftOp>(op) && elementTy.isa<IntegerType>()) {
224  auto result = rewriter.create<arith::ShRSIOp>(loc, resultTypes, args);
225  auto round = op->getAttr("round").cast<BoolAttr>().getValue();
226  if (!round) {
227  return result;
228  }
229 
230  Type i1Ty = IntegerType::get(rewriter.getContext(), /*width=*/1);
231  auto one =
232  rewriter.create<arith::ConstantOp>(loc, IntegerAttr::get(elementTy, 1));
233  auto zero =
234  rewriter.create<arith::ConstantOp>(loc, IntegerAttr::get(elementTy, 0));
235  auto i1one =
236  rewriter.create<arith::ConstantOp>(loc, IntegerAttr::get(i1Ty, 1));
237 
238  // Checking that input2 != 0
239  auto shiftValueGreaterThanZero = rewriter.create<arith::CmpIOp>(
240  loc, arith::CmpIPredicate::sgt, args[1], zero);
241 
242  // Checking for the last bit of input1 to be 1
243  auto subtract =
244  rewriter.create<arith::SubIOp>(loc, resultTypes, args[1], one);
245  auto shifted =
246  rewriter.create<arith::ShRSIOp>(loc, resultTypes, args[0], subtract)
247  ->getResults();
248  auto truncated =
249  rewriter.create<arith::TruncIOp>(loc, i1Ty, shifted, mlir::None);
250  auto isInputOdd =
251  rewriter.create<arith::AndIOp>(loc, i1Ty, truncated, i1one);
252 
253  auto shouldRound = rewriter.create<arith::AndIOp>(
254  loc, i1Ty, shiftValueGreaterThanZero, isInputOdd);
255  auto extended =
256  rewriter.create<arith::ExtUIOp>(loc, resultTypes, shouldRound);
257  return rewriter.create<arith::AddIOp>(loc, resultTypes, result, extended);
258  }
259 
260  // tosa::ClzOp
261  if (isa<tosa::ClzOp>(op) && elementTy.isa<IntegerType>()) {
262  int bitWidth = elementTy.getIntOrFloatBitWidth();
263  auto zero =
264  rewriter.create<arith::ConstantOp>(loc, IntegerAttr::get(elementTy, 0));
265  auto leadingZeros = rewriter.create<arith::ConstantOp>(
266  loc, IntegerAttr::get(elementTy, bitWidth));
267 
268  SmallVector<Value> operands = {args[0], leadingZeros, zero};
269  SmallVector<Type> types = {elementTy, elementTy, elementTy};
270  SmallVector<Location> locations = {loc, loc, loc};
271 
272  auto whileOp = rewriter.create<scf::WhileOp>(loc, types, operands);
273  Block *before =
274  rewriter.createBlock(&whileOp.getBefore(), {}, types, locations);
275  Block *after =
276  rewriter.createBlock(&whileOp.getAfter(), {}, types, locations);
277 
278  // The conditional block of the while loop.
279  {
280  rewriter.setInsertionPointToStart(&whileOp.getBefore().front());
281  Value input = before->getArgument(0);
282  Value zero = before->getArgument(2);
283 
284  Value inputLargerThanZero = rewriter.create<arith::CmpIOp>(
285  loc, arith::CmpIPredicate::ne, input, zero);
286  rewriter.create<scf::ConditionOp>(loc, inputLargerThanZero,
287  before->getArguments());
288  }
289 
290  // The body of the while loop: shift right until reaching a value of 0.
291  {
292  rewriter.setInsertionPointToStart(&whileOp.getAfter().front());
293  Value input = after->getArgument(0);
294  Value leadingZeros = after->getArgument(1);
295 
296  auto one = rewriter.create<arith::ConstantOp>(
297  loc, IntegerAttr::get(elementTy, 1));
298  auto shifted =
299  rewriter.create<arith::ShRUIOp>(loc, resultTypes, input, one);
300  auto leadingZerosMinusOne =
301  rewriter.create<arith::SubIOp>(loc, resultTypes, leadingZeros, one);
302 
303  rewriter.create<scf::YieldOp>(
304  loc,
305  ValueRange({shifted, leadingZerosMinusOne, after->getArgument(2)}));
306  }
307 
308  rewriter.setInsertionPointAfter(whileOp);
309  return whileOp->getResult(1);
310  }

313  if (isa<tosa::LogicalAndOp>(op) && elementTy.isInteger(1))
314  return rewriter.create<arith::AndIOp>(loc, resultTypes, args);
315 
316  // tosa::LogicalNot
317  if (isa<tosa::LogicalNotOp>(op) && elementTy.isInteger(1)) {
318  auto one = rewriter.create<arith::ConstantOp>(

333  return rewriter.create<mlir::math::PowFOp>(loc, resultTypes, args);
334

336  if (isa<tosa::RsqrtOp>(op) && elementTy.isa<FloatType>())
337  return rewriter.create<mlir::math::RsqrtOp>(loc, resultTypes, args);
338 
339  // tosa::LogOp
340  if (isa<tosa::LogOp>(op) && elementTy.isa<FloatType>())
341  return rewriter.create<mlir::math::LogOp>(loc, resultTypes, args);
342

344  if (isa<tosa::ExpOp>(op) && elementTy.isa<FloatType>())
345  return rewriter.create<mlir::math::ExpOp>(loc, resultTypes, args);

348  if (isa<tosa::TanhOp>(op) && elementTy.isa<FloatType>())
349  return rewriter.create<mlir::math::TanhOp>(loc, resultTypes, args);
350

352  if (isa<tosa::GreaterOp>(op) && elementTy.isa<FloatType>())
353  return rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::OGT,
354  args[0], args[1]);
355

357  return rewriter.create<arith::CmpIOp>(loc, arith::CmpIPredicate::sgt,
358  args[0], args[1]);
359 
360  // tosa::GreaterEqualOp
361  if (isa<tosa::GreaterEqualOp>(op) && elementTy.isa<FloatType>())
362  return rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::OGE,
363  args[0], args[1]);
364 
365  if (isa<tosa::GreaterEqualOp>(op) && elementTy.isSignlessInteger())
366  return rewriter.create<arith::CmpIOp>(loc, arith::CmpIPredicate::sge,
367  args[0], args[1]);
368

371  return rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::OEQ,
372  args[0], args[1]);
373 
374  if (isa<tosa::EqualOp>(op) && elementTy.isSignlessInteger())
375  return rewriter.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq,
376  args[0], args[1]);

382  return rewriter.create<arith::SelectOp>(loc, args[0], args[1], args[2]);

422  op->getAttr("min_fp"));

432  op->getAttr("min_int").cast<IntegerAttr>().getValue().getSExtValue());

454  return clampHelper<arith::CmpIOp>(loc, args[0], minVal, maxVal,
455  arith::CmpIPredicate::slt, rewriter);

493 
494  if (srcTy == dstTy)
495  return args.front();
496

498  return rewriter.create<arith::ExtFOp>(loc, resultTypes, args, mlir::None);
499

501  return rewriter.create<arith::TruncFOp>(loc, resultTypes, args,
502  mlir::None);
503

506  return rewriter.create<arith::UIToFPOp>(loc, resultTypes, args,
507  mlir::None);
508 
509  if (srcTy.isInteger(1) && dstTy.isa<IntegerType>() && bitExtend)
510  return rewriter.create<arith::ExtUIOp>(loc, resultTypes, args,
511  mlir::None);
512 
513  // Unsigned integers need an unrealized cast so that they can be passed
514  // to UIToFP.
515  if (srcTy.isUnsignedInteger() && dstTy.isa<FloatType>()) {
516  auto unrealizedCast =
517  rewriter
518  .create<UnrealizedConversionCastOp>(
519  loc, rewriter.getIntegerType(srcTy.getIntOrFloatBitWidth()),
520  args[0])
521  .getResult(0);
522  return rewriter.create<arith::UIToFPOp>(loc, resultTypes[0],
523  unrealizedCast);
524  }
525 
526  // All other si-to-fp conversions should be handled by SIToFP.
527  if (arith::SIToFPOp::areCastCompatible(srcTy, dstTy))
528  return rewriter.create<arith::SIToFPOp>(loc, resultTypes, args,
529  mlir::None);
530 
531  // Casting to boolean, floats need to only be checked as not-equal to zero.
532  if (srcTy.isa<FloatType>() && dstTy.isInteger(1)) {
533  Value zero = rewriter.create<arith::ConstantOp>(
534  loc, rewriter.getFloatAttr(srcTy, 0.0));
535  return rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::UNE,
536  args.front(), zero);
537  }
538 
539  if (arith::FPToSIOp::areCastCompatible(srcTy, dstTy)) {
540  auto zero = rewriter.create<arith::ConstantOp>(
541  loc, rewriter.getF32FloatAttr(0.0f));
542  auto half = rewriter.create<arith::ConstantOp>(
543  loc, rewriter.getF32FloatAttr(0.5f));
544 
545  auto intMin = rewriter.create<arith::ConstantOp>(
546  loc, rewriter.getF32FloatAttr(
547  APInt::getSignedMinValue(dstTy.getIntOrFloatBitWidth())
548  .getSExtValue()));
549 
550  auto intMax = rewriter.create<arith::ConstantOp>(
551  loc, rewriter.getF32FloatAttr(
552  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())
553  .getSExtValue()));
554 
555  auto added = rewriter.create<arith::AddFOp>(loc, args[0], half);
556  auto subbed = rewriter.create<arith::SubFOp>(loc, args[0], half);
557  auto negative = rewriter.create<arith::CmpFOp>(
558  loc, arith::CmpFPredicate::OLT, args[0], zero);
559  auto rounded =
560  rewriter.create<arith::SelectOp>(loc, negative, subbed, added);
561 
562  auto clamped = clampHelper<arith::CmpFOp>(
563  loc, rounded, intMin, intMax, arith::CmpFPredicate::OLT, rewriter);
564 
565  return rewriter.create<arith::FPToSIOp>(loc, dstTy, clamped);
566  }
567 
568  // Casting to boolean, integers need to only be checked as not-equal to
569  // zero.
570  if (srcTy.isa<IntegerType>() && dstTy.isInteger(1)) {
571  Value zero = rewriter.create<arith::ConstantIntOp>(
572  loc, 0, srcTy.getIntOrFloatBitWidth());
573  return rewriter.create<arith::CmpIOp>(loc, arith::CmpIPredicate::ne,
574  args.front(), zero);
575  }
576 
577  if (srcTy.isa<IntegerType>() && dstTy.isa<IntegerType>() && bitExtend)
578  return rewriter.create<arith::ExtSIOp>(loc, resultTypes, args,
579  mlir::None);
580 
581  if (srcTy.isa<IntegerType>() && dstTy.isa<IntegerType>() && !bitExtend) {
582  auto intMin = rewriter.create<arith::ConstantIntOp>(
583  loc,
584  APInt::getSignedMinValue(dstTy.getIntOrFloatBitWidth())
585  .getSExtValue(),
586  srcTy.getIntOrFloatBitWidth());
587 
588  auto intMax = rewriter.create<arith::ConstantIntOp>(
589  loc,
590  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())
591  .getSExtValue(),
592  srcTy.getIntOrFloatBitWidth());
593 
594  auto clamped = clampHelper<arith::CmpIOp>(
595  loc, args[0], intMin, intMax, arith::CmpIPredicate::slt, rewriter);
596  return rewriter.create<arith::TruncIOp>(loc, dstTy, clamped);
597  }
598  }
599 
600  (void)rewriter.notifyMatchFailure(
601  op, "unhandled op for linalg body calculation for elementwise op");
602  return nullptr;
603 }

605 static LogicalResult
606 elementwiseMatchAndRewriteHelper(Operation *operation,
607  PatternRewriter &rewriter) {
608  auto loc = operation->getLoc();
609 
610  assert(operation->getNumResults() == 1 &&
611  "All TOSA elementwise ops should only return a single result.");
612 
613  auto results = operation->getResults();
614  auto resultTy = operation->getResult(0).getType().dyn_cast<ShapedType>();
615 
616  if (!resultTy)
617  return rewriter.notifyMatchFailure(operation,
618  "All results must be a shaped type");
619 
620  unsigned rank = resultTy.getRank();
621 
622  // Construct the indexing maps needed for linalg.generic ops.
623  SmallVector<Type> bodyArgTypes;
624 
625  for (Value in : operation->getOperands())
626  bodyArgTypes.emplace_back(getElementTypeOrSelf(in.getType()));
627 
628  SmallVector<Type> opResultTypes;
629  SmallVector<Value> initTensors;
630 
631  SmallVector<Value> dynDims;
632  dynDims.resize(results.front().getType().cast<ShapedType>().getRank());
633 
634  for (auto arg : operation->getOperands()) {
635  auto operandTy = arg.getType().cast<ShapedType>();
636  for (int i = 0; i < operandTy.getRank(); i++) {
637  if (operandTy.isDynamicDim(i) && !dynDims[i])
638  dynDims[i] = rewriter.create<tensor::DimOp>(loc, arg, i);
639  }
640  }
641 
642  SmallVector<Value> filteredDims = condenseValues(dynDims);
643 
644  for (auto result : results) {
645  auto resultTy = result.getType().template cast<ShapedType>();
646  initTensors.push_back(rewriter.create<linalg::InitTensorOp>(
647  loc, filteredDims, resultTy.getShape(), resultTy.getElementType()));
648  opResultTypes.push_back(result.getType());
649  }
650 
651  auto bodyResultTypes = llvm::to_vector<4>(llvm::map_range(
652  initTensors, [](Value v) { return getElementTypeOrSelf(v); }));
653 
654  SmallVector<Value, 2> operands;
655  SmallVector<AffineMap, 2> indexingMaps;
656  indexingMaps.reserve(operation->getNumOperands() + bodyResultTypes.size());
657 
658  // Input indexing maps may be broadcasted.
659  for (Value operand : operation->getOperands()) {
660  ShapedType type = operand.getType().cast<ShapedType>();
661 
662  if (type.getShape() == resultTy.getShape()) {
663  operands.push_back(operand);
664  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(rank));
665  continue;
666  }
667 
668  SmallVector<int64_t, 5> newShape;
669  SmallVector<AffineExpr, 4> affineExprs;
670  newShape.reserve(type.getRank());
671  for (const auto &it : llvm::enumerate(type.getShape())) {
672  if (it.value() == resultTy.getDimSize(it.index())) {
673  newShape.push_back(it.value());
674  affineExprs.push_back(
675  mlir::getAffineDimExpr(it.index(), rewriter.getContext()));
676  }
677  }
678 
679  if (newShape.size() != rank) {
680  operand = rewriter.create<tosa::ReshapeOp>(
681  loc, RankedTensorType::get(newShape, type.getElementType()), operand,
682  rewriter.getI64ArrayAttr(newShape));
683  }
684 
685  operands.push_back(operand);
686  indexingMaps.push_back(AffineMap::get(
687  /*dimCount=*/type.getRank(), /*symbolCount=*/0, affineExprs,
688  rewriter.getContext()));
689  }
690 
691  indexingMaps.append(operation->getNumResults(),
692  rewriter.getMultiDimIdentityMap(rank));
693 
694  bool didEncounterError = false;
695  auto linalgOp = rewriter.create<linalg::GenericOp>(
696  loc, opResultTypes, operands, initTensors, indexingMaps,
697  getNParallelLoopsAttrs(rank),
698  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange blockArgs) {
699  Value opResult = createLinalgBodyCalculationForElementwiseOp(
700  operation, blockArgs.take_front(operation->getNumOperands()),
701  bodyResultTypes, rewriter);
702  if (!opResult) {
703  didEncounterError = true;
704  return;
705  }
706  nestedBuilder.create<linalg::YieldOp>(loc, opResult);
707  });
708 
709  if (didEncounterError)
710  return failure();
711 
712  rewriter.replaceOp(operation, linalgOp->getResults());
713  return success();
714 }

768 // Creates the body calculation for a reduction. The operations vary depending
769 // on the input type.
770 static Value createLinalgBodyCalculationForReduceOp(Operation *op,
771  ValueRange args,
772  Type elementTy,
773  PatternRewriter &rewriter) {
774  Location loc = op->getLoc();
775  if (isa<tosa::ReduceSumOp>(op) && elementTy.isa<FloatType>()) {
776  return rewriter.create<arith::AddFOp>(loc, args);
777  }
778 
779  if (isa<tosa::ReduceSumOp>(op) && elementTy.isa<IntegerType>()) {
780  return rewriter.create<arith::AddIOp>(loc, args);
781  }
782 
783  if (isa<tosa::ReduceProdOp>(op) && elementTy.isa<FloatType>()) {
784  return rewriter.create<arith::MulFOp>(loc, args);
785  }
786 
787  if (isa<tosa::ReduceProdOp>(op) && elementTy.isa<IntegerType>()) {
788  return rewriter.create<arith::MulIOp>(loc, args);
789  }
790 
791  if (isa<tosa::ReduceMinOp>(op) && elementTy.isa<FloatType>()) {
792  auto predicate = rewriter.create<arith::CmpFOp>(
793  loc, arith::CmpFPredicate::OLT, args[0], args[1]);
794  return rewriter.create<arith::SelectOp>(loc, predicate, args[0], args[1]);
795  }
796 
797  if (isa<tosa::ReduceMinOp>(op) && elementTy.isa<IntegerType>()) {
798  auto predicate = rewriter.create<arith::CmpIOp>(
799  loc, arith::CmpIPredicate::slt, args[0], args[1]);
800  return rewriter.create<arith::SelectOp>(loc, predicate, args[0], args[1]);
801  }
802 
803  if (isa<tosa::ReduceMaxOp>(op) && elementTy.isa<FloatType>()) {
804  auto predicate = rewriter.create<arith::CmpFOp>(
805  loc, arith::CmpFPredicate::OGT, args[0], args[1]);
806  return rewriter.create<arith::SelectOp>(loc, predicate, args[0], args[1]);
807  }
808 
809  if (isa<tosa::ReduceMaxOp>(op) && elementTy.isa<IntegerType>()) {
810  auto predicate = rewriter.create<arith::CmpIOp>(
811  loc, arith::CmpIPredicate::sgt, args[0], args[1]);
812  return rewriter.create<arith::SelectOp>(loc, predicate, args[0], args[1]);
813  }
814 
815  if (isa<tosa::ReduceAllOp>(op) && elementTy.isInteger(1))
816  return rewriter.create<arith::AndIOp>(loc, args);
817 
818  if (isa<tosa::ReduceAnyOp>(op) && elementTy.isInteger(1))
819  return rewriter.create<arith::OrIOp>(loc, args);
820 
821  return {};
822 }

824 // Performs the match and rewrite for reduction operations. This includes
825 // declaring a correctly sized initial value, and the linalg.generic operation
826 // that reduces across the specified axis.
827 static LogicalResult reduceMatchAndRewriteHelper(Operation *op, uint64_t axis,
828  PatternRewriter &rewriter) {
829  auto loc = op->getLoc();
830  auto inputTy = op->getOperand(0).getType().template cast<ShapedType>();
831  auto resultTy = op->getResult(0).getType().template cast<ShapedType>();
832  auto elementTy = resultTy.getElementType();
833  Value input = op->getOperand(0);
834 
835  llvm::SmallVector<int64_t> reduceShape;
836  SmallVector<Value> dynDims;
837  for (unsigned i = 0; i < inputTy.getRank(); i++) {
838  if (axis != i) {
839  reduceShape.push_back(inputTy.getDimSize(i));
840  if (inputTy.isDynamicDim(i))
841  dynDims.push_back(rewriter.create<tensor::DimOp>(loc, input, i));
842  }
843  }
844 
845  Type reduceTy = RankedTensorType::get(reduceShape, resultTy.getElementType());
846 
847  // First fill the output buffer with the init value.
848  auto initTensor = rewriter
849  .create<linalg::InitTensorOp>(loc, dynDims, reduceShape,
850  resultTy.getElementType())
851  .result();
852 
853  auto fillValueAttr = createInitialValueForReduceOp(op, elementTy, rewriter);
854  if (!fillValueAttr)
855  return rewriter.notifyMatchFailure(
856  op, "No initial value found for reduction operation");
857 
858  auto fillValue = rewriter.create<arith::ConstantOp>(loc, fillValueAttr);
859  auto filledTensor = rewriter
860  .create<linalg::FillOp>(loc, ValueRange{fillValue},
861  ValueRange{initTensor})
862  .result();
863 
864  SmallVector<AffineExpr, 2> srcExprs;
865  SmallVector<AffineExpr, 2> dstExprs;
866  SmallVector<StringRef, 4> iteratorTypes;
867  for (unsigned int i = 0, rank = inputTy.getRank(); i != rank; ++i) {
868  srcExprs.push_back(mlir::getAffineDimExpr(i, rewriter.getContext()));
869 
870  iteratorTypes.push_back(axis == i ? getReductionIteratorTypeName()
871  : getParallelIteratorTypeName());
872  if (axis != i)
873  dstExprs.push_back(mlir::getAffineDimExpr(i, rewriter.getContext()));
874  }
875 
876  bool didEncounterError = false;
877  auto maps = AffineMap::inferFromExprList({srcExprs, dstExprs});
878  auto linalgOp = rewriter.create<linalg::GenericOp>(
879  loc, reduceTy, input, filledTensor, maps, iteratorTypes,
880  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange blockArgs) {
881  auto result = createLinalgBodyCalculationForReduceOp(
882  op, blockArgs, elementTy, rewriter);
883  if (result)
884  didEncounterError = true;
885 
886  nestedBuilder.create<linalg::YieldOp>(loc, result);
887  });
888 
889  if (!didEncounterError)
890  return failure();
891 
892  rewriter.replaceOpWithNewOp<tosa::ReshapeOp>(op, resultTy,
893  linalgOp.getResults());
894  return success();
895 }

897 static bool findIntermediateShape(ArrayRef<int64_t> lhsShape,
898  ArrayRef<int64_t> rhsShape,
899  SmallVector<int64_t> &intermediateShape,
900  bool isDynamic) {
901  if (isDynamic) {
902  // TODO (natashaknk): Make dynamic intermediate shape not always be rank-1
903  intermediateShape = {-1};
904  return true;
905  }
906 
907  if (lhsShape.empty() || rhsShape.empty()) {
908  intermediateShape = {};
909  return true;
910  }
911 
912  unsigned currLhsDim = 0, currRhsDim = 0;
913  while (currLhsDim < lhsShape.size() && currRhsDim < rhsShape.size()) {
914  int64_t rhsSize = rhsShape[currRhsDim];
915  int64_t lhsSize = lhsShape[currLhsDim];
916  while (lhsSize != rhsSize && currLhsDim < lhsShape.size() &&
917  currRhsDim < rhsShape.size()) {
918  if (lhsSize < rhsSize) {
919  currLhsDim++;
920  lhsSize *= lhsShape[currLhsDim];
921  } else {
922  currRhsDim++;
923  rhsSize *= rhsShape[currRhsDim];
924  }
925  }
926  if (lhsSize == rhsSize) {
927  intermediateShape.push_back(lhsSize);
928  }
929  currRhsDim++;
930  currLhsDim++;
931  }
932 
933  // If the iterators didn't reach the end and their leftover dimensions are not
934  // equal to 1 an intermediate shape was not found.
935  while (currLhsDim < lhsShape.size()) {
936  if (lhsShape[currLhsDim++] != 1) {
937  return false;
938  }
939  }
940 
941  while (currRhsDim < rhsShape.size()) {
942  if (rhsShape[currRhsDim++] != 1) {
943  return false;
944  }
945  }
946 
947  return true;
948 }

950 static bool createReassociationMapsForCollapse(
951  PatternRewriter &rewriter, ArrayRef<int64_t> srcShape,
952  ArrayRef<int64_t> dstShape,
953  SmallVector<ReassociationExprs, 4> &reassociationMap, bool isDynamic) {
954 
955  // If the shape is dynamic, create a map for collapsing into one dimension.
956  if (isDynamic) {
957  SmallVector<AffineExpr, 2> exprs;
958  for (int i = 0, s = srcShape.size(); i < s; ++i)
959  exprs.push_back(rewriter.getAffineDimExpr(i));
960  reassociationMap = {exprs};
961  return true;
962  }
963 
964  if (dstShape.empty()) {
965  reassociationMap = {};
966  return true;
967  }
968 
969  reassociationMap.resize(dstShape.size());
970  unsigned currSrcDim = 0, currDstDim = 0;
971  while (currSrcDim < srcShape.size() && currDstDim < dstShape.size()) {
972  int64_t dstSize = dstShape[currDstDim];
973  int64_t srcSize = srcShape[currSrcDim];
974  while (srcSize < dstSize && currSrcDim < srcShape.size()) {
975  reassociationMap[currDstDim].push_back(
976  rewriter.getAffineDimExpr(currSrcDim++));
977  srcSize *= srcShape[currSrcDim];
978  }
979  if (srcSize == dstSize) {
980  reassociationMap[currDstDim].push_back(
981  rewriter.getAffineDimExpr(currSrcDim++));
982  // If the next dim in collapsedShape is not 1, treat subsequent dims in
983  // expandedShape which are 1 to be collapsed.
984  if (currDstDim == dstShape.size() - 1 || dstShape[currDstDim + 1] != 1) {
985  while (currSrcDim < srcShape.size() && srcShape[currSrcDim] == 1) {
986  reassociationMap[currDstDim].push_back(
987  rewriter.getAffineDimExpr(currSrcDim++));
988  }
989  }
990  }
991  currDstDim++;
992  }
993 
994  // If both iterators didn't reach the end, we have leftover dimentions which
995  // implies that we have a mismatch in shape.
996  return !(currSrcDim != srcShape.size() || currDstDim != dstShape.size());
997 }

1266  rewriter.getAffineDimExpr(rank - 1)};

1273  /*symbolCount=*/0, shiftExprs,
1274  rewriter.getContext()));

1276  }
1277 
1278  // Indexing maps for output values.
1279  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(rank));
1280

1282  Value initTensor = rewriter.create<linalg::InitTensorOp>(
1283  loc, dynamicDims, outputTy.getShape(), outputTy.getElementType());

1286  loc, outputTy, genericInputs, ValueRange{initTensor}, indexingMaps,

1288  [&](OpBuilder &nestedBuilder, Location nestedLoc,
1289  ValueRange blockArgs) {
1290  Value value = blockArgs[0];
1291  Type valueTy = value.getType();
1292 
1293  // For now we do all of our math in 64-bit. This is not optimal but
1294  // should be correct for now, consider computing correct bit depth
1295  // later.
1296  int32_t inBitwidth = valueTy.getIntOrFloatBitWidth() > 32 ? 48 : 32;
1297 
1298  auto inputZp = createConstFromIntAttribute<int32_t>(
1299  op, "input_zp", nestedBuilder.getIntegerType(inBitwidth),
1300  nestedBuilder);
1301  auto outputZp = createConstFromIntAttribute<int32_t>(
1302  op, "output_zp", nestedBuilder.getI32Type(), nestedBuilder);
1303

1305  : blockArgs[multiplierArg];
1306  Value shift = shiftConstant ? shiftConstant : blockArgs[shiftArg];
1307 
1308  if (valueTy.getIntOrFloatBitWidth() < 32) {
1309  if (valueTy.isUnsignedInteger()) {
1310  value = nestedBuilder
1311  .create<UnrealizedConversionCastOp>(
1312  nestedLoc,
1313  nestedBuilder.getIntegerType(
1314  valueTy.getIntOrFloatBitWidth()),
1315  value)
1316  .getResult(0);
1317  value = nestedBuilder.create<arith::ExtUIOp>(
1318  nestedLoc, nestedBuilder.getI32Type(), value);
1319  } else {
1320  value = nestedBuilder.create<arith::ExtSIOp>(
1321  nestedLoc, nestedBuilder.getI32Type(), value);
1322  }
1323  }
1324 
1325  value =
1326  nestedBuilder.create<arith::SubIOp>(nestedLoc, value, inputZp);
1327 
1328  value = nestedBuilder.create<tosa::ApplyScaleOp>(
1329  loc, nestedBuilder.getI32Type(), value, multiplier, shift,
1330  nestedBuilder.getBoolAttr(doubleRound));
1331 
1332  // Move to the new zero-point.
1333  value =
1334  nestedBuilder.create<arith::AddIOp>(nestedLoc, value, outputZp);
1335 
1336  // Saturate to the output size.
1337  IntegerType outIntType =
1338  blockArgs.back().getType().cast<IntegerType>();
1339  unsigned outBitWidth = outIntType.getWidth();
1340 
1341  int32_t intMin = APInt::getSignedMinValue(outBitWidth).getSExtValue();
1342  int32_t intMax = APInt::getSignedMaxValue(outBitWidth).getSExtValue();

1345  if (outIntType.isUnsignedInteger()) {
1346  intMin = 0;
1347  intMax = APInt::getMaxValue(outBitWidth).getZExtValue();
1348  }
1349 
1350  auto intMinVal = nestedBuilder.create<arith::ConstantOp>(
1351  loc, nestedBuilder.getI32IntegerAttr(intMin));
1352  auto intMaxVal = nestedBuilder.create<arith::ConstantOp>(
1353  loc, nestedBuilder.getI32IntegerAttr(intMax));
1354 
1355  value = clampHelper<arith::CmpIOp>(
1356  nestedLoc, value, intMinVal, intMaxVal, arith::CmpIPredicate::slt,
1357  nestedBuilder);
1358 
1359  if (outIntType.getWidth() < 32) {
1360  value = nestedBuilder.create<arith::TruncIOp>(
1361  nestedLoc, rewriter.getIntegerType(outIntType.getWidth()),
1362  value);
1363 
1364  if (outIntType.isUnsignedInteger()) {
1365  value = nestedBuilder
1366  .create<UnrealizedConversionCastOp>(nestedLoc,
1367  outIntType, value)
1368  .getResult(0);
1369  }
1370  }
1371 
1372  nestedBuilder.create<linalg::YieldOp>(loc, value);
1373  });
1374 
1375  rewriter.replaceOp(op, linalgOp->getResults());
1376  return success();
1377  }
