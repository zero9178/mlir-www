36 /// A result type used to indicate if a change happened. Boolean operations on
37 /// ChangeResult behave as though `Change` is truthy.
38 enum class ChangeResult {
39  NoChange,
40  Change,
41 };
42 inline ChangeResult operator|(ChangeResult lhs, ChangeResult rhs) {
43  return lhs == ChangeResult::Change ? lhs : rhs;
44 }
45 inline ChangeResult &operator|=(ChangeResult &lhs, ChangeResult rhs) {
46  lhs = lhs | rhs;
47  return lhs;
48 }
49 inline ChangeResult operator&(ChangeResult lhs, ChangeResult rhs) {
50  return lhs == ChangeResult::NoChange ? lhs : rhs;
51 }

57 namespace detail {
58 /// This class represents an abstract lattice. A lattice is what gets propagated
59 /// across the IR, and contains the information for a specific Value.
60 class AbstractLatticeElement {
61 public:
62  virtual ~AbstractLatticeElement();
63 
64  /// Returns true if the value of this lattice is uninitialized, meaning that
65  /// it hasn't yet been initialized.
66  virtual bool isUninitialized() const = 0;
67 
68  /// Join the information contained in 'rhs' into this lattice. Returns
69  /// if the value of the lattice changed.
70  virtual ChangeResult join(const AbstractLatticeElement &rhs) = 0;
71 
72  /// Mark the lattice element as having reached a pessimistic fixpoint. This
73  /// means that the lattice may potentially have conflicting value states, and
74  /// only the most conservative value should be relied on.
75  virtual ChangeResult markPessimisticFixpoint() = 0;
76 
77  /// Mark the lattice element as having reached an optimistic fixpoint. This
78  /// means that we optimistically assume the current value is the true state.
79  virtual void markOptimisticFixpoint() = 0;
80 
81  /// Returns true if the lattice has reached a fixpoint. A fixpoint is when the
82  /// information optimistically assumed to be true is the same as the
83  /// information known to be true.
84  virtual bool isAtFixpoint() const = 0;
85 };

92 /// This class represents a lattice holding a specific value of type `ValueT`.
93 /// Lattice values (`ValueT`) are required to adhere to the following:
94 /// * static ValueT join(const ValueT &lhs, const ValueT &rhs);
95 /// - This method conservatively joins the information held by `lhs`
96 /// and `rhs` into a new value. This method is required to be monotonic.
97 /// * static ValueT getPessimisticValueState(MLIRContext *context);
98 /// - This method computes a pessimistic/conservative value state assuming
99 /// no information about the state of the IR.
100 /// * static ValueT getPessimisticValueState(Value value);
101 /// - This method computes a pessimistic/conservative value state for
102 /// `value` assuming only information present in the current IR.
103 /// * bool operator==(const ValueT &rhs) const;
104 ///
105 template <typename ValueT>
106 class LatticeElement final : public detail::AbstractLatticeElement {
107 public:
108  LatticeElement() = delete;
109  LatticeElement(const ValueT &knownValue) : knownValue(knownValue) {}
110 
111  /// Return the value held by this lattice. This requires that the value is
112  /// initialized.
113  ValueT &getValue() {
114  assert(!isUninitialized() && "expected known lattice element");
115  return *optimisticValue;
116  }
117  const ValueT &getValue() const {
118  assert(!isUninitialized() && "expected known lattice element");
119  return *optimisticValue;
120  }
121 
122  /// Returns true if the value of this lattice hasn't yet been initialized.
123  bool isUninitialized() const final { return !optimisticValue.hasValue(); }
124 
125  /// Join the information contained in the 'rhs' lattice into this
126  /// lattice. Returns if the state of the current lattice changed.
127  ChangeResult join(const detail::AbstractLatticeElement &rhs) final {
128  const LatticeElement<ValueT> &rhsLattice =
129  static_cast<const LatticeElement<ValueT> &>(rhs);
130 
131  // If we are at a fixpoint, or rhs is uninitialized, there is nothing to do.
132  if (isAtFixpoint() || rhsLattice.isUninitialized())
133  return ChangeResult::NoChange;
134 
135  // Join the rhs value into this lattice.
136  return join(rhsLattice.getValue());
137  }
138 
139  /// Join the information contained in the 'rhs' value into this
140  /// lattice. Returns if the state of the current lattice changed.
141  ChangeResult join(const ValueT &rhs) {
142  // If the current lattice is uninitialized, copy the rhs value.
143  if (isUninitialized()) {
144  optimisticValue = rhs;
145  return ChangeResult::Change;
146  }
147 
148  // Otherwise, join rhs with the current optimistic value.
149  ValueT newValue = ValueT::join(*optimisticValue, rhs);
150  assert(ValueT::join(newValue, *optimisticValue) == newValue &&
151  "expected `join` to be monotonic");
152  assert(ValueT::join(newValue, rhs) == newValue &&
153  "expected `join` to be monotonic");
154 
155  // Update the current optimistic value if something changed.
156  if (newValue == optimisticValue)
157  return ChangeResult::NoChange;
158 
159  optimisticValue = newValue;
160  return ChangeResult::Change;
161  }
162 
163  /// Mark the lattice element as having reached a pessimistic fixpoint. This
164  /// means that the lattice may potentially have conflicting value states, and
165  /// only the conservatively known value state should be relied on.
166  ChangeResult markPessimisticFixpoint() final {
167  if (isAtFixpoint())
168  return ChangeResult::NoChange;
169 
170  // For this fixed point, we take whatever we knew to be true and set that to
171  // our optimistic value.
172  optimisticValue = knownValue;
173  return ChangeResult::Change;
174  }
175 
176  /// Mark the lattice element as having reached an optimistic fixpoint. This
177  /// means that we optimistically assume the current value is the true state.
178  void markOptimisticFixpoint() final {
179  assert(!isUninitialized() && "expected an initialized value");
180  knownValue = *optimisticValue;
181  }
182 
183  /// Returns true if the lattice has reached a fixpoint. A fixpoint is when the
184  /// information optimistically assumed to be true is the same as the
185  /// information known to be true.
186  bool isAtFixpoint() const final { return optimisticValue == knownValue; }
187 
188 private:
189  /// The value that is conservatively known to be true.
190  ValueT knownValue;
191  /// The currently computed value that is optimistically assumed to be true, or
192  /// None if the lattice element is uninitialized.
193  Optional<ValueT> optimisticValue;
194 };

277 /// This class provides a general forward dataflow analysis driver
278 /// utilizing the lattice classes defined above, to enable the easy definition
279 /// of dataflow analysis algorithms. More specifically this driver is useful for
280 /// defining analyses that are forward, sparse, pessimistic (except along
281 /// unreached backedges) and context-insensitive for the interprocedural
282 /// aspects.
283 template <typename ValueT>
284 class ForwardDataFlowAnalysis : public detail::ForwardDataFlowAnalysisBase {
285 public:
286  ForwardDataFlowAnalysis(MLIRContext *context) : context(context) {}
287 
288  /// Return the MLIR context used when constructing this analysis.
289  MLIRContext *getContext() { return context; }
290 
291  /// Compute the analysis on operations rooted under the given top-level
292  /// operation. Note that the top-level operation is not visited.
293  void run(Operation *topLevelOp) {
294  detail::ForwardDataFlowAnalysisBase::run(topLevelOp);
295  }
296 
297  /// Return the lattice element attached to the given value, or nullptr if no
298  /// lattice for the value has yet been created.
299  LatticeElement<ValueT> *lookupLatticeElement(Value value) {
300  return static_cast<LatticeElement<ValueT> *>(
301  detail::ForwardDataFlowAnalysisBase::lookupLatticeElement(value));
302  }
303 
304 protected:
305  /// Return the lattice element attached to the given value. If a lattice has
306  /// not been added for the given value, a new 'uninitialized' value is
307  /// inserted and returned.
308  LatticeElement<ValueT> &getLatticeElement(Value value) {
309  return static_cast<LatticeElement<ValueT> &>(
310  detail::ForwardDataFlowAnalysisBase::getLatticeElement(value));
311  }
312 
313  /// Mark all of the lattices for the given range of Values as having reached a
314  /// pessimistic fixpoint.
315  ChangeResult markAllPessimisticFixpoint(ValueRange values) {
316  ChangeResult result = ChangeResult::NoChange;
317  for (Value value : values)
318  result |= getLatticeElement(value).markPessimisticFixpoint();
319  return result;
320  }
321 
322  /// Visit the given operation, and join any necessary analysis state
323  /// into the lattices for the results and block arguments owned by this
324  /// operation using the provided set of operand lattice elements (all pointer
325  /// values are guaranteed to be non-null). Returns if any result or block
326  /// argument value lattices changed during the visit. The lattice for a result
327  /// or block argument value can be obtained by using
328  /// `getLatticeElement`.
329  virtual ChangeResult
330  visitOperation(Operation *op,
331  ArrayRef<LatticeElement<ValueT> *> operands) = 0;
332 
333  /// Given a BranchOpInterface, and the current lattice elements that
334  /// correspond to the branch operands (all pointer values are guaranteed to be
335  /// non-null), try to compute a specific set of successors that would be
336  /// selected for the branch. Returns failure if not computable, or if all of
337  /// the successors would be chosen. If a subset of successors can be selected,
338  /// `successors` is populated.
339  virtual LogicalResult
340  getSuccessorsForOperands(BranchOpInterface branch,
341  ArrayRef<LatticeElement<ValueT> *> operands,
342  SmallVectorImpl<Block *> &successors) {
343  return failure();
344  }
345 
346  /// Given a RegionBranchOpInterface, and the current lattice elements that
347  /// correspond to the branch operands (all pointer values are guaranteed to be
348  /// non-null), compute a specific set of region successors that would be
349  /// selected.
350  virtual void
351  getSuccessorsForOperands(RegionBranchOpInterface branch,
352  Optional<unsigned> sourceIndex,
353  ArrayRef<LatticeElement<ValueT> *> operands,
354  SmallVectorImpl<RegionSuccessor> &successors) {
355  SmallVector<Attribute> constantOperands(operands.size());
356  branch.getSuccessorRegions(sourceIndex, constantOperands, successors);
357  }
358 
359  /// Given a operation with successor regions, one of those regions,
360  /// and the lattice elements corresponding to the operation's
361  /// arguments, compute the latice values for block arguments
362  /// that are not accounted for by the branching control flow (ex. the
363  /// bounds of loops). By default, this method marks all such lattice elements
364  /// as having reached a pessimistic fixpoint. The region in the
365  /// RegionSuccessor and the operand latice elements are guaranteed to be
366  /// non-null.
367  virtual ChangeResult
368  visitNonControlFlowArguments(Operation *op, const RegionSuccessor &successor,
369  ArrayRef<LatticeElement<ValueT> *> operands) {
370  ChangeResult result = ChangeResult::NoChange;
371  Region *region = successor.getSuccessor();
372  ValueRange succArgs = successor.getSuccessorInputs();
373  Block *block = &region->front();
374  Block::BlockArgListType arguments = block->getArguments();
375  if (arguments.size() != succArgs.size()) {
376  unsigned firstArgIdx =
377  succArgs.empty() ? 0
378  : succArgs[0].cast<BlockArgument>().getArgNumber();
379  result |= markAllPessimisticFixpoint(arguments.take_front(firstArgIdx));
380  result |= markAllPessimisticFixpoint(
381  arguments.drop_front(firstArgIdx + succArgs.size()));
382  }
383  return result;
384  }
385 
386 private:
387  /// Type-erased wrappers that convert the abstract lattice operands to derived
388  /// lattices and invoke the virtual hooks operating on the derived lattices.
389  ChangeResult
390  visitOperation(Operation *op,
391  ArrayRef<detail::AbstractLatticeElement *> operands) final {
392  LatticeElement<ValueT> *const *derivedOperandBase =
393  reinterpret_cast<LatticeElement<ValueT> *const *>(operands.data());
394  return visitOperation(
395  op, llvm::makeArrayRef(derivedOperandBase, operands.size()));
396  }
397  LogicalResult
398  getSuccessorsForOperands(BranchOpInterface branch,
399  ArrayRef<detail::AbstractLatticeElement *> operands,
400  SmallVectorImpl<Block *> &successors) final {
401  LatticeElement<ValueT> *const *derivedOperandBase =
402  reinterpret_cast<LatticeElement<ValueT> *const *>(operands.data());
403  return getSuccessorsForOperands(
404  branch, llvm::makeArrayRef(derivedOperandBase, operands.size()),
405  successors);
406  }
407  void
408  getSuccessorsForOperands(RegionBranchOpInterface branch,
409  Optional<unsigned> sourceIndex,
410  ArrayRef<detail::AbstractLatticeElement *> operands,
411  SmallVectorImpl<RegionSuccessor> &successors) final {
412  LatticeElement<ValueT> *const *derivedOperandBase =
413  reinterpret_cast<LatticeElement<ValueT> *const *>(operands.data());
414  getSuccessorsForOperands(
415  branch, sourceIndex,
416  llvm::makeArrayRef(derivedOperandBase, operands.size()), successors);
417  }
418  ChangeResult visitNonControlFlowArguments(
419  Operation *op, const RegionSuccessor &region,
420  ArrayRef<detail::AbstractLatticeElement *> operands) final {
421  LatticeElement<ValueT> *const *derivedOperandBase =
422  reinterpret_cast<LatticeElement<ValueT> *const *>(operands.data());
423  return visitNonControlFlowArguments(
424  op, region, llvm::makeArrayRef(derivedOperandBase, operands.size()));
425  }
426 
427  /// Create a new uninitialized lattice element. An optional value is provided,
428  /// which if valid, should be used to initialize the known conservative state
429  /// of the lattice.
430  detail::AbstractLatticeElement *createLatticeElement(Value value) final {
431  ValueT knownValue = value ? ValueT::getPessimisticValueState(value)
432  : ValueT::getPessimisticValueState(context);
433  return new (allocator.Allocate()) LatticeElement<ValueT>(knownValue);
434  }
435 
436  /// An allocator used for new lattice elements.
437  llvm::SpecificBumpPtrAllocator<LatticeElement<ValueT>> allocator;
438 
439  /// The MLIRContext of this solver.
440  MLIRContext *context;
441 };
