70  /// Emit an error and return failure.
71  InFlightDiagnostic emitError(const Twine &message = {}) {
72  // If the error is to be emitted at EOF, move it back one character.
73  if (state.curToken.is(Token::eof)) {
74  return emitError(
75  SMLoc::getFromPointer(state.curToken.getLoc().getPointer() - 1),
76  message);
77  }
78  return emitError(state.curToken.getLoc(), message);
79  }
80  InFlightDiagnostic emitError(SMLoc loc, const Twine &message = {});

82  /// Encode the specified source location information into an attribute for
83  /// attachment to the IR.
84  Location getEncodedSourceLocation(SMLoc loc) {
85  // If there are no active nested parsers, we can get the encoded source
86  // location directly.
87  if (state.parserDepth == 0)
88  return state.lex.getEncodedSourceLocation(loc);
89  // Otherwise, we need to re-encode it to point to the top level buffer.
90  return state.symbols.topLevelLexer->getEncodedSourceLocation(
91  remapLocationToTopLevelBuffer(loc));
92  }
93 
94  /// Remaps the given SMLoc to the top level lexer of the parser. This is used
95  /// to adjust locations of potentially nested parsers to ensure that they can
96  /// be emitted properly as diagnostics.
97  SMLoc remapLocationToTopLevelBuffer(SMLoc loc) {
98  // If there are no active nested parsers, we can return location directly.
99  SymbolState &symbols = state.symbols;
100  if (state.parserDepth == 0)
101  return loc;
102  assert(symbols.topLevelLexer && "expected valid top-level lexer");
103 
104  // Otherwise, we need to remap the location to the main parser. This is
105  // simply offseting the location onto the location of the last nested
106  // parser.
107  size_t offset = loc.getPointer() - state.lex.getBufferBegin();
108  auto *rawLoc =
109  symbols.nestedParserLocs[state.parserDepth - 1].getPointer() + offset;
110  return SMLoc::getFromPointer(rawLoc);
111  }

121  /// If the current token has the specified kind, consume it and return true.
122  /// If not, return false.
123  bool consumeIf(Token::Kind kind) {
124  if (state.curToken.isNot(kind))
125  return false;
126  consumeToken(kind);
127  return true;
128  }

130  /// Advance the current lexer onto the next token.
131  void consumeToken() {
132  assert(state.curToken.isNot(Token::eof, Token::error) &&
133  "shouldn't advance past EOF or errors");
134  state.curToken = state.lex.lexToken();
135  }
136 
137  /// Advance the current lexer onto the next token, asserting what the expected
138  /// current token is. This is preferred to the above method because it leads
139  /// to more self-documenting code with better checking.
140  void consumeToken(Token::Kind kind) {
141  assert(state.curToken.is(kind) && "consumed an unexpected token");
142  consumeToken();
143  }
