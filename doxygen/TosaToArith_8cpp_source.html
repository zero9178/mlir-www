52 // This converts the TOSA ApplyScale operator to a set of arithmetic ops,
53 // using 64-bit operations to perform the necessary multiply, bias, and shift.
54 // Multiple types are used to use minimal bit width operations.
55 class ApplyScaleOpConverter : public OpRewritePattern<tosa::ApplyScaleOp> {
56 public:
57  using OpRewritePattern<tosa::ApplyScaleOp>::OpRewritePattern;
58 
59  LogicalResult matchAndRewrite(tosa::ApplyScaleOp op,
60  PatternRewriter &rewriter) const final {
61  Location loc = op.getLoc();
62  Value value32 = op.value();
63  Value multiplier32 = op.multiplier();
64  Value shift8 = op.shift();
65 
66  bool doubleRound = op.double_round();
67  Type inType = op.value().getType();
68  Type resultTy = op.getType();
69 
70  Type i8Ty = matchContainerType(rewriter.getIntegerType(8), resultTy);
71  Type i32Ty = matchContainerType(rewriter.getI32Type(), resultTy);
72  Type i64Ty = matchContainerType(rewriter.getI64Type(), resultTy);
73 
74  Value one8 = rewriter.create<arith::ConstantOp>(
75  loc, getConstantAttr(i8Ty, 1, rewriter));
76  Value one64 = rewriter.create<arith::ConstantOp>(
77  loc, getConstantAttr(i64Ty, 1, rewriter));
78 
79  Value shiftSubOne8 = rewriter.create<arith::SubIOp>(loc, shift8, one8);
80 
81  // The rounding value semantics below equate to the following code:
82  // int64_t round = 1 << (shift - 1);
83  // if (double_round) {
84  // if (shift > 31 && value >= 0) round += 1<<30;
85  // if (shift > 31 && value < 0) round -= 1<<30;
86  // }
87  //
88  // Note that minimal bitwidth operators are used throughout the block.
89 
90  Value round64 = rewriter.create<arith::ShLIOp>(
91  loc, one64, rewriter.create<arith::ExtSIOp>(loc, i64Ty, shiftSubOne8));
92 
93  // Double rounding is performing a round operation before the shift
94  if (doubleRound) {
95  Value one32 = rewriter.create<arith::ConstantOp>(
96  loc, getConstantAttr(i32Ty, 1, rewriter));
97  Value shift32 = rewriter.create<arith::ExtSIOp>(loc, i32Ty, shift8);
98  Value thirty32 = rewriter.create<arith::ConstantOp>(
99  loc, getConstantAttr(i32Ty, 30, rewriter));
100 
101  Value shiftThirty32 =
102  rewriter.create<arith::ShLIOp>(loc, one32, thirty32);
103  Value shiftThirty64 =
104  rewriter.create<arith::ExtSIOp>(loc, i64Ty, shiftThirty32);
105 
106  // Round value needs to with be added or subtracted depending on the sign
107  // of the input value.
108  Value roundAdd64 =
109  rewriter.create<arith::AddIOp>(loc, round64, shiftThirty64);
110  Value roundSub64 =
111  rewriter.create<arith::SubIOp>(loc, round64, shiftThirty64);
112 
113  Value zero32 =
114  rewriter.create<arith::ConstantOp>(loc, rewriter.getZeroAttr(inType));
115  Value valueGreaterThanZero = rewriter.create<arith::CmpIOp>(
116  loc, arith::CmpIPredicate::sge, value32, zero32);
117 
118  Value doubleRound64 = rewriter.create<arith::SelectOp>(
119  loc, valueGreaterThanZero, roundAdd64, roundSub64);
120 
121  // We only perform double rounding if the shift value is greater than 32.
122  Value thirtyTwo32 = rewriter.create<arith::ConstantOp>(
123  loc, getConstantAttr(i32Ty, 32, rewriter));
124  Value shiftGreaterThanThirtyTwo = rewriter.create<arith::CmpIOp>(
125  loc, arith::CmpIPredicate::sge, shift32, thirtyTwo32);
126  round64 = rewriter.create<arith::SelectOp>(loc, shiftGreaterThanThirtyTwo,
127  doubleRound64, round64);
128  }
129 
130  // The computation below equates to the following pseudocode:
131  // int64_t result = (int64_t)value * multiplier + round;
132  // result = result >> shift;
133  //
134  // Note that multiply and shift need to be perform in i64 to preserve bits.
135 
136  Value value64 = rewriter.create<arith::ExtSIOp>(loc, i64Ty, value32);
137  Value multiplier64 =
138  rewriter.create<arith::ExtSIOp>(loc, i64Ty, multiplier32);
139  Value shift64 = rewriter.create<arith::ExtSIOp>(loc, i64Ty, shift8);
140 
141  // Multiply as a pair of i64 values to guarantee the end value fits.
142  Value result64 = rewriter.create<arith::MulIOp>(loc, value64, multiplier64);
143  result64 = rewriter.create<arith::AddIOp>(loc, result64, round64);
144  result64 = rewriter.create<arith::ShRSIOp>(loc, result64, shift64);
145 
146  Value result32 = rewriter.create<arith::TruncIOp>(loc, resultTy, result64);
147 
148  rewriter.replaceOp(op, result32);
149  return success();
150  }
151 };
