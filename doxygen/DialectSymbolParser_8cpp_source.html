46 /// Parse the body of a pretty dialect symbol, which starts and ends with <>'s,
47 /// and may be recursive. Return with the 'prettyName' StringRef encompassing
48 /// the entire pretty name.
49 ///
50 /// pretty-dialect-sym-body ::= '<' pretty-dialect-sym-contents+ '>'
51 /// pretty-dialect-sym-contents ::= pretty-dialect-sym-body
52 /// | '(' pretty-dialect-sym-contents+ ')'
53 /// | '[' pretty-dialect-sym-contents+ ']'
54 /// | '{' pretty-dialect-sym-contents+ '}'
55 /// | '[^[<({>\])}\0]+'
56 ///
57 ParseResult Parser::parsePrettyDialectSymbolName(StringRef &prettyName) {
58  // Pretty symbol names are a relatively unstructured format that contains a
59  // series of properly nested punctuation, with anything else in the middle.
60  // Scan ahead to find it and consume it if successful, otherwise emit an
61  // error.
62  auto *curPtr = getTokenSpelling().data();
63 
64  SmallVector<char, 8> nestedPunctuation;
65 
66  // Scan over the nested punctuation, bailing out on error and consuming until
67  // we find the end. We know that we're currently looking at the '<', so we
68  // can go until we find the matching '>' character.
69  assert(*curPtr == '<');
70  do {
71  char c = *curPtr++;
72  switch (c) {
73  case '\0':
74  // This also handles the EOF case.
75  return emitError("unexpected nul or EOF in pretty dialect name");
76  case '<':
77  case '[':
78  case '(':
79  case '{':
80  nestedPunctuation.push_back(c);
81  continue;
82 
83  case '-':
84  // The sequence `->` is treated as special token.
85  if (*curPtr == '>')
86  ++curPtr;
87  continue;
88 
89  case '>':
90  if (nestedPunctuation.pop_back_val() != '<')
91  return emitError("unbalanced '>' character in pretty dialect name");
92  break;
93  case ']':
94  if (nestedPunctuation.pop_back_val() != '[')
95  return emitError("unbalanced ']' character in pretty dialect name");
96  break;
97  case ')':
98  if (nestedPunctuation.pop_back_val() != '(')
99  return emitError("unbalanced ')' character in pretty dialect name");
100  break;
101  case '}':
102  if (nestedPunctuation.pop_back_val() != '{')
103  return emitError("unbalanced '}' character in pretty dialect name");
104  break;
105 
106  default:
107  continue;
108  }
109  } while (!nestedPunctuation.empty());
110 
111  // Ok, we succeeded, remember where we stopped, reset the lexer to know it is
112  // consuming all this stuff, and return.
113  state.lex.resetPointer(curPtr);
114 
115  unsigned length = curPtr - prettyName.begin();
116  prettyName = StringRef(prettyName.begin(), length);
117  consumeToken();
118  return success();
119 }

121 /// Parse an extended dialect symbol.
122 template <typename Symbol, typename SymbolAliasMap, typename CreateFn>
123 static Symbol parseExtendedSymbol(Parser &p, Token::Kind identifierTok,
124  SymbolAliasMap &aliases,
125  CreateFn &&createSymbol) {
126  // Parse the dialect namespace.
127  StringRef identifier = p.getTokenSpelling().drop_front();
128  auto loc = p.getToken().getLoc();
129  p.consumeToken(identifierTok);
130 
131  // If there is no '<' token following this, and if the typename contains no
132  // dot, then we are parsing a symbol alias.
133  if (p.getToken().isNot(Token::less) && !identifier.contains('.')) {
134  // Check for an alias for this type.
135  auto aliasIt = aliases.find(identifier);
136  if (aliasIt == aliases.end())
137  return (p.emitError("undefined symbol alias id '" + identifier + "'"),
138  nullptr);
139  return aliasIt->second;
140  }
141 
142  // Otherwise, we are parsing a dialect-specific symbol. If the name contains
143  // a dot, then this is the "pretty" form. If not, it is the verbose form that
144  // looks like <"...">.
145  std::string symbolData;
146  auto dialectName = identifier;
147 
148  // Handle the verbose form, where "identifier" is a simple dialect name.
149  if (!identifier.contains('.')) {
150  // Consume the '<'.
151  if (p.parseToken(Token::less, "expected '<' in dialect type"))
152  return nullptr;
153 
154  // Parse the symbol specific data.
155  if (p.getToken().isNot(Token::string))
156  return (p.emitError("expected string literal data in dialect symbol"),
157  nullptr);
158  symbolData = p.getToken().getStringValue();
159  loc = SMLoc::getFromPointer(p.getToken().getLoc().getPointer() + 1);
160  p.consumeToken(Token::string);
161 
162  // Consume the '>'.
163  if (p.parseToken(Token::greater, "expected '>' in dialect symbol"))
164  return nullptr;
165  } else {
166  // Ok, the dialect name is the part of the identifier before the dot, the
167  // part after the dot is the dialect's symbol, or the start thereof.
168  auto dotHalves = identifier.split('.');
169  dialectName = dotHalves.first;
170  auto prettyName = dotHalves.second;
171  loc = SMLoc::getFromPointer(prettyName.data());
172 
173  // If the dialect's symbol is followed immediately by a <, then lex the body
174  // of it into prettyName.
175  if (p.getToken().is(Token::less) &&
176  prettyName.bytes_end() == p.getTokenSpelling().bytes_begin()) {
177  if (p.parsePrettyDialectSymbolName(prettyName))
178  return nullptr;
179  }
180 
181  symbolData = prettyName.str();
182  }
183 
184  // Record the name location of the type remapped to the top level buffer.
185  SMLoc locInTopLevelBuffer = p.remapLocationToTopLevelBuffer(loc);
186  p.getState().symbols.nestedParserLocs.push_back(locInTopLevelBuffer);
187 
188  // Call into the provided symbol construction function.
189  Symbol sym = createSymbol(dialectName, symbolData, loc);
190 
191  // Pop the last parser location.
192  p.getState().symbols.nestedParserLocs.pop_back();
193  return sym;
194 }

196 /// Parses a symbol, of type 'T', and returns it if parsing was successful. If
197 /// parsing failed, nullptr is returned. The number of bytes read from the input
198 /// string is returned in 'numRead'.
199 template <typename T, typename ParserFn>
200 static T parseSymbol(StringRef inputStr, MLIRContext *context,
201  SymbolState &symbolState, ParserFn &&parserFn,
202  size_t *numRead = nullptr) {
203  SourceMgr sourceMgr;
204  auto memBuffer = MemoryBuffer::getMemBuffer(
205  inputStr, /*BufferName=*/"<mlir_parser_buffer>",
206  /*RequiresNullTerminator=*/false);
207  sourceMgr.AddNewSourceBuffer(std::move(memBuffer), SMLoc());
208  ParserState state(sourceMgr, context, symbolState, /*asmState=*/nullptr);
209  Parser parser(state);
210 
211  Token startTok = parser.getToken();
212  T symbol = parserFn(parser);
213  if (!symbol)
214  return T();
215 
216  // If 'numRead' is valid, then provide the number of bytes that were read.
217  Token endTok = parser.getToken();
218  if (numRead) {
219  *numRead = static_cast<size_t>(endTok.getLoc().getPointer() -
220  startTok.getLoc().getPointer());
221 
222  // Otherwise, ensure that all of the tokens were parsed.
223  } else if (startTok.getLoc() != endTok.getLoc() && endTok.isNot(Token::eof)) {
224  parser.emitError(endTok.getLoc(), "encountered unexpected token");
225  return T();
226  }
227  return symbol;
228 }

230 /// Parse an extended attribute.
231 ///
232 /// extended-attribute ::= (dialect-attribute | attribute-alias)
233 /// dialect-attribute ::= `#` dialect-namespace `<` `"` attr-data `"` `>`
234 /// dialect-attribute ::= `#` alias-name pretty-dialect-sym-body?
235 /// attribute-alias ::= `#` alias-name
236 ///
237 Attribute Parser::parseExtendedAttr(Type type) {
238  Attribute attr = parseExtendedSymbol<Attribute>(
239  *this, Token::hash_identifier, state.symbols.attributeAliasDefinitions,
240  [&](StringRef dialectName, StringRef symbolData,
241  SMLoc loc) -> Attribute {
242  // Parse an optional trailing colon type.
243  Type attrType = type;
244  if (consumeIf(Token::colon) && !(attrType = parseType()))
245  return Attribute();
246 
247  // If we found a registered dialect, then ask it to parse the attribute.
248  if (Dialect *dialect =
249  builder.getContext()->getOrLoadDialect(dialectName)) {
250  return parseSymbol<Attribute>(
251  symbolData, state.context, state.symbols, [&](Parser &parser) {
252  CustomDialectAsmParser customParser(symbolData, parser);
253  return dialect->parseAttribute(customParser, attrType);
254  });
255  }
256 
257  // Otherwise, form a new opaque attribute.
258  return OpaqueAttr::getChecked(
259  [&] { return emitError(loc); },
260  StringAttr::get(state.context, dialectName), symbolData,
261  attrType ? attrType : NoneType::get(state.context));
262  });
263 
264  // Ensure that the attribute has the same type as requested.
265  if (attr && type && attr.getType() != type) {
266  emitError("attribute type different than expected: expected ")
267  << type << ", but got " << attr.getType();
268  return nullptr;
269  }
270  return attr;
271 }

273 /// Parse an extended type.
274 ///
275 /// extended-type ::= (dialect-type | type-alias)
276 /// dialect-type ::= `!` dialect-namespace `<` `"` type-data `"` `>`
277 /// dialect-type ::= `!` alias-name pretty-dialect-attribute-body?
278 /// type-alias ::= `!` alias-name
279 ///
280 Type Parser::parseExtendedType() {
281  return parseExtendedSymbol<Type>(
282  *this, Token::exclamation_identifier, state.symbols.typeAliasDefinitions,
283  [&](StringRef dialectName, StringRef symbolData,
284  SMLoc loc) -> Type {
285  // If we found a registered dialect, then ask it to parse the type.
286  auto *dialect = state.context->getOrLoadDialect(dialectName);
287 
288  if (dialect) {
289  return parseSymbol<Type>(
290  symbolData, state.context, state.symbols, [&](Parser &parser) {
291  CustomDialectAsmParser customParser(symbolData, parser);
292  return dialect->parseType(customParser);
293  });
294  }
295 
296  // Otherwise, form a new opaque type.
297  return OpaqueType::getChecked(
298  [&] { return emitError(loc); },
299  StringAttr::get(state.context, dialectName), symbolData);
300  });
301 }
