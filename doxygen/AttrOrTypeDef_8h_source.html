48 /// A wrapper class for tblgen AttrOrTypeParameter, arrays of which belong to
49 /// AttrOrTypeDefs to parameterize them.
50 class AttrOrTypeParameter {
51 public:
52  explicit AttrOrTypeParameter(const llvm::DagInit *def, unsigned index)
53  : def(def), index(index) {}
54 
55  /// Returns true if the parameter is anonymous (has no name).
56  bool isAnonymous() const;
57 
58  /// Get the parameter name.
59  StringRef getName() const;
60 
61  /// If specified, get the custom allocator code for this parameter.
62  Optional<StringRef> getAllocator() const;
63 
64  /// If specified, get the custom comparator code for this parameter.
65  StringRef getComparator() const;
66 
67  /// Get the C++ type of this parameter.
68  StringRef getCppType() const;
69 
70  /// Get the C++ accessor type of this parameter.
71  StringRef getCppAccessorType() const;
72 
73  /// Get the C++ storage type of this parameter.
74  StringRef getCppStorageType() const;
75 
76  /// Get an optional C++ parameter parser.
77  Optional<StringRef> getParser() const;
78 
79  /// Get an optional C++ parameter printer.
80  Optional<StringRef> getPrinter() const;
81 
82  /// Get a description of this parameter for documentation purposes.
83  Optional<StringRef> getSummary() const;
84 
85  /// Get the assembly syntax documentation.
86  StringRef getSyntax() const;
87 
88  /// Returns true if the parameter is optional.
89  bool isOptional() const;
90 
91  /// Get the default value of the parameter if it has one.
92  Optional<StringRef> getDefaultValue() const;
93 
94  /// Return the underlying def of this parameter.
95  llvm::Init *getDef() const;
96 
97  /// The parameter is pointer-comparable.
98  bool operator==(const AttrOrTypeParameter &other) const {
99  return def == other.def && index == other.index;
100  }
101  bool operator!=(const AttrOrTypeParameter &other) const {
102  return !(*this == other);
103  }
104 
105 private:
106  /// A parameter can be either a string or a def. Get a potentially null value
107  /// from the def.
108  template <typename InitT>
109  auto getDefValue(StringRef name) const;
110 
111  /// The underlying tablegen parameter list this parameter is a part of.
112  const llvm::DagInit *def;
113  /// The index of the parameter within the parameter list (`def`).
114  unsigned index;
115 };
