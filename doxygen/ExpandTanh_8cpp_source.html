21 /// Expands tanh op into
22 /// 1) 1-exp^{-2x} / 1+exp^{-2x}, if x => 0
23 /// 2) exp^{2x}-1 / exp^{2x}+1 , if x < 0
24 static LogicalResult convertTanhOp(math::TanhOp op, PatternRewriter &rewriter) {
25  auto floatType = op.getOperand().getType();
26  Location loc = op.getLoc();
27  auto floatOne = rewriter.getFloatAttr(floatType, 1.0);
28  auto floatTwo = rewriter.getFloatAttr(floatType, 2.0);
29  Value one = rewriter.create<arith::ConstantOp>(loc, floatOne);
30  Value two = rewriter.create<arith::ConstantOp>(loc, floatTwo);
31  Value doubledX = rewriter.create<arith::MulFOp>(loc, op.getOperand(), two);
32 
33  // Case 1: tanh(x) = 1-exp^{-2x} / 1+exp^{-2x}
34  Value negDoubledX = rewriter.create<arith::NegFOp>(loc, doubledX);
35  Value exp2x = rewriter.create<math::ExpOp>(loc, negDoubledX);
36  Value dividend = rewriter.create<arith::SubFOp>(loc, one, exp2x);
37  Value divisor = rewriter.create<arith::AddFOp>(loc, one, exp2x);
38  Value positiveRes = rewriter.create<arith::DivFOp>(loc, dividend, divisor);
39 
40  // Case 2: tanh(x) = exp^{2x}-1 / exp^{2x}+1
41  exp2x = rewriter.create<math::ExpOp>(loc, doubledX);
42  dividend = rewriter.create<arith::SubFOp>(loc, exp2x, one);
43  divisor = rewriter.create<arith::AddFOp>(loc, exp2x, one);
44  Value negativeRes = rewriter.create<arith::DivFOp>(loc, dividend, divisor);
45 
46  // tanh(x) = x >= 0 ? positiveRes : negativeRes
47  auto floatZero = rewriter.getFloatAttr(floatType, 0.0);
48  Value zero = rewriter.create<arith::ConstantOp>(loc, floatZero);
49  Value cmpRes = rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::OGE,
50  op.getOperand(), zero);
51  rewriter.replaceOpWithNewOp<arith::SelectOp>(op, cmpRes, positiveRes,
52  negativeRes);
53  return success();
54 }
