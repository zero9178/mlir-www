139 /// Collect a set of patterns to convert vector.multi_reduction op into
140 /// a sequence of vector.reduction ops. The patterns comprise:
141 /// - InnerOuterDimReductionConversion: rewrites vector.multi_reduction such
142 /// that all reduction dimensions are either innermost or outermost, by adding
143 /// the proper vector.transpose operations.
144 /// - ReduceMultiDimReductionRank: once in innermost or outermost reduction
145 /// form, rewrites n-D vector.multi_reduction into 2-D vector.multi_reduction,
146 /// by introducing vector.shape_cast ops to collapse + multi-reduce + expand
147 /// back.
148 /// - TwoDimMultiReductionToElementWise: once in 2-D vector.multi_reduction
149 /// form, with an **outermost** reduction dimension, unroll the outer dimension
150 /// to obtain a sequence of 1-D vector ops. This also has an opportunity for
151 /// tree-reduction (in the future).
152 /// - TwoDimMultiReductionToReduction: once in 2-D vector.multi_reduction form,
153 /// with an **innermost** reduction dimension, unroll the outer dimension to
154 /// obtain a sequence of extract + vector.reduction + insert. This can further
155 /// lower to horizontal reduction ops.
156 /// - OneDimMultiReductionToTwoDim: for cases that reduce to 1-D vector<k>
157 /// reduction (and are thus missing either a parallel or a reduction), we lift
158 /// them back up to 2-D with a simple vector.shape_cast to vector<1xk> so that
159 /// the other patterns can kick in, thus fully exiting out of the
160 /// vector.multi_reduction abstraction.
161 void populateVectorMultiReductionLoweringPatterns(
162  RewritePatternSet &patterns, VectorMultiReductionLowering options);

180 /// Collect a set of transfer read/write lowering patterns that simplify the
181 /// permutation map (e.g., converting it to a minor identity map) by inserting
182 /// broadcasts and transposes. More specifically:
183 ///
184 /// [TransferReadPermutationLowering]
185 /// Lower transfer_read op with permutation into a transfer_read with a
186 /// permutation map composed of leading zeros followed by a minor identity +
187 /// vector.transpose op.
188 /// Ex:
189 /// vector.transfer_read ...
190 /// permutation_map: (d0, d1, d2) -> (0, d1)
191 /// into:
192 /// %v = vector.transfer_read ...
193 /// permutation_map: (d0, d1, d2) -> (d1, 0)
194 /// vector.transpose %v, [1, 0]
195 ///
196 /// vector.transfer_read ...
197 /// permutation_map: (d0, d1, d2, d3) -> (0, 0, 0, d1, d3)
198 /// into:
199 /// %v = vector.transfer_read ...
200 /// permutation_map: (d0, d1, d2, d3) -> (0, 0, d1, 0, d3)
201 /// vector.transpose %v, [0, 1, 3, 2, 4]
202 /// Note that an alternative is to transform it to linalg.transpose +
203 /// vector.transfer_read to do the transpose in memory instead.
204 ///
205 /// [TransferWritePermutationLowering]
206 /// Lower transfer_write op with permutation into a transfer_write with a
207 /// minor identity permutation map. (transfer_write ops cannot have broadcasts.)
208 /// Ex:
209 /// vector.transfer_write %v ...
210 /// permutation_map: (d0, d1, d2) -> (d2, d0, d1)
211 /// into:
212 /// %tmp = vector.transpose %v, [2, 0, 1]
213 /// vector.transfer_write %tmp ...
214 /// permutation_map: (d0, d1, d2) -> (d0, d1, d2)
215 ///
216 /// vector.transfer_write %v ...
217 /// permutation_map: (d0, d1, d2, d3) -> (d3, d2)
218 /// into:
219 /// %tmp = vector.transpose %v, [1, 0]
220 /// %v = vector.transfer_write %tmp ...
221 /// permutation_map: (d0, d1, d2, d3) -> (d2, d3)
222 ///
223 /// [TransferOpReduceRank]
224 /// Lower transfer_read op with broadcast in the leading dimensions into
225 /// transfer_read of lower rank + vector.broadcast.
226 /// Ex: vector.transfer_read ...
227 /// permutation_map: (d0, d1, d2, d3) -> (0, d1, 0, d3)
228 /// into:
229 /// %v = vector.transfer_read ...
230 /// permutation_map: (d0, d1, d2, d3) -> (d1, 0, d3)
231 /// vector.broadcast %v
232 void populateVectorTransferPermutationMapLoweringPatterns(
233  RewritePatternSet &patterns);

242 /// Populate `patterns` with the following patterns.
243 ///
244 /// [DecomposeDifferentRankInsertStridedSlice]
245 /// ==========================================
246 /// RewritePattern for InsertStridedSliceOp where source and destination vectors
247 /// have different ranks.
248 ///
249 /// When ranks are different, InsertStridedSlice needs to extract a properly
250 /// ranked vector from the destination vector into which to insert. This pattern
251 /// only takes care of this extraction part and forwards the rest to
252 /// [VectorInsertStridedSliceOpSameRankRewritePattern].

255 /// 1. ExtractOp to extract the (unique) (n-1)-D subvector into which to
256 /// insert the k-D source.
257 /// 2. k-D -> (n-1)-D InsertStridedSlice op
258 /// 3. InsertOp that is the reverse of 1.
259 ///
260 /// [DecomposeNDExtractStridedSlice]
261 /// ================================
262 /// For such cases, we can rewrite it to ExtractOp/ExtractElementOp + lower
263 /// rank ExtractStridedSliceOp + InsertOp/InsertElementOp for the n-D case.
264 void populateVectorInsertExtractStridedSliceDecompositionPatterns(

288 /// Collect a set of pattern to unroll vector operations to a smaller shapes.
289 /// `options` structure controls which operations are unrolled and the target
290 /// shape.
291 /// `op` is unrolled to the `targetShape` as follows, for each of its operands:
292 /// 1. the unrolled type `unrolledVectorType` and number of unrolled instances
293 /// `numUnrolledInstances` are computed from the `targetShape`. For now it is
294 /// assumed the unrolling factors divide the vector sizes.
295 /// 2. ExtractStridedSlice are created to break-up the vector operands.
296 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
297 /// result.
298 /// 4. InsertStridedSlice are inserted to re-assemble the slices into the
299 /// original vectore shape.
300 ///
301 /// Example:
302 ///
303 /// opA(operand0, operand1) // numUnrolledInstances = 3
304 ///
305 /// operand0 operand1
306 /// | |
307 /// fork fork
308 /// <----------gather all fork ops --------->
309 /// /|\ /|\
310 /// f00 f01 f02 f10 f11 f12
311 /// <---------- clone op 3 times --------->
312 /// opA0(f00, f10), opA1(f01, f11), opA2(f02, f12)
313 /// \ | /
314 /// <-------------------- join ------------------------->
315 ///
316 /// Other local patterns then kick in iteratively (including DCE) and compose
317 /// to combine the ExtractStridedSlice/InsertStridedSlice.
318 void populateVectorUnrollPatterns(RewritePatternSet &patterns,
319  const UnrollVectorOptions &options);
