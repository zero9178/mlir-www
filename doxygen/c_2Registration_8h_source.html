20 // Dialect registration declarations.
21 // Registration entry-points for each dialect are declared using the common
22 // MLIR_DECLARE_DIALECT_REGISTRATION_CAPI macro, which takes the dialect
23 // API name (i.e. "Func", "Tensor", "Linalg") and namespace (i.e. "func",
24 // "tensor", "linalg"). The following declarations are produced:
25 //
26 // /// Gets the above hook methods in struct form for a dialect by namespace.
27 // /// This is intended to facilitate dynamic lookup and registration of
28 // /// dialects via a plugin facility based on shared library symbol lookup.
29 // const MlirDialectHandle *mlirGetDialectHandle__{NAMESPACE}__();
30 //
31 // This is done via a common macro to facilitate future expansion to
32 // registration schemes.
33 //===----------------------------------------------------------------------===//
34

36  const void *ptr;
37 };
