21 /// Base class for AffineExpr visitors/walkers.
22 ///
23 /// AffineExpr visitors are used when you want to perform different actions
24 /// for different kinds of AffineExprs without having to use lots of casts
25 /// and a big switch instruction.
26 ///
27 /// To define your own visitor, inherit from this class, specifying your
28 /// new type for the 'SubClass' template parameter, and "override" visitXXX
29 /// functions in your class. This class is defined in terms of statically
30 /// resolved overloading, not virtual functions.
31 ///
32 /// For example, here is a visitor that counts the number of for AffineDimExprs
33 /// in an AffineExpr.
34 ///
35 /// /// Declare the class. Note that we derive from AffineExprVisitor
36 /// /// instantiated with our new subclasses_ type.
37 ///
38 /// struct DimExprCounter : public AffineExprVisitor<DimExprCounter> {
39 /// unsigned numDimExprs;
40 /// DimExprCounter() : numDimExprs(0) {}
41 /// void visitDimExpr(AffineDimExpr expr) { ++numDimExprs; }
42 /// };
43 ///
44 /// And this class would be used like this:
45 /// DimExprCounter dec;
46 /// dec.visit(affineExpr);
47 /// numDimExprs = dec.numDimExprs;
48 ///
49 /// AffineExprVisitor provides visit methods for the following binary affine
50 /// op expressions:
51 /// AffineBinaryAddOpExpr, AffineBinaryMulOpExpr,
52 /// AffineBinaryModOpExpr, AffineBinaryFloorDivOpExpr,
53 /// AffineBinaryCeilDivOpExpr. Note that default implementations of these
54 /// methods will call the general AffineBinaryOpExpr method.
55 ///
56 /// In addition, visit methods are provided for the following affine
57 // expressions: AffineConstantExpr, AffineDimExpr, and
58 // AffineSymbolExpr.
59 ///
60 /// Note that if you don't implement visitXXX for some affine expression type,
61 /// the visitXXX method for Instruction superclass will be invoked.
62 ///
63 /// Note that this class is specifically designed as a template to avoid
64 /// virtual function call overhead. Defining and using a AffineExprVisitor is
65 /// just as efficient as having your own switch instruction over the instruction
66 /// opcode.
67 
68 template <typename SubClass, typename RetTy = void> class AffineExprVisitor {
69  //===--------------------------------------------------------------------===//
70  // Interface code - This is the public interface of the AffineExprVisitor
71  // that you use to visit affine expressions...
72 public:
73  // Function to walk an AffineExpr (in post order).
74  RetTy walkPostOrder(AffineExpr expr) {
75  static_assert(std::is_base_of<AffineExprVisitor, SubClass>::value,
76  "Must instantiate with a derived type of AffineExprVisitor");
77  switch (expr.getKind()) {
78  case AffineExprKind::Add: {
79  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
80  walkOperandsPostOrder(binOpExpr);
81  return static_cast<SubClass *>(this)->visitAddExpr(binOpExpr);
82  }
83  case AffineExprKind::Mul: {
84  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
85  walkOperandsPostOrder(binOpExpr);
86  return static_cast<SubClass *>(this)->visitMulExpr(binOpExpr);
87  }
88  case AffineExprKind::Mod: {
89  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
90  walkOperandsPostOrder(binOpExpr);
91  return static_cast<SubClass *>(this)->visitModExpr(binOpExpr);
92  }
93  case AffineExprKind::FloorDiv: {
94  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
95  walkOperandsPostOrder(binOpExpr);
96  return static_cast<SubClass *>(this)->visitFloorDivExpr(binOpExpr);
97  }
98  case AffineExprKind::CeilDiv: {
99  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
100  walkOperandsPostOrder(binOpExpr);
101  return static_cast<SubClass *>(this)->visitCeilDivExpr(binOpExpr);
102  }
103  case AffineExprKind::Constant:
104  return static_cast<SubClass *>(this)->visitConstantExpr(
105  expr.cast<AffineConstantExpr>());
106  case AffineExprKind::DimId:
107  return static_cast<SubClass *>(this)->visitDimExpr(
108  expr.cast<AffineDimExpr>());
109  case AffineExprKind::SymbolId:
110  return static_cast<SubClass *>(this)->visitSymbolExpr(
111  expr.cast<AffineSymbolExpr>());
112  }
113  }
114 
115  // Function to visit an AffineExpr.
116  RetTy visit(AffineExpr expr) {
117  static_assert(std::is_base_of<AffineExprVisitor, SubClass>::value,
118  "Must instantiate with a derived type of AffineExprVisitor");
119  switch (expr.getKind()) {
120  case AffineExprKind::Add: {
121  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
122  return static_cast<SubClass *>(this)->visitAddExpr(binOpExpr);
123  }
124  case AffineExprKind::Mul: {
125  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
126  return static_cast<SubClass *>(this)->visitMulExpr(binOpExpr);
127  }
128  case AffineExprKind::Mod: {
129  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
130  return static_cast<SubClass *>(this)->visitModExpr(binOpExpr);
131  }
132  case AffineExprKind::FloorDiv: {
133  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
134  return static_cast<SubClass *>(this)->visitFloorDivExpr(binOpExpr);
135  }
136  case AffineExprKind::CeilDiv: {
137  auto binOpExpr = expr.cast<AffineBinaryOpExpr>();
138  return static_cast<SubClass *>(this)->visitCeilDivExpr(binOpExpr);
139  }
140  case AffineExprKind::Constant:
141  return static_cast<SubClass *>(this)->visitConstantExpr(
142  expr.cast<AffineConstantExpr>());
143  case AffineExprKind::DimId:
144  return static_cast<SubClass *>(this)->visitDimExpr(
145  expr.cast<AffineDimExpr>());
146  case AffineExprKind::SymbolId:
147  return static_cast<SubClass *>(this)->visitSymbolExpr(
148  expr.cast<AffineSymbolExpr>());
149  }
150  llvm_unreachable("Unknown AffineExpr");
151  }

153  //===--------------------------------------------------------------------===//
154  // Visitation functions... these functions provide default fallbacks in case
155  // the user does not specify what to do for a particular instruction type.
156  // The default behavior is to generalize the instruction type to its subtype
157  // and try visiting the subtype. All of this should be inlined perfectly,
158  // because there are no virtual functions to get in the way.
159  //
160 
161  // Default visit methods. Note that the default op-specific binary op visit
162  // methods call the general visitAffineBinaryOpExpr visit method.
163  RetTy visitAffineBinaryOpExpr(AffineBinaryOpExpr expr) { return RetTy(); }
164  RetTy visitAddExpr(AffineBinaryOpExpr expr) {
165  return static_cast<SubClass *>(this)->visitAffineBinaryOpExpr(expr);
166  }
167  RetTy visitMulExpr(AffineBinaryOpExpr expr) {
168  return static_cast<SubClass *>(this)->visitAffineBinaryOpExpr(expr);
169  }
170  RetTy visitModExpr(AffineBinaryOpExpr expr) {
171  return static_cast<SubClass *>(this)->visitAffineBinaryOpExpr(expr);
172  }
173  RetTy visitFloorDivExpr(AffineBinaryOpExpr expr) {
174  return static_cast<SubClass *>(this)->visitAffineBinaryOpExpr(expr);
175  }
176  RetTy visitCeilDivExpr(AffineBinaryOpExpr expr) {
177  return static_cast<SubClass *>(this)->visitAffineBinaryOpExpr(expr);
178  }
179  RetTy visitConstantExpr(AffineConstantExpr expr) { return RetTy(); }
180  RetTy visitDimExpr(AffineDimExpr expr) { return RetTy(); }
181  RetTy visitSymbolExpr(AffineSymbolExpr expr) { return RetTy(); }
182 
183 private:
184  // Walk the operands - each operand is itself walked in post order.
185  RetTy walkOperandsPostOrder(AffineBinaryOpExpr expr) {
186  walkPostOrder(expr.getLHS());
187  walkPostOrder(expr.getRHS());
188  }

191 // This class is used to flatten a pure affine expression (AffineExpr,
192 // which is in a tree form) into a sum of products (w.r.t constants) when
193 // possible, and in that process simplifying the expression. For a modulo,
194 // floordiv, or a ceildiv expression, an additional identifier, called a local
195 // identifier, is introduced to rewrite the expression as a sum of product
196 // affine expression. Each local identifier is always and by construction a
197 // floordiv of a pure add/mul affine function of dimensional, symbolic, and
198 // other local identifiers, in a non-mutually recursive way. Hence, every local
199 // identifier can ultimately always be recovered as an affine function of
200 // dimensional and symbolic identifiers (involving floordiv's); note however
201 // that by AffineExpr construction, some floordiv combinations are converted to
202 // mod's. The result of the flattening is a flattened expression and a set of
203 // constraints involving just the local variables.
204 //
205 // d2 + (d0 + d1) floordiv 4 is flattened to d2 + q where 'q' is the local
206 // variable introduced, with localVarCst containing 4*q <= d0 + d1 <= 4*q + 3.
207 //
208 // The simplification performed includes the accumulation of contributions for
209 // each dimensional and symbolic identifier together, the simplification of
210 // floordiv/ceildiv/mod expressions and other simplifications that in turn
211 // happen as a result. A simplification that this flattening naturally performs
212 // is of simplifying the numerator and denominator of floordiv/ceildiv, and
213 // folding a modulo expression to a zero, if possible. Three examples are below:
214 //
215 // (d0 + 3 * d1) + d0) - 2 * d1) - d0 simplified to d0 + d1
216 // (d0 - d0 mod 4 + 4) mod 4 simplified to 0
217 // (3*d0 + 2*d1 + d0) floordiv 2 + d1 simplified to 2*d0 + 2*d1
218 //
219 // The way the flattening works for the second example is as follows: d0 % 4 is
220 // replaced by d0 - 4*q with q being introduced: the expression then simplifies
221 // to: (d0 - (d0 - 4q) + 4) = 4q + 4, modulo of which w.r.t 4 simplifies to
222 // zero. Note that an affine expression may not always be expressible purely as
223 // a sum of products involving just the original dimensional and symbolic
224 // identifiers due to the presence of modulo/floordiv/ceildiv expressions that
225 // may not be eliminated after simplification; in such cases, the final
226 // expression can be reconstructed by replacing the local identifiers with their
227 // corresponding explicit form stored in 'localExprs' (note that each of the
228 // explicit forms itself would have been simplified).
229 //
230 // The expression walk method here performs a linear time post order walk that
231 // performs the above simplifications through visit methods, with partial
232 // results being stored in 'operandExprStack'. When a parent expr is visited,
233 // the flattened expressions corresponding to its two operands would already be
234 // on the stack - the parent expression looks at the two flattened expressions
235 // and combines the two. It pops off the operand expressions and pushes the
236 // combined result (although this is done in-place on its LHS operand expr).
237 // When the walk is completed, the flattened form of the top-level expression
238 // would be left on the stack.
239 //
240 // A flattener can be repeatedly used for multiple affine expressions that bind
241 // to the same operands, for example, for all result expressions of an
242 // AffineMap or AffineValueMap. In such cases, using it for multiple expressions
243 // is more efficient than creating a new flattener for each expression since
244 // common identical div and mod expressions appearing across different
245 // expressions are mapped to the same local identifier (same column position in
246 // 'localVarCst').
247 class SimpleAffineExprFlattener
248  : public AffineExprVisitor<SimpleAffineExprFlattener> {
249 public:
250  // Flattend expression layout: [dims, symbols, locals, constant]
251  // Stack that holds the LHS and RHS operands while visiting a binary op expr.
252  // In future, consider adding a prepass to determine how big the SmallVector's
253  // will be, and linearize this to std::vector<int64_t> to prevent
254  // SmallVector moves on re-allocation.
255  std::vector<SmallVector<int64_t, 8>> operandExprStack;
256 
257  unsigned numDims;
258  unsigned numSymbols;
259 
260  // Number of newly introduced identifiers to flatten mod/floordiv/ceildiv's.
261  unsigned numLocals;
262 
263  // AffineExpr's corresponding to the floordiv/ceildiv/mod expressions for
264  // which new identifiers were introduced; if the latter do not get canceled
265  // out, these expressions can be readily used to reconstruct the AffineExpr
266  // (tree) form. Note that these expressions themselves would have been
267  // simplified (recursively) by this pass. Eg. d0 + (d0 + 2*d1 + d0) ceildiv 4
268  // will be simplified to d0 + q, where q = (d0 + d1) ceildiv 2. (d0 + d1)
269  // ceildiv 2 would be the local expression stored for q.
270  SmallVector<AffineExpr, 4> localExprs;
