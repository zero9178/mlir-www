82 /// Replaces the given op with the contents of the given single-block region,
83 /// using the operands of the block terminator to replace operation results.
84 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,
85  Region &region, ValueRange blockArgs = {}) {
86  assert(llvm::hasSingleElement(region) && "expected single-region block");
87  Block *block = &region.front();
88  Operation *terminator = block->getTerminator();
89  ValueRange results = terminator->getOperands();
90  rewriter.mergeBlockBefore(block, op, blockArgs);
91  rewriter.replaceOp(op, results);
92  rewriter.eraseOp(terminator);
93 }

193 // cf.br ^bb3(%x : i64)
194 // ^bb2: // pred: ^bb0
195 // %y = "test.val2"() : () -> i64
196 // cf.br ^bb3(%y : i64)
197 // ^bb3(%z: i64): // 2 preds: ^bb1, ^bb2
198 // "test.bar"(%z) : (i64) -> ()
199 // return
200 // }

202 struct MultiBlockExecuteInliner : public OpRewritePattern<ExecuteRegionOp> {
203  using OpRewritePattern<ExecuteRegionOp>::OpRewritePattern;
204 
205  LogicalResult matchAndRewrite(ExecuteRegionOp op,
206  PatternRewriter &rewriter) const override {
207  if (!isa<FunctionOpInterface, ExecuteRegionOp>(op->getParentOp()))
208  return failure();
209 
210  Block *prevBlock = op->getBlock();
211  Block *postBlock = rewriter.splitBlock(prevBlock, op->getIterator());
212  rewriter.setInsertionPointToEnd(prevBlock);
213 
214  rewriter.create<cf::BranchOp>(op.getLoc(), &op.getRegion().front());
215 
216  for (Block &blk : op.getRegion()) {
217  if (YieldOp yieldOp = dyn_cast<YieldOp>(blk.getTerminator())) {
218  rewriter.setInsertionPoint(yieldOp);
219  rewriter.create<cf::BranchOp>(yieldOp.getLoc(), postBlock,
220  yieldOp.getResults());
221  rewriter.eraseOp(yieldOp);
222  }
223  }
224 
225  rewriter.inlineRegionBefore(op.getRegion(), postBlock);
226  SmallVector<Value> blockArgs;
227 
228  for (auto res : op.getResults())
229  blockArgs.push_back(postBlock->addArgument(res.getType(), res.getLoc()));
230 
231  rewriter.replaceOp(op, blockArgs);
232  return success();
233  }
234 };

273 void ForOp::build(OpBuilder &builder, OperationState &result, Value lb,
274  Value ub, Value step, ValueRange iterArgs,
275  BodyBuilderFn bodyBuilder) {
276  result.addOperands({lb, ub, step});
277  result.addOperands(iterArgs);
278  for (Value v : iterArgs)
279  result.addTypes(v.getType());
280  Region *bodyRegion = result.addRegion();
281  bodyRegion->push_back(new Block);
282  Block &bodyBlock = bodyRegion->front();
283  bodyBlock.addArgument(builder.getIndexType(), result.location);
284  for (Value v : iterArgs)
285  bodyBlock.addArgument(v.getType(), v.getLoc());
286 
287  // Create the default terminator if the builder is not provided and if the
288  // iteration arguments are not provided. Otherwise, leave this to the caller
289  // because we don't know which values to return from the loop.
290  if (iterArgs.empty() && !bodyBuilder) {
291  ForOp::ensureTerminator(*bodyRegion, builder, result.location);
292  } else if (bodyBuilder) {
293  OpBuilder::InsertionGuard guard(builder);
294  builder.setInsertionPointToStart(&bodyBlock);
295  bodyBuilder(builder, result.location, bodyBlock.getArgument(0),
296  bodyBlock.getArguments().drop_front());
297  }
298 }

318  // Check that the body defines as single block argument for the induction
319  // variable.
320  auto *body = getBody();
321  if (!body->getArgument(0).getType().isIndex())
322  return emitOpError(

327  if (opNumResults == 0)
328  return success();

331  return emitOpError(

400 ParseResult ForOp::parse(OpAsmParser &parser, OperationState &result) {
401  auto &builder = parser.getBuilder();
402  Type indexType = builder.getIndexType();
403 
404  OpAsmParser::Argument inductionVariable;
405  inductionVariable.type = indexType;
406  OpAsmParser::UnresolvedOperand lb, ub, step;
407 
408  // Parse the induction variable followed by '='.
409  if (parser.parseArgument(inductionVariable) || parser.parseEqual() ||
410  // Parse loop bounds.
411  parser.parseOperand(lb) ||
412  parser.resolveOperand(lb, indexType, result.operands) ||
413  parser.parseKeyword("to") || parser.parseOperand(ub) ||
414  parser.resolveOperand(ub, indexType, result.operands) ||
415  parser.parseKeyword("step") || parser.parseOperand(step) ||
416  parser.resolveOperand(step, indexType, result.operands))
417  return failure();
418 
419  // Parse the optional initial iteration arguments.
420  SmallVector<OpAsmParser::Argument, 4> regionArgs;
421  SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
422  regionArgs.push_back(inductionVariable);
423 
424  if (succeeded(parser.parseOptionalKeyword("iter_args"))) {
425  // Parse assignment list and results type list.
426  if (parser.parseAssignmentList(regionArgs, operands) ||
427  parser.parseArrowTypeList(result.types))
428  return failure();
429 
430  // Resolve input operands.
431  for (auto argOperandType :
432  llvm::zip(llvm::drop_begin(regionArgs), operands, result.types)) {
433  Type type = std::get<2>(argOperandType);
434  std::get<0>(argOperandType).type = type;
435  if (parser.resolveOperand(std::get<1>(argOperandType), type,
436  result.operands))
437  return failure();
438  }
439  }
440 
441  if (regionArgs.size() != result.types.size() + 1)
442  return parser.emitError(
443  parser.getNameLoc(),
444  "mismatch in number of loop-carried values and defined values");
445 
446  // Parse the body region.
447  Region *body = result.addRegion();
448  if (parser.parseRegion(*body, regionArgs))
449  return failure();
450 
451  ForOp::ensureTerminator(*body, builder, result.location);
452 
453  // Parse the optional attribute list.
454  if (parser.parseOptionalAttrDict(result.attributes))
455  return failure();
456 
457  return success();
458 }

504 LoopNest mlir::scf::buildLoopNest(
505  OpBuilder &builder, Location loc, ValueRange lbs, ValueRange ubs,
506  ValueRange steps, ValueRange iterArgs,
507  function_ref<ValueVector(OpBuilder &, Location, ValueRange, ValueRange)>
508  bodyBuilder) {
509  assert(lbs.size() == ubs.size() &&
510  "expected the same number of lower and upper bounds");
511  assert(lbs.size() == steps.size() &&
512  "expected the same number of lower bounds and steps");
513 
514  // If there are no bounds, call the body-building function and return early.
515  if (lbs.empty()) {
516  ValueVector results =
517  bodyBuilder ? bodyBuilder(builder, loc, ValueRange(), iterArgs)
518  : ValueVector();
519  assert(results.size() == iterArgs.size() &&
520  "loop nest body must return as many values as loop has iteration "
521  "arguments");
522  return LoopNest();
523  }
524 
525  // First, create the loop structure iteratively using the body-builder
526  // callback of `ForOp::build`. Do not create `YieldOp`s yet.
527  OpBuilder::InsertionGuard guard(builder);
528  SmallVector<scf::ForOp, 4> loops;
529  SmallVector<Value, 4> ivs;
530  loops.reserve(lbs.size());
531  ivs.reserve(lbs.size());
532  ValueRange currentIterArgs = iterArgs;
533  Location currentLoc = loc;
534  for (unsigned i = 0, e = lbs.size(); i < e; ++i) {
535  auto loop = builder.create<scf::ForOp>(
536  currentLoc, lbs[i], ubs[i], steps[i], currentIterArgs,
537  [&](OpBuilder &nestedBuilder, Location nestedLoc, Value iv,
538  ValueRange args) {
539  ivs.push_back(iv);
540  // It is safe to store ValueRange args because it points to block
541  // arguments of a loop operation that we also own.
542  currentIterArgs = args;
543  currentLoc = nestedLoc;
544  });
545  // Set the builder to point to the body of the newly created loop. We don't
546  // do this in the callback because the builder is reset when the callback
547  // returns.
548  builder.setInsertionPointToStart(loop.getBody());
549  loops.push_back(loop);
550  }
551 
552  // For all loops but the innermost, yield the results of the nested loop.
553  for (unsigned i = 0, e = loops.size() - 1; i < e; ++i) {
554  builder.setInsertionPointToEnd(loops[i].getBody());
555  builder.create<scf::YieldOp>(loc, loops[i + 1].getResults());
556  }
557 
558  // In the body of the innermost loop, call the body building function if any
559  // and yield its results.
560  builder.setInsertionPointToStart(loops.back().getBody());
561  ValueVector results = bodyBuilder
562  ? bodyBuilder(builder, currentLoc, ivs,
563  loops.back().getRegionIterArgs())
564  : ValueVector();
565  assert(results.size() == iterArgs.size() &&
566  "loop nest body must return as many values as loop has iteration "
567  "arguments");
568  builder.setInsertionPointToEnd(loops.back().getBody());
569  builder.create<scf::YieldOp>(loc, results);
570 
571  // Return the loops.
572  LoopNest res;
573  res.loops.assign(loops.begin(), loops.end());
574  return res;
575 }

593 // Fold away ForOp iter arguments when:
594 // 1) The op yields the iter arguments.
595 // 2) The iter arguments have no use and the corresponding outer region
596 // iterators (inputs) are yielded.
597 // 3) The iter arguments have no use and the corresponding (operation) results
598 // have no use.
599 //
600 // These arguments must be defined outside of
601 // the ForOp region and can just be forwarded after simplifying the op inits,
602 // yields and returns.
603 //
604 // The implementation uses `mergeBlockBefore` to steal the content of the
605 // original ForOp and avoid cloning.
606 struct ForOpIterArgsFolder : public OpRewritePattern<scf::ForOp> {
607  using OpRewritePattern<scf::ForOp>::OpRewritePattern;
608 
609  LogicalResult matchAndRewrite(scf::ForOp forOp,
610  PatternRewriter &rewriter) const final {
611  bool canonicalize = false;
612  Block &block = forOp.getRegion().front();
613  auto yieldOp = cast<scf::YieldOp>(block.getTerminator());
614 
615  // An internal flat vector of block transfer
616  // arguments `newBlockTransferArgs` keeps the 1-1 mapping of original to
617  // transformed block argument mappings. This plays the role of a
618  // BlockAndValueMapping for the particular use case of calling into
619  // `mergeBlockBefore`.
620  SmallVector<bool, 4> keepMask;
621  keepMask.reserve(yieldOp.getNumOperands());
622  SmallVector<Value, 4> newBlockTransferArgs, newIterArgs, newYieldValues,
623  newResultValues;
624  newBlockTransferArgs.reserve(1 + forOp.getNumIterOperands());
625  newBlockTransferArgs.push_back(Value()); // iv placeholder with null value
626  newIterArgs.reserve(forOp.getNumIterOperands());
627  newYieldValues.reserve(yieldOp.getNumOperands());
628  newResultValues.reserve(forOp.getNumResults());
629  for (auto it : llvm::zip(forOp.getIterOperands(), // iter from outside
630  forOp.getRegionIterArgs(), // iter inside region
631  forOp.getResults(), // op results
632  yieldOp.getOperands() // iter yield
633  )) {
634  // Forwarded is `true` when:
635  // 1) The region `iter` argument is yielded.
636  // 2) The region `iter` argument has no use, and the corresponding iter
637  // operand (input) is yielded.
638  // 3) The region `iter` argument has no use, and the corresponding op
639  // result has no use.
640  bool forwarded = ((std::get<1>(it) == std::get<3>(it)) ||
641  (std::get<1>(it).use_empty() &&
642  (std::get<0>(it) == std::get<3>(it) ||
643  std::get<2>(it).use_empty())));
644  keepMask.push_back(!forwarded);
645  canonicalize |= forwarded;
646  if (forwarded) {
647  newBlockTransferArgs.push_back(std::get<0>(it));
648  newResultValues.push_back(std::get<0>(it));
649  continue;
650  }
651  newIterArgs.push_back(std::get<0>(it));
652  newYieldValues.push_back(std::get<3>(it));
653  newBlockTransferArgs.push_back(Value()); // placeholder with null value
654  newResultValues.push_back(Value()); // placeholder with null value
655  }
656 
657  if (!canonicalize)
658  return failure();
659 
660  scf::ForOp newForOp = rewriter.create<scf::ForOp>(
661  forOp.getLoc(), forOp.getLowerBound(), forOp.getUpperBound(),
662  forOp.getStep(), newIterArgs);
663  Block &newBlock = newForOp.getRegion().front();
664 
665  // Replace the null placeholders with newly constructed values.
666  newBlockTransferArgs[0] = newBlock.getArgument(0); // iv
667  for (unsigned idx = 0, collapsedIdx = 0, e = newResultValues.size();
668  idx != e; ++idx) {
669  Value &blockTransferArg = newBlockTransferArgs[1 + idx];
670  Value &newResultVal = newResultValues[idx];
671  assert((blockTransferArg && newResultVal) ||
672  (!blockTransferArg && !newResultVal));
673  if (!blockTransferArg) {
674  blockTransferArg = newForOp.getRegionIterArgs()[collapsedIdx];
675  newResultVal = newForOp.getResult(collapsedIdx++);
676  }
677  }
678 
679  Block &oldBlock = forOp.getRegion().front();
680  assert(oldBlock.getNumArguments() == newBlockTransferArgs.size() &&
681  "unexpected argument size mismatch");
682 
683  // No results case: the scf::ForOp builder already created a zero
684  // result terminator. Merge before this terminator and just get rid of the
685  // original terminator that has been merged in.
686  if (newIterArgs.empty()) {
687  auto newYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
688  rewriter.mergeBlockBefore(&oldBlock, newYieldOp, newBlockTransferArgs);
689  rewriter.eraseOp(newBlock.getTerminator()->getPrevNode());
690  rewriter.replaceOp(forOp, newResultValues);
691  return success();
692  }
693 
694  // No terminator case: merge and rewrite the merged terminator.
695  auto cloneFilteredTerminator = [&](scf::YieldOp mergedTerminator) {
696  OpBuilder::InsertionGuard g(rewriter);
697  rewriter.setInsertionPoint(mergedTerminator);
698  SmallVector<Value, 4> filteredOperands;
699  filteredOperands.reserve(newResultValues.size());
700  for (unsigned idx = 0, e = keepMask.size(); idx < e; ++idx)
701  if (keepMask[idx])
702  filteredOperands.push_back(mergedTerminator.getOperand(idx));
703  rewriter.create<scf::YieldOp>(mergedTerminator.getLoc(),
704  filteredOperands);
705  };
706 
707  rewriter.mergeBlocks(&oldBlock, &newBlock, newBlockTransferArgs);
708  auto mergedYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
709  cloneFilteredTerminator(mergedYieldOp);
710  rewriter.eraseOp(mergedYieldOp);
711  rewriter.replaceOp(forOp, newResultValues);
712  return success();
713  }
714 };

716 /// Rewriting pattern that erases loops that are known not to iterate, replaces
717 /// single-iteration loops with their bodies, and removes empty loops that
718 /// iterate at least once and only return values defined outside of the loop.
719 struct SimplifyTrivialLoops : public OpRewritePattern<ForOp> {
720  using OpRewritePattern<ForOp>::OpRewritePattern;
721 
722  LogicalResult matchAndRewrite(ForOp op,
723  PatternRewriter &rewriter) const override {
724  // If the upper bound is the same as the lower bound, the loop does not
725  // iterate, just remove it.
726  if (op.getLowerBound() == op.getUpperBound()) {
727  rewriter.replaceOp(op, op.getIterOperands());
728  return success();
729  }
730 
731  auto lb = op.getLowerBound().getDefiningOp<arith::ConstantOp>();
732  auto ub = op.getUpperBound().getDefiningOp<arith::ConstantOp>();
733  if (!lb || !ub)
734  return failure();
735 
736  // If the loop is known to have 0 iterations, remove it.
737  llvm::APInt lbValue = lb.getValue().cast<IntegerAttr>().getValue();
738  llvm::APInt ubValue = ub.getValue().cast<IntegerAttr>().getValue();
739  if (lbValue.sge(ubValue)) {
740  rewriter.replaceOp(op, op.getIterOperands());
741  return success();
742  }
743 
744  auto step = op.getStep().getDefiningOp<arith::ConstantOp>();
745  if (!step)
746  return failure();
747 
748  // If the loop is known to have 1 iteration, inline its body and remove the
749  // loop.
750  llvm::APInt stepValue = step.getValue().cast<IntegerAttr>().getValue();
751  if ((lbValue + stepValue).sge(ubValue)) {
752  SmallVector<Value, 4> blockArgs;
753  blockArgs.reserve(op.getNumIterOperands() + 1);
754  blockArgs.push_back(op.getLowerBound());
755  llvm::append_range(blockArgs, op.getIterOperands());
756  replaceOpWithRegion(rewriter, op, op.getLoopBody(), blockArgs);
757  return success();
758  }
759 
760  // Now we are left with loops that have more than 1 iterations.
761  Block &block = op.getRegion().front();
762  if (!llvm::hasSingleElement(block))
763  return failure();
764  // If the loop is empty, iterates at least once, and only returns values
765  // defined outside of the loop, remove it and replace it with yield values.
766  auto yieldOp = cast<scf::YieldOp>(block.getTerminator());
767  auto yieldOperands = yieldOp.getOperands();
768  if (llvm::any_of(yieldOperands,
769  [&](Value v) { return !op.isDefinedOutsideOfLoop(v); }))
770  return failure();
771  rewriter.replaceOp(op, yieldOperands);
772  return success();
773  }

776 /// Perform a replacement of one iter OpOperand of an scf.for to the
777 /// `replacement` value which is expected to be the source of a tensor.cast.
778 /// tensor.cast ops are inserted inside the block to account for the type cast.
779 static ForOp replaceTensorCastForOpIterArg(PatternRewriter &rewriter,
780  OpOperand &operand,
781  Value replacement) {
782  Type oldType = operand.get().getType(), newType = replacement.getType();
783  assert(oldType.isa<RankedTensorType>() && newType.isa<RankedTensorType>() &&
784  "expected ranked tensor types");
785 
786  // 1. Create new iter operands, exactly 1 is replaced.
787  ForOp forOp = cast<ForOp>(operand.getOwner());
788  assert(operand.getOperandNumber() >= forOp.getNumControlOperands() &&
789  "expected an iter OpOperand");
790  if (operand.get().getType() == replacement.getType())
791  return forOp;
792  SmallVector<Value> newIterOperands;
793  for (OpOperand &opOperand : forOp.getIterOpOperands()) {
794  if (opOperand.getOperandNumber() == operand.getOperandNumber()) {
795  newIterOperands.push_back(replacement);
796  continue;
797  }
798  newIterOperands.push_back(opOperand.get());
799  }
800 
801  // 2. Create the new forOp shell.
802  scf::ForOp newForOp = rewriter.create<scf::ForOp>(
803  forOp.getLoc(), forOp.getLowerBound(), forOp.getUpperBound(),
804  forOp.getStep(), newIterOperands);
805  Block &newBlock = newForOp.getRegion().front();
806  SmallVector<Value, 4> newBlockTransferArgs(newBlock.getArguments().begin(),
807  newBlock.getArguments().end());
808 
809  // 3. Inject an incoming cast op at the beginning of the block for the bbArg
810  // corresponding to the `replacement` value.
811  OpBuilder::InsertionGuard g(rewriter);
812  rewriter.setInsertionPoint(&newBlock, newBlock.begin());
813  BlockArgument newRegionIterArg = newForOp.getRegionIterArgForOpOperand(
814  newForOp->getOpOperand(operand.getOperandNumber()));
815  Value castIn = rewriter.create<tensor::CastOp>(newForOp.getLoc(), oldType,
816  newRegionIterArg);
817  newBlockTransferArgs[newRegionIterArg.getArgNumber()] = castIn;
818 
819  // 4. Steal the old block ops, mapping to the newBlockTransferArgs.
820  Block &oldBlock = forOp.getRegion().front();
821  rewriter.mergeBlocks(&oldBlock, &newBlock, newBlockTransferArgs);
822 
823  // 5. Inject an outgoing cast op at the end of the block and yield it instead.
824  auto clonedYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
825  rewriter.setInsertionPoint(clonedYieldOp);
826  unsigned yieldIdx =
827  newRegionIterArg.getArgNumber() - forOp.getNumInductionVars();
828  Value castOut = rewriter.create<tensor::CastOp>(
829  newForOp.getLoc(), newType, clonedYieldOp.getOperand(yieldIdx));
830  SmallVector<Value> newYieldOperands = clonedYieldOp.getOperands();
831  newYieldOperands[yieldIdx] = castOut;
832  rewriter.create<scf::YieldOp>(newForOp.getLoc(), newYieldOperands);
833  rewriter.eraseOp(clonedYieldOp);
834 
835  // 6. Inject an outgoing cast op after the forOp.
836  rewriter.setInsertionPointAfter(newForOp);
837  SmallVector<Value> newResults = newForOp.getResults();
838  newResults[yieldIdx] = rewriter.create<tensor::CastOp>(
839  newForOp.getLoc(), oldType, newResults[yieldIdx]);
840 
841  return newForOp;
842 }

844 /// Fold scf.for iter_arg/result pairs that go through incoming/ougoing
845 /// a tensor.cast op pair so as to pull the tensor.cast inside the scf.for:
846 ///
847 /// ```
848 /// %0 = tensor.cast %t0 : tensor<32x1024xf32> to tensor<?x?xf32>
849 /// %1 = scf.for %i = %c0 to %c1024 step %c32 iter_args(%iter_t0 = %0)
850 /// -> (tensor<?x?xf32>) {
851 /// %2 = call @do(%iter_t0) : (tensor<?x?xf32>) -> tensor<?x?xf32>
852 /// scf.yield %2 : tensor<?x?xf32>
853 /// }
854 /// %2 = tensor.cast %1 : tensor<?x?xf32> to tensor<32x1024xf32>
855 /// use_of(%2)
856 /// ```
857 ///
858 /// folds into:
859 ///
860 /// ```
861 /// %0 = scf.for %arg2 = %c0 to %c1024 step %c32 iter_args(%arg3 = %arg0)
862 /// -> (tensor<32x1024xf32>) {
863 /// %2 = tensor.cast %arg3 : tensor<32x1024xf32> to tensor<?x?xf32>
864 /// %3 = call @do(%2) : (tensor<?x?xf32>) -> tensor<?x?xf32>
865 /// %4 = tensor.cast %3 : tensor<?x?xf32> to tensor<32x1024xf32>
866 /// scf.yield %4 : tensor<32x1024xf32>
867 /// }
868 /// use_of(%0)
869 /// ```
870 struct ForOpTensorCastFolder : public OpRewritePattern<ForOp> {
871  using OpRewritePattern<ForOp>::OpRewritePattern;
872 
873  LogicalResult matchAndRewrite(ForOp op,
874  PatternRewriter &rewriter) const override {
875  for (auto it : llvm::zip(op.getIterOpOperands(), op.getResults())) {
876  OpOperand &iterOpOperand = std::get<0>(it);
877  auto incomingCast = iterOpOperand.get().getDefiningOp<tensor::CastOp>();
878  if (!incomingCast)
879  continue;
880  if (!std::get<1>(it).hasOneUse())
881  continue;
882  auto outgoingCastOp =
883  dyn_cast<tensor::CastOp>(*std::get<1>(it).user_begin());
884  if (!outgoingCastOp)
885  continue;
886 
887  // Must be a tensor.cast op pair with matching types.
888  if (outgoingCastOp.getResult().getType() !=
889  incomingCast.source().getType())
890  continue;
891 
892  // Create a new ForOp with that iter operand replaced.
893  auto newForOp = replaceTensorCastForOpIterArg(rewriter, iterOpOperand,
894  incomingCast.source());
895 
896  // Insert outgoing cast and use it to replace the corresponding result.
897  rewriter.setInsertionPointAfter(newForOp);
898  SmallVector<Value> replacements = newForOp.getResults();
899  unsigned returnIdx =
900  iterOpOperand.getOperandNumber() - op.getNumControlOperands();
901  replacements[returnIdx] = rewriter.create<tensor::CastOp>(
902  op.getLoc(), incomingCast.dest().getType(), replacements[returnIdx]);
903  rewriter.replaceOp(op, replacements);
904  return success();
905  }
906  return failure();
907  }

910 /// Canonicalize the iter_args of an scf::ForOp that involve a
911 /// `bufferization.to_tensor` and for which only the last loop iteration is
912 /// actually visible outside of the loop. The canonicalization looks for a
913 /// pattern such as:
914 /// ```
915 /// %t0 = ... : tensor_type
916 /// %0 = scf.for ... iter_args(%bb0 : %t0) -> (tensor_type) {
917 /// ...
918 /// // %m is either buffer_cast(%bb00) or defined above the loop
919 /// %m... : memref_type
920 /// ... // uses of %m with potential inplace updates
921 /// %new_tensor = bufferization.to_tensor %m : memref_type
922 /// ...
923 /// scf.yield %new_tensor : tensor_type
924 /// }
925 /// ```
926 ///
927 /// `%bb0` may have either 0 or 1 use. If it has 1 use it must be exactly a
928 /// `%m = buffer_cast %bb0` op that feeds into the yielded
929 /// `bufferization.to_tensor` op.
930 ///
931 /// If no aliasing write to the memref `%m`, from which `%new_tensor`is loaded,
932 /// occurs between `bufferization.to_tensor and yield then the value %0
933 /// visible outside of the loop is the last `bufferization.to_tensor`
934 /// produced in the loop.
935 ///
936 /// For now, we approximate the absence of aliasing by only supporting the case
937 /// when the bufferization.to_tensor is the operation immediately preceding
938 /// the yield.
939 //
940 /// The canonicalization rewrites the pattern as:
941 /// ```
942 /// // %m is either a buffer_cast or defined above
943 /// %m... : memref_type
944 /// scf.for ... iter_args(%bb0 : %t0) -> (tensor_type) {
945 /// ... // uses of %m with potential inplace updates
946 /// scf.yield %bb0: tensor_type
947 /// }
948 /// %0 = bufferization.to_tensor %m : memref_type
949 /// ```
950 ///
951 /// A later bbArg canonicalization will further rewrite as:
952 /// ```
953 /// // %m is either a buffer_cast or defined above
954 /// %m... : memref_type
955 /// scf.for ... { // no iter_args
956 /// ... // uses of %m with potential inplace updates
957 /// }
958 /// %0 = bufferization.to_tensor %m : memref_type
959 /// ```
960 struct LastTensorLoadCanonicalization : public OpRewritePattern<ForOp> {
961  using OpRewritePattern<ForOp>::OpRewritePattern;
962 
963  LogicalResult matchAndRewrite(ForOp forOp,
964  PatternRewriter &rewriter) const override {
965  assert(std::next(forOp.getRegion().begin()) == forOp.getRegion().end() &&
966  "unexpected multiple blocks");
967 
968  Location loc = forOp.getLoc();
969  DenseMap<Value, Value> replacements;
970  for (BlockArgument bbArg : forOp.getRegionIterArgs()) {
971  unsigned idx = bbArg.getArgNumber() - /*numIv=*/1;
972  auto yieldOp =
973  cast<scf::YieldOp>(forOp.getRegion().front().getTerminator());
974  Value yieldVal = yieldOp->getOperand(idx);
975  auto tensorLoadOp = yieldVal.getDefiningOp<bufferization::ToTensorOp>();
976  bool isTensor = bbArg.getType().isa<TensorType>();
977 
978  bufferization::ToMemrefOp tensorToMemref;
979  // Either bbArg has no use or it has a single buffer_cast use.
980  if (bbArg.hasOneUse())
981  tensorToMemref =
982  dyn_cast<bufferization::ToMemrefOp>(*bbArg.getUsers().begin());
983  if (!isTensor || !tensorLoadOp || (!bbArg.use_empty() && !tensorToMemref))
984  continue;
985  // If tensorToMemref is present, it must feed into the `ToTensorOp`.
986  if (tensorToMemref && tensorLoadOp.memref() != tensorToMemref)
987  continue;
988  // TODO: Any aliasing write of tensorLoadOp.memref() nested under `forOp`
989  // must be before `ToTensorOp` in the block so that the lastWrite
990  // property is not subject to additional side-effects.
991  // For now, we only support the case when ToTensorOp appears
992  // immediately before the terminator.
993  if (tensorLoadOp->getNextNode() != yieldOp)
994  continue;
995 
996  // Clone the optional tensorToMemref before forOp.
997  if (tensorToMemref) {
998  rewriter.setInsertionPoint(forOp);
999  rewriter.replaceOpWithNewOp<bufferization::ToMemrefOp>(
1000  tensorToMemref, tensorToMemref.memref().getType(),
1001  tensorToMemref.tensor());
1002  }
1003 
1004  // Clone the tensorLoad after forOp.
1005  rewriter.setInsertionPointAfter(forOp);
1006  Value newTensorLoad = rewriter.create<bufferization::ToTensorOp>(
1007  loc, tensorLoadOp.memref());
1008  Value forOpResult = forOp.getResult(bbArg.getArgNumber() - /*iv=*/1);
1009  replacements.insert(std::make_pair(forOpResult, newTensorLoad));
1010 
1011  // Make the terminator just yield the bbArg, the old tensorLoadOp + the
1012  // old bbArg (that is now directly yielded) will canonicalize away.
1013  rewriter.startRootUpdate(yieldOp);
1014  yieldOp.setOperand(idx, bbArg);
1015  rewriter.finalizeRootUpdate(yieldOp);
1016  }
1017  if (replacements.empty())
1018  return failure();
1019 
1020  // We want to replace a subset of the results of `forOp`. rewriter.replaceOp
1021  // replaces the whole op and erase it unconditionally. This is wrong for
1022  // `forOp` as it generally contains ops with side effects.
1023  // Instead, use `rewriter.replaceOpWithIf`.
1024  SmallVector<Value> newResults;
1025  newResults.reserve(forOp.getNumResults());
1026  for (Value v : forOp.getResults()) {
1027  auto it = replacements.find(v);
1028  newResults.push_back((it != replacements.end()) ? it->second : v);
1029  }
1030  unsigned idx = 0;
1031  rewriter.replaceOpWithIf(forOp, newResults, [&](OpOperand &op) {
1032  return op.get() != newResults[idx++];
1033  });
1034  return success();
1035  }
1036 };

1049 bool mlir::scf::insideMutuallyExclusiveBranches(Operation *a, Operation *b) {
1050  assert(a && "expected non-empty operation");
1051  assert(b && "expected non-empty operation");
1052 
1053  IfOp ifOp = a->getParentOfType<IfOp>();
1054  while (ifOp) {
1055  // Check if b is inside ifOp. (We already know that a is.)
1056  if (ifOp->isProperAncestor(b))
1057  // b is contained in ifOp. a and b are in mutually exclusive branches if
1058  // they are in different blocks of ifOp.
1059  return static_cast<bool>(ifOp.thenBlock()->findAncestorOpInBlock(*a)) !=
1060  static_cast<bool>(ifOp.thenBlock()->findAncestorOpInBlock(*b));
1061  // Check next enclosing IfOp.
1062  ifOp = ifOp->getParentOfType<IfOp>();
1063  }
1064 
1065  // Could not find a common IfOp among a's and b's ancestors.
1066  return false;
1067 }

1775  // If the outer scf.if yields a value produced by the inner scf.if,
1776  // only permit combining if the value yielded when the condition
1777  // is false in the outer scf.if is the same value yielded when the
1778  // inner scf.if condition is false.
1779  // Note that the array access to elseYield will not go out of bounds
1780  // since it must have the same length as thenYield, since they both
1781  // come from the same scf.if.
1782  for (const auto &tup : llvm::enumerate(thenYield)) {
1783  if (tup.value().getDefiningOp() == nestedIf) {
1784  auto nestedIdx = tup.value().cast<OpResult>().getResultNumber();
1785  if (nestedIf.elseYield().getOperand(nestedIdx) !=
1786  elseYield[tup.index()]) {
1787  return failure();
1788  }
1789  // If the correctness test passes, we will yield
1790  // corresponding value from the inner scf.if
1791  thenYield[tup.index()] = nestedIf.thenYield().getOperand(nestedIdx);
1792  continue;
1793  }
1794 
1795  // Otherwise, we need to ensure the else block of the combined
1796  // condition still returns the same value when the outer condition is
1797  // true and the inner condition is false. This can be accomplished if
1798  // the then value is defined outside the outer scf.if and we replace the
1799  // value with a select that considers just the outer condition. Since
1800  // the else region contains just the yield, its yielded value is
1801  // defined outside the scf.if, by definition.
1802 
1803  // If the then value is defined within the scf.if, bail.
1804  if (tup.value().getParentRegion() == &op.getThenRegion()) {
1805  return failure();
1806  }
1807  elseYieldsToUpgradeToSelect.push_back(tup.index());
1808  }
1809 
1810  Location loc = op.getLoc();
1811  Value newCondition = rewriter.create<arith::AndIOp>(
1812  loc, op.getCondition(), nestedIf.getCondition());
1813  auto newIf = rewriter.create<IfOp>(loc, op.getResultTypes(), newCondition);
1814 
1815  SmallVector<Value> results;
1816  llvm::append_range(results, newIf.getResults());
1817  rewriter.setInsertionPoint(newIf);
1818 
1819  for (auto idx : elseYieldsToUpgradeToSelect)
1820  results[idx] = rewriter.create<arith::SelectOp>(
1821  op.getLoc(), op.getCondition(), thenYield[idx], elseYield[idx]);
1822 
1823  Block *newIfBlock = newIf.thenBlock();
1824  if (newIfBlock)
1825  rewriter.eraseOp(newIfBlock->getTerminator());
1826  else
1827  newIfBlock = rewriter.createBlock(&newIf.getThenRegion());
1828  rewriter.mergeBlocks(nestedIf.thenBlock(), newIfBlock);
1829  rewriter.setInsertionPointToEnd(newIf.thenBlock());
1830  rewriter.replaceOpWithNewOp<YieldOp>(newIf.thenYield(), thenYield);
1831  if (!elseYield.empty()) {
1832  rewriter.createBlock(&newIf.getElseRegion());
1833  rewriter.setInsertionPointToEnd(newIf.elseBlock());
1834  rewriter.create<YieldOp>(loc, elseYield);
1835  }
1836  rewriter.replaceOp(op, results);
1837  return success();
1838  }

2096  dyn_cast_or_null<arith::ConstantIndexOp>(step.getDefiningOp());

2101  (upperBoundConstant.value() - lowerBoundConstant.value()) <=
2102  stepConstant.value()) {
2103  mapping.map(iv, lowerBound);
2104  } else {
2105  newLowerBounds.push_back(lowerBound);
2106  newUpperBounds.push_back(upperBound);
2107  newSteps.push_back(step);
2108  }

2111  if (newLowerBounds.size() == op.getLowerBound().size())
2112  return failure();

2116  // loop body and nested ReduceOp's

2529 /// A before block argument is considered loop invariant if :-
2530 /// 1. i-th yield operand is equal to the i-th while operand.
2531 /// 2. i-th yield operand is k-th after block argument which is (k+1)-th
2532 /// condition operand AND this (k+1)-th condition operand is equal to i-th

2543 /// scf.condition(%cond) %arg1_before, %arg0_before,

2547 /// ..., %argK_after):
