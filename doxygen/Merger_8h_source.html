126 /// A class to handle all iteration lattice operations. This class abstracts
127 /// away from some implementation details of storing iteration lattices and
128 /// tensor expressions. This allows for fine-tuning performance characteristics
129 /// independently from the basic algorithm if bottlenecks are identified.
130 class Merger {
131 public:
132  /// Constructs a merger for the given number of tensors and loops. The
133  /// user supplies the number of tensors involved in the kernel, with the
134  /// last tensor in this set denoting the output tensor. The merger adds an
135  /// additional synthetic tensor at the end of this set to represent all
136  /// invariant expressions in the kernel.
137  Merger(unsigned t, unsigned l)
138  : outTensor(t - 1), syntheticTensor(t), numTensors(t + 1), numLoops(l),
139  hasSparseOut(false), dims(t + 1, std::vector<Dim>(l, Dim::kUndef)) {}
140 
141  /// Adds a tensor expression. Returns its index.
142  unsigned addExp(Kind k, unsigned e0, unsigned e1 = -1u, Value v = Value(),
143  Operation *op = nullptr);
144  unsigned addExp(Kind k, unsigned e, Value v, Operation *op = nullptr) {
145  return addExp(k, e, -1u, v, op);
146  }
147  unsigned addExp(Kind k, Value v, Operation *op = nullptr) {
148  return addExp(k, -1u, -1u, v, op);
149  }
150 
151  /// Adds an iteration lattice point. Returns its index.
152  unsigned addLat(unsigned t, unsigned i, unsigned e);
153 
154  /// Adds a new, initially empty, set. Returns its index.
155  unsigned addSet();
156 
157  /// Computes a single conjunction of two lattice points by taking the "union"
158  /// of loop indices (effectively constructing a larger "intersection" of those
159  /// indices) with a newly constructed tensor (sub)expression of given kind.
160  /// Returns the index of the new lattice point.
161  unsigned conjLatPoint(Kind kind, unsigned p0, unsigned p1,
162  Operation *op = nullptr);
163 
164  /// Conjunctive merge of two lattice sets L0 and L1 is conjunction of
165  /// cartesian product. Returns the index of the new set.
166  unsigned takeConj(Kind kind, unsigned s0, unsigned s1,
167  Operation *op = nullptr);
168 
169  /// Disjunctive merge of two lattice sets L0 and L1 is (L0 /\_op L1, L0, L1).
170  /// Returns the index of the new set.
171  unsigned takeDisj(Kind kind, unsigned s0, unsigned s1,
172  Operation *op = nullptr);
173 
174  /// Disjunctive merge of two lattice sets L0 and L1 with custom handling of
175  /// the overlap, left, and right regions. Any region may be left missing in
176  /// the output. Returns the index of the new set.
177  unsigned takeCombi(Kind kind, unsigned s0, unsigned s1, Operation *orig,
178  bool includeLeft, Kind ltrans, Operation *opleft,
179  bool includeRight, Kind rtrans, Operation *opright);
180 
181  /// Maps the unary operator over the lattice set of the operand, i.e. each
182  /// lattice point on an expression E is simply copied over, but with OP E
183  /// as new expression. Returns the index of the new set.
184  unsigned mapSet(Kind kind, unsigned s0, Value v = Value(),
185  Operation *op = nullptr);
186 
187  /// Optimizes the iteration lattice points in the given set. This
188  /// method should be called right before code generation to avoid
189  /// generating redundant loops and conditions.
190  unsigned optimizeSet(unsigned s0);
191 
192  /// Simplifies the conditions in a conjunction of a given lattice point
193  /// within the given set using just two basic rules:
194  /// (1) multiple dense conditions are reduced to single dense, and
195  /// (2) a *singleton* sparse/dense is reduced to sparse/random access.
196  BitVector simplifyCond(unsigned s0, unsigned p0);
197 
198  /// Returns true if Li > Lj.
199  bool latGT(unsigned i, unsigned j) const;
200 
201  /// Returns true if Li and Lj only differ in dense.
202  bool onlyDenseDiff(unsigned i, unsigned j);
203 
204  /// Bit translation.
205  unsigned tensor(unsigned b) const { return b % numTensors; }
206  unsigned index(unsigned b) const { return b / numTensors; }
207 
208  /// Returns true if bit corresponds to queried dim.
209  bool isDim(unsigned b, Dim d) const { return isDim(tensor(b), index(b), d); }
210 
211  /// Returns true if bit corresponds to index of output tensor.
212  bool isOutTensor(unsigned b, unsigned i) const {
213  return tensor(b) == outTensor && index(b) == i;
214  }
215 
216  /// Returns true if tensor access at given index has queried dim.
217  bool isDim(unsigned t, unsigned i, Dim d) const {
218  assert(t < numTensors && i < numLoops);
219  return dims[t][i] == d;
220  }
221 
222  /// Returns true if any set bit corresponds to queried dim.
223  bool hasAnyDimOf(const BitVector &bits, Dim d) const;
224 
225  /// Returns true if given tensor iterates *only* in the given tensor
226  /// expression. For the output tensor, this defines a "simply dynamic"
227  /// operation [Bik96]. For instance: a(i) *= 2.0 or a(i) += a(i) for
228  /// sparse vector a.
229  bool isSingleCondition(unsigned t, unsigned e) const;
230 
231  /// Dimension setter.
232  void setDim(unsigned t, unsigned i, Dim d) { dims[t][i] = d; }
233 
234  // Has sparse output tensor setter.
235  void setHasSparseOut(bool s) { hasSparseOut = s; }
236 
237  /// Convenience getters to immediately access the stored nodes.
238  /// Typically it is inadvisible to keep the reference around, as in
239  /// "TensorExpr &te = merger.exp(e))", since insertions into the merger
240  /// may cause data movement and invalidate the underlying memory address.
241  TensorExp &exp(unsigned e) { return tensorExps[e]; }
242  LatPoint &lat(unsigned l) { return latPoints[l]; }
243  SmallVector<unsigned, 16> &set(unsigned s) { return latSets[s]; }
244 
245 #ifndef NDEBUG
246  /// Print methods (for debugging).
247  void dumpExp(unsigned e) const;
248  void dumpLat(unsigned p) const;
249  void dumpSet(unsigned s) const;
250  void dumpBits(const BitVector &bits) const;
251 #endif
252 
253  /// Builds the iteration lattices in a bottom-up traversal given the remaining
254  /// tensor (sub)expression and the next loop index in the iteration graph.
255  /// Returns index of the root expression.
256  unsigned buildLattices(unsigned e, unsigned i);
257 
258  /// Builds a tensor expression from the given Linalg operation.
259  /// Returns index of the root expression on success.
260  Optional<unsigned> buildTensorExpFromLinalg(linalg::GenericOp op);
261 
262  /// Rebuilds SSA format from a tensor expression.
263  Value buildExp(PatternRewriter &rewriter, Location loc, unsigned e, Value v0,
264  Value v1);
265 
266 private:
267  /// Private helpers.
268  bool maybeZero(unsigned e) const;
269  bool isInvariant(unsigned e) const;
270  Type inferType(unsigned e, Value src);
271 
272  /// Traverses the SSA tree (possibly a DAG) to build a tensor expression.
273  Optional<unsigned> buildTensorExp(linalg::GenericOp op, Value v);
274 
275  /// Merger data structures.
276  const unsigned outTensor;
277  const unsigned syntheticTensor;
278  const unsigned numTensors;
279  const unsigned numLoops;
280  bool hasSparseOut;
281  std::vector<std::vector<Dim>> dims;
282  llvm::SmallVector<TensorExp, 32> tensorExps;
283  llvm::SmallVector<LatPoint, 16> latPoints;
284  llvm::SmallVector<SmallVector<unsigned, 16>, 8> latSets;
285 };
