19 /// This class contains a collection of compilation information for files
20 /// provided to the language server, such as the available include directories.
21 /// This database acts as an aggregate in-memory form of compilation databases
22 /// used by the current language client. The textual form of a compilation
23 /// database is a YAML file containing documents of the following form:
24 ///
25 /// --- !FileInfo:
26 /// filepath: <string> - Absolute file path of the file.
27 /// includes: <string> - Semi-colon delimited list of include directories.
28 ///
29 class CompilationDatabase {
30 public:
31  /// Compilation information for a specific file within the database.
32  struct FileInfo {
33  /// The absolute path to the file.
34  std::string filename;
35  /// The include directories available for the file.
36  std::vector<std::string> includeDirs;
37  };
38 
39  /// Construct a compilation database from the provided files containing YAML
40  /// descriptions of the database.
41  CompilationDatabase(ArrayRef<std::string> databases);
42 
43  /// Get the compilation information for the provided file, or nullptr if the
44  /// database doesn't include information for `filename`.
45  const FileInfo *getFileInfo(StringRef filename) const;
46 
47 private:
48  /// Load the given database file into this database.
49  void loadDatabase(StringRef filename);
50 
51  /// A map of filename to file information for each known file within the
52  /// databases.
53  llvm::StringMap<FileInfo> files;
54 };
