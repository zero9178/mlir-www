39 /// Converts func.return to spv.Return.
40 class ReturnOpPattern final : public OpConversionPattern<func::ReturnOp> {
41 public:
42  using OpConversionPattern<func::ReturnOp>::OpConversionPattern;
43 
44  LogicalResult
45  matchAndRewrite(func::ReturnOp returnOp, OpAdaptor adaptor,
46  ConversionPatternRewriter &rewriter) const override {
47  if (returnOp.getNumOperands() > 1)
48  return failure();
49 
50  if (returnOp.getNumOperands() == 1) {
51  rewriter.replaceOpWithNewOp<spirv::ReturnValueOp>(
52  returnOp, adaptor.getOperands()[0]);
53  } else {
54  rewriter.replaceOpWithNewOp<spirv::ReturnOp>(returnOp);
55  }
56  return success();
57  }
58 };
59 
60 /// Converts func.call to spv.FunctionCall.
61 class CallOpPattern final : public OpConversionPattern<func::CallOp> {
62 public:
63  using OpConversionPattern<func::CallOp>::OpConversionPattern;
64 
65  LogicalResult
66  matchAndRewrite(func::CallOp callOp, OpAdaptor adaptor,
67  ConversionPatternRewriter &rewriter) const override {
68  // multiple results func was not converted to spv.func
69  if (callOp.getNumResults() > 1)
70  return failure();
71  if (callOp.getNumResults() == 1) {
72  auto resultType =
73  getTypeConverter()->convertType(callOp.getResult(0).getType());
74  if (!resultType)
75  return failure();
76  rewriter.replaceOpWithNewOp<spirv::FunctionCallOp>(
77  callOp, resultType, adaptor.getOperands(), callOp->getAttrs());
78  } else {
79  rewriter.replaceOpWithNewOp<spirv::FunctionCallOp>(
80  callOp, TypeRange(), adaptor.getOperands(), callOp->getAttrs());
81  }
82  return success();
83  }
84 };
