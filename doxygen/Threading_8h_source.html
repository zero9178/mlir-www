27 /// Invoke the given function on the elements between [begin, end)
28 /// asynchronously. If the given function returns a failure when processing any
29 /// of the elements, execution is stopped and a failure is returned from this
30 /// function. This means that in the case of failure, not all elements of the
31 /// range will be processed. Diagnostics emitted during processing are ordered
32 /// relative to the element's position within [begin, end). If the provided
33 /// context does not have multi-threading enabled, this function always
34 /// processes elements sequentially.
35 template <typename IteratorT, typename FuncT>
36 LogicalResult failableParallelForEach(MLIRContext *context, IteratorT begin,
37  IteratorT end, FuncT &&func) {
38  unsigned numElements = static_cast<unsigned>(std::distance(begin, end));
39  if (numElements == 0)
40  return success();
41 
42  // If multithreading is disabled or there is a small number of elements,
43  // process the elements directly on this thread.
44  // FIXME: ThreadPool should allow work stealing to avoid deadlocks when
45  // scheduling work within a worker thread.
46  if (!context->isMultithreadingEnabled() || numElements <= 1 ||
47  context->getThreadPool().isWorkerThread()) {
48  for (; begin != end; ++begin)
49  if (failed(func(*begin)))
50  return failure();
51  return success();
52  }
53 
54  // Build a wrapper processing function that properly initializes a parallel
55  // diagnostic handler.
56  ParallelDiagnosticHandler handler(context);
57  std::atomic<unsigned> curIndex(0);
58  std::atomic<bool> processingFailed(false);
59  auto processFn = [&] {
60  while (!processingFailed) {
61  unsigned index = curIndex++;
62  if (index >= numElements)
63  break;
64  handler.setOrderIDForThread(index);
65  if (failed(func(*std::next(begin, index))))
66  processingFailed = true;
67  handler.eraseOrderIDForThread();
68  }
69  };
70 
71  // Otherwise, process the elements in parallel.
72  llvm::ThreadPool &threadPool = context->getThreadPool();
73  size_t numActions = std::min(numElements, threadPool.getThreadCount());
74  SmallVector<std::shared_future<void>> threadFutures;
75  threadFutures.reserve(numActions - 1);
76  for (unsigned i = 1; i < numActions; ++i)
77  threadFutures.emplace_back(threadPool.async(processFn));
78  processFn();
79 
80  // Wait for all of the threads to finish.
81  for (std::shared_future<void> &future : threadFutures)
82  future.wait();
83  return failure(processingFailed);
84 }

86 /// Invoke the given function on the elements in the provided range
87 /// asynchronously. If the given function returns a failure when processing any
88 /// of the elements, execution is stopped and a failure is returned from this
89 /// function. This means that in the case of failure, not all elements of the
90 /// range will be processed. Diagnostics emitted during processing are ordered
91 /// relative to the element's position within the range. If the provided context
92 /// does not have multi-threading enabled, this function always processes
93 /// elements sequentially.

95 LogicalResult failableParallelForEach(MLIRContext *context, RangeT &&range,
96  FuncT &&func) {

116 /// Invoke the given function on the elements between [begin, end)
117 /// asynchronously. Diagnostics emitted during processing are ordered relative
118 /// to the element's position within [begin, end). If the provided context does
119 /// not have multi-threading enabled, this function always processes elements
120 /// sequentially.
121 template <typename IteratorT, typename FuncT>
122 void parallelForEach(MLIRContext *context, IteratorT begin, IteratorT end,
123  FuncT &&func) {
124  (void)failableParallelForEach(context, begin, end, [&](auto &&value) {
125  return func(std::forward<decltype(value)>(value)), success();
126  });
127 }
