245 struct Position {
246  Position(int line = 0, int character = 0)
247  : line(line), character(character) {}
248 
249  /// Construct a position from the given source location.
250  Position(llvm::SourceMgr &mgr, SMLoc loc) {
251  std::pair<unsigned, unsigned> lineAndCol = mgr.getLineAndColumn(loc);
252  line = lineAndCol.first - 1;
253  character = lineAndCol.second - 1;
254  }
255 
256  /// Line position in a document (zero-based).
257  int line = 0;
258 
259  /// Character offset on a line in a document (zero-based).
260  int character = 0;
261 
262  friend bool operator==(const Position &lhs, const Position &rhs) {
263  return std::tie(lhs.line, lhs.character) ==
264  std::tie(rhs.line, rhs.character);
265  }
266  friend bool operator!=(const Position &lhs, const Position &rhs) {
267  return !(lhs == rhs);
268  }
269  friend bool operator<(const Position &lhs, const Position &rhs) {
270  return std::tie(lhs.line, lhs.character) <
271  std::tie(rhs.line, rhs.character);
272  }
273  friend bool operator<=(const Position &lhs, const Position &rhs) {
274  return std::tie(lhs.line, lhs.character) <=
275  std::tie(rhs.line, rhs.character);
276  }
277 
278  /// Convert this position into a source location in the main file of the given
279  /// source manager.
280  SMLoc getAsSMLoc(llvm::SourceMgr &mgr) const {
281  return mgr.FindLocForLineAndColumn(mgr.getMainFileID(), line + 1,
282  character + 1);
283  }
284 };

536 /// Represents programming constructs like variables, classes, interfaces etc.
537 /// that appear in a document. Document symbols can be hierarchical and they
538 /// have two ranges: one that encloses its definition and one that points to its
539 /// most interesting range, e.g. the range of an identifier.

541  DocumentSymbol() = default;
542  DocumentSymbol(DocumentSymbol &&) = default;
543  DocumentSymbol(const Twine &name, SymbolKind kind, Range range,
544  Range selectionRange)
545  : name(name.str()), kind(kind), range(range),
546  selectionRange(selectionRange) {}
547 
548  /// The name of this symbol.
549  std::string name;
550 
551  /// More detail for this symbol, e.g the signature of a function.
552  std::string detail;
553 
554  /// The kind of this symbol.
555  SymbolKind kind;
556 
557  /// The range enclosing this symbol not including leading/trailing whitespace
558  /// but everything else like comments. This information is typically used to
559  /// determine if the clients cursor is inside the symbol to reveal in the
560  /// symbol in the UI.
561  Range range;
562 
563  /// The range that should be selected and revealed when this symbol is being
564  /// picked, e.g the name of a function. Must be contained by the `range`.

567  /// Children of this symbol, e.g. properties of a class.
568  std::vector<DocumentSymbol> children;

744 /// Defines whether the insert text in a completion item should be interpreted
745 /// as plain text or a snippet.

748  /// The primary text to be inserted is treated as a plain string.

754  /// of the snippet. Placeholders with equal identifiers are linked, that is
755  /// typing in one will update others too.
756  ///
757  /// See also:
758  /// https//github.com/Microsoft/vscode/blob/master/src/vs/editor/contrib/snippet/common/snippet.md
