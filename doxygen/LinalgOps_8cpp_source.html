90 /// Creates a structured operation given `inputs`, `outputs`, and `attributes`.
91 /// The result types are derived automatically if `resultTensorTypes` is none.
92 /// The body of the operation is filled using `regionBuilder`. All ods-gen
93 /// created structured operations use the method to implement their builders.
94 static void buildStructuredOp(OpBuilder &b, OperationState &state,
95  llvm::Optional<TypeRange> resultTensorTypes,
96  ValueRange inputs, ValueRange outputs,
97  ArrayRef<NamedAttribute> attributes,
98  RegionBuilderFn regionBuilder) {
99  // Derive the result types if needed.
100  SmallVector<Type> derivedResultTypes =
101  resultTensorTypes.getValueOr(TypeRange());
102  if (!resultTensorTypes.hasValue())
103  copy_if(outputs.getTypes(), std::back_inserter(derivedResultTypes),
104  [](Type type) { return type.isa<RankedTensorType>(); });
105 
106  state.addOperands(inputs);
107  state.addOperands(outputs);
108  state.addTypes(derivedResultTypes);
109  state.addAttributes(attributes);
110  state.addAttribute(
111  "operand_segment_sizes",
112  b.getI32VectorAttr({static_cast<int32_t>(inputs.size()),
113  static_cast<int32_t>(outputs.size())}));
114 
115  // Create and fill the region of the structured operation.
116  Region &region = *state.addRegion();
117  fillStructuredOpRegion(b, region, TypeRange(inputs), TypeRange(outputs),
118  state.attributes.getAttrs(), regionBuilder);
119 }

121 /// Common parsing used for both named structured ops created by ods-gen and by
122 /// manually defined C++ ops. Does not handle regions.
123 static ParseResult
124 parseCommonStructuredOpParts(OpAsmParser &parser, OperationState &result,
125  SmallVectorImpl<Type> &inputTypes,
126  SmallVectorImpl<Type> &outputTypes) {
127  SMLoc inputsOperandsLoc, outputsOperandsLoc;
128  SmallVector<OpAsmParser::UnresolvedOperand, 4> inputsOperands,
129  outputsOperands;
130 
131  parser.parseOptionalAttrDict(result.attributes);
132 
133  if (succeeded(parser.parseOptionalKeyword("ins"))) {
134  if (parser.parseLParen())
135  return failure();
136 
137  inputsOperandsLoc = parser.getCurrentLocation();
138  if (parser.parseOperandList(inputsOperands) ||
139  parser.parseColonTypeList(inputTypes) || parser.parseRParen())
140  return failure();
141  }
142 
143  if (succeeded(parser.parseOptionalKeyword("outs"))) {
144  outputsOperandsLoc = parser.getCurrentLocation();
145  if (parser.parseLParen() || parser.parseOperandList(outputsOperands) ||
146  parser.parseColonTypeList(outputTypes) || parser.parseRParen())
147  return failure();
148  }
149 
150  if (parser.resolveOperands(inputsOperands, inputTypes, inputsOperandsLoc,
151  result.operands) ||
152  parser.resolveOperands(outputsOperands, outputTypes, outputsOperandsLoc,
153  result.operands))
154  return failure();
155 
156  result.addAttribute("operand_segment_sizes",
157  parser.getBuilder().getI32VectorAttr(
158  {static_cast<int32_t>(inputsOperands.size()),
159  static_cast<int32_t>(outputsOperands.size())}));
160  return success();
161 }

202  OperationState &result,
203  unsigned numRegionArgs,
204  RegionBuilderFn regionBuilder) {
205  // TODO: Enable when ods-gen supports captures.
206  SmallVector<Type, 1> inputTypes, outputTypes;

269 /// Helper function to find if there is atleast one dimension in an AffineMap
270 /// testMap that is contained in `testMapLocation` of `maps` but not in any
271 /// other locations
272 static bool hasaUniqueDim(ArrayRef<AffineMap> maps, unsigned testMapLocation) {
273  AffineMap testMap = maps[testMapLocation];
274  llvm::SmallDenseSet<unsigned> dimsToCheck;
275  for (auto result : testMap.getResults()) {
276  auto expr = result.dyn_cast<AffineDimExpr>();
277  if (expr != nullptr)
278  dimsToCheck.insert(expr.getPosition());
279  }
280  for (const auto &it : llvm::enumerate(maps)) {
281  if (it.index() == testMapLocation)
282  continue;
283  auto map = it.value();
284  for (auto result : map.getResults()) {
285  auto expr = result.dyn_cast<AffineDimExpr>();
286  if (expr != nullptr) {
287  dimsToCheck.erase(expr.getPosition());
288  }
289  if (dimsToCheck.empty())
290  return false;
291  }
292  }
293  return true;
294 }

334  return builder.create<math::LogOp>(arg.getLoc(), arg);
335  case UnaryFn::abs:
336  return builder.create<math::AbsOp>(arg.getLoc(), arg);
337  case UnaryFn::ceil:
338  return builder.create<math::CeilOp>(arg.getLoc(), arg);
339  case UnaryFn::floor:

347  // Build the binary functions defined by OpDSL.
348  Value buildBinaryFn(BinaryFn binaryFn, Value arg0, Value arg1) {
349  bool allFloatingPoint = isFloatingPoint(arg0) && isFloatingPoint(arg1);
350  bool allInteger = isInteger(arg0) && isInteger(arg1);
351  if (!allFloatingPoint && !allInteger)
352  llvm_unreachable("unsupported non numeric type");
353  OpBuilder builder = getBuilder();
354  switch (binaryFn) {
355  case BinaryFn::add:
356  if (allFloatingPoint)
357  return builder.create<arith::AddFOp>(arg0.getLoc(), arg0, arg1);
358  return builder.create<arith::AddIOp>(arg0.getLoc(), arg0, arg1);
359  case BinaryFn::sub:
360  if (allFloatingPoint)
361  return builder.create<arith::SubFOp>(arg0.getLoc(), arg0, arg1);
362  return builder.create<arith::SubIOp>(arg0.getLoc(), arg0, arg1);
363  case BinaryFn::mul:
364  if (allFloatingPoint)
365  return builder.create<arith::MulFOp>(arg0.getLoc(), arg0, arg1);
366  return builder.create<arith::MulIOp>(arg0.getLoc(), arg0, arg1);
367  case BinaryFn::max_signed:
368  if (allFloatingPoint)
369  return builder.create<arith::MaxFOp>(arg0.getLoc(), arg0, arg1);
370  return builder.create<arith::MaxSIOp>(arg0.getLoc(), arg0, arg1);
371  case BinaryFn::min_signed:
372  if (allFloatingPoint)
373  return builder.create<arith::MinFOp>(arg0.getLoc(), arg0, arg1);
374  return builder.create<arith::MinSIOp>(arg0.getLoc(), arg0, arg1);
375  case BinaryFn::max_unsigned:
376  if (allFloatingPoint)
377  return builder.create<arith::MaxFOp>(arg0.getLoc(), arg0, arg1);
378  return builder.create<arith::MaxUIOp>(arg0.getLoc(), arg0, arg1);

425  // Generates operations to cast the given operand to a specified type.
426  // If the cast cannot be performed, a warning will be issued and the
427  // operand returned as-is (which will presumably yield a verification
428  // issue downstream).
429  Value cast(Type toType, Value operand, bool isUnsignedCast) {
430  OpBuilder builder = getBuilder();
431  auto loc = operand.getLoc();
432 
433  if (operand.getType() == toType)
434  return operand;
435  if (auto toIntType = toType.dyn_cast<IntegerType>()) {
436  // If operand is floating point, cast directly to the int type.
437  if (operand.getType().isa<FloatType>()) {
438  if (isUnsignedCast)
439  return builder.create<arith::FPToUIOp>(loc, toType, operand);
440  return builder.create<arith::FPToSIOp>(loc, toType, operand);
441  }
442  // Cast index operands directly to the int type.
443  if (operand.getType().isIndex())
444  return builder.create<arith::IndexCastOp>(loc, toType, operand);
445  if (auto fromIntType = operand.getType().dyn_cast<IntegerType>()) {
446  // Either extend or truncate.
447  if (toIntType.getWidth() > fromIntType.getWidth()) {
448  if (isUnsignedCast)
449  return builder.create<arith::ExtUIOp>(loc, toType, operand);
450  return builder.create<arith::ExtSIOp>(loc, toType, operand);
451  }
452  if (toIntType.getWidth() < fromIntType.getWidth())
453  return builder.create<arith::TruncIOp>(loc, toType, operand);
454  }
455  } else if (auto toFloatType = toType.dyn_cast<FloatType>()) {
456  // If operand is integer, cast directly to the float type.
457  // Note that it is unclear how to cast from BF16<->FP16.
458  if (operand.getType().isa<IntegerType>()) {
459  if (isUnsignedCast)
460  return builder.create<arith::UIToFPOp>(loc, toFloatType, operand);
461  return builder.create<arith::SIToFPOp>(loc, toFloatType, operand);
462  }
463  if (auto fromFloatType = operand.getType().dyn_cast<FloatType>()) {
464  if (toFloatType.getWidth() > fromFloatType.getWidth())
465  return builder.create<arith::ExtFOp>(loc, toFloatType, operand);
466  if (toFloatType.getWidth() < fromFloatType.getWidth())
467  return builder.create<arith::TruncFOp>(loc, toFloatType, operand);
468  }
469  }
470 
471  emitWarning(operand.getLoc()) << "could not cast operand of type "
472  << operand.getType() << " to " << toType;
473  return operand;
474  }

523 struct FoldFillWithPad final : public OpRewritePattern<tensor::PadOp> {

526  LogicalResult matchAndRewrite(tensor::PadOp padOp,
527  PatternRewriter &rewriter) const override {
528  auto fillOp = padOp.source().getDefiningOp<linalg::FillOp>();
529  if (!fillOp)
530  return failure();
531 
532  // We can only fold if the padding value is the same as the original
533  // filling value.

547  ShapedType::kDynamicSize);

560 /// Fold tensor.insert_slice(tensor.pad(<input>), linalg.fill) into
561 /// tensor.insert_slice(<input>, linalg.fill) if the padding value and the
562 /// filling value are the same.
563 struct FoldInsertPadIntoFill : public OpRewritePattern<tensor::InsertSliceOp> {
564  using OpRewritePattern::OpRewritePattern;
565 
566  LogicalResult matchAndRewrite(tensor::InsertSliceOp insertOp,
567  PatternRewriter &rewriter) const override {
568  auto srcPadOp = insertOp.source().getDefiningOp<tensor::PadOp>();
569  if (!srcPadOp)
570  return failure();
571 
572  if (insertOp.getType().getRank() != insertOp.getSourceType().getRank())
573  return failure();
574 
575  // Walk back the tensor.insert_slice chain and find the first destination
576  // value at the start of the chain.
577  Value firstDest = insertOp.dest();
578  while (auto prevOp = firstDest.getDefiningOp<tensor::InsertSliceOp>()) {
579  if (prevOp.getType().getRank() != prevOp.getSourceType().getRank())
580  return failure();
581 
582  // Make sure the range of values accessed are disjoint. Without this, we
583  // cannot fold tensor.pad away.
584  bool disjoint = false;
585  for (int i = 0, e = prevOp.getType().getRank(); i < e; ++i) {
586  // If the dimension has dynamic offset/size, we cannot guarantee
587  // disjoint. So just skip it.
588  if (insertOp.isDynamicOffset(i) || insertOp.isDynamicSize(i) ||
589  insertOp.isDynamicStride(i) || prevOp.isDynamicOffset(i) ||
590  prevOp.isDynamicSize(i) || prevOp.isDynamicStride(i))
591  continue;
592 
593  // Get the range start and end, inclusively for both.
594  int64_t prevStart = prevOp.getStaticOffset(i);
595  int64_t prevEnd = prevStart + (prevOp.getStaticSize(i) - 1) *
596  prevOp.getStaticStride(i);
597  int64_t nextStart = insertOp.getStaticOffset(i);
598  int64_t nextEnd = nextStart + (insertOp.getStaticSize(i) - 1) *
599  insertOp.getStaticStride(i);
600  if (prevEnd < nextStart || nextEnd < prevStart) {
601  disjoint = true;
602  break;
603  }
604  }
605 
606  if (!disjoint)
607  break;
608  firstDest = prevOp.dest();
609  }
610 
611  // Check whether the first destination is a fill op. For overlapped cases,
612  // this also cannot be true.
613  auto dstFillOp = firstDest.getDefiningOp<linalg::FillOp>();
614  if (!dstFillOp)
615  return failure();
616 
617  // We can only fold if the padding value is the same as the original
618  // filling value.
619  Value padValue = srcPadOp.getConstantPaddingValue();
620  if (!padValue || dstFillOp.value() != padValue)
621  return failure();
622 
623  SmallVector<OpFoldResult> lowPads = srcPadOp.getMixedLowPad();
624  SmallVector<OpFoldResult> oldOffsets = insertOp.getMixedOffsets();
625 
626  Location loc = insertOp.getLoc();
627  MLIRContext *context = getContext();
628 
629  AffineExpr sym0, sym1;
630  bindSymbols(context, sym0, sym1);
631  auto addMap = AffineMap::get(0, 2, {sym0 + sym1}, context);
632 
633  // Calculate the new offsets for the insert. It should be the old offsets
634  // plus low padding sizes.
635  SmallVector<OpFoldResult, 4> newOffsets;
636  for (const auto &p : llvm::zip(lowPads, oldOffsets)) {
637  Value padValue = getValueOrCreateConstantIndexOp(
638  rewriter, srcPadOp.getLoc(), std::get<0>(p));
639  Value offsetValue = getValueOrCreateConstantIndexOp(
640  rewriter, insertOp.getLoc(), std::get<1>(p));
641  newOffsets.push_back(
642  applyMapToValues(rewriter, loc, addMap, {offsetValue, padValue})[0]);
643  }
644 
645  SmallVector<OpFoldResult, 4> newSizes;
646  for (int i = 0, e = srcPadOp.getSourceType().getRank(); i < e; ++i) {
647  newSizes.push_back(
648  rewriter.create<tensor::DimOp>(loc, srcPadOp.source(), i).result());
649  }
650 
651  rewriter.replaceOpWithNewOp<tensor::InsertSliceOp>(
652  insertOp, srcPadOp.source(), insertOp.dest(), newOffsets, newSizes,
653  insertOp.getMixedStrides());
654  return success();
655  }
656 };

847 // Deduplicate redundant args of a linalg generic op.
848 // An arg is redundant if it has the same Value and indexing map as another.
849 struct DeduplicateGenericOpInputs : public OpRewritePattern<GenericOp> {
850  using OpRewritePattern<GenericOp>::OpRewritePattern;
851 
852  LogicalResult matchAndRewrite(GenericOp genericOp,
853  PatternRewriter &rewriter) const override {
854  // Associate each input to an equivalent "canonical" input that has the same
855  // Value and indexing map.
856  //
857  // In the non-duplicate case, input `i` will have canonical input `i`. But
858  // in the case of duplicated inputs, the canonical input could be some other
859  // input `< i`. That is, a later input will have some earlier input as its
860  // canonical input.
861  llvm::SmallDenseMap<std::pair<Value, AffineMap>, unsigned> canonicalInput;
862  // For later remapping tasks like deduplicating payload block arguments,
863  // having a simple "inputIndex -> canonicalInputIndex" integer mapping is
864  // convenient.
865  SmallVector<unsigned> canonicalInputIndices;
866  for (OpOperand *opOperand : genericOp.getInputOperands()) {
867  AffineMap indexingMap = genericOp.getTiedIndexingMap(opOperand);
868  // STL-like maps have a convenient behavior for our use case here. In the
869  // case of duplicate keys, the insertion is rejected, and the returned
870  // iterator gives access to the value already in the map.
871  auto pair = canonicalInput.insert(
872  {{opOperand->get(), indexingMap}, opOperand->getOperandNumber()});
873  canonicalInputIndices.push_back(pair.first->second);
874  }
875 
876  // If there are no duplicate args, then bail out.
877  if (canonicalInput.size() == genericOp.getNumInputs())
878  return failure();
879 
880  // The operands for the newly canonicalized op.
881  SmallVector<Value> newInputOperands;
882  for (OpOperand *opOperand : genericOp.getInputOperands())
883  if (canonicalInputIndices[opOperand->getOperandNumber()] ==
884  opOperand->getOperandNumber())
885  newInputOperands.push_back(opOperand->get());
886 
887  // Repair the indexing maps by filtering out the ones that have been
888  // eliminated.
889  SmallVector<AffineMap> newIndexingMaps;
890  for (OpOperand *opOperand : genericOp.getInputOperands())
891  if (canonicalInputIndices[opOperand->getOperandNumber()] ==
892  opOperand->getOperandNumber())
893  newIndexingMaps.push_back(genericOp.getTiedIndexingMap(opOperand));
894  for (OpOperand *opOperand : genericOp.getOutputOperands())
895  newIndexingMaps.push_back(genericOp.getTiedIndexingMap(opOperand));
896 
897  // Clone the old op with new operands.
898  SmallVector<Value> outputOperands = genericOp.getOutputOperands();
899  auto newOp = rewriter.create<GenericOp>(
900  genericOp.getLoc(), genericOp->getResultTypes(), newInputOperands,
901  outputOperands, rewriter.getAffineMapArrayAttr(newIndexingMaps),
902  genericOp.iterator_types(), genericOp.docAttr(),
903  genericOp.library_callAttr());
904 
905  // Copy over unknown attributes. They might be load bearing for some flow.
906  ArrayRef<StringRef> odsAttrs = genericOp.getAttributeNames();
907  for (NamedAttribute kv : genericOp->getAttrs()) {
908  if (!llvm::is_contained(odsAttrs, kv.getName().getValue())) {
909  newOp->setAttr(kv.getName(), kv.getValue());
910  }
911  }
912 
913  rewriter.inlineRegionBefore(genericOp.region(), newOp.region(),
914  newOp.region().begin());
915 
916  // Repair the payload entry block by RAUW'ing redundant arguments and
917  // erasing them.
918  Block &payload = newOp.region().front();
919  SmallVector<OpOperand *> inputOperands = genericOp.getInputOperands();
920  for (OpOperand *opOperand : llvm::reverse(inputOperands)) {
921  // Iterate in reverse, so that we erase later args first, preventing the
922  // argument list from shifting unexpectedly and invalidating all our
923  // indices.
924  unsigned operandNumber = opOperand->getOperandNumber();
925  if (canonicalInputIndices[operandNumber] == operandNumber)
926  continue;
927  payload.getArgument(operandNumber)
928  .replaceAllUsesWith(
929  payload.getArgument(canonicalInputIndices[operandNumber]));
930  payload.eraseArgument(operandNumber);
931  }
932 
933  rewriter.replaceOp(genericOp, newOp->getResults());
934  return success();
935  }
936 };

938 /// Remove generic operations (on tensors) that are just copying
939 /// the values from inputs to the results. Requirements are
940 /// 1) All iterator types are parallel
941 /// 2) The body contains just a yield operation with the yielded values being
942 /// the arguments corresponding to the operands.
943 struct EraseIdentityGenericOp : public OpRewritePattern<GenericOp> {
944  using OpRewritePattern<GenericOp>::OpRewritePattern;
945 
946  LogicalResult matchAndRewrite(GenericOp genericOp,
947  PatternRewriter &rewriter) const override {
948  // Check all indexing maps are identity.
949  if (llvm::any_of(genericOp.getIndexingMaps(),
950  [](AffineMap map) { return !map.isIdentity(); }))
951  return failure();
952 
953  // Check that the body of the linalg operation is just a linalg.yield
954  // operation.
955  Block &body = genericOp.region().front();
956  if (!llvm::hasSingleElement(body))
957  return failure();
958  auto yieldOp = dyn_cast<linalg::YieldOp>(body.getTerminator());
959  if (!yieldOp)
960  return failure();
961 
962  // In the buffer case, we need to check exact buffer equality.
963  if (genericOp.hasBufferSemantics()) {
964  if (genericOp.getNumInputs() == 1 && genericOp.getNumOutputs() == 1 &&
965  genericOp.getInputOperand(0)->get() ==
966  genericOp.getOutputOperand(0)->get()) {
967  rewriter.eraseOp(genericOp);
968  return success();
969  }
970  return failure();
971  }
972 
973  // Get the argument number of the returned values. That is the operand
974  // number to use for replacing uses of this operation.
975  SmallVector<Value> returnedArgs;
976  for (const auto &yieldVal : llvm::enumerate(yieldOp.values())) {
977  auto yieldArg = yieldVal.value().dyn_cast<BlockArgument>();
978  if (!yieldArg || yieldArg.getOwner() != &body)
979  return failure();
980  unsigned argumentNumber = yieldArg.getArgNumber();
981  Value returnedArg = genericOp->getOperand(argumentNumber);
982  Type resultType = genericOp->getResult(yieldVal.index()).getType();
983  // The input can have a different type than the result, e.g. a dynamic
984  // input dimension can be turned into a static output dimension.
985  Type returnType = returnedArg.getType();
986  if (returnType != resultType) {
987  // Distinguish between sparse conversion or dense tensor casting.
988  // TODO: unify the two ops?
989  if (sparse_tensor::getSparseTensorEncoding(returnType) ||
990  sparse_tensor::getSparseTensorEncoding(resultType))
991  returnedArg = rewriter.create<sparse_tensor::ConvertOp>(
992  genericOp.getLoc(), resultType, returnedArg);
993  else {
994  if (!tensor::CastOp::areCastCompatible(returnedArg.getType(),
995  resultType))
996  return failure();
997  returnedArg = rewriter.create<tensor::CastOp>(
998  genericOp.getLoc(), resultType, returnedArg);
999  }
1000  }
1001  returnedArgs.push_back(returnedArg);
1002  }
1003 
1004  if (returnedArgs.size() != genericOp->getNumResults())
1005  return failure();
1006  rewriter.replaceOp(genericOp, returnedArgs);
1007  return success();
1008  }
1009 };

1011 /// Drop dead args of a linalg generic op.
1012 /// An arg is dead if it has zero uses in the op region.
1013 struct DeadArgsGenericOpInputs : public OpRewritePattern<GenericOp> {
1014  using OpRewritePattern<GenericOp>::OpRewritePattern;
1015  LogicalResult matchAndRewrite(GenericOp genericOp,
1016  PatternRewriter &rewriter) const override {
1017  SmallVector<AffineMap> oldIndexingMaps = genericOp.getIndexingMaps();
1018  // Maps must be projected permutations.
1019  if (llvm::any_of(genericOp.getIndexingMaps(), [](AffineMap map) {
1020  return !map.isProjectedPermutation();
1021  }))
1022  return failure();
1023  Block &payload = genericOp.region().front();
1024  SmallVector<Value> newInputOperands;
1025  SmallVector<AffineMap> newIndexingMaps;
1026  bool deadArgFound = false;
1027  int inputSize = genericOp.getInputOperands().size();
1028  for (int i = inputSize - 1; i >= 0; i--) {
1029  OpOperand *opOperand = genericOp.getInputOperand(i);
1030  // Iterate in reverse, so that we erase later args first, preventing the
1031  // argument list from shifting unexpectedly and invalidating all our
1032  // indices.
1033  if (payload.getArgument(i).use_empty() &&
1034  !hasaUniqueDim(oldIndexingMaps, i)) {
1035  payload.eraseArgument(i);
1036  deadArgFound = true;
1037  // remove this indexing map out of consideration for hasaUniqueDim check
1038  oldIndexingMaps.erase(oldIndexingMaps.begin() + i);
1039  } else {
1040  newInputOperands.insert(newInputOperands.begin(), opOperand->get());
1041  newIndexingMaps.insert(newIndexingMaps.begin(),
1042  genericOp.getTiedIndexingMap(opOperand));
1043  }
1044  }
1045  // Bail out if there are no dead args.
1046  if (!deadArgFound)
1047  return failure();
1048  for (OpOperand *opOperand : genericOp.getOutputOperands())
1049  newIndexingMaps.push_back(genericOp.getTiedIndexingMap(opOperand));
1050  SmallVector<Value> outputOperands = genericOp.getOutputOperands();
1051 
1052  auto newOp = rewriter.create<GenericOp>(
1053  genericOp.getLoc(), genericOp->getResultTypes(), newInputOperands,
1054  outputOperands, rewriter.getAffineMapArrayAttr(newIndexingMaps),
1055  genericOp.iterator_types(), genericOp.docAttr(),
1056  genericOp.library_callAttr());
1057  // Copy over unknown attributes. They might be load bearing for some flow.
1058  ArrayRef<StringRef> odsAttrs = genericOp.getAttributeNames();
1059  for (NamedAttribute kv : genericOp->getAttrs()) {
1060  if (!llvm::is_contained(odsAttrs, kv.getName().getValue())) {
1061  newOp->setAttr(kv.getName(), kv.getValue());
1062  }
1063  }
1064  rewriter.inlineRegionBefore(genericOp.region(), newOp.region(),
1065  newOp.region().begin());
1066  rewriter.replaceOp(genericOp, newOp->getResults());
1067  return success();
1068  }

1170  // constant value to find the static size to use.
1171  unsigned operandNum = op.getIndexOfDynamicSize(i);
1172  Value sizeOperand = op.getOperand(operandNum);
1173  if (auto constantIndexOp =
1174  sizeOperand.getDefiningOp<arith::ConstantIndexOp>()) {
1175  staticSizes.push_back(constantIndexOp.value());
1176  continue;
1177  }
1178

1180  dynamicSizes.push_back(sizeOperand);
1181  staticSizes.push_back(ShapedType::kDynamicSize);
1182  }

1184  RankedTensorType::get(staticSizes, op.getType().getElementType());
1185  if (newType == op.getType())
1186  return failure();
1187  auto newOp =
1188  rewriter.create<InitTensorOp>(op.getLoc(), newType, dynamicSizes,
1189  rewriter.getI64ArrayAttr(staticSizes));
1190  rewriter.replaceOpWithNewOp<tensor::CastOp>(op, op.getType(), newOp);
1191  return success();
1192  }

1265 /// Canonicalize
1266 ///
1267 /// ```mlir
1268 /// %0 = linalg.init_tensor [%d0, %d1] : tensor<?x?xf32>
1269 /// %1 = tensor.cast %0 : tensor<?x?xf32> to tensor<4x?xf32>
1270 /// ```
1271 ///
1272 /// into
1273 ///
1274 /// ```mlir
1275 /// %0 = linalg.init_tensor [4, %d1] : tensor<4x?xf32>
1276 /// ```
1277 ///
1278 /// This assumes the input program is correct in terms of its shape. So it
1279 /// is safe to assume that `%d0` is in fact 4. If that was not the case, the
1280 /// input program is wrong to begin with, so its undefined behavior anyway (i.e.
1281 /// this optimization can still triggering without violating program semantics).
1282 struct FoldInitTensorWithTensorCastOp
1283  : public OpRewritePattern<tensor::CastOp> {
1284  using OpRewritePattern<tensor::CastOp>::OpRewritePattern;
1285 
1286  LogicalResult matchAndRewrite(tensor::CastOp castOp,
1287  PatternRewriter &rewriter) const override {
1288  if (!canFoldIntoProducerOp(castOp))
1289  return failure();
1290  auto producer = castOp.source().getDefiningOp<InitTensorOp>();
1291  if (!producer)
1292  return failure();
1293 
1294  auto resultType = castOp->getResult(0).getType().cast<RankedTensorType>();
1295  ArrayRef<int64_t> resultShape = resultType.getShape();
1296  SmallVector<OpFoldResult> currMixedSizes = producer.getMixedSizes();
1297  SmallVector<OpFoldResult> newMixedSizes;
1298  newMixedSizes.reserve(currMixedSizes.size());
1299  assert(resultShape.size() == currMixedSizes.size() &&
1300  "mismatch in result shape and sizes of init_tensor op");
1301  for (auto it : llvm::zip(resultShape, currMixedSizes)) {
1302  int64_t newDim = std::get<0>(it);
1303  OpFoldResult currDim = std::get<1>(it);
1304  // Case 1: The init tensor dim is static. Check that the tensor cast
1305  // result dim matches.
1306  if (auto attr = currDim.dyn_cast<Attribute>()) {
1307  if (ShapedType::isDynamic(newDim) ||
1308  newDim != attr.cast<IntegerAttr>().getInt()) {
1309  // Something is off, the cast result shape cannot be more dynamic than
1310  // the init tensor result shape (enforced by `canFoldIntoProducer`).
1311  // Abort for now.
1312  return rewriter.notifyMatchFailure(
1313  producer, "mismatch in static value of shape of init "
1314  "tensor result and cast result");
1315  }
1316  newMixedSizes.push_back(attr);
1317  continue;
1318  }
1319 
1320  // Case 2 : The tensor cast shape is static, but init tensor result shape
1321  // is dynamic.
1322  if (!ShapedType::isDynamic(newDim)) {
1323  newMixedSizes.push_back(rewriter.getIndexAttr(newDim));
1324  continue;
1325  }
1326 
1327  // Case 3 : The tensor cast shape is dynamic and init tensor result shape
1328  // is dynamic. Use the dynamic value from the init tensor op.
1329  newMixedSizes.push_back(currDim);
1330  }
1331 
1332  rewriter.replaceOpWithNewOp<InitTensorOp>(castOp, newMixedSizes,
1333  resultType.getElementType());
1334  return success();
1335  }
1336 };

1605 /// result that is more static than the linalg op.
1606 struct FoldTensorCastConsumerOp : public OpRewritePattern<tensor::CastOp> {
1607  using OpRewritePattern<tensor::CastOp>::OpRewritePattern;
1608 
1609  LogicalResult matchAndRewrite(tensor::CastOp castOp,
1610  PatternRewriter &rewriter) const override {
1611  if (!tensor::canFoldIntoProducerOp(castOp))
1612  return failure();
1613  auto linalgOp = castOp.source().getDefiningOp<LinalgOp>();
1614  if (!linalgOp)
1615  return failure();
1616 
1617  OpBuilder::InsertionGuard guard(rewriter);
1618  rewriter.setInsertionPoint(linalgOp);
1619 
1620  Location loc = linalgOp.getLoc();
1621  OpResult resultValue = castOp.source().cast<OpResult>();
1622  unsigned resultNumber = resultValue.getResultNumber();
1623  auto resultType = castOp->getResult(0).getType().cast<RankedTensorType>();
1624  // Replace the `outs` for the result with a `tensor.cast`. This cast is now
1625  // going from a more dynamic shape to a less dynamic shape. If the producer
1626  // for this cast, i.e. producer of the out operand, is also an operation
1627  // that folds with tensor.cast consumer (like this pattern), the cast will
1628  // continue to propagate as far up the stack as it can go.
1629  OpOperand *outOperand = linalgOp.getOutputOperand(resultNumber);
1630  Value newOperand =
1631  rewriter.create<tensor::CastOp>(loc, resultType, outOperand->get());
1632  SmallVector<Value> newOperands = linalgOp.getInputOperands();
1633  SmallVector<Value> outputOperands = linalgOp.getOutputOperands();
1634  outputOperands[resultNumber] = newOperand;
1635  newOperands.append(outputOperands.begin(), outputOperands.end());
1636 
1637  SmallVector<Type> resultTypes(linalgOp->result_type_begin(),
1638  linalgOp->result_type_end());
1639  resultTypes[resultNumber] = resultType;
1640  Operation *newOp = linalgOp.clone(rewriter, loc, resultTypes, newOperands);
1641 
1642  // Create a tensor.cast operation back to the original type.
1643  Value castBack = rewriter.create<tensor::CastOp>(
1644  loc, resultValue.getType(), newOp->getResult(resultNumber));
1645 
1646  SmallVector<Value> results(newOp->result_begin(), newOp->result_end());
1647  results[resultNumber] = castBack;
1648  rewriter.replaceOp(linalgOp, results);
1649  rewriter.replaceOp(castOp, newOp->getResult(resultNumber));
1650  return success();

1690 /// Creates new operand w.r.t 'opOperand' of `linalgOp` with static sizes
1691 /// mapped in `affineExprToSize`. New operands are created in `newOperands` and
1692 /// their result types is stored in `resultTypes`. If `opOperand` requires no
1693 /// change then `changeNeeded` is false and same operand is added in the
1694 /// `newOperands` list.
1695 static void createNewOperandWithStaticSizes(
1696  Location loc, PatternRewriter &rewriter, OpOperand *opOperand,
1697  llvm::DenseMap<AffineExpr, int64_t> &affineExprToSize, LinalgOp linalgOp,
1698  SmallVector<Value> &newOperands, SmallVector<Type> &resultTypes,
1699  bool &changeNeeded) {
1700  Value src = opOperand->get();
1701  newOperands.push_back(src);
1702  if (linalgOp.isScalar(opOperand))

1704  auto sourceType = src.getType().cast<RankedTensorType>();

1714  // true.
1715  bool newOperandNeeded = false;

1744 /// have a static shape. This can be done by referring to the affine dim

1748 
1749  LogicalResult matchAndRewrite(LinalgOp linalgOp,
1750  PatternRewriter &rewriter) const override {
1751  if (!linalgOp.hasTensorSemantics())
1752  return failure();
1753 
1754  // Maps must be projected permutations.
1755  if (llvm::any_of(linalgOp.getIndexingMaps(), [](AffineMap map) {
1756  return !map.isProjectedPermutation();
1757  }))
1758  return failure();
1759 
1760  // Maps affine dim expressions to the static size of that dimension.
1761  llvm::DenseMap<AffineExpr, int64_t> affineExprToSize;
1762  Location loc = linalgOp.getLoc();
1763 
1764  // For each of the affine dim expression, check if the size is known. If
1765  // known add that in the map.
1766  populateMap(linalgOp, linalgOp.getInputAndOutputOperands(),
1767  affineExprToSize);
1768 
1769  SmallVector<Value> newOperands;
1770  SmallVector<Type> resultTypes;
1771

1773  // change in their types.
1774  bool changeNeeded = false;
1775  newOperands.reserve(linalgOp.getNumInputsAndOutputs());
1776  resultTypes.reserve(linalgOp.getNumOutputs());

1780  createNewOperandWithStaticSizes(loc, rewriter, opOperand,
1781  affineExprToSize, linalgOp, newOperands,
1782  resultTypes, changeNeeded);
1783  }

1786  // canonicalization needed.
1787  if (!changeNeeded)
1788  return failure();

1790  // Clone op.
1791  Operation *newOp =
1792  linalgOp.clone(rewriter, linalgOp->getLoc(), resultTypes, newOperands);
1793  SmallVector<Value> replacements;
1794  replacements.reserve(newOp->getNumResults());

1796  Value newResult = std::get<1>(it);

1803  : newResult);
1804  }

1807  }
