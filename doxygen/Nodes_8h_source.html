77  /// Add a new decl to the scope.

79 
80  /// Lookup a decl with the given name starting from this scope. Returns
81  /// nullptr if no decl could be found.

104 /// This class represents a base AST node. All AST nodes are derived from this
105 /// class, and it contains many of the base functionality for interacting with
106 /// nodes.
107 class Node {
108 public:
109  /// This CRTP class provides several utilies when defining new AST nodes.
110  template <typename T, typename BaseT> class NodeBase : public BaseT {
111  public:
112  using Base = NodeBase<T, BaseT>;
113 
114  /// Provide type casting support.
115  static bool classof(const Node *node) {
116  return node->getTypeID() == TypeID::get<T>();
117  }
118 
119  protected:
120  template <typename... Args>
121  explicit NodeBase(SMRange loc, Args &&...args)
122  : BaseT(TypeID::get<T>(), loc, std::forward<Args>(args)...) {}
123  };
124 
125  /// Return the type identifier of this node.
126  TypeID getTypeID() const { return typeID; }
127 
128  /// Return the location of this node.
129  SMRange getLoc() const { return loc; }
130 
131  /// Print this node to the given stream.
132  void print(raw_ostream &os) const;
133 
134  /// Walk all of the nodes including, and nested under, this node in pre-order.
135  void walk(function_ref<void(const Node *)> walkFn) const;
136  template <typename WalkFnT, typename ArgT = typename llvm::function_traits<
137  WalkFnT>::template arg_t<0>>
138  std::enable_if_t<!std::is_convertible<const Node *, ArgT>::value>
139  walk(WalkFnT &&walkFn) const {
140  walk([&](const Node *node) {
141  if (const ArgT *derivedNode = dyn_cast<ArgT>(node))
142  walkFn(derivedNode);
143  });
144  }
145 
146 protected:
147  Node(TypeID typeID, SMRange loc) : typeID(typeID), loc(loc) {}
148 
149 private:
150  /// A unique type identifier for this node.
151  TypeID typeID;
152 
153  /// The location of this node.
154  SMRange loc;
155 };
