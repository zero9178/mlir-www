148 static llvm::FastMathFlags getFastmathFlags(FastmathFlagsInterface &op) {
149  using llvmFMF = llvm::FastMathFlags;
150  using FuncT = void (llvmFMF::*)(bool);
151  const std::pair<FastmathFlags, FuncT> handlers[] = {
152  // clang-format off
153  {FastmathFlags::nnan, &llvmFMF::setNoNaNs},
154  {FastmathFlags::ninf, &llvmFMF::setNoInfs},
155  {FastmathFlags::nsz, &llvmFMF::setNoSignedZeros},
156  {FastmathFlags::arcp, &llvmFMF::setAllowReciprocal},
157  {FastmathFlags::contract, &llvmFMF::setAllowContract},
158  {FastmathFlags::afn, &llvmFMF::setApproxFunc},
159  {FastmathFlags::reassoc, &llvmFMF::setAllowReassoc},
160  {FastmathFlags::fast, &llvmFMF::setFast},
161  // clang-format on
162  };
163  llvm::FastMathFlags ret;
164  auto fmf = op.getFastmathFlags();
165  for (auto it : handlers)
166  if (bitEnumContains(fmf, it.first))
167  (ret.*(it.second))(true);
168  return ret;
169 }

171 /// Returns an LLVM metadata node corresponding to a loop option. This metadata
172 /// is attached to an llvm.loop node.
173 static llvm::MDNode *getLoopOptionMetadata(llvm::LLVMContext &ctx,
174  LoopOptionCase option,
175  int64_t value) {
176  StringRef name;
177  llvm::Constant *cstValue = nullptr;
178  switch (option) {
179  case LoopOptionCase::disable_licm:
180  name = "llvm.licm.disable";
181  cstValue = llvm::ConstantInt::getBool(ctx, value);
182  break;
183  case LoopOptionCase::disable_unroll:
184  name = "llvm.loop.unroll.disable";
185  cstValue = llvm::ConstantInt::getBool(ctx, value);
186  break;
187  case LoopOptionCase::interleave_count:
188  name = "llvm.loop.interleave.count";
189  cstValue = llvm::ConstantInt::get(
190  llvm::IntegerType::get(ctx, /*NumBits=*/32), value);
191  break;
192  case LoopOptionCase::disable_pipeline:
193  name = "llvm.loop.pipeline.disable";
194  cstValue = llvm::ConstantInt::getBool(ctx, value);
195  break;
196  case LoopOptionCase::pipeline_initiation_interval:
197  name = "llvm.loop.pipeline.initiationinterval";
198  cstValue = llvm::ConstantInt::get(
199  llvm::IntegerType::get(ctx, /*NumBits=*/32), value);
200  break;
201  }
202  return llvm::MDNode::get(ctx, {llvm::MDString::get(ctx, name),
203  llvm::ConstantAsMetadata::get(cstValue)});
204 }

206 static void setLoopMetadata(Operation &opInst, llvm::Instruction &llvmInst,
207  llvm::IRBuilderBase &builder,
208  LLVM::ModuleTranslation &moduleTranslation) {
209  if (Attribute attr = opInst.getAttr(LLVMDialect::getLoopAttrName())) {
210  llvm::Module *module = builder.GetInsertBlock()->getModule();
211  llvm::MDNode *loopMD = moduleTranslation.lookupLoopOptionsMetadata(attr);
212  if (!loopMD) {
213  llvm::LLVMContext &ctx = module->getContext();
214 
215  SmallVector<llvm::Metadata *> loopOptions;
216  // Reserve operand 0 for loop id self reference.
217  auto dummy = llvm::MDNode::getTemporary(ctx, llvm::None);
218  loopOptions.push_back(dummy.get());
219 
220  auto loopAttr = attr.cast<DictionaryAttr>();
221  auto parallelAccessGroup =
222  loopAttr.getNamed(LLVMDialect::getParallelAccessAttrName());
223  if (parallelAccessGroup.hasValue()) {
224  SmallVector<llvm::Metadata *> parallelAccess;
225  parallelAccess.push_back(
226  llvm::MDString::get(ctx, "llvm.loop.parallel_accesses"));
227  for (SymbolRefAttr accessGroupRef : parallelAccessGroup->getValue()
228  .cast<ArrayAttr>()
229  .getAsRange<SymbolRefAttr>())
230  parallelAccess.push_back(
231  moduleTranslation.getAccessGroup(opInst, accessGroupRef));
232  loopOptions.push_back(llvm::MDNode::get(ctx, parallelAccess));
233  }
234 
235  if (auto loopOptionsAttr = loopAttr.getAs<LoopOptionsAttr>(
236  LLVMDialect::getLoopOptionsAttrName())) {
237  for (auto option : loopOptionsAttr.getOptions())
238  loopOptions.push_back(
239  getLoopOptionMetadata(ctx, option.first, option.second));
240  }
241 
242  // Create loop options and set the first operand to itself.
243  loopMD = llvm::MDNode::get(ctx, loopOptions);
244  loopMD->replaceOperandWith(0, loopMD);
245 
246  // Store a map from this Attribute to the LLVM metadata in case we
247  // encounter it again.
248  moduleTranslation.mapLoopOptionsMetadata(attr, loopMD);
249  }
250 
251  llvmInst.setMetadata(module->getMDKindID("llvm.loop"), loopMD);
252  }
253 }

255 static LogicalResult
256 convertOperationImpl(Operation &opInst, llvm::IRBuilderBase &builder,
257  LLVM::ModuleTranslation &moduleTranslation) {
258  auto extractPosition = [](ArrayAttr attr) {
259  SmallVector<unsigned, 4> position;
260  position.reserve(attr.size());
261  for (Attribute v : attr)
262  position.push_back(v.cast<IntegerAttr>().getValue().getZExtValue());
263  return position;
264  };
265 
266  llvm::IRBuilder<>::FastMathFlagGuard fmfGuard(builder);
267  if (auto fmf = dyn_cast<FastmathFlagsInterface>(opInst))
268  builder.setFastMathFlags(getFastmathFlags(fmf));
269 
270 #include "mlir/Dialect/LLVMIR/LLVMConversions.inc"
271 #include "mlir/Dialect/LLVMIR/LLVMIntrinsicConversions.inc"
272 
273  // Emit function calls. If the "callee" attribute is present, this is a
274  // direct function call and we also need to look up the remapped function
275  // itself. Otherwise, this is an indirect call and the callee is the first
276  // operand, look it up as a normal value. Return the llvm::Value representing
277  // the function result, which may be of llvm::VoidTy type.
278  auto convertCall = [&](Operation &op) -> llvm::Value * {
279  auto operands = moduleTranslation.lookupValues(op.getOperands());
280  ArrayRef<llvm::Value *> operandsRef(operands);
281  if (auto attr = op.getAttrOfType<FlatSymbolRefAttr>("callee"))
282  return builder.CreateCall(
283  moduleTranslation.lookupFunction(attr.getValue()), operandsRef);
284  auto calleeType =
285  op.getOperands().front().getType().cast<LLVMPointerType>();
286  auto *calleeFunctionType = cast<llvm::FunctionType>(
287  moduleTranslation.convertType(calleeType.getElementType()));
288  return builder.CreateCall(calleeFunctionType, operandsRef.front(),
289  operandsRef.drop_front());
290  };
291 
292  // Emit calls. If the called function has a result, remap the corresponding
293  // value. Note that LLVM IR dialect CallOp has either 0 or 1 result.
294  if (isa<LLVM::CallOp>(opInst)) {
295  llvm::Value *result = convertCall(opInst);
296  if (opInst.getNumResults() != 0) {
297  moduleTranslation.mapValue(opInst.getResult(0), result);
298  return success();
299  }
300  // Check that LLVM call returns void for 0-result functions.
301  return success(result->getType()->isVoidTy());
302  }
303 
304  if (auto inlineAsmOp = dyn_cast<LLVM::InlineAsmOp>(opInst)) {
305  // TODO: refactor function type creation which usually occurs in std-LLVM
306  // conversion.
307  SmallVector<Type, 8> operandTypes;
308  llvm::append_range(operandTypes, inlineAsmOp.getOperands().getTypes());
309 
310  Type resultType;
311  if (inlineAsmOp.getNumResults() == 0) {
312  resultType = LLVM::LLVMVoidType::get(&moduleTranslation.getContext());
313  } else {
314  assert(inlineAsmOp.getNumResults() == 1);
315  resultType = inlineAsmOp.getResultTypes()[0];
316  }
317  auto ft = LLVM::LLVMFunctionType::get(resultType, operandTypes);
318  llvm::InlineAsm *inlineAsmInst =
319  inlineAsmOp.getAsmDialect().hasValue()
320  ? llvm::InlineAsm::get(
321  static_cast<llvm::FunctionType *>(
322  moduleTranslation.convertType(ft)),
323  inlineAsmOp.getAsmString(), inlineAsmOp.getConstraints(),
324  inlineAsmOp.getHasSideEffects(),
325  inlineAsmOp.getIsAlignStack(),
326  convertAsmDialectToLLVM(*inlineAsmOp.getAsmDialect()))
327  : llvm::InlineAsm::get(static_cast<llvm::FunctionType *>(
328  moduleTranslation.convertType(ft)),
329  inlineAsmOp.getAsmString(),
330  inlineAsmOp.getConstraints(),
331  inlineAsmOp.getHasSideEffects(),
332  inlineAsmOp.getIsAlignStack());
333  llvm::CallInst *inst = builder.CreateCall(
334  inlineAsmInst,
335  moduleTranslation.lookupValues(inlineAsmOp.getOperands()));
336  if (auto maybeOperandAttrs = inlineAsmOp.getOperandAttrs()) {
337  llvm::AttributeList attrList;
338  for (const auto &it : llvm::enumerate(*maybeOperandAttrs)) {
339  Attribute attr = it.value();
340  if (!attr)
341  continue;
342  DictionaryAttr dAttr = attr.cast<DictionaryAttr>();
343  TypeAttr tAttr =
344  dAttr.get(InlineAsmOp::getElementTypeAttrName()).cast<TypeAttr>();
345  llvm::AttrBuilder b(moduleTranslation.getLLVMContext());
346  llvm::Type *ty = moduleTranslation.convertType(tAttr.getValue());
347  b.addTypeAttr(llvm::Attribute::ElementType, ty);
348  // shift to account for the returned value (this is always 1 aggregate
349  // value in LLVM).
350  int shift = (opInst.getNumResults() > 0) ? 1 : 0;
351  attrList = attrList.addAttributesAtIndex(
352  moduleTranslation.getLLVMContext(), it.index() + shift, b);
353  }
354  inst->setAttributes(attrList);
355  }
356 
357  if (opInst.getNumResults() != 0)
358  moduleTranslation.mapValue(opInst.getResult(0), inst);
359  return success();
360  }
361 
362  if (auto invOp = dyn_cast<LLVM::InvokeOp>(opInst)) {
363  auto operands = moduleTranslation.lookupValues(invOp.getCalleeOperands());
364  ArrayRef<llvm::Value *> operandsRef(operands);
365  llvm::Instruction *result;
366  if (auto attr = opInst.getAttrOfType<FlatSymbolRefAttr>("callee")) {
367  result = builder.CreateInvoke(
368  moduleTranslation.lookupFunction(attr.getValue()),
369  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
370  moduleTranslation.lookupBlock(invOp.getSuccessor(1)), operandsRef);
371  } else {
372  auto calleeType =
373  invOp.getCalleeOperands().front().getType().cast<LLVMPointerType>();
374  auto *calleeFunctionType = cast<llvm::FunctionType>(
375  moduleTranslation.convertType(calleeType.getElementType()));
376  result = builder.CreateInvoke(
377  calleeFunctionType, operandsRef.front(),
378  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
379  moduleTranslation.lookupBlock(invOp.getSuccessor(1)),
380  operandsRef.drop_front());
381  }
382  moduleTranslation.mapBranch(invOp, result);
383  // InvokeOp can only have 0 or 1 result
384  if (invOp->getNumResults() != 0) {
385  moduleTranslation.mapValue(opInst.getResult(0), result);
386  return success();
387  }
388  return success(result->getType()->isVoidTy());
389  }
390 
391  if (auto lpOp = dyn_cast<LLVM::LandingpadOp>(opInst)) {
392  llvm::Type *ty = moduleTranslation.convertType(lpOp.getType());
393  llvm::LandingPadInst *lpi =
394  builder.CreateLandingPad(ty, lpOp.getNumOperands());
395  lpi->setCleanup(lpOp.getCleanup());
396 
397  // Add clauses
398  for (llvm::Value *operand :
399  moduleTranslation.lookupValues(lpOp.getOperands())) {
400  // All operands should be constant - checked by verifier
401  if (auto *constOperand = dyn_cast<llvm::Constant>(operand))
402  lpi->addClause(constOperand);
403  }
404  moduleTranslation.mapValue(lpOp.getResult(), lpi);
405  return success();
406  }
407 
408  // Emit branches. We need to look up the remapped blocks and ignore the block
409  // arguments that were transformed into PHI nodes.
410  if (auto brOp = dyn_cast<LLVM::BrOp>(opInst)) {
411  llvm::BranchInst *branch =
412  builder.CreateBr(moduleTranslation.lookupBlock(brOp.getSuccessor()));
413  moduleTranslation.mapBranch(&opInst, branch);
414  setLoopMetadata(opInst, *branch, builder, moduleTranslation);
415  return success();
416  }
417  if (auto condbrOp = dyn_cast<LLVM::CondBrOp>(opInst)) {
418  llvm::MDNode *branchWeights = nullptr;
419  if (auto weights = condbrOp.getBranchWeights()) {
420  // Map weight attributes to LLVM metadata.
421  auto weightValues = weights->getValues<APInt>();
422  auto trueWeight = weightValues[0].getSExtValue();
423  auto falseWeight = weightValues[1].getSExtValue();
424  branchWeights =
425  llvm::MDBuilder(moduleTranslation.getLLVMContext())
426  .createBranchWeights(static_cast<uint32_t>(trueWeight),
427  static_cast<uint32_t>(falseWeight));
428  }
429  llvm::BranchInst *branch = builder.CreateCondBr(
430  moduleTranslation.lookupValue(condbrOp.getOperand(0)),
431  moduleTranslation.lookupBlock(condbrOp.getSuccessor(0)),
432  moduleTranslation.lookupBlock(condbrOp.getSuccessor(1)), branchWeights);
433  moduleTranslation.mapBranch(&opInst, branch);
434  setLoopMetadata(opInst, *branch, builder, moduleTranslation);
435  return success();
436  }
437  if (auto switchOp = dyn_cast<LLVM::SwitchOp>(opInst)) {
438  llvm::MDNode *branchWeights = nullptr;
439  if (auto weights = switchOp.getBranchWeights()) {
440  llvm::SmallVector<uint32_t> weightValues;
441  weightValues.reserve(weights->size());
442  for (llvm::APInt weight : weights->cast<DenseIntElementsAttr>())
443  weightValues.push_back(weight.getLimitedValue());
444  branchWeights = llvm::MDBuilder(moduleTranslation.getLLVMContext())
445  .createBranchWeights(weightValues);
446  }
447 
448  llvm::SwitchInst *switchInst = builder.CreateSwitch(
449  moduleTranslation.lookupValue(switchOp.getValue()),
450  moduleTranslation.lookupBlock(switchOp.getDefaultDestination()),
451  switchOp.getCaseDestinations().size(), branchWeights);
452 
453  auto *ty = llvm::cast<llvm::IntegerType>(
454  moduleTranslation.convertType(switchOp.getValue().getType()));
455  for (auto i :
456  llvm::zip(switchOp.getCaseValues()->cast<DenseIntElementsAttr>(),
457  switchOp.getCaseDestinations()))
458  switchInst->addCase(
459  llvm::ConstantInt::get(ty, std::get<0>(i).getLimitedValue()),
460  moduleTranslation.lookupBlock(std::get<1>(i)));
461 
462  moduleTranslation.mapBranch(&opInst, switchInst);
463  return success();
464  }
465 
466  // Emit addressof. We need to look up the global value referenced by the
467  // operation and store it in the MLIR-to-LLVM value mapping. This does not
468  // emit any LLVM instruction.
469  if (auto addressOfOp = dyn_cast<LLVM::AddressOfOp>(opInst)) {
470  LLVM::GlobalOp global = addressOfOp.getGlobal();
471  LLVM::LLVMFuncOp function = addressOfOp.getFunction();
472 
473  // The verifier should not have allowed this.
474  assert((global || function) &&
475  "referencing an undefined global or function");
476 
477  moduleTranslation.mapValue(
478  addressOfOp.getResult(),
479  global ? moduleTranslation.lookupGlobal(global)
480  : moduleTranslation.lookupFunction(function.getName()));
481  return success();
482  }
483 
484  return failure();
485 }
