539  /// Walk the operation by calling the callback for each nested operation
540  /// (including this one), block or region, depending on the callback provided.
541  /// Regions, blocks and operations at the same nesting level are visited in
542  /// lexicographical order. The walk order for enclosing regions, blocks and
543  /// operations with respect to their nested ones is specified by 'Order'
544  /// (post-order by default). A callback on a block or operation is allowed to
545  /// erase that block or operation if either:
546  /// * the walk is in post-order, or
547  /// * the walk is in pre-order and the walk is skipped after the erasure.
548  ///
549  /// The callback method can take any of the following forms:
550  /// void(Operation*) : Walk all operations opaquely.
551  /// * op->walk([](Operation *nestedOp) { ...});
552  /// void(OpT) : Walk all operations of the given derived type.
553  /// * op->walk([](ReturnOp returnOp) { ...});
554  /// WalkResult(Operation*|OpT) : Walk operations, but allow for
555  /// interruption/skipping.
556  /// * op->walk([](... op) {
557  /// // Skip the walk of this op based on some invariant.
558  /// if (some_invariant)
559  /// return WalkResult::skip();
560  /// // Interrupt, i.e cancel, the walk based on some invariant.
561  /// if (another_invariant)
562  /// return WalkResult::interrupt();
563  /// return WalkResult::advance();
564  /// });
565  template <WalkOrder Order = WalkOrder::PostOrder, typename FnT,
566  typename RetT = detail::walkResultType<FnT>>
567  typename std::enable_if<
568  llvm::function_traits<std::decay_t<FnT>>::num_args == 1, RetT>::type
569  walk(FnT &&callback) {
570  return detail::walk<Order>(this, std::forward<FnT>(callback));
571  }
572 
573  /// Generic walker with a stage aware callback. Walk the operation by calling
574  /// the callback for each nested operation (including this one) N+1 times,
575  /// where N is the number of regions attached to that operation.
576  ///
577  /// The callback method can take any of the following forms:
578  /// void(Operation *, const WalkStage &) : Walk all operation opaquely
579  /// * op->walk([](Operation *nestedOp, const WalkStage &stage) { ...});
580  /// void(OpT, const WalkStage &) : Walk all operations of the given derived
581  /// type.
582  /// * op->walk([](ReturnOp returnOp, const WalkStage &stage) { ...});
583  /// WalkResult(Operation*|OpT, const WalkStage &stage) : Walk operations,
584  /// but allow for interruption/skipping.
585  /// * op->walk([](... op, const WalkStage &stage) {
586  /// // Skip the walk of this op based on some invariant.
587  /// if (some_invariant)
588  /// return WalkResult::skip();
589  /// // Interrupt, i.e cancel, the walk based on some invariant.
590  /// if (another_invariant)
591  /// return WalkResult::interrupt();
592  /// return WalkResult::advance();
593  /// });
594  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
595  typename std::enable_if<
596  llvm::function_traits<std::decay_t<FnT>>::num_args == 2, RetT>::type
597  walk(FnT &&callback) {
598  return detail::walk(this, std::forward<FnT>(callback));
599  }
