35 class Type {
36 public:
37  /// This class represents the internal storage of the Type class.
38  struct Storage;
39 
40  /// This class provides several utilities when defining derived type classes.
41  template <typename ImplT, typename BaseT = Type>
42  class TypeBase : public BaseT {
43  public:
44  using Base = TypeBase<ImplT, BaseT>;
45  using ImplTy = ImplT;
46  using BaseT::BaseT;
47 
48  /// Provide type casting support.
49  static bool classof(Type type) {
50  return type.getTypeID() == TypeID::get<ImplTy>();
51  }
52  };
53 
54  Type(Storage *impl = nullptr) : impl(impl) {}
55 
56  bool operator==(const Type &other) const { return impl == other.impl; }
57  bool operator!=(const Type &other) const { return !(*this == other); }
58  explicit operator bool() const { return impl; }
59 
60  /// Provide type casting support.
61  template <typename U> bool isa() const {
62  assert(impl && "isa<> used on a null type.");
63  return U::classof(*this);
64  }
65  template <typename U, typename V, typename... Others> bool isa() const {
66  return isa<U>() || isa<V, Others...>();
67  }
68  template <typename U> U dyn_cast() const {
69  return isa<U>() ? U(impl) : U(nullptr);
70  }
71  template <typename U> U dyn_cast_or_null() const {
72  return (impl && isa<U>()) ? U(impl) : U(nullptr);
73  }
74  template <typename U> U cast() const {
75  assert(isa<U>());
76  return U(impl);
77  }
78 
79  /// Return the internal storage instance of this type.
80  Storage *getImpl() const { return impl; }
81 
82  /// Return the TypeID instance of this type.
83  TypeID getTypeID() const;
84 
85  /// Print this type to the given stream.
86  void print(raw_ostream &os) const;
87 
88  /// Try to refine this type with the one provided. Given two compatible types,
89  /// this will return a merged type contains as much detail from the two types.
90  /// For example, if refining two operation types and one contains a name,
91  /// while the other doesn't, the refined type contains the name. If the two
92  /// types are incompatible, null is returned.
93  Type refineWith(Type other) const;
94 
95 protected:
96  /// Return the internal storage instance of this type reinterpreted as the
97  /// given derived storage type.
98  template <typename T> const T *getImplAs() const {
99  return static_cast<const T *>(impl);
100  }
101 
102 private:
103  Storage *impl;
104 };
105 
106 inline llvm::hash_code hash_value(Type type) {
107  return DenseMapInfo<Type::Storage *>::getHashValue(type.getImpl());
108 }
109 
110 inline raw_ostream &operator<<(raw_ostream &os, Type type) {
111  type.print(os);
112  return os;
113 }
