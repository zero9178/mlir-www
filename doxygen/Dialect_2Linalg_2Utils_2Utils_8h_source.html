248 /// Scheme used to distribute loops to processors.
249 enum class DistributionMethod {
250  /// Cyclic distribution where no assumption is made about the dynamic
251  /// relationship between number of processors and number of iterations of the
252  /// distributed loop. Distributes the following loop
253  ///
254  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
255  ///
256  /// to
257  ///
258  /// scf.parallel(%iv)= (%lb + %procId * %step) to (%ub) step (%step * %nprocs)
259  Cyclic = 0,
260 
261  /// Cyclic distribution where the number of processors can be assumed to be
262  /// more than or equal to the number of iterations of the distributed loop. In
263  /// such cases, a simple in-bounds check is enough (instead of materializing a
264  /// loop). Distributes the following loop
265  ///
266  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
267  ///
268  /// to
269  ///
270  /// %iv = %lb + %procId * %step
271  /// %cond = arith.cmpi "slt", %iv, %ub
272  /// scf.if %cond {
273  /// ...
274  /// }
275  CyclicNumProcsGeNumIters = 1,
276 
277  /// Cyclic distribution where the number of processors can be assumed to be
278  /// equal to the number of iterations of the distributed loop. In such cases,
279  /// no bounds check is needed. Distributes the following loop
280  ///
281  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
282  ///
283  /// to
284  ///
285  /// %iv = %lb + %procId * %step
286  CyclicNumProcsEqNumIters = 2
287 };

300 /// Options that allow distribution of loops generated in Linalg transforms to
301 /// processors while generating the loops.
302 struct LinalgLoopDistributionOptions {
303  /// Callback function that returns the Values for processor ID (`procId`), and
304  /// number of processors (`nprocs`) used to execute the parallel loops. The
305  /// number of `{procId, nprocs}` pairs returned must be equal to the number of
