24 /// This class represents the result of operations optimizing something subject
25 /// to some constraints. If the constraints were not satisfiable the, kind will
26 /// be Empty. If the optimum is unbounded, the kind is Unbounded, and if the
27 /// optimum is bounded, the kind will be Bounded and `optimum` holds the optimal
28 /// value.
29 enum class OptimumKind { Empty, Unbounded, Bounded };
30 template <typename T>
31 class MaybeOptimum {
32 public:
33 private:
34  OptimumKind kind = OptimumKind::Empty;
35  T optimum;
36 
37 public:
38  MaybeOptimum() = default;
39  MaybeOptimum(OptimumKind kind) : kind(kind) {
40  assert(kind != OptimumKind::Bounded &&
41  "Bounded optima should be constructed by specifying the optimum!");
42  }
43  MaybeOptimum(const T &optimum)
44  : kind(OptimumKind::Bounded), optimum(optimum) {}
45 
46  OptimumKind getKind() const { return kind; }
47  bool isBounded() const { return kind == OptimumKind::Bounded; }
48  bool isUnbounded() const { return kind == OptimumKind::Unbounded; }
49  bool isEmpty() const { return kind == OptimumKind::Empty; }
50 
51  Optional<T> getOptimumIfBounded() const { return optimum; }
52  const T &getBoundedOptimum() const {
53  assert(kind == OptimumKind::Bounded &&
54  "This should be called only for bounded optima");
55  return optimum;
56  }
57  T &getBoundedOptimum() {
58  assert(kind == OptimumKind::Bounded &&
59  "This should be called only for bounded optima");
60  return optimum;
61  }
62  const T &operator*() const { return getBoundedOptimum(); }
63  T &operator*() { return getBoundedOptimum(); }
64  const T *operator->() const { return &getBoundedOptimum(); }
65  T *operator->() { return &getBoundedOptimum(); }
66  bool operator==(const MaybeOptimum<T> &other) const {
67  if (kind != other.kind)
68  return false;
69  if (kind != OptimumKind::Bounded)
70  return true;
71  return optimum == other.optimum;
72  }
73 
74  // Given f that takes a T and returns a U, convert this `MaybeOptimum<T>` to
75  // a `MaybeOptimum<U>` by applying `f` to the bounded optimum if it exists, or
76  // returning a MaybeOptimum of the same kind otherwise.
77  template <class Function>
78  auto map(const Function &f) const & -> MaybeOptimum<decltype(f(optimum))> {
79  if (kind == OptimumKind::Bounded)
80  return f(optimum);
81  return kind;
82  }
83 };

88 /// `MaybeLocalRepr` contains the indices of the contraints that can be
89 /// expressed as a floordiv of an affine function. If it's an `equality`
90 /// contraint `equalityIdx` is set, in case of `inequality` the `lowerBoundIdx`
91 /// and `upperBoundIdx` is set. By default the kind attribute is set to None.
92 struct MaybeLocalRepr {
93  ReprKind kind = ReprKind::None;
94  explicit operator bool() const { return kind != ReprKind::None; }
95  union {
96  unsigned equalityIdx;

98  unsigned lowerBoundIdx, upperBoundIdx;
99  } inequalityPair;
100  } repr;

117 /// Given dividends of divisions `divs` and denominators `denoms`, detects and
118 /// removes duplicate divisions. `localOffset` is the offset in dividend of a
119 /// division from where local identifiers start.
120 ///
121 /// On every possible duplicate division found, `merge(i, j)`, where `i`, `j`
122 /// are current index of the duplicate divisions, is called and division at
123 /// index `j` is merged into division at index `i`. If `merge(i, j)` returns
124 /// `true`, the divisions are merged i.e. `j^th` division gets eliminated and
125 /// it's each instance is replaced by `i^th` division. If it returns `false`,
126 /// the divisions are not merged. `merge` can also do side effects, For example
127 /// it can merge the local identifiers in IntegerRelation.
128 void removeDuplicateDivs(
129  std::vector<SmallVector<int64_t, 8>> &divs,
130  SmallVectorImpl<unsigned> &denoms, unsigned localOffset,
131  llvm::function_ref<bool(unsigned i, unsigned j)> merge);

133 /// Given two relations, A and B, add additional local ids to the sets such
134 /// that both have the union of the local ids in each set, without changing
135 /// the set of points that lie in A and B.
136 ///
137 /// While taking union, if a local id in any set has a division representation
138 /// which is a duplicate of division representation, of another local id in any
139 /// set, it is not added to the final union of local ids and is instead merged.
140 ///
141 /// On every possible merge, `merge(i, j)` is called. `i`, `j` are position
142 /// of local identifiers in both sets which are being merged. If `merge(i, j)`
143 /// returns true, the divisions are merged, otherwise the divisions are not
144 /// merged.
145 void mergeLocalIds(IntegerRelation &relA, IntegerRelation &relB,
146  llvm::function_ref<bool(unsigned i, unsigned j)> merge);
