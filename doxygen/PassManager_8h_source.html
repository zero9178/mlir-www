48 /// This class represents a pass manager that runs passes on a specific
49 /// operation type. This class is not constructed directly, but nested within
50 /// other OpPassManagers or the top-level PassManager.
51 class OpPassManager {
52 public:
53  enum class Nesting { Implicit, Explicit };
54  OpPassManager(StringAttr name, Nesting nesting = Nesting::Explicit);
55  OpPassManager(StringRef name, Nesting nesting = Nesting::Explicit);
56  OpPassManager(OpPassManager &&rhs);
57  OpPassManager(const OpPassManager &rhs);
58  ~OpPassManager();
59  OpPassManager &operator=(const OpPassManager &rhs);

61  /// Iterator over the passes in this pass manager.
62  using pass_iterator =
63  llvm::pointee_iterator<MutableArrayRef<std::unique_ptr<Pass>>::iterator>;
64  pass_iterator begin();
65  pass_iterator end();
66  iterator_range<pass_iterator> getPasses() { return {begin(), end()}; }
67 
68  using const_pass_iterator =
69  llvm::pointee_iterator<ArrayRef<std::unique_ptr<Pass>>::const_iterator>;
70  const_pass_iterator begin() const;
71  const_pass_iterator end() const;
72  iterator_range<const_pass_iterator> getPasses() const {
73  return {begin(), end()};
74  }
75 
76  /// Returns true if the pass manager has no passes.
77  bool empty() const { return begin() == end(); }

79  /// Nest a new operation pass manager for the given operation kind under this
80  /// pass manager.
81  OpPassManager &nest(StringAttr nestedName);

238  /// A configuration struct provided to the IR printer instrumentation.
239  class IRPrinterConfig {
240  public:
241  using PrintCallbackFn = function_ref<void(raw_ostream &)>;
242 
243  /// Initialize the configuration.
244  /// * 'printModuleScope' signals if the top-level module IR should always be
245  /// printed. This should only be set to true when multi-threading is
246  /// disabled, otherwise we may try to print IR that is being modified
247  /// asynchronously.
248  /// * 'printAfterOnlyOnChange' signals that when printing the IR after a
249  /// pass, in the case of a non-failure, we should first check if any
250  /// potential mutations were made. This allows for reducing the number of
251  /// logs that don't contain meaningful changes.
252  /// * 'printAfterOnlyOnFailure' signals that when printing the IR after a
253  /// pass, we only print in the case of a failure.
254  /// - This option should *not* be used with the other `printAfter` flags
255  /// above.
256  /// * 'opPrintingFlags' sets up the printing flags to use when printing the
257  /// IR.
258  explicit IRPrinterConfig(
259  bool printModuleScope = false, bool printAfterOnlyOnChange = false,
260  bool printAfterOnlyOnFailure = false,
261  OpPrintingFlags opPrintingFlags = OpPrintingFlags());
262  virtual ~IRPrinterConfig();
263 
264  /// A hook that may be overridden by a derived config that checks if the IR
265  /// of 'operation' should be dumped *before* the pass 'pass' has been
266  /// executed. If the IR should be dumped, 'printCallback' should be invoked
267  /// with the stream to dump into.
268  virtual void printBeforeIfEnabled(Pass *pass, Operation *operation,
269  PrintCallbackFn printCallback);
270 
271  /// A hook that may be overridden by a derived config that checks if the IR
272  /// of 'operation' should be dumped *after* the pass 'pass' has been
273  /// executed. If the IR should be dumped, 'printCallback' should be invoked
274  /// with the stream to dump into.
275  virtual void printAfterIfEnabled(Pass *pass, Operation *operation,
276  PrintCallbackFn printCallback);
277 
278  /// Returns true if the IR should always be printed at the top-level scope.
279  bool shouldPrintAtModuleScope() const { return printModuleScope; }
280 
281  /// Returns true if the IR should only printed after a pass if the IR
282  /// "changed".
283  bool shouldPrintAfterOnlyOnChange() const { return printAfterOnlyOnChange; }
284 
285  /// Returns true if the IR should only printed after a pass if the pass
286  /// "failed".
287  bool shouldPrintAfterOnlyOnFailure() const {
288  return printAfterOnlyOnFailure;
289  }
290 
291  /// Returns the printing flags to be used to print the IR.
292  OpPrintingFlags getOpPrintingFlags() const { return opPrintingFlags; }
293 
294  private:
295  /// A flag that indicates if the IR should be printed at module scope.
296  bool printModuleScope;
297 
298  /// A flag that indicates that the IR after a pass should only be printed if
299  /// a change is detected.
300  bool printAfterOnlyOnChange;
301 
302  /// A flag that indicates that the IR after a pass should only be printed if
303  /// the pass failed.
304  bool printAfterOnlyOnFailure;
305 
306  /// Flags to control printing behavior.
307  OpPrintingFlags opPrintingFlags;
308  };

314  /// Add an instrumentation to print the IR before and after pass execution,
315  /// using the provided fields to generate a default configuration:
316  /// * 'shouldPrintBeforePass' and 'shouldPrintAfterPass' correspond to filter

319  /// * 'printModuleScope' signals if the module IR should be printed, even
320  /// for non module passes.
321  /// * 'printAfterOnlyOnChange' signals that when printing the IR after a
322  /// pass, in the case of a non-failure, we should first check if any
323  /// potential mutations were made.
324  /// * 'printAfterOnlyOnFailure' signals that when printing the IR after a
325  /// pass, we only print in the case of a failure.
326  /// - This option should *not* be used with the other `printAfter` flags
327  /// above.
328  /// * 'out' corresponds to the stream to output the printed IR to.
329  /// * 'opPrintingFlags' sets up the printing flags to use when printing the
330  /// IR.
331  void enableIRPrinting(
332  std::function<bool(Pass *, Operation *)> shouldPrintBeforePass =
333  [](Pass *, Operation *) { return true; },
334  std::function<bool(Pass *, Operation *)> shouldPrintAfterPass =
335  [](Pass *, Operation *) { return true; },
336  bool printModuleScope = true, bool printAfterOnlyOnChange = true,
337  bool printAfterOnlyOnFailure = false, raw_ostream &out = llvm::errs(),
