81 extern "C" int64_t _mlir_ciface_nano_time() {
82  auto now = std::chrono::high_resolution_clock::now();
83  auto duration = now.time_since_epoch();
84  auto nanoseconds =
85  std::chrono::duration_cast<std::chrono::nanoseconds>(duration);
