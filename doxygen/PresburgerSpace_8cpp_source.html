44 unsigned PresburgerSpace::getIdKindOverlap(IdKind kind, unsigned idStart,
45  unsigned idLimit) const {
46  unsigned idRangeStart = getIdKindOffset(kind);
47  unsigned idRangeEnd = getIdKindEnd(kind);
48 
49  // Compute number of elements in intersection of the ranges [idStart, idLimit)
50  // and [idRangeStart, idRangeEnd).
51  unsigned overlapStart = std::max(idStart, idRangeStart);
52  unsigned overlapEnd = std::min(idLimit, idRangeEnd);
53 
54  if (overlapStart > overlapEnd)
55  return 0;
56  return overlapEnd - overlapStart;
57 }
