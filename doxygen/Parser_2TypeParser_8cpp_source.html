149 /// Parse the offset and strides from a strided layout specification.
150 ///
151 /// strided-layout ::= `offset:` dimension `,` `strides: ` stride-list
152 ///
153 ParseResult Parser::parseStridedLayout(int64_t &offset,
154  SmallVectorImpl<int64_t> &strides) {
155  // Parse offset.
156  consumeToken(Token::kw_offset);
157  if (!consumeIf(Token::colon))
158  return emitError("expected colon after `offset` keyword");
159  auto maybeOffset = getToken().getUnsignedIntegerValue();
160  bool question = getToken().is(Token::question);
161  if (!maybeOffset && !question)
162  return emitError("invalid offset");
163  offset = maybeOffset ? static_cast<int64_t>(maybeOffset.getValue())
164  : MemRefType::getDynamicStrideOrOffset();
165  consumeToken();
166 
167  if (!consumeIf(Token::comma))
168  return emitError("expected comma after offset value");
169 
170  // Parse stride list.
171  if (parseToken(Token::kw_strides,
172  "expected `strides` keyword after offset specification") ||
173 
174  parseToken(Token::colon, "expected colon after `strides` keyword") ||
175  parseStrideList(strides))
176  return failure();
177  return success();
178 }

180 /// Parse a memref type.
181 ///
182 /// memref-type ::= ranked-memref-type | unranked-memref-type
183 ///
184 /// ranked-memref-type ::= `memref` `<` dimension-list-ranked type
185 /// (`,` layout-specification)? (`,` memory-space)? `>`
186 ///
187 /// unranked-memref-type ::= `memref` `<*x` type (`,` memory-space)? `>`
188 ///
189 /// stride-list ::= `[` (dimension (`,` dimension)*)? `]`
190 /// strided-layout ::= `offset:` dimension `,` `strides: ` stride-list
191 /// layout-specification ::= semi-affine-map | strided-layout | attribute
192 /// memory-space ::= integer-literal | attribute
193 ///
194 Type Parser::parseMemRefType() {
195  SMLoc loc = getToken().getLoc();
196  consumeToken(Token::kw_memref);
197 
198  if (parseToken(Token::less, "expected '<' in memref type"))
199  return nullptr;
200 
201  bool isUnranked;
202  SmallVector<int64_t, 4> dimensions;
203 
204  if (consumeIf(Token::star)) {
205  // This is an unranked memref type.
206  isUnranked = true;
207  if (parseXInDimensionList())
208  return nullptr;
209 
210  } else {
211  isUnranked = false;
212  if (parseDimensionListRanked(dimensions))
213  return nullptr;
214  }
215 
216  // Parse the element type.
217  auto typeLoc = getToken().getLoc();
218  auto elementType = parseType();
219  if (!elementType)
220  return nullptr;
221 
222  // Check that memref is formed from allowed types.
223  if (!BaseMemRefType::isValidElementType(elementType))
224  return emitError(typeLoc, "invalid memref element type"), nullptr;
225 
226  MemRefLayoutAttrInterface layout;
227  Attribute memorySpace;
228 
229  auto parseElt = [&]() -> ParseResult {
230  // Check for AffineMap as offset/strides.
231  if (getToken().is(Token::kw_offset)) {
232  int64_t offset;
233  SmallVector<int64_t, 4> strides;
234  if (failed(parseStridedLayout(offset, strides)))
235  return failure();
236  // Construct strided affine map.
237  AffineMap map =
238  makeStridedLinearLayoutMap(strides, offset, state.context);
239  layout = AffineMapAttr::get(map);
240  } else {
241  // Either it is MemRefLayoutAttrInterface or memory space attribute.
242  Attribute attr = parseAttribute();
243  if (!attr)
244  return failure();
245 
246  if (attr.isa<MemRefLayoutAttrInterface>()) {
247  layout = attr.cast<MemRefLayoutAttrInterface>();
248  } else if (memorySpace) {
249  return emitError("multiple memory spaces specified in memref type");
250  } else {
251  memorySpace = attr;
252  return success();
253  }
254  }
255 
256  if (isUnranked)
257  return emitError("cannot have affine map for unranked memref type");
258  if (memorySpace)
259  return emitError("expected memory space to be last in memref type");
260 
261  return success();
262  };
263 
264  // Parse a list of mappings and address space if present.
265  if (!consumeIf(Token::greater)) {
266  // Parse comma separated list of affine maps, followed by memory space.
267  if (parseToken(Token::comma, "expected ',' or '>' in memref type") ||
268  parseCommaSeparatedListUntil(Token::greater, parseElt,
269  /*allowEmptyList=*/false)) {
270  return nullptr;
271  }
272  }
273 
274  if (isUnranked)
275  return getChecked<UnrankedMemRefType>(loc, elementType, memorySpace);
276 
277  return getChecked<MemRefType>(loc, dimensions, elementType, layout,
278  memorySpace);
279 }

281 /// Parse any type except the function type.
282 ///
283 /// non-function-type ::= integer-type
284 /// | index-type
285 /// | float-type
286 /// | extended-type
287 /// | vector-type
288 /// | tensor-type
289 /// | memref-type
290 /// | complex-type
291 /// | tuple-type
292 /// | none-type
293 ///
294 /// index-type ::= `index`
295 /// float-type ::= `f16` | `bf16` | `f32` | `f64` | `f80` | `f128`
296 /// none-type ::= `none`
297 ///
298 Type Parser::parseNonFunctionType() {
299  switch (getToken().getKind()) {
300  default:
301  return (emitError("expected non-function type"), nullptr);
302  case Token::kw_memref:
303  return parseMemRefType();
304  case Token::kw_tensor:
305  return parseTensorType();
306  case Token::kw_complex:
307  return parseComplexType();
308  case Token::kw_tuple:
309  return parseTupleType();
310  case Token::kw_vector:
311  return parseVectorType();
312  // integer-type
313  case Token::inttype: {
314  auto width = getToken().getIntTypeBitwidth();
315  if (!width.hasValue())
316  return (emitError("invalid integer width"), nullptr);
317  if (width.getValue() > IntegerType::kMaxWidth) {
318  emitError(getToken().getLoc(), "integer bitwidth is limited to ")
319  << IntegerType::kMaxWidth << " bits";
320  return nullptr;
321  }
322 
323  IntegerType::SignednessSemantics signSemantics = IntegerType::Signless;
324  if (Optional<bool> signedness = getToken().getIntTypeSignedness())
325  signSemantics = *signedness ? IntegerType::Signed : IntegerType::Unsigned;
326 
327  consumeToken(Token::inttype);
328  return IntegerType::get(getContext(), width.getValue(), signSemantics);
329  }
330 
331  // float-type
332  case Token::kw_bf16:
333  consumeToken(Token::kw_bf16);
334  return builder.getBF16Type();
335  case Token::kw_f16:
336  consumeToken(Token::kw_f16);
337  return builder.getF16Type();
338  case Token::kw_f32:
339  consumeToken(Token::kw_f32);
340  return builder.getF32Type();
341  case Token::kw_f64:
342  consumeToken(Token::kw_f64);
343  return builder.getF64Type();
344  case Token::kw_f80:
345  consumeToken(Token::kw_f80);
346  return builder.getF80Type();
347  case Token::kw_f128:
348  consumeToken(Token::kw_f128);
349  return builder.getF128Type();
350 
351  // index-type
352  case Token::kw_index:
353  consumeToken(Token::kw_index);
354  return builder.getIndexType();
355 
356  // none-type
357  case Token::kw_none:
358  consumeToken(Token::kw_none);
359  return builder.getNoneType();
360 
361  // extended type
362  case Token::exclamation_identifier:
363  return parseExtendedType();
364  }
365 }

367 /// Parse a tensor type.
368 ///
369 /// tensor-type ::= `tensor` `<` dimension-list type `>`
370 /// dimension-list ::= dimension-list-ranked | `*x`
371 ///
372 Type Parser::parseTensorType() {
373  consumeToken(Token::kw_tensor);
374 
375  if (parseToken(Token::less, "expected '<' in tensor type"))
376  return nullptr;
377 
378  bool isUnranked;
379  SmallVector<int64_t, 4> dimensions;
380 
381  if (consumeIf(Token::star)) {
382  // This is an unranked tensor type.
383  isUnranked = true;
384 
385  if (parseXInDimensionList())
386  return nullptr;
387 
388  } else {
389  isUnranked = false;
390  if (parseDimensionListRanked(dimensions))
391  return nullptr;
392  }
393 
394  // Parse the element type.
395  auto elementTypeLoc = getToken().getLoc();
396  auto elementType = parseType();
397 
398  // Parse an optional encoding attribute.
399  Attribute encoding;
400  if (consumeIf(Token::comma)) {
401  encoding = parseAttribute();
402  if (auto v = encoding.dyn_cast_or_null<VerifiableTensorEncoding>()) {
403  if (failed(v.verifyEncoding(dimensions, elementType,
404  [&] { return emitError(); })))
405  return nullptr;
406  }
407  }
408 
409  if (!elementType || parseToken(Token::greater, "expected '>' in tensor type"))
410  return nullptr;
411  if (!TensorType::isValidElementType(elementType))
412  return emitError(elementTypeLoc, "invalid tensor element type"), nullptr;
413 
414  if (isUnranked) {
415  if (encoding)
416  return emitError("cannot apply encoding to unranked tensor"), nullptr;
417  return UnrankedTensorType::get(elementType);
418  }
419  return RankedTensorType::get(dimensions, elementType, encoding);
420 }

480 /// Parse a dimension list in a vector type. This populates the dimension list,
481 /// and returns the number of scalable dimensions in `numScalableDims`.
482 ///
483 /// vector-dim-list := (static-dim-list `x`)? (`[` static-dim-list `]` `x`)?
484 /// static-dim-list ::= decimal-literal (`x` decimal-literal)*
485 ///
486 ParseResult
487 Parser::parseVectorDimensionList(SmallVectorImpl<int64_t> &dimensions,
488  unsigned &numScalableDims) {
489  numScalableDims = 0;
490  // If there is a set of fixed-length dimensions, consume it
491  while (getToken().is(Token::integer)) {
492  int64_t value;
493  if (parseIntegerInDimensionList(value))
494  return failure();
495  dimensions.push_back(value);
496  // Make sure we have an 'x' or something like 'xbf32'.
497  if (parseXInDimensionList())
498  return failure();
499  }
500  // If there is a set of scalable dimensions, consume it
501  if (consumeIf(Token::l_square)) {
502  while (getToken().is(Token::integer)) {
503  int64_t value;
504  if (parseIntegerInDimensionList(value))
505  return failure();
506  dimensions.push_back(value);
507  numScalableDims++;
508  // Check if we have reached the end of the scalable dimension list
509  if (consumeIf(Token::r_square)) {
510  // Make sure we have something like 'xbf32'.
511  if (parseXInDimensionList())
512  return failure();
513  return success();
514  }
515  // Make sure we have an 'x'
516  if (parseXInDimensionList())
517  return failure();
518  }
519  // If we make it here, we've finished parsing the dimension list
520  // without finding ']' closing the set of scalable dimensions
521  return emitError("missing ']' closing set of scalable dimensions");
522  }
523 
524  return success();
525 }

527 /// Parse a dimension list of a tensor or memref type. This populates the
528 /// dimension list, using -1 for the `?` dimensions if `allowDynamic` is set and
529 /// errors out on `?` otherwise.
530 ///
531 /// dimension-list-ranked ::= (dimension `x`)*
532 /// dimension ::= `?` | decimal-literal
533 ///
534 /// When `allowDynamic` is not set, this is used to parse:
535 ///
536 /// static-dimension-list ::= (decimal-literal `x`)*
537 ParseResult
538 Parser::parseDimensionListRanked(SmallVectorImpl<int64_t> &dimensions,
539  bool allowDynamic) {
540  while (getToken().isAny(Token::integer, Token::question)) {
541  if (consumeIf(Token::question)) {
542  if (!allowDynamic)
543  return emitError("expected static shape");
544  dimensions.push_back(-1);
545  } else {
546  int64_t value;
547  if (parseIntegerInDimensionList(value))
548  return failure();
549  dimensions.push_back(value);
550  }
551  // Make sure we have an 'x' or something like 'xbf32'.
552  if (parseXInDimensionList())
553  return failure();
554  }
555 
556  return success();
557 }

559 ParseResult Parser::parseIntegerInDimensionList(int64_t &value) {
560  // Hexadecimal integer literals (starting with `0x`) are not allowed in
561  // aggregate type declarations. Therefore, `0xf32` should be processed as
562  // a sequence of separate elements `0`, `x`, `f32`.
563  if (getTokenSpelling().size() > 1 && getTokenSpelling()[1] == 'x') {
564  // We can get here only if the token is an integer literal. Hexadecimal
565  // integer literals can only start with `0x` (`1x` wouldn't lex as a
566  // literal, just `1` would, at which point we don't get into this
567  // branch).
568  assert(getTokenSpelling()[0] == '0' && "invalid integer literal");
569  value = 0;
570  state.lex.resetPointer(getTokenSpelling().data() + 1);
571  consumeToken();
572  } else {
573  // Make sure this integer value is in bound and valid.
574  Optional<uint64_t> dimension = getToken().getUInt64IntegerValue();
575  if (!dimension ||
576  *dimension > (uint64_t)std::numeric_limits<int64_t>::max())
577  return emitError("invalid dimension");
578  value = (int64_t)dimension.getValue();
579  consumeToken(Token::integer);
580  }
581  return success();
582 }

584 /// Parse an 'x' token in a dimension list, handling the case where the x is
585 /// juxtaposed with an element type, as in "xf32", leaving the "f32" as the next
586 /// token.
587 ParseResult Parser::parseXInDimensionList() {
588  if (getToken().isNot(Token::bare_identifier) || getTokenSpelling()[0] != 'x')
589  return emitError("expected 'x' in dimension list");
590 
591  // If we had a prefix of 'x', lex the next token immediately after the 'x'.
592  if (getTokenSpelling().size() != 1)
593  state.lex.resetPointer(getTokenSpelling().data() + 1);
594 
595  // Consume the 'x'.
596  consumeToken(Token::bare_identifier);
597 
598  return success();
599 }
