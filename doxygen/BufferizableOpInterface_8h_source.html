52  /// An op filter entry. Filters can be used to specify which ops should be

58  /// Filter type: A filter can either be a DENY filter or an ALLOW filter.

193  /// If set to `true`, an `getAliasingOpResult` will return the corresponding
194  /// "out"/"dest" OpOperand for every op that has the notion of an "out"/"dest"
195  /// operand. I.e., the aliasing OpOperand of the i-th tensor OpResult is
196  /// usually the i-th "out" tensor OpOperand. This is in line with
197  /// destination-passing style and the default behavior. Op interface
198  /// implementations must follow this contract to avoid surprising behavior.
199  ///
200  /// If set to `false`, BufferizableOpInterface implementations can try to be
201  /// smart and choose to alias with "in" operands or other operands. E.g., the
202  /// result of a `linalg.generic` op could bufferize in-place with an "in"
203  /// OpOperand if the corresponding "out" operand is not used within the
204  /// computation. Whether this pays off or not can be very input IR-specific.
205  bool alwaysAliasingWithDest = true;
206 
207  /// If set to `true`, try to hoist allocations out of blocks as much as
208  /// possible. An allocation is not hoisted across allocation hoisting barriers
209  /// as indicated by `BufferizableOpInterface::isAllocationHoistingBarrier`.
210  ///
211  /// Examples of allocation hoisting barriers are parallel loops or ops where
212  /// SSA values cannot be captured from the outside.
213  bool hoistAllocations = true;

298  /// Starting from `value`, follow the use-def chain in reverse, always
299  /// selecting the aliasing OpOperands. Find and return Values for which
300  /// `condition` evaluates to true. OpOperands of such matching Values are not
301  /// traversed any further.
302  ///
303  /// When reaching the end of a chain (BlockArgument or Value without aliasing
304  /// OpOperands), also return the last Value of that chain.
305  ///
306  /// Example:
307  ///
308  /// 8
309  /// |
310  /// 6* 7* +-----+----+
311  /// | | | |
312  /// 2* 3 4* 5
313  /// | | | |
314  /// +----------+----------+----------+
315  /// |
316  /// 1
317  ///
318  /// In the above example, Values with a star satisfy the condition. When
319  /// starting the traversal from Value 1, the resulting SetVector is:
320  /// { 2, 7, 8, 5 }
321  SetVector<Value> findValueInReverseUseDefChain(
322  Value value, llvm::function_ref<bool(Value)> condition) const;
323 
324  /// Find the Values of the last preceding write of a given Value.
325  ///
326  /// Note: Unknown ops are handled conservatively and assumed to be writes.
327  /// Furthermore, BlockArguments are also assumed to be writes. There is no
328  /// analysis across block boundaries.
329  ///
330  /// Note: When reaching an end of the reverse SSA use-def chain, that value
331  /// is returned regardless of whether it is a memory write or not.
332  SetVector<Value> findLastPrecedingWrite(Value value) const;

418 /// BufferizationState provides helper functions for performing bufferization
419 /// rewrites and handling memref buffers.

423  BufferizationState(const AnalysisState &analysisState)

426  /// Creates a memref allocation for the given shaped value. `dealloc`
427  /// indicates whether the buffer should be deallocated or not. When `dealloc`
428  /// is `false`, this would create a memory leak, unless the buffer is
429  /// deallocated through some other mechanism.
430  ///
431  /// `dealloc` is optional. By default, this function will figure out by itself
432  /// if it is safe to deallocate the buffer. In essence, when returning the
433  /// buffer from a block, it is not safe to deallocate the buffer. This
434  /// information is queried via `AnalysisState::isTensorYielded`.
435  ///
436  /// Note: `shapedValue` is typically a tensor value. However, if it is a
437  /// memref value, `dealloc` is no longer optional and must be specified.
438  FailureOr<Value> createAlloc(OpBuilder &b, Location loc, Value shapedValue,
439  Optional<bool> dealloc = None);
440 
441  /// Return the buffer (memref) for a given OpOperand (tensor). Allocate
442  /// a new buffer and copy over data from the existing buffer if out-of-place
443  /// bufferization was decided.
444  ///
445  /// Whether a buffer is in-place or out-of-place is queried from the analysis
446  /// state. Some analyses may always conservatively opt for out-of-place
447  /// bufferization. Inplacability decisions can be overridden with the optional
448  /// `overrideInPlace` parameter.
449  FailureOr<Value>
450  getBuffer(RewriterBase &rewriter, OpOperand &opOperand,
451  Optional<ForceInPlacability> overrideInPlace = None,
452  Optional<Operation *> customCopyInsertionPoint = None);
453 
454  /// Return the buffer type for a given OpOperand (tensor) after bufferization.
455  BaseMemRefType getBufferType(OpOperand &opOperand) const;
456 
457  /// Return a reference to the BufferizationOptions.
458  const BufferizationOptions &getOptions() const {
459  return analysisState.getOptions();
460  }
461 
462  const AnalysisState &getAnalysisState() const { return analysisState; }
463 
464 protected:
465  // BufferizationState should be passed as a reference.
466  BufferizationState(const BufferizationState &) = delete;
467 
468 private:

470 };
