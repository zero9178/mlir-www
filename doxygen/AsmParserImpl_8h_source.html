271  /// Parse a keyword if it is one of the 'allowedKeywords'.
272  ParseResult
273  parseOptionalKeyword(StringRef *keyword,
274  ArrayRef<StringRef> allowedKeywords) override {
275  // Check that the current token is a keyword.
276  if (!isCurrentTokenAKeyword())
277  return failure();
278 
279  StringRef currentKeyword = parser.getTokenSpelling();
280  if (llvm::is_contained(allowedKeywords, currentKeyword)) {
281  *keyword = currentKeyword;
282  parser.consumeToken();
283  return success();
284  }
285 
286  return failure();
287  }

300  /// Parse a floating point value from the stream.
301  ParseResult parseFloat(double &result) override {
302  bool isNegative = parser.consumeIf(Token::minus);
303  Token curTok = parser.getToken();
304  SMLoc loc = curTok.getLoc();
305 
306  // Check for a floating point value.
307  if (curTok.is(Token::floatliteral)) {
308  auto val = curTok.getFloatingPointValue();
309  if (!val.hasValue())
310  return emitError(loc, "floating point value too large");
311  parser.consumeToken(Token::floatliteral);
312  result = isNegative ? -*val : *val;
313  return success();
314  }
315 
316  // Check for a hexadecimal float value.
317  if (curTok.is(Token::integer)) {
318  Optional<APFloat> apResult;
319  if (failed(parser.parseFloatFromIntegerLiteral(
320  apResult, curTok, isNegative, APFloat::IEEEdouble(),
321  /*typeSizeInBits=*/64)))
322  return failure();
323 
324  parser.consumeToken(Token::integer);
325  result = apResult->convertToDouble();
326  return success();
327  }
328 
329  return emitError(loc, "expected floating point literal");
330  }

421  /// Parse an optional @-identifier and store it (without the '@' symbol) in a
422  /// string attribute named 'attrName'.
423  ParseResult parseOptionalSymbolName(StringAttr &result, StringRef attrName,
424  NamedAttrList &attrs) override {
425  Token atToken = parser.getToken();
426  if (atToken.isNot(Token::at_identifier))
427  return failure();
428 
429  result = getBuilder().getStringAttr(atToken.getSymbolReference());
430  attrs.push_back(getBuilder().getNamedAttr(attrName, result));
431  parser.consumeToken();
432

434  // reference.
435  if (parser.getState().asmState) {
436  parser.getState().asmState->addUses(SymbolRefAttr::get(result),
437  atToken.getLocRange());
438  }
439  return success();
440  }
