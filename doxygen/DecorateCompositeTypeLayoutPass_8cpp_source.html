27 class SPIRVGlobalVariableOpLayoutInfoDecoration
28  : public OpRewritePattern<spirv::GlobalVariableOp> {
29 public:
30  using OpRewritePattern<spirv::GlobalVariableOp>::OpRewritePattern;
31 
32  LogicalResult matchAndRewrite(spirv::GlobalVariableOp op,
33  PatternRewriter &rewriter) const override {
34  SmallVector<NamedAttribute, 4> globalVarAttrs;
35 
36  auto ptrType = op.type().cast<spirv::PointerType>();
37  auto structType = VulkanLayoutUtils::decorateType(
38  ptrType.getPointeeType().cast<spirv::StructType>());
39 
40  if (!structType)
41  return failure();
42 
43  auto decoratedType =
44  spirv::PointerType::get(structType, ptrType.getStorageClass());
45 
46  // Save all named attributes except "type" attribute.
47  for (const auto &attr : op->getAttrs()) {
48  if (attr.getName() == "type")
49  continue;
50  globalVarAttrs.push_back(attr);
51  }
52 
53  rewriter.replaceOpWithNewOp<spirv::GlobalVariableOp>(
54  op, TypeAttr::get(decoratedType), globalVarAttrs);
55  return success();
56  }
57 };

108 void DecorateSPIRVCompositeTypeLayoutPass::runOnOperation() {
109  auto module = getOperation();
110  RewritePatternSet patterns(module.getContext());
111  populateSPIRVLayoutInfoPatterns(patterns);
112  ConversionTarget target(*(module.getContext()));
113  target.addLegalDialect<spirv::SPIRVDialect>();
114  target.addLegalOp<func::FuncOp>();
115  target.addDynamicallyLegalOp<spirv::GlobalVariableOp>(
116  [](spirv::GlobalVariableOp op) {
117  return VulkanLayoutUtils::isLegalType(op.type());
118  });
119 
120  // Change the type for the direct users.
121  target.addDynamicallyLegalOp<spirv::AddressOfOp>([](spirv::AddressOfOp op) {
122  return VulkanLayoutUtils::isLegalType(op.pointer().getType());
123  });
124 
125  // Change the type for the indirect users.
126  target.addDynamicallyLegalOp<spirv::AccessChainOp, spirv::LoadOp,
127  spirv::StoreOp>([&](Operation *op) {
128  for (Value operand : op->getOperands()) {
129  auto addrOp = operand.getDefiningOp<spirv::AddressOfOp>();
130  if (addrOp && !VulkanLayoutUtils::isLegalType(addrOp.pointer().getType()))
131  return false;
132  }
133  return true;
134  });
135 
136  FrozenRewritePatternSet frozenPatterns(std::move(patterns));
137  for (auto spirvModule : module.getOps<spirv::ModuleOp>())
138  if (failed(applyFullConversion(spirvModule, target, frozenPatterns)))
139  signalPassFailure();
140 }
