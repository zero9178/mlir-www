33 static mlir::Value applyPad(Location loc, Value input, ArrayRef<int64_t> pad,
34  Attribute padAttr, OpBuilder &rewriter) {
35  // Input should be padded if necessary.
36  if (llvm::all_of(pad, [](int64_t p) { return p == 0; }))
37  return input;
38 
39  ShapedType inputTy = input.getType().cast<ShapedType>();
40  Type inputETy = inputTy.getElementType();
41  auto inputShape = inputTy.getShape();
42 
43  assert((inputShape.size() * 2) == pad.size());
44 
45  SmallVector<int64_t, 4> paddedShape;
46  SmallVector<OpFoldResult, 8> lowIndices;
47  SmallVector<OpFoldResult, 8> highIndices;
48  for (int i = 0, s = inputShape.size(); i < s; i++) {
49  auto lowPad = pad[i * 2];
50  auto highPad = pad[i * 2 + 1];
51  if (ShapedType::isDynamic(inputShape[i]))
52  paddedShape.push_back(inputShape[i]);
53  else
54  paddedShape.push_back(inputShape[i] + highPad + lowPad);
55  lowIndices.push_back(rewriter.getIndexAttr(lowPad));
56  highIndices.push_back(rewriter.getIndexAttr(highPad));
57  }
58 
59  Value padValue = rewriter.create<arith::ConstantOp>(loc, padAttr);
60 
61  return tensor::createPadScalarOp(RankedTensorType::get(paddedShape, inputETy),
62  input, padValue, lowIndices, highIndices,
63  /*nofold=*/false, loc, rewriter)
64  .result();
65 }

67 static mlir::Value reifyConstantDim(Attribute attr,
68  ImplicitLocOpBuilder &builder) {
69  return builder.createOrFold<arith::IndexCastOp>(
70  builder.getIndexType(), builder.create<arith::ConstantOp>(attr));
71 }

73 // Calculating the output width/height using the formula:
74 // H = ((IH+pad_top+pad_bottom-(dilation_y*(KH-1)+1))/stride_y)+1
75 // W = ((IW+pad_left+pad_right-(dilation_x*(KW-1)+1))/stride_x)+1
76 static mlir::Value
77 getConvOutputDim(Location loc, Value initDim, Attribute padBeforeAttr,
78  Attribute padAfterAttr, Value kernelDim, Attribute strideAttr,
79  Attribute dilationAttr, Type inputETy, OpBuilder &rewriter) {
80  ImplicitLocOpBuilder builder(loc, rewriter);
81  auto one = rewriter.create<arith::ConstantOp>(
82  loc, IntegerAttr::get(initDim.getType(), 1));
83  Value padBefore = reifyConstantDim(padBeforeAttr, builder);
84  Value paddedBefore = builder.create<arith::AddIOp>(initDim, padBefore);
85  Value padAfter = reifyConstantDim(padAfterAttr, builder);
86  Value paddedAfter = builder.create<arith::AddIOp>(paddedBefore, padAfter);
87 
88  Value subOne = builder.create<arith::SubIOp>(kernelDim, one);
89  Value dilation = reifyConstantDim(dilationAttr, builder);
90  Value dilated = builder.create<arith::MulIOp>(dilation, subOne);
91  Value addOne = builder.create<arith::AddIOp>(dilated, one);
92 
93  Value subtract = builder.create<arith::SubIOp>(paddedAfter, addOne);
94  Value stride = reifyConstantDim(strideAttr, builder);
95  Value divide = builder.create<arith::DivUIOp>(subtract, stride);
96  return builder.create<arith::SubIOp>(divide, one);
97 }

99 // Creates a vector of the dynamic output dims for Conv2D and Depthwise_Conv2D
100 static SmallVector<Value> inferDynamicDimsForConv(
101  Location loc, Value input, Value weight, ShapedType resultTy,
102  ArrayAttr padAttr, ArrayAttr strideAttr, ArrayAttr dilationAttr,
103  int64_t weightHDim, int64_t weightWDim, OpBuilder &rewriter) {
104  ShapedType inputTy = input.getType().cast<ShapedType>();
105  Type inputETy = inputTy.getElementType();
106  int64_t inputRank = inputTy.getRank();
107  int64_t heightDim = 1;
108  int64_t weightDim = 2;
109 
110  SmallVector<Value> dynDims;
111  dynDims.resize(resultTy.getRank());
112  for (int i = 0; i < inputRank; i++) {
113  if (inputTy.isDynamicDim(i) && i != heightDim && i != weightDim)
114  dynDims[i] = rewriter.create<tensor::DimOp>(loc, input, i);
115  }
116 
117  // Dynamic input height
118  if (inputTy.isDynamicDim(heightDim)) {
119  Value initHDim =
120  rewriter.create<tensor::DimOp>(loc, input, heightDim).getResult();
121  Value kernelHDim =
122  rewriter.create<tensor::DimOp>(loc, weight, weightHDim).getResult();
123  // H = F(IH, pad_top, pad_bottom, dilation_y, KH, stride_y)
124  dynDims[heightDim] = getConvOutputDim(
125  loc, initHDim, padAttr.getValue()[0], padAttr.getValue()[1], kernelHDim,
126  strideAttr.getValue()[0], dilationAttr.getValue()[0], inputETy,
127  rewriter);
128  }
129 
130  // Dynamic input weight
131  if (inputTy.isDynamicDim(weightDim)) {
132  Value initWDim =
133  rewriter.create<tensor::DimOp>(loc, input, weightDim).getResult();
134  Value kernelWDim =
135  rewriter.create<tensor::DimOp>(loc, weight, weightWDim).getResult();
136  // W = F(IW, pad_left, pad_right, dilation_x, KW, stride_x)
137  dynDims[weightDim] = getConvOutputDim(
138  loc, initWDim, padAttr.getValue()[2], padAttr.getValue()[3], kernelWDim,
139  strideAttr.getValue()[1], dilationAttr.getValue()[1], inputETy,
140  rewriter);
141  }
142 
143  SmallVector<Value> filteredDims = condenseValues(dynDims);
144  return filteredDims;
145 }

149 class ConvConverter : public OpConversionPattern<tosa::Conv2DOp> {
150 public:
151  using OpConversionPattern<tosa::Conv2DOp>::OpConversionPattern;
152  LogicalResult
153  matchAndRewrite(tosa::Conv2DOp op, OpAdaptor adaptor,
154  ConversionPatternRewriter &rewriter) const final {
155  Location loc = op->getLoc();
156  Value input = op->getOperand(0);
157  Value weight = op->getOperand(1);
158  Value bias = op->getOperand(2);
159 
160  ShapedType inputTy = input.getType().cast<ShapedType>();
161  ShapedType weightTy = weight.getType().cast<ShapedType>();
162  ShapedType biasTy = bias.getType().cast<ShapedType>();
163  ShapedType resultTy = op->getResult(0).getType().cast<ShapedType>();
164 
165  Type inputETy = inputTy.getElementType();
166  Type resultETy = resultTy.getElementType();
167 
168  auto padAttr = op->getAttr("pad").cast<ArrayAttr>();
169  auto strideTosaAttr = op->getAttr("stride").cast<ArrayAttr>();
170  auto dilationTosaAttr = op->getAttr("dilation").cast<ArrayAttr>();
171  bool isQuantized = op->hasAttr("quantization_info");
172 
173  if (!weightTy.hasStaticShape() || !biasTy.hasStaticShape())
174  return rewriter.notifyMatchFailure(
175  op, "tosa.conv ops require static shapes for weight and bias");
176 
177  if (inputETy.isUnsignedInteger())
178  return rewriter.notifyMatchFailure(
179  op, "tosa.conv ops does not support unsigned integer input");
180 
181  SmallVector<Value> filteredDims = inferDynamicDimsForConv(
182  loc, input, weight, resultTy, padAttr, strideTosaAttr, dilationTosaAttr,
183  /*weightHDim=*/1, /*weightWDim=*/2, rewriter);
184 
185  auto weightShape = weightTy.getShape();
186 
187  // Apply padding as necessary.
188  Attribute zeroAttr = rewriter.getZeroAttr(inputETy);
189  if (isQuantized) {
190  auto quantizationInfo =
191  op->getAttr("quantization_info").cast<tosa::ConvOpQuantizationAttr>();
192  auto iZp = quantizationInfo.input_zp().getValue().getSExtValue();
193 
194  int64_t intMin =
195  APInt::getSignedMinValue(inputETy.getIntOrFloatBitWidth())
196  .getSExtValue();
197  int64_t intMax =
198  APInt::getSignedMaxValue(inputETy.getIntOrFloatBitWidth())
199  .getSExtValue();
200 
201  if (iZp < intMin || iZp > intMax)
202  return rewriter.notifyMatchFailure(
203  op, "tosa.conv op quantization has zp outside of input range");
204 
205  zeroAttr = rewriter.getIntegerAttr(inputETy, iZp);
206  }
207 
208  llvm::SmallVector<int64_t> pad;
209  pad.resize(2, 0);
210  getValuesFromIntArrayAttribute(padAttr, pad);
211  pad.resize(pad.size() + 2, 0);
212  input = applyPad(loc, input, pad, zeroAttr, rewriter);
213 
214  // Transpose the kernel to match dimension ordering of the linalg
215  // convolution operation.
216  // TODO(suderman): See if this can be efficiently folded - check whether
217  // the input is used anywhere else, if not fold the constant.
218  SmallVector<int64_t> weightPerm{1, 2, 3, 0};
219  SmallVector<int64_t> newWeightShape{weightShape[1], weightShape[2],
220  weightShape[3], weightShape[0]};
221  auto weightPermAttr = DenseIntElementsAttr::get(
222  RankedTensorType::get({4}, rewriter.getI64Type()), weightPerm);
223  Value weightPermValue =
224  rewriter.create<arith::ConstantOp>(loc, weightPermAttr);
225  Type newWeightTy =
226  RankedTensorType::get(newWeightShape, weightTy.getElementType());
227  weight = rewriter.create<tosa::TransposeOp>(loc, newWeightTy, weight,
228  weightPermValue);
229 
230  Attribute resultZeroAttr = rewriter.getZeroAttr(resultETy);
231  Value initTensor = rewriter.create<linalg::InitTensorOp>(
232  loc, filteredDims, resultTy.getShape(), resultETy);
233  Value zero = rewriter.create<arith::ConstantOp>(loc, resultZeroAttr);
234  Value zeroTensor = rewriter
235  .create<linalg::FillOp>(loc, ValueRange{zero},
236  ValueRange{initTensor})
237  .result();
238 
239  // Extract the attributes for convolution.
240  llvm::SmallVector<int64_t> stride, dilation;
241  getValuesFromIntArrayAttribute(strideTosaAttr, stride);
242  getValuesFromIntArrayAttribute(dilationTosaAttr, dilation);
243 
244  // Create the convolution op.
245  auto strideAttr = DenseIntElementsAttr::get(
246  RankedTensorType::get({2}, rewriter.getI64Type()), stride);
247  auto dilationAttr = DenseIntElementsAttr::get(
248  RankedTensorType::get({2}, rewriter.getI64Type()), dilation);
249 
250  // Create maps for the bias broadcasting
251  SmallVector<AffineMap, 4> indexingMaps;
252  indexingMaps.push_back(AffineMap::get(
253  /*dimCount=*/resultTy.getRank(), /*symbolCount=*/0,
254  {rewriter.getAffineDimExpr(3)}, rewriter.getContext()));
255  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(resultTy.getRank()));
256  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(resultTy.getRank()));
257 
258  Value biasInitTensor = rewriter.create<linalg::InitTensorOp>(
259  loc, filteredDims, resultTy.getShape(), resultETy);
260 
261  if (isQuantized) {
262  auto quantizationInfo =
263  op->getAttr("quantization_info").cast<tosa::ConvOpQuantizationAttr>();
264  auto iZp = rewriter.getI32IntegerAttr(
265  quantizationInfo.input_zp().getValue().getSExtValue());
266  auto kZp = rewriter.getI32IntegerAttr(
267  quantizationInfo.weight_zp().getValue().getSExtValue());
268 
269  auto iZpVal = rewriter.create<arith::ConstantOp>(loc, iZp);
270  auto kZpVal = rewriter.create<arith::ConstantOp>(loc, kZp);
271  Value conv =
272  rewriter
273  .create<linalg::Conv2DNhwcHwcfQOp>(
274  loc, resultTy, ValueRange{input, weight, iZpVal, kZpVal},
275  ValueRange{zeroTensor}, strideAttr, dilationAttr)
276  ->getResult(0);
277 
278  Value result =
279  rewriter
280  .create<linalg::GenericOp>(
281  loc, resultTy, ValueRange({bias, conv}), biasInitTensor,
282  indexingMaps, getNParallelLoopsAttrs(resultTy.getRank()),
283  [&](OpBuilder &nestedBuilder, Location nestedLoc,
284  ValueRange args) {
285  Value added = nestedBuilder.create<arith::AddIOp>(
286  loc, args[0], args[1]);
287  nestedBuilder.create<linalg::YieldOp>(nestedLoc, added);
288  })
289  .getResult(0);
290  rewriter.replaceOp(op, result);
291  return success();
292  }
293 
294  Value conv = rewriter
295  .create<linalg::Conv2DNhwcHwcfOp>(
296  loc, resultTy, ValueRange{input, weight},
297  ValueRange{zeroTensor}, strideAttr, dilationAttr)
298  ->getResult(0);
299 
300  Value result =
301  rewriter
302  .create<linalg::GenericOp>(
303  loc, resultTy, ValueRange({bias, conv}), biasInitTensor,
304  indexingMaps, getNParallelLoopsAttrs(resultTy.getRank()),
305  [&](OpBuilder &nestedBuilder, Location nestedLoc,
306  ValueRange args) {
307  Value added = nestedBuilder.create<arith::AddFOp>(
308  loc, args[0], args[1]);
309  nestedBuilder.create<linalg::YieldOp>(nestedLoc, added);
310  })
311  .getResult(0);
312 
313  rewriter.replaceOp(op, result);
314  return success();
315  }
316 };

320 public:

323  matchAndRewrite(tosa::DepthwiseConv2DOp op, OpAdaptor adaptor,
324  ConversionPatternRewriter &rewriter) const final {

326  Value input = op->getOperand(0);
327  Value weight = op->getOperand(1);

329

331  ShapedType weightTy = weight.getType().cast<ShapedType>();
332  ShapedType biasTy = bias.getType().cast<ShapedType>();

364  auto resultShape = resultTy.getShape();

366  // Apply padding as necessary.

368  if (isQuantized) {
369  auto quantizationInfo =
370  op->getAttr("quantization_info").cast<tosa::ConvOpQuantizationAttr>();
371  auto iZp = quantizationInfo.input_zp().getValue().getSExtValue();
372 
373  int64_t intMin =
374  APInt::getSignedMinValue(inputETy.getIntOrFloatBitWidth())
375  .getSExtValue();

377  APInt::getSignedMaxValue(inputETy.getIntOrFloatBitWidth())

386  }

388  llvm::SmallVector<int64_t> pad;
389  pad.resize(2, 0);
390  getValuesFromIntArrayAttribute(padAttr, pad);
391  pad.resize(pad.size() + 2, 0);

394 
395  // Extract the attributes for convolution.
396  llvm::SmallVector<int64_t> stride, dilation;
397  getValuesFromIntArrayAttribute(strideTosaAttr, stride);
398  getValuesFromIntArrayAttribute(dilationTosaAttr, dilation);

402  RankedTensorType::get({2}, rewriter.getI64Type()), stride);
403  auto dilationAttr = DenseIntElementsAttr::get(
404  RankedTensorType::get({2}, rewriter.getI64Type()), dilation);
405  ShapedType linalgConvTy =
406  RankedTensorType::get({resultShape[0], resultShape[1], resultShape[2],
407  weightShape[2], weightShape[3]},
408  resultETy);
409 
410  // Broadcast the initial value to the output tensor before convolving.
411  SmallVector<AffineMap, 4> indexingMaps;
412  indexingMaps.push_back(AffineMap::get(
413  /*dimCount=*/resultTy.getRank(), /*symbolCount=*/0,

423  .create<linalg::FillOp>(loc, ValueRange{zero},

449  .getResult(0);
450  rewriter.replaceOp(op, result);
451  } else {
452  auto iZpVal = rewriter.create<arith::ConstantOp>(loc, iZp);
453  auto kZpVal = rewriter.create<arith::ConstantOp>(loc, kZp);
454  Value conv =
455  rewriter
456  .create<linalg::DepthwiseConv2DNhwcHwcmQOp>(
457  loc, linalgConvTy, ValueRange{input, weight, iZpVal, kZpVal},
458  ValueRange{zeroTensor}, strideAttr, dilationAttr)
459  .getResult(0);

481 class MatMulConverter : public OpConversionPattern<tosa::MatMulOp> {
482 public:
483  using OpConversionPattern<tosa::MatMulOp>::OpConversionPattern;
484  LogicalResult
485  matchAndRewrite(tosa::MatMulOp op, OpAdaptor adaptor,
486  ConversionPatternRewriter &rewriter) const final {
487  Location loc = op.getLoc();
488 
489  auto outputTy = op.getType().cast<ShapedType>();
490  auto outputElementTy = outputTy.getElementType();
491 
492  auto firstOperandTy = op->getOperand(0).getType().cast<ShapedType>();
493  auto secondOperandTy = op->getOperand(1).getType().cast<ShapedType>();
494 
495  SmallVector<Value> dynDims;
496  dynDims.resize(op->getResult(0).getType().cast<ShapedType>().getRank());
497 
498  if (!firstOperandTy.hasRank() || firstOperandTy.isDynamicDim(0)) {
499  dynDims[0] = rewriter.create<tensor::DimOp>(loc, op->getOperand(0), 0);
500  }
501 
502  if (!firstOperandTy.hasRank() || firstOperandTy.isDynamicDim(1)) {
503  dynDims[1] = rewriter.create<tensor::DimOp>(loc, op->getOperand(0), 1);
504  }
505 
506  if (!secondOperandTy.hasRank() || secondOperandTy.isDynamicDim(2)) {
507  dynDims[2] = rewriter.create<tensor::DimOp>(loc, op->getOperand(1), 2);
508  }
509 
510  SmallVector<Value> filteredDims = condenseValues(dynDims);
511 
512  auto zeroAttr = rewriter.getZeroAttr(outputElementTy);
513  Value zero = rewriter.create<arith::ConstantOp>(loc, zeroAttr);
514  auto initTensor = rewriter.create<linalg::InitTensorOp>(
515  loc, filteredDims, outputTy.getShape(), outputTy.getElementType());
516  Value zeroTensor = rewriter
517  .create<linalg::FillOp>(loc, ValueRange{zero},
518  ValueRange{initTensor})
519  .result();
520  if (!op.quantization_info()) {
521  rewriter.replaceOpWithNewOp<linalg::BatchMatmulOp>(
522  op, TypeRange{op.getType()}, ValueRange{adaptor.a(), adaptor.b()},
523  ValueRange{zeroTensor});
524  return success();
525  }
526 
527  auto quantizationInfo = op.quantization_info().getValue();
528  auto aZp = rewriter.create<arith::ConstantOp>(
529  loc, rewriter.getI32IntegerAttr(
530  quantizationInfo.a_zp().getValue().getSExtValue()));
531  auto bZp = rewriter.create<arith::ConstantOp>(
532  loc, rewriter.getI32IntegerAttr(
533  quantizationInfo.b_zp().getValue().getSExtValue()));
534  rewriter.replaceOpWithNewOp<linalg::QuantizedBatchMatmulOp>(
535  op, TypeRange{op.getType()},
536  ValueRange{adaptor.a(), adaptor.b(), aZp, bZp}, zeroTensor);
537 
538  return success();
539  }

544 public:

547  matchAndRewrite(tosa::FullyConnectedOp op, OpAdaptor adaptor,
548  ConversionPatternRewriter &rewriter) const final {
549  Location loc = op.getLoc();
550  auto outputTy = op.getType().cast<ShapedType>();
551  auto input = op.input();
552  auto inputTy = input.getType().cast<ShapedType>();
553 
554  auto bias = op.bias();
555 
556  auto weight = op.weight();
557  auto weightTy = weight.getType().cast<ShapedType>();
558  auto weightShape = weightTy.getShape();
559 
560  auto outputETy = outputTy.getElementType();
561

564 
565  if (!inputTy.hasRank() || inputTy.isDynamicDim(0)) {
566  dynDims[0] = rewriter.create<tensor::DimOp>(loc, input, 0);
567  }
568 
569  if (!weightTy.hasRank() || weightTy.isDynamicDim(0)) {
570  dynDims[1] = rewriter.create<tensor::DimOp>(loc, weight, 0);
571  }
572 
573  SmallVector<Value> filteredDims = condenseValues(dynDims);
574 
575  // Creating maps for the output of MatMul and the bias
576  SmallVector<AffineMap, 4> indexingMaps;
577 
578  // Broadcast the bias.
579  indexingMaps.push_back(AffineMap::get(/*dimCount=*/2, /*symbolCount=*/0,
580  {rewriter.getAffineDimExpr(1)},
581  rewriter.getContext()));
582 
583  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(outputTy.getRank()));
584  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(outputTy.getRank()));
585 
586  auto initTensor = rewriter.create<linalg::InitTensorOp>(
587  loc, filteredDims, outputTy.getShape(), outputTy.getElementType());
588 
589  // When quantized, the input elemeny type is not the same as the output
590  Attribute resultZeroAttr = rewriter.getZeroAttr(outputETy);
591  Value zero = rewriter.create<arith::ConstantOp>(loc, resultZeroAttr);
592  Value zeroTensor = rewriter
593  .create<linalg::FillOp>(loc, ValueRange{zero},
594  ValueRange{initTensor})
595  .result();
596 
597  SmallVector<int64_t> permutation{1, 0};
598  auto permutationAttr = DenseIntElementsAttr::get(
599  RankedTensorType::get({2}, rewriter.getI64Type()), permutation);
600  Value permutationValue =
601  rewriter.create<arith::ConstantOp>(loc, permutationAttr);
602 
603  SmallVector<int64_t> newWeightShape{weightShape[1], weightShape[0]};
604  Type newWeightTy =
605  RankedTensorType::get(newWeightShape, weightTy.getElementType());
606 
607  Value transposedWeight = rewriter.create<tosa::TransposeOp>(
608  loc, newWeightTy, weight, permutationValue);
609 
610  auto biasInitTensor =
611  rewriter
612  .create<linalg::InitTensorOp>(loc, filteredDims,
613  outputTy.getShape(), outputETy)
614  ->getResults();
615 
616  if (!op.quantization_info()) {
617  Value matmul = rewriter
618  .create<linalg::MatmulOp>(
619  loc, TypeRange{op.getType()},
620  ValueRange{input, transposedWeight}, zeroTensor)
621  ->getResult(0);
622 
623  Value result =
624  rewriter
625  .create<linalg::GenericOp>(
626  loc, outputTy, ValueRange({bias, matmul}), biasInitTensor,
627  indexingMaps, getNParallelLoopsAttrs(outputTy.getRank()),
628  [&](OpBuilder &nestedBuilder, Location nestedLoc,
629  ValueRange args) {
630  Value added = nestedBuilder.create<arith::AddFOp>(
631  loc, args[0], args[1]);
632  nestedBuilder.create<linalg::YieldOp>(nestedLoc, added);
633  })
634  .getResult(0);
635  rewriter.replaceOp(op, result);
636  return success();
637  }
638 
639  auto quantizationInfo = op.quantization_info().getValue();
640  auto inputZp = rewriter.create<arith::ConstantOp>(
641  loc, rewriter.getI32IntegerAttr(
642  quantizationInfo.input_zp().getValue().getSExtValue()));
643  auto outputZp = rewriter.create<arith::ConstantOp>(
644  loc, rewriter.getI32IntegerAttr(
645  quantizationInfo.weight_zp().getValue().getSExtValue()));
646  Value matmul =
647  rewriter
648  .create<linalg::QuantizedMatmulOp>(
649  loc, TypeRange{op.getType()},
650  ValueRange{input, transposedWeight, inputZp, outputZp},
651  zeroTensor)
652  ->getResult(0);
653  Value result =
654  rewriter
655  .create<linalg::GenericOp>(
656  loc, outputTy, ValueRange({bias, matmul}), biasInitTensor,
657  indexingMaps, getNParallelLoopsAttrs(outputTy.getRank()),
658  [&](OpBuilder &nestedBuilder, Location nestedLoc,
659  ValueRange args) {
660  Value added = nestedBuilder.create<arith::AddIOp>(
661  loc, args[0], args[1]);
662  nestedBuilder.create<linalg::YieldOp>(nestedLoc, added);
663  })
664  .getResult(0);
665  rewriter.replaceOp(op, result);
666  return success();

742 class AvgPool2dConverter : public OpRewritePattern<tosa::AvgPool2dOp> {
743 public:
744  using OpRewritePattern<tosa::AvgPool2dOp>::OpRewritePattern;
745 
746  LogicalResult matchAndRewrite(tosa::AvgPool2dOp op,
747  PatternRewriter &rewriter) const final {
748  Location loc = op.getLoc();
749  Value input = op.input();
750  ShapedType inputTy = input.getType().cast<ShapedType>();
751  Type inElementTy = inputTy.getElementType();
752 
753  ShapedType resultTy = op.getType().template cast<ShapedType>();
754  Type resultETy = op.getType().cast<ShapedType>().getElementType();
755 
756  Type accETy =
757  inElementTy.isa<IntegerType>() ? rewriter.getI32Type() : inElementTy;
758  ShapedType accTy = resultTy.clone(accETy);
759 
760  auto dynamicDimsOr =
761  checkHasDynamicBatchDims(rewriter, op, {input, op.output()});
762  if (!dynamicDimsOr.hasValue())
763  return failure();
764  SmallVector<Value> dynamicDims = dynamicDimsOr.getValue();
765 
766  // Apply padding as necessary.
767  llvm::SmallVector<int64_t> pad;
768  pad.resize(2, 0);
769  getValuesFromIntArrayAttribute(op.pad(), pad);
770  pad.resize(pad.size() + 2, 0);
771  Attribute padAttr = rewriter.getZeroAttr(inElementTy);
772  Value paddedInput = applyPad(loc, input, pad, padAttr, rewriter);
773 
774  Attribute initialAttr = rewriter.getZeroAttr(accETy);
775  Value initialValue = rewriter.create<arith::ConstantOp>(loc, initialAttr);
776 
777  SmallVector<int64_t> kernel, stride;
778  getValuesFromIntArrayAttribute(op.kernel(), kernel);
779  getValuesFromIntArrayAttribute(op.stride(), stride);
780 
781  Attribute strideAttr = rewriter.getI64VectorAttr(stride);
782  Attribute dilationAttr = rewriter.getI64VectorAttr({1, 1});
783 
784  // Create the linalg op that performs pooling.
785  Value poolInitTensor = rewriter.create<linalg::InitTensorOp>(
786  loc, dynamicDims, accTy.getShape(), accETy);
787 
788  Value filledInitTensor =
789  rewriter
790  .create<linalg::FillOp>(loc, ValueRange{initialValue},
791  ValueRange{poolInitTensor})
792  .result();
793 
794  Value fakeWindowDims =
795  rewriter.create<linalg::InitTensorOp>(loc, kernel, accETy);
796 
797  // Sum across the pooled region.
798  Value poolingOp = rewriter
799  .create<linalg::PoolingNhwcSumOp>(
800  loc, ArrayRef<Type>{accTy},
801  ValueRange{paddedInput, fakeWindowDims},
802  filledInitTensor, strideAttr, dilationAttr)
803  .getResult(0);
804 
805  // Normalize the summed value by the number of elements grouped in each
806  // pool.
807  auto poolingOpTy = poolingOp.getType().cast<ShapedType>();
808  auto affineMap = rewriter.getMultiDimIdentityMap(resultTy.getRank());
809 
810  Value genericInitTensor = rewriter.create<linalg::InitTensorOp>(
811  loc, dynamicDims, resultTy.getShape(), resultETy);
812 
813  auto genericOp = rewriter.create<linalg::GenericOp>(
814  loc, ArrayRef<Type>({resultTy}), ValueRange{poolingOp},
815  ValueRange{genericInitTensor},
816  ArrayRef<AffineMap>({affineMap, affineMap}),
817  getNParallelLoopsAttrs(resultTy.getRank()),
818  [&](OpBuilder &b, Location loc, ValueRange args) {
819  auto zero = rewriter.create<arith::ConstantIndexOp>(loc, 0);
820  auto one = rewriter.create<arith::ConstantIndexOp>(loc, 1);
821  auto iH = rewriter.create<arith::ConstantIndexOp>(
822  loc, poolingOpTy.getDimSize(1) - 1);
823  auto iW = rewriter.create<arith::ConstantIndexOp>(
824  loc, poolingOpTy.getDimSize(2) - 1);
825 
826  // Compute the indices from either end.
827  auto y0 = rewriter.create<linalg::IndexOp>(loc, 1);
828  auto x0 = rewriter.create<linalg::IndexOp>(loc, 2);
829  auto y1 = rewriter.create<arith::SubIOp>(loc, iH, y0);
830  auto x1 = rewriter.create<arith::SubIOp>(loc, iW, x0);
831 
832  // Determines what the portion of valid input is covered by the
833  // kernel.
834  auto padFn = [&](Value v, Value x, int64_t pad) -> Value {
835  if (pad == 0)
836  return v;
837 
838  auto padVal = rewriter.create<arith::ConstantIndexOp>(loc, pad);
839  Value dx = rewriter.create<arith::SubIOp>(loc, x, padVal);
840 
841  Value cmp = rewriter.create<arith::CmpIOp>(
842  loc, arith::CmpIPredicate::slt, dx, zero);
843  Value offset = rewriter.create<arith::SelectOp>(loc, cmp, dx, zero);
844  return rewriter.create<arith::AddIOp>(loc, v, offset)->getResult(0);
845  };
846 
847  // Compute the vertical component of coverage.
848  auto kH0 = rewriter.create<arith::ConstantIndexOp>(loc, kernel[0]);
849  auto kH1 = padFn(kH0, y0, pad[2]);
850  auto kH2 = padFn(kH1, y1, pad[3]);
851  auto kHCmp = rewriter.create<arith::CmpIOp>(
852  loc, arith::CmpIPredicate::slt, kH2, one);
853  auto kH3 = rewriter.create<arith::SelectOp>(loc, kHCmp, one, kH2);
854 
855  // compute the horizontal component of coverage.
856  auto kW0 = rewriter.create<arith::ConstantIndexOp>(loc, kernel[1]);
857  auto kW1 = padFn(kW0, x0, pad[4]);
858  auto kW2 = padFn(kW1, x1, pad[5]);
859  auto kWCmp = rewriter.create<arith::CmpIOp>(
860  loc, arith::CmpIPredicate::slt, kW2, one);
861  auto kW3 = rewriter.create<arith::SelectOp>(loc, kWCmp, one, kW2);
862 
863  // Compute the total number of elements and normalize.
864  Value count = rewriter.create<arith::MulIOp>(loc, kH3, kW3);
865  auto countI = rewriter.create<arith::IndexCastOp>(
866  loc, rewriter.getI32Type(), count);
867 
868  // Divide by the number of summed values. For floats this is just
869  // a div however for quantized values input normalization had
870  // to be applied.
871  Value poolVal = args[0];
872  if (accETy.isa<FloatType>()) {
873  auto countF = rewriter.create<arith::SIToFPOp>(loc, accETy, countI);
874  poolVal = rewriter.create<arith::DivFOp>(loc, poolVal, countF)
875  ->getResult(0);
876  } else {
877 
878  // If we have quantization information we need to apply an offset
879  // for the input zp value.
880  if (op.quantization_info()) {
881  auto quantizationInfo = op.quantization_info().getValue();
882  auto inputZp = rewriter.create<arith::ConstantOp>(
883  loc, quantizationInfo.input_zp());
884  Value offset =
885  rewriter.create<arith::MulIOp>(loc, accETy, countI, inputZp);
886  poolVal =
887  rewriter.create<arith::SubIOp>(loc, accETy, poolVal, offset);
888  }
889 
890  // Compute the multiplier and shift values for the quantization
891  // normalization. Preferably we would want to compute more bits
892  // however 32-bits should be enough for compute. Honestly we
893  // should probably straight divide.
894  int64_t numerator = ((1 << 30) + 1);
895  int64_t shift = 30;
896 
897  Value numeratorVal = rewriter.create<arith::ConstantOp>(
898  loc, rewriter.getI32IntegerAttr(numerator));
899  Value multiplierVal =
900  rewriter
901  .create<arith::DivUIOp>(loc, rewriter.getI32Type(),
902  numeratorVal, countI)
903  .getResult();
904  Value shiftVal = rewriter.create<arith::ConstantOp>(
905  loc, rewriter.getI8IntegerAttr(shift));
906 
907  auto scaled =
908  rewriter
909  .create<tosa::ApplyScaleOp>(
910  loc, rewriter.getI32Type(), poolVal, multiplierVal,
911  shiftVal, rewriter.getBoolAttr(false))
912  .getResult();
913 
914  // If we have quantization information we need to apply output
915  // zeropoint.
916  if (op.quantization_info()) {
917  auto quantizationInfo = op.quantization_info().getValue();
918  auto outputZp = rewriter.create<arith::ConstantOp>(
919  loc, quantizationInfo.output_zp());
920  scaled = rewriter.create<arith::AddIOp>(loc, scaled, outputZp)
921  .getResult();
922  }
923 
924  // Apply Clip.
925  int64_t outBitwidth = resultETy.getIntOrFloatBitWidth();
926 
927  auto min = rewriter.create<arith::ConstantIntOp>(
928  loc, APInt::getSignedMinValue(outBitwidth).getSExtValue(),
929  accETy);
930  auto max = rewriter.create<arith::ConstantIntOp>(
931  loc, APInt::getSignedMaxValue(outBitwidth).getSExtValue(),
932  accETy);
933  auto clamp = clampHelper<arith::CmpIOp>(
934  loc, scaled, min, max, arith::CmpIPredicate::slt, rewriter);
935 
936  poolVal = clamp;
937  // Convert type.
938  if (resultETy != clamp.getType()) {
939  poolVal =
940  rewriter.create<arith::TruncIOp>(loc, resultETy, poolVal);
941  }
942  }
943 
944  rewriter.create<linalg::YieldOp>(loc, poolVal);
945  });
946 
947  rewriter.replaceOp(op, genericOp.getResult(0));
948  return success();
949  }
950 };
