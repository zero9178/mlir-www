26 /// This is a class to represent a resizable matrix.
27 ///
28 /// More columns and rows can be reserved than are currently used. The data is
29 /// stored as a single 1D array, viewed as a 2D matrix with nRows rows and
30 /// nReservedColumns columns, stored in row major form. Thus the element at
31 /// (i, j) is stored at data[i*nReservedColumns + j]. The reserved but unused
32 /// columns always have all zero values. The reserved rows are just reserved
33 /// space in the underlying SmallVector's capacity.
34 class Matrix {
35 public:
36  Matrix() = delete;
37 
38  /// Construct a matrix with the specified number of rows and columns.
39  /// The number of reserved rows and columns will be at least the number
40  /// specified, and will always be sufficient to accomodate the number of rows
41  /// and columns specified.
42  ///
43  /// Initially, the entries are initialized to ero.
44  Matrix(unsigned rows, unsigned columns, unsigned reservedRows = 0,
45  unsigned reservedColumns = 0);
46 
47  /// Return the identity matrix of the specified dimension.
48  static Matrix identity(unsigned dimension);

51  int64_t &at(unsigned row, unsigned column);
52  int64_t at(unsigned row, unsigned column) const;
53  int64_t &operator()(unsigned row, unsigned column);
54  int64_t operator()(unsigned row, unsigned column) const;
55 
56  /// Swap the given columns.
57  void swapColumns(unsigned column, unsigned otherColumn);
58 
59  /// Swap the given rows.
60  void swapRows(unsigned row, unsigned otherRow);

62  unsigned getNumRows() const;
63 
64  unsigned getNumColumns() const;
65

67  /// incurring a reallocation.
68  unsigned getNumReservedRows() const;
69  unsigned getNumReservedColumns() const;

72  /// reallocations.
73  void reserveRows(unsigned rows);

79  /// Insert columns having positions pos, pos + 1, ... pos + count - 1.
80  /// Columns that were at positions 0 to pos - 1 will stay where they are;
81  /// columns that were at positions pos to nColumns - 1 will be pushed to the
82  /// right. pos should be at most nColumns.
83  void insertColumns(unsigned pos, unsigned count);
84  void insertColumn(unsigned pos);
85 
86  /// Insert rows having positions pos, pos + 1, ... pos + count - 1.
87  /// Rows that were at positions 0 to pos - 1 will stay where they are;
88  /// rows that were at positions pos to nColumns - 1 will be pushed to the
89  /// right. pos should be at most nRows.
90  void insertRows(unsigned pos, unsigned count);
91  void insertRow(unsigned pos);
92 
93  /// Remove the columns having positions pos, pos + 1, ... pos + count - 1.
94  /// Rows that were at positions 0 to pos - 1 will stay where they are;
95  /// columns that were at positions pos + count - 1 or later will be pushed to
96  /// the right. The columns to be deleted must be valid rows: pos + count - 1
97  /// must be at most nColumns - 1.
98  void removeColumns(unsigned pos, unsigned count);
99  void removeColumn(unsigned pos);
100 
101  /// Remove the rows having positions pos, pos + 1, ... pos + count - 1.
102  /// Rows that were at positions 0 to pos - 1 will stay where they are;
103  /// rows that were at positions pos + count - 1 or later will be pushed to the
104  /// right. The rows to be deleted must be valid rows: pos + count - 1 must be
105  /// at most nRows - 1.
106  void removeRows(unsigned pos, unsigned count);
107  void removeRow(unsigned pos);

141  /// Resize the matrix to the specified dimensions. If a dimension is smaller,
142  /// the values are truncated; if it is bigger, the new values are initialized
143  /// to zero.
144  ///
145  /// Due to the representation of the matrix, resizing vertically (adding rows)
146  /// is less expensive than increasing the number of columns beyond

157

160  void dump() const;
161 
162  /// Return whether the Matrix is in a consistent state with all its
163  /// invariants satisfied.
164  bool hasConsistentState() const;
165 
166 private:
167  /// The current number of rows, columns, and reserved columns. The underlying
168  /// data vector is viewed as an nRows x nReservedColumns matrix, of which the
169  /// first nColumns columns are currently in use, and the remaining are
170  /// reserved columns filled with zeros.
171  unsigned nRows, nColumns, nReservedColumns;
172 
173  /// Stores the data. data.size() is equal to nRows * nReservedColumns.
174  /// data.capacity() / nReservedColumns is the number of reserved rows.
175  SmallVector<int64_t, 64> data;
176 };
