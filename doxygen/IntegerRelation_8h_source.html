27 /// An IntegerRelation represents the set of points from a PresburgerSpace that
28 /// satisfy a list of affine constraints. Affine constraints can be inequalities
29 /// or equalities in the form:
30 ///
31 /// Inequality: c_0*x_0 + c_1*x_1 + .... + c_{n-1}*x_{n-1} + c_n >= 0
32 /// Equality : c_0*x_0 + c_1*x_1 + .... + c_{n-1}*x_{n-1} + c_n == 0
33 ///
34 /// where c_0, c_1, ..., c_n are integers and n is the total number of
35 /// identifiers in the space.
36 ///
37 /// Such a relation corresponds to the set of integer points lying in a convex
38 /// polyhedron. For example, consider the relation:
39 /// (x) -> (y) : (1 <= x <= 7, x = 2y)
40 /// These can be thought of as points in the polyhedron:
41 /// (x, y) : (1 <= x <= 7, x = 2y)
42 /// This relation contains the pairs (2, 1), (4, 2), and (6, 3).
43 ///
44 /// Since IntegerRelation makes a distinction between dimensions, IdKind::Range
45 /// and IdKind::Domain should be used to refer to dimension identifiers.
46 class IntegerRelation {
47 public:
48  /// All derived classes of IntegerRelation.
49  enum class Kind {
50  FlatAffineConstraints,
51  FlatAffineValueConstraints,
52  IntegerRelation,
53  IntegerPolyhedron,
54  };

610 /// An IntegerPolyhedron represents the set of points from a PresburgerSpace
611 /// that satisfy a list of affine constraints. Affine constraints can be
612 /// inequalities or equalities in the form:
613 ///
614 /// Inequality: c_0*x_0 + c_1*x_1 + .... + c_{n-1}*x_{n-1} + c_n >= 0
615 /// Equality : c_0*x_0 + c_1*x_1 + .... + c_{n-1}*x_{n-1} + c_n == 0
616 ///
617 /// where c_0, c_1, ..., c_n are integers and n is the total number of
618 /// identifiers in the space.
619 ///
620 /// An IntegerPolyhedron is similar to an IntegerRelation but it does not make a
621 /// distinction between Domain and Range identifiers. Internally,
622 /// IntegerPolyhedron is implemented as a IntegerRelation with zero domain ids.
623 ///
624 /// Since IntegerPolyhedron does not make a distinction between kinds of
625 /// dimensions, IdKind::SetDim should be used to refer to dimension identifiers.
626 class IntegerPolyhedron : public IntegerRelation {
627 public:
628  /// Constructs a set reserving memory for the specified number
629  /// of constraints and identifiers.
630  IntegerPolyhedron(unsigned numReservedInequalities,
631  unsigned numReservedEqualities, unsigned numReservedCols,
632  const PresburgerSpace &space)
633  : IntegerRelation(numReservedInequalities, numReservedEqualities,
634  numReservedCols, space) {
635  assert(space.getNumDomainIds() == 0 &&
636  "Number of domain id's should be zero in Set kind space.");
637  }
638 
639  /// Constructs a relation with the specified number of dimensions and
640  /// symbols.
641  explicit IntegerPolyhedron(const PresburgerSpace &space)
642  : IntegerPolyhedron(/*numReservedInequalities=*/0,
643  /*numReservedEqualities=*/0,
644  /*numReservedCols=*/space.getNumIds() + 1, space) {}
645 
646  /// Return a system with no constraints, i.e., one which is satisfied by all
647  /// points.
648  static IntegerPolyhedron getUniverse(const PresburgerSpace &space) {
649  return IntegerPolyhedron(space);
650  }
651 
652  /// Return the kind of this IntegerRelation.
653  Kind getKind() const override { return Kind::IntegerPolyhedron; }
654 
655  static bool classof(const IntegerRelation *cst) {
656  return cst->getKind() == Kind::IntegerPolyhedron;
657  }
658 
659  // Clones this object.
660  std::unique_ptr<IntegerPolyhedron> clone() const;
661 
662  /// Insert `num` identifiers of the specified kind at position `pos`.
663  /// Positions are relative to the kind of identifier. Return the absolute
664  /// column position (i.e., not relative to the kind of identifier) of the
665  /// first added identifier.
666  unsigned insertId(IdKind kind, unsigned pos, unsigned num = 1) override;
667 
668  /// Compute the symbolic integer lexmin of the polyhedron.
669  /// This finds, for every assignment to the symbols, the lexicographically
670  /// minimum value attained by the dimensions. For example, the symbolic lexmin
671  /// of the set
672  ///
673  /// (x, y)[a, b, c] : (a <= x, b <= x, x <= c)
674  ///
675  /// can be written as
676  ///
677  /// x = a if b <= a, a <= c
678  /// x = b if a < b, b <= c
679  ///
680  /// This function is stored in the `lexmin` function in the result.
681  /// Some assignments to the symbols might make the set empty.
682  /// Such points are not part of the function's domain.
683  /// In the above example, this happens when max(a, b) > c.
684  ///
685  /// For some values of the symbols, the lexmin may be unbounded.
686  /// `SymbolicLexMin` stores these parts of the symbolic domain in a separate
687  /// `PresburgerSet`, `unboundedDomain`.
688  SymbolicLexMin findSymbolicIntegerLexMin() const;
