27 /// This class provides support for indexing into the element range of an
28 /// ElementsAttr. It is used to opaquely wrap either a contiguous range, via
29 /// `ElementsAttrIndexer::contiguous`, or a non-contiguous range, via
30 /// `ElementsAttrIndexer::nonContiguous`, A contiguous range is an array-like
31 /// range, where all of the elements are layed out sequentially in memory. A
32 /// non-contiguous range implies no contiguity, and elements may even be
33 /// materialized when indexing, such as the case for a mapped_range.
34 struct ElementsAttrIndexer {
35 public:
36  ElementsAttrIndexer()
37  : ElementsAttrIndexer(/*isContiguous=*/true, /*isSplat=*/true) {}
38  ElementsAttrIndexer(ElementsAttrIndexer &&rhs)
39  : isContiguous(rhs.isContiguous), isSplat(rhs.isSplat) {
40  if (isContiguous)
41  conState = rhs.conState;
42  else
43  new (&nonConState) NonContiguousState(std::move(rhs.nonConState));
44  }
45  ElementsAttrIndexer(const ElementsAttrIndexer &rhs)
46  : isContiguous(rhs.isContiguous), isSplat(rhs.isSplat) {
47  if (isContiguous)
48  conState = rhs.conState;
49  else
50  new (&nonConState) NonContiguousState(rhs.nonConState);
51  }
52  ~ElementsAttrIndexer() {
53  if (!isContiguous)
54  nonConState.~NonContiguousState();
55  }
56 
57  /// Construct an indexer for a non-contiguous range starting at the given
58  /// iterator. A non-contiguous range implies no contiguity, and elements may
59  /// even be materialized when indexing, such as the case for a mapped_range.
60  template <typename IteratorT>
61  static ElementsAttrIndexer nonContiguous(bool isSplat, IteratorT &&iterator) {
62  ElementsAttrIndexer indexer(/*isContiguous=*/false, isSplat);
63  new (&indexer.nonConState)
64  NonContiguousState(std::forward<IteratorT>(iterator));
65  return indexer;
66  }
67 
68  // Construct an indexer for a contiguous range starting at the given element
69  // pointer. A contiguous range is an array-like range, where all of the
70  // elements are layed out sequentially in memory.
71  template <typename T>
72  static ElementsAttrIndexer contiguous(bool isSplat, const T *firstEltPtr) {
73  ElementsAttrIndexer indexer(/*isContiguous=*/true, isSplat);
74  new (&indexer.conState) ContiguousState(firstEltPtr);
75  return indexer;
76  }
77 
78  /// Access the element at the given index.
79  template <typename T> T at(uint64_t index) const {
80  if (isSplat)
81  index = 0;
82  return isContiguous ? conState.at<T>(index) : nonConState.at<T>(index);
83  }
84 
85 private:
86  ElementsAttrIndexer(bool isContiguous, bool isSplat)
87  : isContiguous(isContiguous), isSplat(isSplat), conState(nullptr) {}

89  /// This class contains all of the state necessary to index a contiguous
90  /// range.
91  class ContiguousState {
92  public:
93  ContiguousState(const void *firstEltPtr) : firstEltPtr(firstEltPtr) {}
94 
95  /// Access the element at the given index.
96  template <typename T> const T &at(uint64_t index) const {
97  return *(reinterpret_cast<const T *>(firstEltPtr) + index);
98  }
99 
100  private:
101  const void *firstEltPtr;
102  };

104  /// This class contains all of the state necessary to index a non-contiguous
105  /// range.
106  class NonContiguousState {
107  private:
108  /// This class is used to represent the abstract base of an opaque iterator.
109  /// This allows for all iterator and element types to be completely
110  /// type-erased.
111  struct OpaqueIteratorBase {
112  virtual ~OpaqueIteratorBase() = default;
113  virtual std::unique_ptr<OpaqueIteratorBase> clone() const = 0;
114  };
115  /// This class is used to represent the abstract base of an opaque iterator
116  /// that iterates over elements of type `T`. This allows for all iterator
117  /// types to be completely type-erased.
118  template <typename T>
119  struct OpaqueIteratorValueBase : public OpaqueIteratorBase {
120  virtual T at(uint64_t index) = 0;
121  };
122  /// This class is used to represent an opaque handle to an iterator of type
123  /// `IteratorT` that iterates over elements of type `T`.
124  template <typename IteratorT, typename T>
125  struct OpaqueIterator : public OpaqueIteratorValueBase<T> {
126  template <typename ItTy, typename FuncTy, typename FuncReturnTy>
127  static void isMappedIteratorTestFn(
128  llvm::mapped_iterator<ItTy, FuncTy, FuncReturnTy>) {}
129  template <typename U, typename... Args>
130  using is_mapped_iterator =
131  decltype(isMappedIteratorTestFn(std::declval<U>()));
132  template <typename U>
133  using detect_is_mapped_iterator =
134  llvm::is_detected<is_mapped_iterator, U>;
135 
136  /// Access the element within the iterator at the given index.
137  template <typename ItT>
138  static std::enable_if_t<!detect_is_mapped_iterator<ItT>::value, T>
139  atImpl(ItT &&it, uint64_t index) {
140  return *std::next(it, index);
141  }
142  template <typename ItT>
143  static std::enable_if_t<detect_is_mapped_iterator<ItT>::value, T>
144  atImpl(ItT &&it, uint64_t index) {
145  // Special case mapped_iterator to avoid copying the function.
146  return it.getFunction()(*std::next(it.getCurrent(), index));
147  }
148 
149  public:
150  template <typename U>
151  OpaqueIterator(U &&iterator) : iterator(std::forward<U>(iterator)) {}
152  std::unique_ptr<OpaqueIteratorBase> clone() const final {
153  return std::make_unique<OpaqueIterator<IteratorT, T>>(iterator);
154  }
155 
156  /// Access the element at the given index.
157  T at(uint64_t index) final { return atImpl(iterator, index); }
158 
159  private:
160  IteratorT iterator;
161  };
162 
163  public:
164  /// Construct the state with the given iterator type.
165  template <typename IteratorT, typename T = typename llvm::remove_cvref_t<
166  decltype(*std::declval<IteratorT>())>>
167  NonContiguousState(IteratorT iterator)
168  : iterator(std::make_unique<OpaqueIterator<IteratorT, T>>(iterator)) {}
169  NonContiguousState(const NonContiguousState &other)
170  : iterator(other.iterator->clone()) {}
171  NonContiguousState(NonContiguousState &&other) = default;
172 
173  /// Access the element at the given index.
174  template <typename T> T at(uint64_t index) const {
175  auto *valueIt = static_cast<OpaqueIteratorValueBase<T> *>(iterator.get());
176  return valueIt->at(index);
177  }
178 
179  /// The opaque iterator state.
180  std::unique_ptr<OpaqueIteratorBase> iterator;
181  };
182 
183  /// A boolean indicating if this range is contiguous or not.
184  bool isContiguous;
185  /// A boolean indicating if this range is a splat.
186  bool isSplat;
187  /// The underlying range state.
188  union {
189  ContiguousState conState;
190  NonContiguousState nonConState;
191  };
192 };

194 /// This class implements a generic iterator for ElementsAttr.
195 template <typename T>
196 class ElementsAttrIterator
197  : public llvm::iterator_facade_base<ElementsAttrIterator<T>,
198  std::random_access_iterator_tag, T,
199  std::ptrdiff_t, T, T> {
200 public:
201  ElementsAttrIterator(ElementsAttrIndexer indexer, size_t dataIndex)
202  : indexer(std::move(indexer)), index(dataIndex) {}
203 
204  // Boilerplate iterator methods.
205  ptrdiff_t operator-(const ElementsAttrIterator &rhs) const {
206  return index - rhs.index;
207  }
208  bool operator==(const ElementsAttrIterator &rhs) const {
209  return index == rhs.index;
210  }
211  bool operator<(const ElementsAttrIterator &rhs) const {
212  return index < rhs.index;
213  }
214  ElementsAttrIterator &operator+=(ptrdiff_t offset) {
215  index += offset;
216  return *this;
217  }
218  ElementsAttrIterator &operator-=(ptrdiff_t offset) {
219  index -= offset;
220  return *this;
221  }
222 
223  /// Return the value at the current iterator position.
224  T operator*() const { return indexer.at<T>(index); }

296 /// Return the elements of this attribute as a value of type 'T'.
297 template <typename T>
298 auto ElementsAttr::value_begin() const -> DefaultValueCheckT<T, iterator<T>> {
299  if (Optional<iterator<T>> iterator = try_value_begin<T>())
300  return std::move(*iterator);
301  llvm::errs()
302  << "ElementsAttr does not provide iteration facilities for type `"
303  << llvm::getTypeName<T>() << "`, see attribute: " << *this << "\n";
304  llvm_unreachable("invalid `T` for ElementsAttr::getValues");
305 }
306 template <typename T>
307 auto ElementsAttr::try_value_begin() const
308  -> DefaultValueCheckT<T, Optional<iterator<T>>> {
309  FailureOr<detail::ElementsAttrIndexer> indexer =
310  getValuesImpl(TypeID::get<T>());
311  if (failed(indexer))
312  return llvm::None;
313  return iterator<T>(std::move(*indexer), 0);
314 }
