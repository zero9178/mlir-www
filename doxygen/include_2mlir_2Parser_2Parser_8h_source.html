31 /// Given a block containing operations that have just been parsed, if the block
32 /// contains a single operation of `ContainerOpT` type then remove it from the
33 /// block and return it. If the block does not contain just that operation,
34 /// create a new operation instance of `ContainerOpT` and move all of the
35 /// operations within `parsedBlock` into the first block of the first region.
36 /// `ContainerOpT` is required to have a single region containing a single
37 /// block, and must implement the `SingleBlockImplicitTerminator` trait.
38 template <typename ContainerOpT>
39 inline OwningOpRef<ContainerOpT> constructContainerOpForParserIfNecessary(
40  Block *parsedBlock, MLIRContext *context, Location sourceFileLoc) {
41  static_assert(
42  ContainerOpT::template hasTrait<OpTrait::OneRegion>() &&
43  (ContainerOpT::template hasTrait<OpTrait::NoTerminator>() ||
44  OpTrait::template hasSingleBlockImplicitTerminator<
45  ContainerOpT>::value),
46  "Expected `ContainerOpT` to have a single region with a single "
47  "block that has an implicit terminator or does not require one");
48 
49  // Check to see if we parsed a single instance of this operation.
50  if (llvm::hasSingleElement(*parsedBlock)) {
51  if (ContainerOpT op = dyn_cast<ContainerOpT>(parsedBlock->front())) {
52  op->remove();
53  return op;
54  }
55  }
56 
57  // If not, then build a new one to contain the parsed operations.
58  OpBuilder builder(context);
59  ContainerOpT op = builder.create<ContainerOpT>(sourceFileLoc);
60  OwningOpRef<ContainerOpT> opRef(op);
61  assert(op->getNumRegions() == 1 && llvm::hasSingleElement(op->getRegion(0)) &&
62  "expected generated operation to have a single region with a single "
63  "block");
64  Block *opBlock = &op->getRegion(0).front();
65  opBlock->getOperations().splice(opBlock->begin(),
66  parsedBlock->getOperations());
67 
68  // After splicing, verify just this operation to ensure it can properly
69  // contain the operations inside of it.
70  if (failed(op.verifyInvariants()))
71  return OwningOpRef<ContainerOpT>();
72  return opRef;
73 }

129 /// This parses the file specified by the indicated SourceMgr. If the source IR
130 /// contained a single instance of `ContainerOpT`, it is returned. Otherwise, a
131 /// new instance of `ContainerOpT` is constructed containing all of the parsed
132 /// operations. If parsing was not successful, null is returned and an error
133 /// message is emitted through the error handler registered in the context, and
134 /// failure is returned. `ContainerOpT` is required to have a single region
135 /// containing a single block, and must implement the
136 /// `SingleBlockImplicitTerminator` trait.
137 template <typename ContainerOpT>
138 inline OwningOpRef<ContainerOpT>
139 parseSourceFile(const llvm::SourceMgr &sourceMgr, MLIRContext *context) {
140  LocationAttr sourceFileLoc;
141  Block block;
142  if (failed(parseSourceFile(sourceMgr, &block, context, &sourceFileLoc)))
143  return OwningOpRef<ContainerOpT>();
144  return detail::constructContainerOpForParserIfNecessary<ContainerOpT>(
145  &block, context, sourceFileLoc);
146 }
