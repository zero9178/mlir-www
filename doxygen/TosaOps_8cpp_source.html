148  ArrayAttr newShape = op.new_shape();

151  DenseElementsAttr inputAttr;
152  if (!matchPattern(input, m_Constant(&inputAttr)))
153  return failure();
154 
155  // Check if has >1 consumer and is not splat

157  return failure();

168  inputAttr.reshape(inputShape.clone(newShapeValues));
169  rewriter.replaceOpWithNewOp<tosa::ConstOp>(op, outputAttr.getType(),

171  return success();
172  }

192 struct ConstantTransposeOptimization
193  : public OpRewritePattern<tosa::TransposeOp> {
194  using OpRewritePattern::OpRewritePattern;
195 
196  LogicalResult matchAndRewrite(tosa::TransposeOp op,
197  PatternRewriter &rewriter) const override {
198  auto outputType = op.getType().cast<ShapedType>();
199  ArrayRef<int64_t> outputShape = outputType.getShape();
200  // TOSA supports quantized types.
201  if (!outputType.getElementType().isIntOrIndexOrFloat())
202  return failure();
203 
204  DenseElementsAttr inputValues;
205  if (!matchPattern(op.input1(), m_Constant(&inputValues)))
206  return failure();
207  // Make sure the input is a constant that has a single user.
208  if (!llvm::hasSingleElement(op.input1().getDefiningOp()->getUsers()))
209  return failure();
210 
211  DenseIntElementsAttr permAttr;
212  if (!matchPattern(op.perms(), m_Constant(&permAttr)))
213  return failure();
214  auto permValues = llvm::to_vector<6>(llvm::map_range(
215  // TOSA allows both 32- and 64-bit integer tensors here.
216  permAttr.getValues<APInt>(),
217  [](const APInt &val) { return val.getZExtValue(); }));
218 
219  auto inputType = op.input1().getType().cast<ShapedType>();
220  ArrayRef<int64_t> inputShape = inputType.getShape();
221  int64_t numElements = inputType.getNumElements();
222 
223  SmallVector<Attribute, 4> outputValues;
224  outputValues.resize(numElements);
225 
226  // Transpose the input constant. Because we don't know its rank in advance,
227  // we need to loop over the range [0, element count) and delinearize the
228  // index.
229  auto attrValues = inputValues.getValues<Attribute>();
230  for (int srcLinearIndex = 0; srcLinearIndex < numElements;
231  ++srcLinearIndex) {
232  SmallVector<uint64_t, 6> srcIndices(inputType.getRank(), 0);
233  int totalCount = srcLinearIndex;
234  for (int dim = inputType.getRank() - 1; dim >= 0; --dim) {
235  srcIndices[dim] = totalCount % inputShape[dim];
236  totalCount /= inputShape[dim];
237  }
238 
239  SmallVector<uint64_t, 6> dstIndices(outputType.getRank(), 0);
240  for (int dim = outputType.getRank() - 1; dim >= 0; --dim)
241  dstIndices[dim] = srcIndices[permValues[dim]];
242 
243  uint64_t dstLinearIndex = dstIndices.front();
244  for (int dim = 1; dim < outputType.getRank(); ++dim)
245  dstLinearIndex = dstLinearIndex * outputShape[dim] + dstIndices[dim];
246 
247  outputValues[dstLinearIndex] = attrValues[srcIndices];
248  }
249 
250  rewriter.replaceOpWithNewOp<tosa::ConstOp>(
251  op, outputType, DenseElementsAttr::get(outputType, outputValues));
252  return success();
253  }
254 };

375 struct MaterializePadValue : public OpRewritePattern<tosa::PadOp> {
376  using OpRewritePattern::OpRewritePattern;
377 
378  LogicalResult matchAndRewrite(tosa::PadOp op,
379  PatternRewriter &rewriter) const override {
380  if (op.pad_const())
381  return failure();
382 
383  auto input = op.input1();
384  auto padding = op.padding();
385 
386  ShapedType inputTy = input.getType().cast<ShapedType>();
387  Type elementTy = inputTy.getElementType();
388 
389  Attribute constantAttr;
390  if (elementTy.isa<FloatType>())
391  constantAttr = rewriter.getFloatAttr(elementTy, 0.0);
392  else if (elementTy.isa<IntegerType>() && !op.quantization_info())
393  constantAttr = rewriter.getIntegerAttr(elementTy, 0);
394  else if (elementTy.isa<IntegerType>() && op.quantization_info()) {
395  auto value = op.quantization_info().getValue().input_zp().getValue();
396  constantAttr = rewriter.getIntegerAttr(elementTy, value.getZExtValue());
397  }
398 
399  if (!constantAttr) {
400  return rewriter.notifyMatchFailure(
401  op,
402  "tosa.pad to linalg lowering encountered an unknown element type");
403  }
404 
405  auto denseAttr = DenseElementsAttr::get(
406  RankedTensorType::get({}, elementTy), constantAttr);
407  auto constantVal = rewriter.create<tosa::ConstOp>(
408  op.getLoc(), denseAttr.getType(), denseAttr);
409 
410  rewriter.replaceOpWithNewOp<tosa::PadOp>(
411  op, op.getType(), ValueRange{input, padding, constantVal},
412  op->getAttrs());
413  return success();
414  }
415 };

480  return success();

497  rewriter.replaceOp(op, input);

693

695  bool weightIsQuant = !weightEType.template isa<FloatType>();

712  }

1446  SmallVector<int64_t> &outShape) {

1448  for (int i = 0, e = operands.size(); i != e; ++i) {

1450  if (!shape.hasRank()) {
1451  return failure();

1454  }

1456  outShape.resize(outRank, 1);
1457 
1458  for (int i = 0, e = operands.size(); i != e; ++i) {
1459  auto shape = operands.getShape(i);
1460  auto rankDiff = outShape.size() - shape.getRank();
1461

1463  auto dim1 = outShape[i + rankDiff];
1464  auto dim2 = shape.getDimSize(i);
1465  auto resolvedDim = dim1;
1466 
1467  if (dim1 == 1) {
1468  resolvedDim = dim2;
1469  } else if (dim2 == 1) {
1470  resolvedDim = dim1;
1471  } else if (dim1 != dim2) {
1472  return failure();
1473  }
1474  outShape[i + rankDiff] = resolvedDim;
1475  }
1476  }
1477
