21 /// The state maintained across applications of various ops implementing the
22 /// TransformOpInterface. The operations implementing this interface and the
23 /// surrounding structure are referred to as transform IR. The operations to
24 /// which transformations apply are referred to as payload IR. The state thus
25 /// contains the mapping between values defined in the transform IR ops and
26 /// payload IR ops. It assumes that each value in the transform IR can be used
27 /// at most once (since transformations are likely to change the payload IR ops
28 /// the value corresponds to). Checks that transform IR values correspond to
29 /// disjoint sets of payload IR ops throughout the transformation.
30 ///
31 /// A reference to this class is passed as an argument to "apply" methods of the
32 /// transform op interface. Thus the "apply" method can call
33 /// `state.getPayloadOps( getSomeOperand() )` to obtain the list of operations
34 /// associated with its operand and subject to transformation. The method is
35 /// expected to populate the `TransformResults` class instance in order to
36 /// update the mapping. The `applyTransform` method takes care of propagating
37 /// the state of `TransformResults` into the instance of this class.
38 ///
39 /// When applying transform IR operations with regions, the client is expected
40 /// to create a RegionScope RAII object to create a new "stack frame" for
41 /// values defined inside the region. The mappings from and to these values will
42 /// be automatically dropped when the object goes out of scope, typically at the
43 /// end of the "apply" function of the parent operation. If a region contains
44 /// blocks with arguments, the client can map those arguments to payload IR ops
45 /// using "mapBlockArguments".
46 class TransformState {

48  /// operations in the payload IR.

107  /// A RAII object maintaining a "stack frame" for a transform IR region. When
108  /// applying a transform IR operation that contains a region, the caller is
109  /// expected to create a RegionScope before applying the ops contained in the
110  /// region. This ensures that the mappings between values defined in the
111  /// transform IR region and payload IR operations are cleared when the region
112  /// processing ends; such values cannot be accessed outside the region.
113  class RegionScope {
114  public:
115  /// Forgets the mapping from or to values defined in the associated
116  /// transform IR region.
117  ~RegionScope() {
118  state.mappings.erase(region);
119 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
120  state.regionStack.pop_back();
121 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
122  }
123 
124  private:
125  /// Creates a new scope for mappings between values defined in the given
126  /// transform IR region and payload IR operations.
127  RegionScope(TransformState &state, Region &region)
128  : state(state), region(&region) {
129  auto res = state.mappings.try_emplace(this->region);
130  assert(res.second && "the region scope is already present");
131  (void)res;
132 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
133  assert(state.regionStack.back()->isProperAncestor(&region) &&
134  "scope started at a non-nested region");
135  state.regionStack.push_back(&region);
136 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
137  }

149  /// Base class for TransformState extensions that allow TransformState to
150  /// contain user-specified information in the state object. Clients are
151  /// expected to derive this class, add the desired fields, and make the
152  /// derived class compatible with the MLIR TypeID mechanism:
153  ///
154  /// ```mlir
155  /// class MyExtension final : public TransformState::Extension {
156  /// public:
157  /// MyExtension(TranfsormState &state, int myData)
158  /// : Extension(state) {...}
159  /// private:
160  /// int mySupplementaryData;
161  /// };
162  /// ```
163  ///
164  /// Instances of this and derived classes are not expected to be created by
165  /// the user, instead they are directly constructed within a TransformState. A
166  /// TransformState can only contain one extension with the given TypeID.
167  /// Extensions can be obtained from a TransformState instance, and can be
168  /// removed when they are no longer required.
169  ///
170  /// ```mlir
171  /// transformState.addExtension<MyExtension>(/*myData=*/42);
172  /// MyExtension *ext = transformState.getExtension<MyExtension>();
173  /// ext->doSomething();
174  /// ```

176  // Allow TransformState to allocate Extensions.
177  friend class TransformState;

181  // Out-of-line definition ensures symbols are emitted in a single object
182  // file.
183  virtual ~Extension();

186  /// Constructs an extension of the given TransformState object.
187  Extension(TransformState &state) : state(state) {}
188 
189  /// Provides read-only access to the parent TransformState object.
190  const TransformState &getTransformState() const { return state; }

192  /// Replaces the given payload op with another op. If the replacement op is

201  /// Adds a new Extension of the type specified as template parameter,
202  /// constructing it with the arguments provided. The extension is owned by the
203  /// TransformState. It is expected that the state does not already have an
204  /// extension of the same type. Extension constructors are expected to take
205  /// a reference to TransformState as first argument, automatically supplied
206  /// by this call.
207  template <typename Ty, typename... Args>
208  Ty &addExtension(Args &&...args) {
209  static_assert(
210  std::is_base_of<Extension, Ty>::value,
211  "only an class derived from TransformState::Extension is allowed here");
212  auto ptr = std::make_unique<Ty>(*this, std::forward<Args>(args)...);
213  auto result = extensions.try_emplace(TypeID::get<Ty>(), std::move(ptr));
214  assert(result.second && "extension already added");
215  return *static_cast<Ty *>(result.first->second.get());
216  }
217 
218  /// Returns the extension of the specified type.
219  template <typename Ty>
220  Ty *getExtension() {
221  static_assert(
222  std::is_base_of<Extension, Ty>::value,
223  "only an class derived from TransformState::Extension is allowed here");
224  auto iter = extensions.find(TypeID::get<Ty>());
225  if (iter == extensions.end())
226  return nullptr;
227  return static_cast<Ty *>(iter->second.get());
228  }
229 
230  /// Removes the extension of the specified type.
231  template <typename Ty>
232  void removeExtension() {
233  static_assert(
234  std::is_base_of<Extension, Ty>::value,
235  "only an class derived from TransformState::Extension is allowed here");
236  extensions.erase(TypeID::get<Ty>());
237  }

364 /// This trait is supposed to be attached to Transform dialect operations that
365 /// can be standalone top-level transforms. Such operations typically contain
366 /// other Transform dialect operations that can be executed following some
367 /// control flow logic specific to the current operation. The operations with
368 /// this trait are expected to have exactly one single-block region with one
369 /// argument of PDL Operation type. The operations are also expected to be valid
370 /// without operands, in which case they are considered top-level, and with one
371 /// or more arguments, in which case they are considered nested. Top-level
372 /// operations have the block argument of the entry block in the Transform IR
373 /// correspond to the root operation of Payload IR. Nested operations have the
374 /// block argument of the entry block in the Transform IR correspond to a list
375 /// of Payload IR operations mapped to the first operand of the Transform IR
376 /// operation. The operation must implement TransformOpInterface.
377 template <typename OpTy>
378 class PossibleTopLevelTransformOpTrait
379  : public OpTrait::TraitBase<OpTy, PossibleTopLevelTransformOpTrait> {
380 public:
381  /// Verifies that `op` satisfies the invariants of this trait. Not expected to
382  /// be called directly.
383  static LogicalResult verifyTrait(Operation *op) {
384  return detail::verifyPossibleTopLevelTransformOpTrait(op);
385  }
386 
387  /// Returns the single block of the op's only region.
388  Block *getBodyBlock() { return &this->getOperation()->getRegion(0).front(); }
389 
390  /// Sets up the mapping between the entry block of the only region of this op
391  /// and the relevant list of Payload IR operations in the given state. The
392  /// state is expected to be already scoped at the region of this operation.
393  /// Returns failure if the mapping failed, e.g., the value is already mapped.
394  LogicalResult mapBlockArguments(TransformState &state) {
395  return detail::mapPossibleTopLevelTransformOpBlockArguments(
396  state, this->getOperation());
397  }

400 /// Side effect resource corresponding to the mapping between Transform IR
401 /// values and Payload IR operations. An Allocate effect from this resource
402 /// means creating a new mapping entry, it is always accompanied by a Write
403 /// effet. A Read effect from this resource means accessing the mapping. A Free
404 /// effect on this resource indicates the removal of the mapping entry,
405 /// typically after a transformation that modifies the Payload IR operations
406 /// associated with one of the Transform IR operation's operands. It is always
407 /// accompanied by a Read effect. Read-after-Free and double-Free are not
408 /// allowed (they would be problematic with "regular" memory effects too) as
409 /// they indicate an attempt to access Payload IR operations that have been
410 /// modified, potentially erased, by the previous tranfsormations.
411 // TODO: consider custom effects if these are not enabling generic passes such
412 // as CSE/DCE to work.
413 struct TransformMappingResource
414  : public SideEffects::Resource::Base<TransformMappingResource> {
415  StringRef getName() override { return "transform.mapping"; }
416 };
417 
418 /// Side effect resource corresponding to the Payload IR itself. Only Read and
419 /// Write effects are expected on this resource, with Write always accompanied
420 /// by a Read (short of fully replacing the top-level Payload IR operation, one
421 /// cannot modify the Payload IR without reading it first). This is intended
422 /// to disallow reordering of Transform IR operations that mutate the Payload IR
423 /// while still allowing the reordering of those that only access it.
