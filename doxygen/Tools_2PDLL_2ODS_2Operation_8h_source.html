115 /// This class provides an ODS representation of a specific operation. This
116 /// includes all of the information necessary for use by the PDL frontend for
117 /// generating code for a pattern rewrite.
118 class Operation {
119 public:

121  SMRange getLoc() const { return location; }
122 
123  /// Append an attribute to this operation.
124  void appendAttribute(StringRef name, bool optional,
125  const AttributeConstraint &constraint) {
126  attributes.emplace_back(Attribute(name, optional, constraint));
127  }
128 
129  /// Append an operand to this operation.

131  const TypeConstraint &constraint) {
132  operands.emplace_back(

134  }
135 
136  /// Append a result to this operation.

138  const TypeConstraint &constraint) {

140  }
