83 template <typename ReshapeOpTy, typename InverseReshapeOpTy>
84 static OpFoldResult foldReshapeOp(ReshapeOpTy reshapeOp,
85  ArrayRef<Attribute> operands) {
86  // Fold producer-consumer reshape ops that where the operand type of the
87  // producer is same as the return type of the consumer.
88  auto reshapeSrcOp =
89  reshapeOp.src().template getDefiningOp<InverseReshapeOpTy>();
90  if (reshapeSrcOp && reshapeSrcOp.getSrcType() == reshapeOp.getResultType())
91  return reshapeSrcOp.src();
92  // Reshape of a constant can be replaced with a new constant.
93  if (auto elements = operands.front().dyn_cast_or_null<DenseElementsAttr>()) {
94  return elements.reshape(
95  reshapeOp.getResult().getType().template cast<ShapedType>());
96  }
97  return nullptr;
98 }

100 /// Common verifier for reshape-like types. Fills `expandedType` and
101 ///`collapsedType` with the proper `src` or `result` type.
102 template <typename Op, typename T>
103 static LogicalResult verifyReshapeLikeTypes(Op op, T expandedType,
104  T collapsedType, bool isExpansion) {
105  unsigned expandedRank = expandedType.getRank();
106  unsigned collapsedRank = collapsedType.getRank();
107  if (expandedRank < collapsedRank)
108  return op.emitOpError("expected the type ")
109  << expandedType
110  << " to have higher rank than the type = " << collapsedType;
111  if (expandedRank == 0)
112  return op.emitOpError("expected non-zero memref ranks");
113  if (expandedRank == collapsedRank)
114  return op.emitOpError("expected to collapse or expand dims");
115 
116  if (collapsedRank == 0) {
117  // If collapsed rank is 0, then expanded type must be static shaped and of
118  // sizes 1.
119  if (llvm::any_of(expandedType.getShape(),
120  [](int64_t dim) -> bool { return dim != 1; }))
121  return op.emitOpError("invalid to reshape tensor/memref with non-unit "
122  "extent dimensions to zero-rank tensor/memref");
123  return success();
124  }
125  if (collapsedRank != op.reassociation().size())
126  return op.emitOpError("expected rank of the collapsed type(")
127  << collapsedRank << ") to be the number of reassociation maps("
128  << op.reassociation().size() << ")";
129  auto maps = op.getReassociationMaps();
130  for (auto it : llvm::enumerate(maps))
131  if (it.value().getNumDims() != expandedRank)
132  return op.emitOpError("expected reassociation map #")
133  << it.index() << " of same rank as expanded memref("
134  << expandedRank << "), but got " << it.value().getNumDims();
135  int invalidIdx = 0;
136  if (!isReassociationValid(maps, &invalidIdx))
137  return op.emitOpError("expected reassociation map #")
138  << invalidIdx << " to be valid and contiguous";
139  return verifyReshapeLikeShapes(op, collapsedType, expandedType, isExpansion);
140 }

142 /// Verify that shapes of the reshaped types using following rules
143 /// 1) if a dimension in the collapsed type is static, then the corresponding
144 /// dimensions in the expanded shape should be
145 /// a) static
146 /// b) the product should be same as the collaped shape.
147 /// 2) if a dimension in the collaped type is dynamic, one and only one of the
148 /// corresponding dimensions in the expanded type should be dynamic. This
149 /// rule is only needed with reshape operations that are expanding.
150 LogicalResult reshapeLikeShapesAreCompatible(
151  function_ref<LogicalResult(const Twine &)> emitError,
152  ArrayRef<int64_t> collapsedShape, ArrayRef<int64_t> expandedShape,
153  ArrayRef<ReassociationIndices> reassociationMaps, bool isExpandingReshape);
154 
155 template <typename OpTy>
156 static LogicalResult verifyReshapeLikeShapes(OpTy op, ShapedType collapsedType,
157  ShapedType expandedType,
158  bool isExpandingReshape) {
159  return reshapeLikeShapesAreCompatible(
160  [&](const Twine &msg) { return op->emitOpError(msg); },
161  collapsedType.getShape(), expandedType.getShape(),
162  op.getReassociationIndices(), isExpandingReshape);
163 }

168 /// Pattern to collapse producer/consumer reshape ops that are both collapsing
169 /// dimensions or are both expanding dimensions.
170 template <typename ReshapeOpTy>
171 struct ComposeReassociativeReshapeOps : public OpRewritePattern<ReshapeOpTy> {
172  using OpRewritePattern<ReshapeOpTy>::OpRewritePattern;
173  LogicalResult matchAndRewrite(ReshapeOpTy reshapeOp,
174  PatternRewriter &rewriter) const override {
175  auto srcReshapeOp = reshapeOp.src().template getDefiningOp<ReshapeOpTy>();
176  if (!srcReshapeOp)
177  return failure();
178 
179  ShapedType resultType = reshapeOp.getResultType();
180 
181  if (hasNonIdentityLayout(srcReshapeOp.src().getType()) ||
182  hasNonIdentityLayout(reshapeOp.src().getType()) ||
183  hasNonIdentityLayout(reshapeOp.result().getType()))
184  return failure();
185 
186  Optional<SmallVector<ReassociationIndices>> reassociationIndices =
187  composeReassociationIndices(srcReshapeOp.getReassociationIndices(),
188  reshapeOp.getReassociationIndices(),
189  rewriter.getContext());
190  if (!reassociationIndices)
191  return failure();
192  rewriter.replaceOpWithNewOp<ReshapeOpTy>(
193  reshapeOp, resultType, srcReshapeOp.src(), *reassociationIndices);
194  return success();
195  }
196 };

198 /// Pattern to compose
199 /// `collapse_shape(expand_shape(%src, reassociation_1), reassociation_2)`.
200 /// In that case both `srcType` and `resultType` can be expressed as a function
201 /// of `intermediateType`.
202 /// In order to demonstrate the approach, let's assume that `rank(srcType) >
203 /// `rank(resultType)`, i.e. the resulting operation should be `collapse_shape`.
204 /// In that case, we can iterate over every set of indices in `reassociation_2`
205 /// and try to find ids of sets of indices in `reassociation_1` that cover it
206 /// completely.
207 ///
208 /// Example:
209 ///
210 /// %0 = tensor.expand_shape %arg [[0], [1], [2, 3]]
211 /// : tensor<?x?x?xi64> into tensor<?x?x?x1xi64>
212 /// %1 = tensor.collapse_shape %0 [[0, 1], [2, 3]]
213 /// : tensor<?x?x?x1xi64> into tensor<?x?xi64>
214 ///
215 /// can be canonicalized into
216 ///
217 /// %0 = tensor.collapse_shape %arg [[0, 1], [2]]
218 /// : tensor<?x?x?xi64> into tensor<?x?xi64>
219 ///
220 /// because [0] and [1] from `expand_shape` reassociation cover completely
221 /// `[0, 1]` from `collapse_shape`. If it is impossible to find such union of
222 /// indices, then we fail.
223 //
224 /// When `rank(srcType) < rank(resultType)`, then we just swap `reassociation_1`
225 /// `reassociation_2` and produce `expand_shape`.
226 template <typename CollapseOpTy, typename ExpandOpTy>
227 struct ComposeCollapseOfExpandOp : public OpRewritePattern<CollapseOpTy> {
228  using OpRewritePattern<CollapseOpTy>::OpRewritePattern;
229  LogicalResult matchAndRewrite(CollapseOpTy collapseOp,
230  PatternRewriter &rewriter) const override {
231  auto expandOp = collapseOp.src().template getDefiningOp<ExpandOpTy>();
232  if (!expandOp)
233  return failure();
234 
235  ShapedType srcType = expandOp.getSrcType();
236  ShapedType resultType = collapseOp.getResultType();
237 
238  if (hasNonIdentityLayout(collapseOp.src().getType()) ||
239  hasNonIdentityLayout(expandOp.src().getType()) ||
240  hasNonIdentityLayout(expandOp.result().getType()))
241  return failure();
242 
243  int64_t srcRank = srcType.getRank();
244  int64_t resultRank = resultType.getRank();
245  if (srcType == resultType)
246  return failure();
247 
248  SmallVector<ReassociationIndices, 4> higherRankReassociation,
249  lowerRankReassociation;
250 
251  bool isResultCollapsed = srcRank > resultRank;
252  if (isResultCollapsed) {
253  higherRankReassociation = expandOp.getReassociationIndices();
254  lowerRankReassociation = collapseOp.getReassociationIndices();
255  } else {
256  higherRankReassociation = collapseOp.getReassociationIndices();
257  lowerRankReassociation = expandOp.getReassociationIndices();
258  }
259 
260  size_t higherRankIndicesID = 0;
261  SmallVector<ReassociationIndices, 4> composedReassociation;
262  for (const auto &lowerRankIndices : lowerRankReassociation) {
263  ReassociationIndices composedIndices;
264  while (higherRankIndicesID < higherRankReassociation.size()) {
265  auto rightmostIndex =
266  higherRankReassociation[higherRankIndicesID].back();
267  if (rightmostIndex > lowerRankIndices.back())
268  return failure();
269  composedIndices.push_back(higherRankIndicesID++);
270  if (rightmostIndex == lowerRankIndices.back())
271  break;
272  }
273  composedReassociation.push_back(composedIndices);
274  }
275  if (isResultCollapsed)
276  rewriter.replaceOpWithNewOp<CollapseOpTy>(
277  collapseOp, resultType, expandOp.src(), composedReassociation);
278  else
279  rewriter.replaceOpWithNewOp<ExpandOpTy>(
280  collapseOp, resultType, expandOp.src(), composedReassociation);
281  return success();
282  }
283 };

285 template <typename ExpandOpTy, typename CollapseOpTy>
286 struct ComposeExpandOfCollapseOp : public OpRewritePattern<ExpandOpTy> {
287  using OpRewritePattern<ExpandOpTy>::OpRewritePattern;
288  LogicalResult matchAndRewrite(ExpandOpTy expandOp,
289  PatternRewriter &rewriter) const override {
290  auto collapseOp = expandOp.src().template getDefiningOp<CollapseOpTy>();
291  if (!collapseOp)
292  return failure();
293 
294  ShapedType srcType = collapseOp.getSrcType();
295  ShapedType resultType = expandOp.getResultType();
296 
297  if (hasNonIdentityLayout(expandOp.src().getType()) ||
298  hasNonIdentityLayout(collapseOp.src().getType()) ||
299  hasNonIdentityLayout(collapseOp.result().getType()))
300  return failure();
301 
302  int64_t srcRank = srcType.getRank();
303  int64_t resultRank = resultType.getRank();
304  if (srcType == resultType)
305  return failure();
306 
307  auto srcReassociation = collapseOp.getReassociationIndices();
308  auto resultReassociation = expandOp.getReassociationIndices();
309  if (srcRank > resultRank) {
310  auto composedReassociation = findCollapsingReassociation(
311  srcReassociation, resultReassociation, srcType.getShape(),
312  resultType.getShape());
313  if (!composedReassociation.hasValue())
314  return failure();
315 
316  rewriter.replaceOpWithNewOp<CollapseOpTy>(
317  expandOp, resultType, collapseOp.src(), *composedReassociation);
318  return success();
319  }
320  auto composedReassociation =
321  findCollapsingReassociation(resultReassociation, srcReassociation,
322  resultType.getShape(), srcType.getShape());
323  if (!composedReassociation.hasValue())
324  return failure();
325 
326  rewriter.replaceOpWithNewOp<ExpandOpTy>(
327  expandOp, resultType, collapseOp.src(), *composedReassociation);
328  return success();
329  }
330 
331 private:
332  // Attempts to find a way to collapse `srcShape` to `resultShape` by
333  // collapsing subshapes defined by the reassociation indices.
334  Optional<SmallVector<ReassociationIndices>> findCollapsingReassociation(
335  ArrayRef<ReassociationIndices> srcReassociation,
336  ArrayRef<ReassociationIndices> resultReassociation,
337  ArrayRef<int64_t> srcShape, ArrayRef<int64_t> resultShape) const {
338  SmallVector<ReassociationIndices, 4> composedReassociation;
339 
340  if (srcReassociation.empty())
341  return {getReassociationIndicesForCollapse(srcShape, resultShape)};
342 
343  for (auto item : llvm::zip(srcReassociation, resultReassociation)) {
344  auto &srcIndices = std::get<0>(item);
345  auto &resultIndices = std::get<1>(item);
346  auto srcSubShape = srcShape.slice(srcIndices.front(), srcIndices.size());
347  auto resultSubShape =
348  resultShape.slice(resultIndices.front(), resultIndices.size());
349 
350  if (srcSubShape.size() == resultSubShape.size()) {
351  if (srcSubShape == resultSubShape)
352  composedReassociation.push_back(srcIndices);
353  else
354  return llvm::None;
355  }
356 
357  // Find reassociation to collapse `srcSubShape` into `resultSubShape`.
358  auto subShapeReassociation =
359  getReassociationIndicesForCollapse(srcSubShape, resultSubShape);
360  if (!subShapeReassociation.hasValue())
361  return llvm::None;
362 
363  // Remap the subshape indices back to the original srcShape.
364  for (auto &subshape_indices : *subShapeReassociation) {
365  ReassociationIndices shape_indices;
366  for (int64_t index : subshape_indices)
367  shape_indices.push_back(srcIndices.front() + index);
368  composedReassociation.push_back(shape_indices);
369  }
370  }
371  return {std::move(composedReassociation)};
372  }
373 };
