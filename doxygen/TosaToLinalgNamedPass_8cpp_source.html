40  void runOnOperation() override {
41  RewritePatternSet patterns(&getContext());
42  ConversionTarget target(getContext());
43  target.addLegalDialect<linalg::LinalgDialect, tosa::TosaDialect,
44  tensor::TensorDialect, scf::SCFDialect>();
45 
46  // Not every TOSA op can be legalized to linalg.
47  target.addIllegalOp<tosa::Conv2DOp>();
48  target.addIllegalOp<tosa::DepthwiseConv2DOp>();
49  target.addIllegalOp<tosa::MaxPool2dOp>();
50  target.addIllegalOp<tosa::AvgPool2dOp>();
51  target.addIllegalOp<tosa::MatMulOp>();
52  target.addIllegalOp<tosa::FullyConnectedOp>();
53 
54  target.markUnknownOpDynamicallyLegal([](Operation *) { return true; });
55 
56  FunctionOpInterface func = getOperation();
57  mlir::tosa::populateTosaToLinalgNamedConversionPatterns(&patterns);
58  if (failed(applyFullConversion(func, target, std::move(patterns))))
59  signalPassFailure();
60  }
