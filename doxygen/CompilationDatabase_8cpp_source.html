28 struct MappingTraits<CompilationDatabase::FileInfo> {
29  static void mapping(IO &io, CompilationDatabase::FileInfo &info) {
30  // Parse the filename and normalize it to the form we will expect from
31  // incoming URIs.
32  io.mapRequired("filepath", info.filename);
33 
34  // Normalize the filename to avoid incompatability with incoming URIs.
35  if (Expected<lsp::URIForFile> uri =
36  lsp::URIForFile::fromFile(info.filename))
37  info.filename = uri->file().str();
38 
39  // Parse the includes from the yaml stream. These are in the form of a
40  // semi-colon delimited list.
41  std::string combinedIncludes;
42  io.mapRequired("includes", combinedIncludes);
43  for (StringRef include : llvm::split(combinedIncludes, ";")) {
44  if (!include.empty())
45  info.includeDirs.push_back(include.str());
46  }
47  }
48 };
49 } // end namespace yaml
50 } // end namespace llvm

63 void CompilationDatabase::loadDatabase(StringRef filename) {
64  if (filename.empty())
65  return;
66 
67  // Set up the input file.
68  std::string errorMessage;
69  std::unique_ptr<llvm::MemoryBuffer> inputFile =
70  openInputFile(filename, &errorMessage);
71  if (!inputFile) {
72  Logger::error("Failed to open compilation database: {0}", errorMessage);
73  return;
74  }
75  llvm::yaml::Input yaml(inputFile->getBuffer());
76 
77  // Parse the yaml description and add any new files to the database.
78  std::vector<FileInfo> parsedFiles;
79  yaml >> parsedFiles;
80  for (auto &file : parsedFiles) {
81  auto it = files.try_emplace(file.filename, std::move(file));
82 
83  // If we encounter a duplicate file, log a warning and ignore it.
84  if (!it.second) {
85  Logger::info("Duplicate .pdll file in compilation database: {0}",
86  file.filename);
87  }
88  }
89 }
