54 /// Conversion pattern that converts a 1-D vector transfer read/write.
55 /// Note that this conversion pass only converts vector x2 or x4 f32
56 /// types. For unsupported cases, they will fall back to the vector to
57 /// llvm conversion pattern.
58 template <typename ConcreteOp>
59 class VectorTransferConversion : public ConvertOpToLLVMPattern<ConcreteOp> {
60 public:
61  using ConvertOpToLLVMPattern<ConcreteOp>::ConvertOpToLLVMPattern;
62 
63  LogicalResult
64  matchAndRewrite(ConcreteOp xferOp, typename ConcreteOp::Adaptor adaptor,
65  ConversionPatternRewriter &rewriter) const override {
66  // TODO: support 0-d corner case.
67  if (xferOp.getTransferRank() == 0)
68  return failure();
69 
70  if (xferOp.getVectorType().getRank() > 1 ||
71  llvm::size(xferOp.getIndices()) == 0)
72  return failure();
73 
74  if (!xferOp.getPermutationMap().isMinorIdentity())
75  return failure();
76 
77  // Have it handled in vector->llvm conversion pass.
78  if (xferOp.isDimInBounds(0))
79  return failure();
80 
81  auto toLLVMTy = [&](Type t) {
82  return this->getTypeConverter()->convertType(t);
83  };
84  auto vecTy = toLLVMTy(xferOp.getVectorType());
85  unsigned vecWidth = LLVM::getVectorNumElements(vecTy).getFixedValue();
86  Location loc = xferOp->getLoc();
87 
88  // The backend result vector scalarization have trouble scalarize
89  // <1 x ty> result, exclude the x1 width from the lowering.
90  if (vecWidth != 2 && vecWidth != 4)
91  return failure();
92 
93  // Obtain dataPtr and elementType from the memref.
94  auto memRefType = xferOp.getShapedType().template dyn_cast<MemRefType>();
95  if (!memRefType)
96  return failure();
97  // MUBUF instruction operate only on addresspace 0(unified) or 1(global)
98  // In case of 3(LDS): fall back to vector->llvm pass
99  // In case of 5(VGPR): wrong
100  if ((memRefType.getMemorySpaceAsInt() != 0) &&
101  (memRefType.getMemorySpaceAsInt() != 1))
102  return failure();
103 
104  // Note that the dataPtr starts at the offset address specified by
105  // indices, so no need to calculate offset size in bytes again in
106  // the MUBUF instruction.
107  Value dataPtr = this->getStridedElementPtr(
108  loc, memRefType, adaptor.getSource(), adaptor.getIndices(), rewriter);
109 
110  // 1. Create and fill a <4 x i32> dwordConfig with:
111  // 1st two elements holding the address of dataPtr.
112  // 3rd element: -1.
113  // 4th element: 0x27000.
114  SmallVector<int32_t, 4> constConfigAttr{0, 0, -1, 0x27000};
115  Type i32Ty = rewriter.getIntegerType(32);
116  VectorType i32Vecx4 = VectorType::get(4, i32Ty);
117  Value constConfig = rewriter.create<LLVM::ConstantOp>(
118  loc, toLLVMTy(i32Vecx4),
119  DenseElementsAttr::get(i32Vecx4, ArrayRef<int32_t>(constConfigAttr)));
120 
121  // Treat first two element of <4 x i32> as i64, and save the dataPtr
122  // to it.
123  Type i64Ty = rewriter.getIntegerType(64);
124  Value i64x2Ty = rewriter.create<LLVM::BitcastOp>(
125  loc, LLVM::getFixedVectorType(toLLVMTy(i64Ty), 2), constConfig);
126  Value dataPtrAsI64 = rewriter.create<LLVM::PtrToIntOp>(
127  loc, toLLVMTy(i64Ty).template cast<Type>(), dataPtr);
128  Value zero = this->createIndexConstant(rewriter, loc, 0);
129  Value dwordConfig = rewriter.create<LLVM::InsertElementOp>(
130  loc, LLVM::getFixedVectorType(toLLVMTy(i64Ty), 2), i64x2Ty,
131  dataPtrAsI64, zero);
132  dwordConfig =
133  rewriter.create<LLVM::BitcastOp>(loc, toLLVMTy(i32Vecx4), dwordConfig);
134 
135  // 2. Rewrite op as a buffer read or write.
136  Value int1False = rewriter.create<LLVM::ConstantOp>(
137  loc, toLLVMTy(rewriter.getIntegerType(1)),
138  rewriter.getIntegerAttr(rewriter.getIntegerType(1), 0));
139  Value int32Zero = rewriter.create<LLVM::ConstantOp>(
140  loc, toLLVMTy(i32Ty),
141  rewriter.getIntegerAttr(rewriter.getIntegerType(32), 0));
142  return replaceTransferOpWithMubuf(
143  rewriter, adaptor.getOperands(), *this->getTypeConverter(), loc, xferOp,
144  vecTy, dwordConfig, int32Zero, int32Zero, int1False, int1False);
145  }
146 };
147 } // namespace

162 void LowerVectorToROCDLPass::runOnOperation() {
163  LLVMTypeConverter converter(&getContext());
164  RewritePatternSet patterns(&getContext());
165 
166  populateVectorToROCDLConversionPatterns(converter, patterns);
167  populateMemRefToLLVMConversionPatterns(converter, patterns);
168  populateFuncToLLVMConversionPatterns(converter, patterns);
169 
170  LLVMConversionTarget target(getContext());
171  target.addLegalDialect<ROCDL::ROCDLDialect>();
172 
173  if (failed(
174  applyPartialConversion(getOperation(), target, std::move(patterns))))
175  signalPassFailure();
176 }
