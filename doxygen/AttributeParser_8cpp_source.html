25 /// Parse an arbitrary attribute.
26 ///
27 /// attribute-value ::= `unit`
28 /// | bool-literal
29 /// | integer-literal (`:` (index-type | integer-type))?
30 /// | float-literal (`:` float-type)?
31 /// | string-literal (`:` type)?
32 /// | type
33 /// | `[` (attribute-value (`,` attribute-value)*)? `]`
34 /// | `{` (attribute-entry (`,` attribute-entry)*)? `}`
35 /// | symbol-ref-id (`::` symbol-ref-id)*
36 /// | `dense` `<` tensor-literal `>` `:`
37 /// (tensor-type | vector-type)
38 /// | `sparse` `<` attribute-value `,` attribute-value `>`
39 /// `:` (tensor-type | vector-type)
40 /// | `opaque` `<` dialect-namespace `,` hex-string-literal
41 /// `>` `:` (tensor-type | vector-type)
42 /// | extended-attribute
43 ///
44 Attribute Parser::parseAttribute(Type type) {
45  switch (getToken().getKind()) {
46  // Parse an AffineMap or IntegerSet attribute.
47  case Token::kw_affine_map: {
48  consumeToken(Token::kw_affine_map);
49 
50  AffineMap map;
51  if (parseToken(Token::less, "expected '<' in affine map") ||
52  parseAffineMapReference(map) ||
53  parseToken(Token::greater, "expected '>' in affine map"))
54  return Attribute();
55  return AffineMapAttr::get(map);
56  }
57  case Token::kw_affine_set: {
58  consumeToken(Token::kw_affine_set);
59 
60  IntegerSet set;
61  if (parseToken(Token::less, "expected '<' in integer set") ||
62  parseIntegerSetReference(set) ||
63  parseToken(Token::greater, "expected '>' in integer set"))
64  return Attribute();
65  return IntegerSetAttr::get(set);
66  }
67 
68  // Parse an array attribute.
69  case Token::l_square: {
70  SmallVector<Attribute, 4> elements;
71  auto parseElt = [&]() -> ParseResult {
72  elements.push_back(parseAttribute());
73  return elements.back() ? success() : failure();
74  };
75 
76  if (parseCommaSeparatedList(Delimiter::Square, parseElt))
77  return nullptr;
78  return builder.getArrayAttr(elements);
79  }
80 
81  // Parse a boolean attribute.
82  case Token::kw_false:
83  consumeToken(Token::kw_false);
84  return builder.getBoolAttr(false);
85  case Token::kw_true:
86  consumeToken(Token::kw_true);
87  return builder.getBoolAttr(true);
88 
89  // Parse a dense elements attribute.
90  case Token::kw_dense:
91  return parseDenseElementsAttr(type);
92 
93  // Parse a dictionary attribute.
94  case Token::l_brace: {
95  NamedAttrList elements;
96  if (parseAttributeDict(elements))
97  return nullptr;
98  return elements.getDictionary(getContext());
99  }
100 
101  // Parse an extended attribute, i.e. alias or dialect attribute.
102  case Token::hash_identifier:
103  return parseExtendedAttr(type);
104 
105  // Parse floating point and integer attributes.
106  case Token::floatliteral:
107  return parseFloatAttr(type, /*isNegative=*/false);
108  case Token::integer:
109  return parseDecOrHexAttr(type, /*isNegative=*/false);
110  case Token::minus: {
111  consumeToken(Token::minus);
112  if (getToken().is(Token::integer))
113  return parseDecOrHexAttr(type, /*isNegative=*/true);
114  if (getToken().is(Token::floatliteral))
115  return parseFloatAttr(type, /*isNegative=*/true);
116 
117  return (emitError("expected constant integer or floating point value"),
118  nullptr);
119  }
120 
121  // Parse a location attribute.
122  case Token::kw_loc: {
123  consumeToken(Token::kw_loc);
124 
125  LocationAttr locAttr;
126  if (parseToken(Token::l_paren, "expected '(' in inline location") ||
127  parseLocationInstance(locAttr) ||
128  parseToken(Token::r_paren, "expected ')' in inline location"))
129  return Attribute();
130  return locAttr;
131  }
132 
133  // Parse an opaque elements attribute.
134  case Token::kw_opaque:
135  return parseOpaqueElementsAttr(type);
136 
137  // Parse a sparse elements attribute.
138  case Token::kw_sparse:
139  return parseSparseElementsAttr(type);
140 
141  // Parse a string attribute.
142  case Token::string: {
143  auto val = getToken().getStringValue();
144  consumeToken(Token::string);
145  // Parse the optional trailing colon type if one wasn't explicitly provided.
146  if (!type && consumeIf(Token::colon) && !(type = parseType()))
147  return Attribute();
148 
149  return type ? StringAttr::get(val, type)
150  : StringAttr::get(getContext(), val);
151  }
152 
153  // Parse a symbol reference attribute.
154  case Token::at_identifier: {
155  // When populating the parser state, this is a list of locations for all of
156  // the nested references.
157  SmallVector<SMRange> referenceLocations;
158  if (state.asmState)
159  referenceLocations.push_back(getToken().getLocRange());
160 
161  // Parse the top-level reference.
162  std::string nameStr = getToken().getSymbolReference();
163  consumeToken(Token::at_identifier);
164 
165  // Parse any nested references.
166  std::vector<FlatSymbolRefAttr> nestedRefs;
167  while (getToken().is(Token::colon)) {
168  // Check for the '::' prefix.
169  const char *curPointer = getToken().getLoc().getPointer();
170  consumeToken(Token::colon);
171  if (!consumeIf(Token::colon)) {
172  if (getToken().isNot(Token::eof, Token::error)) {
173  state.lex.resetPointer(curPointer);
174  consumeToken();
175  }
176  break;
177  }
178  // Parse the reference itself.
179  auto curLoc = getToken().getLoc();
180  if (getToken().isNot(Token::at_identifier)) {
181  emitError(curLoc, "expected nested symbol reference identifier");
182  return Attribute();
183  }
184 
185  // If we are populating the assembly state, add the location for this
186  // reference.
187  if (state.asmState)
188  referenceLocations.push_back(getToken().getLocRange());
189 
190  std::string nameStr = getToken().getSymbolReference();
191  consumeToken(Token::at_identifier);
192  nestedRefs.push_back(SymbolRefAttr::get(getContext(), nameStr));
193  }
194  SymbolRefAttr symbolRefAttr =
195  SymbolRefAttr::get(getContext(), nameStr, nestedRefs);
196 
197  // If we are populating the assembly state, record this symbol reference.
198  if (state.asmState)
199  state.asmState->addUses(symbolRefAttr, referenceLocations);
200  return symbolRefAttr;
201  }
202 
203  // Parse a 'unit' attribute.
204  case Token::kw_unit:
205  consumeToken(Token::kw_unit);
206  return builder.getUnitAttr();
207 
208  default:
209  // Parse a type attribute.
210  if (Type type = parseType())
211  return TypeAttr::get(type);
212  return nullptr;
213  }
214 }

216 /// Parse an optional attribute with the provided type.
217 OptionalParseResult Parser::parseOptionalAttribute(Attribute &attribute,
218  Type type) {
219  switch (getToken().getKind()) {
220  case Token::at_identifier:
221  case Token::floatliteral:
222  case Token::integer:
223  case Token::hash_identifier:
224  case Token::kw_affine_map:
225  case Token::kw_affine_set:
226  case Token::kw_dense:
227  case Token::kw_false:
228  case Token::kw_loc:
229  case Token::kw_opaque:
230  case Token::kw_sparse:
231  case Token::kw_true:
232  case Token::kw_unit:
233  case Token::l_brace:
234  case Token::l_square:
235  case Token::minus:
236  case Token::string:
237  attribute = parseAttribute(type);
238  return success(attribute != nullptr);
239 
240  default:
241  // Parse an optional type attribute.
242  Type type;
243  OptionalParseResult result = parseOptionalType(type);
244  if (result.hasValue() && succeeded(*result))
245  attribute = TypeAttr::get(type);
246  return result;
247  }

258 /// Attribute dictionary.
259 ///
260 /// attribute-dict ::= `{` `}`
261 /// | `{` attribute-entry (`,` attribute-entry)* `}`
262 /// attribute-entry ::= (bare-id | string-literal) `=` attribute-value
263 ///
264 ParseResult Parser::parseAttributeDict(NamedAttrList &attributes) {
265  llvm::SmallDenseSet<StringAttr> seenKeys;
266  auto parseElt = [&]() -> ParseResult {
267  // The name of an attribute can either be a bare identifier, or a string.
268  Optional<StringAttr> nameId;
269  if (getToken().is(Token::string))
270  nameId = builder.getStringAttr(getToken().getStringValue());
271  else if (getToken().isAny(Token::bare_identifier, Token::inttype) ||
272  getToken().isKeyword())
273  nameId = builder.getStringAttr(getTokenSpelling());
274  else
275  return emitError("expected attribute name");
276 
277  if (nameId->size() == 0)
278  return emitError("expected valid attribute name");
279 
280  if (!seenKeys.insert(*nameId).second)
281  return emitError("duplicate key '")
282  << nameId->getValue() << "' in dictionary attribute";
283  consumeToken();
284 
285  // Lazy load a dialect in the context if there is a possible namespace.
286  auto splitName = nameId->strref().split('.');
287  if (!splitName.second.empty())
288  getContext()->getOrLoadDialect(splitName.first);
289 
290  // Try to parse the '=' for the attribute value.
291  if (!consumeIf(Token::equal)) {
292  // If there is no '=', we treat this as a unit attribute.
293  attributes.push_back({*nameId, builder.getUnitAttr()});
294  return success();
295  }
296 
297  auto attr = parseAttribute();
298  if (!attr)
299  return failure();
300  attributes.push_back({*nameId, attr});
301  return success();
302  };
303 
304  return parseCommaSeparatedList(Delimiter::Braces, parseElt,
305  " in attribute dictionary");
306 }

308 /// Parse a float attribute.
309 Attribute Parser::parseFloatAttr(Type type, bool isNegative) {
310  auto val = getToken().getFloatingPointValue();
311  if (!val.hasValue())
312  return (emitError("floating point value too large for attribute"), nullptr);
313  consumeToken(Token::floatliteral);
314  if (!type) {
315  // Default to F64 when no type is specified.
316  if (!consumeIf(Token::colon))
317  type = builder.getF64Type();
318  else if (!(type = parseType()))
319  return nullptr;
320  }
321  if (!type.isa<FloatType>())
322  return (emitError("floating point value not valid for specified type"),
323  nullptr);
324  return FloatAttr::get(type, isNegative ? -val.getValue() : val.getValue());
325 }

327 /// Construct an APint from a parsed value, a known attribute type and
328 /// sign.
329 static Optional<APInt> buildAttributeAPInt(Type type, bool isNegative,
330  StringRef spelling) {
331  // Parse the integer value into an APInt that is big enough to hold the value.
332  APInt result;
333  bool isHex = spelling.size() > 1 && spelling[1] == 'x';
334  if (spelling.getAsInteger(isHex ? 0 : 10, result))
335  return llvm::None;
336 
337  // Extend or truncate the bitwidth to the right size.
338  unsigned width = type.isIndex() ? IndexType::kInternalStorageBitWidth
339  : type.getIntOrFloatBitWidth();
340 
341  if (width > result.getBitWidth()) {
342  result = result.zext(width);
343  } else if (width < result.getBitWidth()) {
344  // The parser can return an unnecessarily wide result with leading zeros.
345  // This isn't a problem, but truncating off bits is bad.
346  if (result.countLeadingZeros() < result.getBitWidth() - width)
347  return llvm::None;
348 
349  result = result.trunc(width);
350  }
351 
352  if (width == 0) {
353  // 0 bit integers cannot be negative and manipulation of their sign bit will
354  // assert, so short-cut validation here.
355  if (isNegative)
356  return llvm::None;
357  } else if (isNegative) {
358  // The value is negative, we have an overflow if the sign bit is not set
359  // in the negated apInt.
360  result.negate();
361  if (!result.isSignBitSet())
362  return llvm::None;
363  } else if ((type.isSignedInteger() || type.isIndex()) &&
364  result.isSignBitSet()) {
365  // The value is a positive signed integer or index,
366  // we have an overflow if the sign bit is set.
367  return llvm::None;
368  }
369 
370  return result;
371 }

373 /// Parse a decimal or a hexadecimal literal, which can be either an integer
374 /// or a float attribute.
375 Attribute Parser::parseDecOrHexAttr(Type type, bool isNegative) {
376  Token tok = getToken();
377  StringRef spelling = tok.getSpelling();
378  SMLoc loc = tok.getLoc();
379 
380  consumeToken(Token::integer);
381  if (!type) {
382  // Default to i64 if not type is specified.
383  if (!consumeIf(Token::colon))
384  type = builder.getIntegerType(64);
385  else if (!(type = parseType()))
386  return nullptr;
387  }
388 
389  if (auto floatType = type.dyn_cast<FloatType>()) {
390  Optional<APFloat> result;
391  if (failed(parseFloatFromIntegerLiteral(result, tok, isNegative,
392  floatType.getFloatSemantics(),
393  floatType.getWidth())))
394  return Attribute();
395  return FloatAttr::get(floatType, *result);
396  }
397 
398  if (!type.isa<IntegerType, IndexType>())
399  return emitError(loc, "integer literal not valid for specified type"),
400  nullptr;
401 
402  if (isNegative && type.isUnsignedInteger()) {
403  emitError(loc,
404  "negative integer literal not valid for unsigned integer type");
405  return nullptr;
406  }
407 
408  Optional<APInt> apInt = buildAttributeAPInt(type, isNegative, spelling);
409  if (!apInt)
410  return emitError(loc, "integer constant out of range for attribute"),
411  nullptr;
412  return builder.getIntegerAttr(type, *apInt);
413 }

494 /// Parse the elements of a tensor literal. If 'allowHex' is true, the parser
495 /// may also parse a tensor literal that is store as a hex string.
496 ParseResult TensorLiteralParser::parse(bool allowHex) {
497  // If hex is allowed, check for a string literal.
498  if (allowHex && p.getToken().is(Token::string)) {
499  hexStorage = p.getToken();
500  p.consumeToken(Token::string);
501  return success();
502  }
503  // Otherwise, parse a list or an individual element.
504  if (p.getToken().is(Token::l_square))
505  return parseList(shape);
506  return parseElement();
507 }

509 /// Build a dense attribute instance with the parsed elements and the given
510 /// shaped type.
511 DenseElementsAttr TensorLiteralParser::getAttr(SMLoc loc,
512  ShapedType type) {
513  Type eltType = type.getElementType();
514 
515  // Check to see if we parse the literal from a hex string.
516  if (hexStorage.hasValue() &&
517  (eltType.isIntOrIndexOrFloat() || eltType.isa<ComplexType>()))
518  return getHexAttr(loc, type);
519 
520  // Check that the parsed storage size has the same number of elements to the
521  // type, or is a known splat.
522  if (!shape.empty() && getShape() != type.getShape()) {
523  p.emitError(loc) << "inferred shape of elements literal ([" << getShape()
524  << "]) does not match type ([" << type.getShape() << "])";
525  return nullptr;
526  }
527 
528  // Handle the case where no elements were parsed.
529  if (!hexStorage.hasValue() && storage.empty() && type.getNumElements()) {
530  p.emitError(loc) << "parsed zero elements, but type (" << type
531  << ") expected at least 1";
532  return nullptr;
533  }
534 
535  // Handle complex types in the specific element type cases below.
536  bool isComplex = false;
537  if (ComplexType complexTy = eltType.dyn_cast<ComplexType>()) {
538  eltType = complexTy.getElementType();
539  isComplex = true;
540  }
541 
542  // Handle integer and index types.
543  if (eltType.isIntOrIndex()) {
544  std::vector<APInt> intValues;
545  if (failed(getIntAttrElements(loc, eltType, intValues)))
546  return nullptr;
547  if (isComplex) {
548  // If this is a complex, treat the parsed values as complex values.
549  auto complexData = llvm::makeArrayRef(
550  reinterpret_cast<std::complex<APInt> *>(intValues.data()),
551  intValues.size() / 2);
552  return DenseElementsAttr::get(type, complexData);
553  }
554  return DenseElementsAttr::get(type, intValues);
555  }
556  // Handle floating point types.
557  if (FloatType floatTy = eltType.dyn_cast<FloatType>()) {
558  std::vector<APFloat> floatValues;
559  if (failed(getFloatAttrElements(loc, floatTy, floatValues)))
560  return nullptr;
561  if (isComplex) {
562  // If this is a complex, treat the parsed values as complex values.
563  auto complexData = llvm::makeArrayRef(
564  reinterpret_cast<std::complex<APFloat> *>(floatValues.data()),
565  floatValues.size() / 2);
566  return DenseElementsAttr::get(type, complexData);
567  }
568  return DenseElementsAttr::get(type, floatValues);
569  }
570 
571  // Other types are assumed to be string representations.
572  return getStringAttr(loc, type, type.getElementType());
573 }

575 /// Build a Dense Integer attribute for the given type.
576 ParseResult
577 TensorLiteralParser::getIntAttrElements(SMLoc loc, Type eltTy,
578  std::vector<APInt> &intValues) {
579  intValues.reserve(storage.size());
580  bool isUintType = eltTy.isUnsignedInteger();
581  for (const auto &signAndToken : storage) {
582  bool isNegative = signAndToken.first;
583  const Token &token = signAndToken.second;
584  auto tokenLoc = token.getLoc();
585 
586  if (isNegative && isUintType) {
587  return p.emitError(tokenLoc)
588  << "expected unsigned integer elements, but parsed negative value";
589  }
590 
591  // Check to see if floating point values were parsed.
592  if (token.is(Token::floatliteral)) {
593  return p.emitError(tokenLoc)
594  << "expected integer elements, but parsed floating-point";
595  }
596 
597  assert(token.isAny(Token::integer, Token::kw_true, Token::kw_false) &&
598  "unexpected token type");
599  if (token.isAny(Token::kw_true, Token::kw_false)) {
600  if (!eltTy.isInteger(1)) {
601  return p.emitError(tokenLoc)
602  << "expected i1 type for 'true' or 'false' values";
603  }
604  APInt apInt(1, token.is(Token::kw_true), /*isSigned=*/false);
605  intValues.push_back(apInt);
606  continue;
607  }
608 
609  // Create APInt values for each element with the correct bitwidth.
610  Optional<APInt> apInt =
611  buildAttributeAPInt(eltTy, isNegative, token.getSpelling());
612  if (!apInt)
613  return p.emitError(tokenLoc, "integer constant out of range for type");
614  intValues.push_back(*apInt);
615  }
616  return success();
617 }

619 /// Build a Dense Float attribute for the given type.
620 ParseResult
621 TensorLiteralParser::getFloatAttrElements(SMLoc loc, FloatType eltTy,
622  std::vector<APFloat> &floatValues) {
623  floatValues.reserve(storage.size());
624  for (const auto &signAndToken : storage) {
625  bool isNegative = signAndToken.first;
626  const Token &token = signAndToken.second;
627 
628  // Handle hexadecimal float literals.
629  if (token.is(Token::integer) && token.getSpelling().startswith("0x")) {
630  Optional<APFloat> result;
631  if (failed(p.parseFloatFromIntegerLiteral(result, token, isNegative,
632  eltTy.getFloatSemantics(),
633  eltTy.getWidth())))
634  return failure();
635 
636  floatValues.push_back(*result);
637  continue;
638  }
639 
640  // Check to see if any decimal integers or booleans were parsed.
641  if (!token.is(Token::floatliteral))
642  return p.emitError()
643  << "expected floating-point elements, but parsed integer";
644 
645  // Build the float values from tokens.
646  auto val = token.getFloatingPointValue();
647  if (!val.hasValue())
648  return p.emitError("floating point value too large for attribute");
649 
650  APFloat apVal(isNegative ? -*val : *val);
651  if (!eltTy.isF64()) {
652  bool unused;
653  apVal.convert(eltTy.getFloatSemantics(), APFloat::rmNearestTiesToEven,
654  &unused);
655  }
656  floatValues.push_back(apVal);
657  }
658  return success();
659 }

662 DenseElementsAttr TensorLiteralParser::getStringAttr(SMLoc loc,
663  ShapedType type,
664  Type eltTy) {
665  if (hexStorage.hasValue()) {
666  auto stringValue = hexStorage.getValue().getStringValue();

668  }
669 
670  std::vector<std::string> stringValues;

723 ParseResult TensorLiteralParser::parseElement() {
724  switch (p.getToken().getKind()) {
725  // Parse a boolean element.
726  case Token::kw_true:
727  case Token::kw_false:
728  case Token::floatliteral:
729  case Token::integer:
730  storage.emplace_back(/*isNegative=*/false, p.getToken());
731  p.consumeToken();
732  break;
733 
734  // Parse a signed integer or a negative floating-point element.
735  case Token::minus:
736  p.consumeToken(Token::minus);
737  if (!p.getToken().isAny(Token::floatliteral, Token::integer))
738  return p.emitError("expected integer or floating point literal");
739  storage.emplace_back(/*isNegative=*/true, p.getToken());
740  p.consumeToken();
741  break;
742 
743  case Token::string:
744  storage.emplace_back(/*isNegative=*/false, p.getToken());
745  p.consumeToken();
746  break;
747 
748  // Parse a complex element of the form '(' element ',' element ')'.
749  case Token::l_paren:
750  p.consumeToken(Token::l_paren);
751  if (parseElement() ||
752  p.parseToken(Token::comma, "expected ',' between complex elements") ||
753  parseElement() ||
754  p.parseToken(Token::r_paren, "expected ')' after complex elements"))
755  return failure();
756  break;
757 
758  default:
759  return p.emitError("expected element literal of primitive type");
760  }
761 
762  return success();
763 }

765 /// Parse a list of either lists or elements, returning the dimensions of the
766 /// parsed sub-tensors in dims. For example:
767 /// parseList([1, 2, 3]) -> Success, [3]
768 /// parseList([[1, 2], [3, 4]]) -> Success, [2, 2]
769 /// parseList([[1, 2], 3]) -> Failure
770 /// parseList([[1, [2, 3]], [4, [5]]]) -> Failure
771 ParseResult TensorLiteralParser::parseList(SmallVectorImpl<int64_t> &dims) {
772  auto checkDims = [&](const SmallVectorImpl<int64_t> &prevDims,
773  const SmallVectorImpl<int64_t> &newDims) -> ParseResult {
774  if (prevDims == newDims)
775  return success();
776  return p.emitError("tensor literal is invalid; ranks are not consistent "
777  "between elements");
778  };
779 
780  bool first = true;
781  SmallVector<int64_t, 4> newDims;
782  unsigned size = 0;
783  auto parseOneElement = [&]() -> ParseResult {
784  SmallVector<int64_t, 4> thisDims;
785  if (p.getToken().getKind() == Token::l_square) {
786  if (parseList(thisDims))
787  return failure();
788  } else if (parseElement()) {
789  return failure();
790  }
791  ++size;
792  if (!first)
793  return checkDims(newDims, thisDims);
794  newDims = thisDims;
795  first = false;
796  return success();
797  };
798  if (p.parseCommaSeparatedList(Parser::Delimiter::Square, parseOneElement))
799  return failure();
800 
801  // Return the sublists' dimensions with 'size' prepended.
802  dims.clear();
803  dims.push_back(size);
804  dims.append(newDims.begin(), newDims.end());
805  return success();
806 }

894 /// Parse a sparse elements attribute.
895 Attribute Parser::parseSparseElementsAttr(Type attrType) {
896  SMLoc loc = getToken().getLoc();
897  consumeToken(Token::kw_sparse);
898  if (parseToken(Token::less, "Expected '<' after 'sparse'"))
899  return nullptr;
900 
901  // Check for the case where all elements are sparse. The indices are
902  // represented by a 2-dimensional shape where the second dimension is the rank
903  // of the type.
904  Type indiceEltType = builder.getIntegerType(64);
905  if (consumeIf(Token::greater)) {
906  ShapedType type = parseElementsLiteralType(attrType);
907  if (!type)
908  return nullptr;
909 
910  // Construct the sparse elements attr using zero element indice/value
911  // attributes.
912  ShapedType indicesType =
913  RankedTensorType::get({0, type.getRank()}, indiceEltType);
914  ShapedType valuesType = RankedTensorType::get({0}, type.getElementType());
915  return getChecked<SparseElementsAttr>(
916  loc, type, DenseElementsAttr::get(indicesType, ArrayRef<Attribute>()),
917  DenseElementsAttr::get(valuesType, ArrayRef<Attribute>()));
918  }
919 
920  /// Parse the indices. We don't allow hex values here as we may need to use
921  /// the inferred shape.
922  auto indicesLoc = getToken().getLoc();
923  TensorLiteralParser indiceParser(*this);
924  if (indiceParser.parse(/*allowHex=*/false))
925  return nullptr;
926 
927  if (parseToken(Token::comma, "expected ','"))
928  return nullptr;
929 
930  /// Parse the values.
931  auto valuesLoc = getToken().getLoc();
932  TensorLiteralParser valuesParser(*this);
933  if (valuesParser.parse(/*allowHex=*/true))
934  return nullptr;
935 
936  if (parseToken(Token::greater, "expected '>'"))
937  return nullptr;
938 
939  auto type = parseElementsLiteralType(attrType);
940  if (!type)
941  return nullptr;
942 
943  // If the indices are a splat, i.e. the literal parser parsed an element and
944  // not a list, we set the shape explicitly. The indices are represented by a
945  // 2-dimensional shape where the second dimension is the rank of the type.
946  // Given that the parsed indices is a splat, we know that we only have one
947  // indice and thus one for the first dimension.
948  ShapedType indicesType;
949  if (indiceParser.getShape().empty()) {
950  indicesType = RankedTensorType::get({1, type.getRank()}, indiceEltType);
951  } else {
952  // Otherwise, set the shape to the one parsed by the literal parser.
953  indicesType = RankedTensorType::get(indiceParser.getShape(), indiceEltType);
954  }
955  auto indices = indiceParser.getAttr(indicesLoc, indicesType);
956 
957  // If the values are a splat, set the shape explicitly based on the number of
958  // indices. The number of indices is encoded in the first dimension of the
959  // indice shape type.
960  auto valuesEltType = type.getElementType();
961  ShapedType valuesType =
962  valuesParser.getShape().empty()
963  ? RankedTensorType::get({indicesType.getDimSize(0)}, valuesEltType)
964  : RankedTensorType::get(valuesParser.getShape(), valuesEltType);
965  auto values = valuesParser.getAttr(valuesLoc, valuesType);
966 
967  // Build the sparse elements attribute by the indices and values.
968  return getChecked<SparseElementsAttr>(loc, type, indices, values);
969 }
