70 void AsmParserState::Impl::resolveSymbolUses() {
71  SmallVector<Operation *> symbolOps;
72  for (auto &opAndUseMapIt : symbolTableOperations) {
73  for (auto &it : *opAndUseMapIt.second) {
74  symbolOps.clear();
75  if (failed(symbolTable.lookupSymbolIn(
76  opAndUseMapIt.first, it.first.cast<SymbolRefAttr>(), symbolOps)))
77  continue;
78 
79  for (ArrayRef<SMRange> useRange : it.second) {
80  for (const auto &symIt : llvm::zip(symbolOps, useRange)) {
81  auto opIt = operationToIdx.find(std::get<0>(symIt));
82  if (opIt != operationToIdx.end())
83  operations[opIt->second]->symbolUses.push_back(std::get<1>(symIt));
84  }
85  }
86  }
87  }
88 }

125 /// Lex a string token whose contents start at the given `curPtr`. Returns the
126 /// position at the end of the string, after a terminal or invalid character
127 /// (e.g. `"` or `\0`).
128 static const char *lexLocStringTok(const char *curPtr) {
129  while (char c = *curPtr++) {
130  // Check for various terminal characters.
131  if (StringRef("\"\n\v\f").contains(c))
132  return curPtr;
133 
134  // Check for escape sequences.
135  if (c == '\\') {
136  // Check a few known escapes and \xx hex digits.
137  if (*curPtr == '"' || *curPtr == '\\' || *curPtr == 'n' || *curPtr == 't')
138  ++curPtr;
139  else if (llvm::isHexDigit(*curPtr) && llvm::isHexDigit(curPtr[1]))
140  curPtr += 2;
141  else
142  return curPtr;
143  }
144  }
145 
146  // If we hit this point, we've reached the end of the buffer. Update the end
147  // pointer to not point past the buffer.
148  return curPtr - 1;
149 }

151 SMRange AsmParserState::convertIdLocToRange(SMLoc loc) {
152  if (!loc.isValid())
153  return SMRange();
154  const char *curPtr = loc.getPointer();
155 
156  // Check if this is a string token.
157  if (*curPtr == '"') {
158  curPtr = lexLocStringTok(curPtr + 1);
159 
160  // Otherwise, default to handling an identifier.
161  } else {
162  // Return if the given character is a valid identifier character.
163  auto isIdentifierChar = [](char c) {
164  return isalnum(c) || c == '$' || c == '.' || c == '_' || c == '-';
165  };
166 
167  while (*curPtr && isIdentifierChar(*(++curPtr)))
168  continue;
169  }
170 
171  return SMRange(loc, SMLoc::getFromPointer(curPtr));
172 }

177 void AsmParserState::initialize(Operation *topLevelOp) {
178  startOperationDefinition(topLevelOp->getName());

186 void AsmParserState::finalize(Operation *topLevelOp) {
187  assert(!impl->partialOperations.empty() &&
188  "expected valid partial operation definition");
189  Impl::PartialOpDef partialOpDef = impl->partialOperations.pop_back_val();
190 
191  // If this operation is a symbol table, resolve any symbol uses.
192  if (partialOpDef.isSymbolTable()) {
193  impl->symbolTableOperations.emplace_back(
194  topLevelOp, std::move(partialOpDef.symbolTable));
195  }
196  impl->resolveSymbolUses();
197 }

203 void AsmParserState::finalizeOperationDefinition(
204  Operation *op, SMRange nameLoc, SMLoc endLoc,
205  ArrayRef<std::pair<unsigned, SMLoc>> resultGroups) {
206  assert(!impl->partialOperations.empty() &&
207  "expected valid partial operation definition");
208  Impl::PartialOpDef partialOpDef = impl->partialOperations.pop_back_val();
209 
210  // Build the full operation definition.
211  std::unique_ptr<OperationDefinition> def =
212  std::make_unique<OperationDefinition>(op, nameLoc, endLoc);
213  for (auto &resultGroup : resultGroups)
214  def->resultGroups.emplace_back(resultGroup.first,
215  convertIdLocToRange(resultGroup.second));
216  impl->operationToIdx.try_emplace(op, impl->operations.size());
217  impl->operations.emplace_back(std::move(def));
218 
219  // If this operation is a symbol table, resolve any symbol uses.
220  if (partialOpDef.isSymbolTable()) {
221  impl->symbolTableOperations.emplace_back(
222  op, std::move(partialOpDef.symbolTable));
223  }
224 }

248 void AsmParserState::addDefinition(Block *block, SMLoc location) {
249  auto it = impl->blocksToIdx.find(block);
250  if (it == impl->blocksToIdx.end()) {
251  impl->blocksToIdx.try_emplace(block, impl->blocks.size());
252  impl->blocks.emplace_back(std::make_unique<BlockDefinition>(
253  block, convertIdLocToRange(location)));
254  return;
255  }
256 
257  // If an entry already exists, this was a forward declaration that now has a
258  // proper definition.
259  impl->blocks[it->second]->definition.loc = convertIdLocToRange(location);
260 }

262 void AsmParserState::addDefinition(BlockArgument blockArg,
263  SMLoc location) {
264  auto it = impl->blocksToIdx.find(blockArg.getOwner());
265  assert(it != impl->blocksToIdx.end() &&
266  "expected owner block to have an entry");
267  BlockDefinition &def = *impl->blocks[it->second];
268  unsigned argIdx = blockArg.getArgNumber();
269 
270  if (def.arguments.size() <= argIdx)
271  def.arguments.resize(argIdx + 1);
272  def.arguments[argIdx] = SMDefinition(convertIdLocToRange(location));
273 }

275 void AsmParserState::addUses(Value value, ArrayRef<SMLoc> locations) {
276  // Handle the case where the value is an operation result.
277  if (OpResult result = value.dyn_cast<OpResult>()) {
278  // Check to see if a definition for the parent operation has been recorded.
279  // If one hasn't, we treat the provided value as a placeholder value that
280  // will be refined further later.
281  Operation *parentOp = result.getOwner();
282  auto existingIt = impl->operationToIdx.find(parentOp);
283  if (existingIt == impl->operationToIdx.end()) {
284  impl->placeholderValueUses[value].append(locations.begin(),
285  locations.end());
286  return;
287  }
288 
289  // If a definition does exist, locate the value's result group and add the
290  // use. The result groups are ordered by increasing start index, so we just
291  // need to find the last group that has a smaller/equal start index.
292  unsigned resultNo = result.getResultNumber();
293  OperationDefinition &def = *impl->operations[existingIt->second];
294  for (auto &resultGroup : llvm::reverse(def.resultGroups)) {
295  if (resultNo >= resultGroup.startIndex) {
296  for (SMLoc loc : locations)
297  resultGroup.definition.uses.push_back(convertIdLocToRange(loc));
298  return;
299  }
300  }
301  llvm_unreachable("expected valid result group for value use");
302  }
303 
304  // Otherwise, this is a block argument.
305  BlockArgument arg = value.cast<BlockArgument>();
306  auto existingIt = impl->blocksToIdx.find(arg.getOwner());
307  assert(existingIt != impl->blocksToIdx.end() &&
308  "expected valid block definition for block argument");
309  BlockDefinition &blockDef = *impl->blocks[existingIt->second];
310  SMDefinition &argDef = blockDef.arguments[arg.getArgNumber()];
311  for (SMLoc loc : locations)
312  argDef.uses.emplace_back(convertIdLocToRange(loc));
313 }

315 void AsmParserState::addUses(Block *block, ArrayRef<SMLoc> locations) {
316  auto it = impl->blocksToIdx.find(block);
317  if (it == impl->blocksToIdx.end()) {
318  it = impl->blocksToIdx.try_emplace(block, impl->blocks.size()).first;
319  impl->blocks.emplace_back(std::make_unique<BlockDefinition>(block));
320  }
321 
322  BlockDefinition &def = *impl->blocks[it->second];
323  for (SMLoc loc : locations)
324  def.definition.uses.push_back(convertIdLocToRange(loc));
325 }

327 void AsmParserState::addUses(SymbolRefAttr refAttr,
328  ArrayRef<SMRange> locations) {
329  // Ignore this symbol if no scopes are active.
330  if (impl->symbolUseScopes.empty())
331  return;
332 
333  assert((refAttr.getNestedReferences().size() + 1) == locations.size() &&
334  "expected the same number of references as provided locations");
335  (*impl->symbolUseScopes.back())[refAttr].emplace_back(locations.begin(),
336  locations.end());
337 }

339 void AsmParserState::refineDefinition(Value oldValue, Value newValue) {
340  auto it = impl->placeholderValueUses.find(oldValue);
341  assert(it != impl->placeholderValueUses.end() &&
342  "expected `oldValue` to be a placeholder");
343  addUses(newValue, it->second);
344  impl->placeholderValueUses.erase(oldValue);
