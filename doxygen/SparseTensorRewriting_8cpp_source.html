9 // This file implements linalg dialect rewriting specific to sparse tensors.
10 //
11 // Sparsity should be mostly transparent to the linalg dialect optimizations
12 // (i.e., the dense and sparse take the same path). However, in some cases,
13 // optimizations only make sense in the context of sparse tensors. This file
14 // implements such sparsity specific rewriting rules.
15 //

34 // Helper to detect a sparse tensor type operand.
35 static bool isSparseTensor(OpOperand *op) {
36  if (auto enc = getSparseTensorEncoding(op->get().getType())) {
37  ArrayRef<SparseTensorEncodingAttr::DimLevelType> dimTypes =
38  enc.getDimLevelType();
39  for (auto dimType : dimTypes)
40  if (dimType == SparseTensorEncodingAttr::DimLevelType::Compressed)
41  return true; // at least one compressed
42  }
43  return false;
44 }

46 // Helper method to find zero or empty initialization.
47 static bool isEmptyInit(OpOperand *op) {
48  Value val = op->get();
49  return matchPattern(val, m_Zero()) || matchPattern(val, m_AnyZeroFloat()) ||
50  val.getDefiningOp<InitTensorOp>() || val.getDefiningOp<InitOp>();
51 }

53 // Helper to detect sampling operation.
54 static bool isSampling(GenericOp op) {
55  auto yieldOp = cast<linalg::YieldOp>(op.region().front().getTerminator());
56  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
57  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def)) {
58  // Both scalar input arguments used exactly once.
59  Value s1 = op.getBlock()->getArgument(0);
60  Value s2 = op.getBlock()->getArgument(1);
61  return (def->getOperand(0) == s1 && def->getOperand(1) == s2) ||
62  (def->getOperand(1) == s1 && def->getOperand(0) == s2);
63  }
64  }
65  return false;
66 }
67 
68 // Helper to detect chain of multiplications that do not involve x.
69 static bool isMulChain(Value val, Value x) {
70  if (auto arg = val.dyn_cast<BlockArgument>())
71  return arg != x;
72  if (auto *def = val.getDefiningOp()) {
73  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def))
74  return isMulChain(def->getOperand(0), x) &&
75  isMulChain(def->getOperand(1), x);
76  }
77  return false;
78 }
79 
80 // Helper to detect x = x + <multiplications>.
81 static bool isSumOfMul(GenericOp op) {
82  auto yieldOp = cast<linalg::YieldOp>(op.region().front().getTerminator());
83  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
84  if (isa<arith::AddFOp>(def) || isa<arith::AddIOp>(def)) {
85  Value x = op.getBlock()->getArguments().back();
86  return (def->getOperand(0) == x && isMulChain(def->getOperand(1), x)) ||
87  (def->getOperand(1) == x && isMulChain(def->getOperand(0), x));
88  }
89  }
90  return false;
91 }

98 /// Rewriting rule that converts two kernels:
99 ///
100 /// T(i,j) = SUM(k, A(i,j,k) * B(i,j,k) * ... )
101 /// X(i,j) = S(i,j) * T(i,j)
102 ///
103 /// into a single kernel, using distributive law:
104 ///
105 /// X(i,j) = SUM(k, S(i,j) * A(i,j,k) * B(i,j,k) * ... )
106 ///
107 /// This kind of fusion (merging two ops into one but using arithmetic
108 /// equalities that may not hold for floating-point computations) would
109 /// be undesirable in the dense case, since we distribute the multiplication
110 /// into the reduction loop. However, for sparse sampling tensor S, such
111 /// a fusion may actually reduce the asymptotic complexity of the kernel,
112 /// since intermediate results may be nullified.
113 struct FuseSparseMultiplyOverAdd : public OpRewritePattern<GenericOp> {
114  using OpRewritePattern<GenericOp>::OpRewritePattern;
115 
116  LogicalResult matchAndRewrite(GenericOp op,
117  PatternRewriter &rewriter) const override {
118  // Check consumer.
119  if (!op.hasTensorSemantics() || op.getNumInputs() != 2 ||
120  op.getNumResults() != 1 ||
121  op.getNumParallelLoops() != op.getNumLoops() ||
122  !op.getTiedIndexingMap(op.getOutputOperand(0)).isIdentity() ||
123  !op.getTiedIndexingMap(op.getInputOperand(0)).isIdentity() ||
124  !op.getTiedIndexingMap(op.getInputOperand(1)).isIdentity())
125  return failure();
126  // Find consuming OP2(sparse, other) or OP2(other, sparse). The other
127  // operand can be sparse or dense, since the point of this rewriting rule
128  // is detecting a situation in which *more* sparsity is introduced into
129  // a computation, be it already sparse or still dense.
130  unsigned other = 0;
131  if (isSparseTensor(op.getInputOperand(0)))
132  other = 1;
133  else if (!isSparseTensor(op.getInputOperand(1)))
134  return failure();
135  // Check producer.
136  auto prod = dyn_cast_or_null<GenericOp>(
137  op.getInputOperand(other)->get().getDefiningOp());
138  if (!prod || !prod.hasTensorSemantics() || prod.getNumResults() != 1 ||
139  !prod.getResult(0).hasOneUse())
140  return failure();
141  // Sampling consumer and sum of multiplication chain producer.
142  if (!isEmptyInit(op.getOutputOperand(0)) ||
143  !isEmptyInit(prod.getOutputOperand(0)) || !isSampling(op) ||
144  !isSumOfMul(prod))
145  return failure();
146  // Modify operand structure of producer and consumer.
147  Location loc = prod.getLoc();
148  SmallVector<Value> inputOps = prod.getInputOperands();
149  SmallVector<Value> outputOps = op.getOutputOperands();
150  SmallVector<AffineMap> fusedIndexMaps = prod.getIndexingMaps();
151  inputOps.push_back(op.getInputOperand(1 - other)->get());
152  fusedIndexMaps.push_back(fusedIndexMaps.back()); // mimic other
153  // Fuse producer and consumer into a new generic op.
154  auto fusedOp = rewriter.create<GenericOp>(
155  loc, op.getResult(0).getType(), inputOps, outputOps,
156  rewriter.getAffineMapArrayAttr(fusedIndexMaps), prod.iterator_types(),
157  /*doc=*/nullptr, /*library_call=*/nullptr);
158  Block &prodBlock = prod.region().front();
159  Block &consBlock = op.region().front();
160  BlockAndValueMapping mapper;
161  Block *fusedBlock = new Block();
162  fusedOp.region().push_back(fusedBlock);
163  unsigned num = prodBlock.getNumArguments();
164  for (unsigned i = 0; i < num - 1; i++)
165  addArg(mapper, fusedBlock, prodBlock.getArgument(i));
166  addArg(mapper, fusedBlock, consBlock.getArgument(1 - other));
167  addArg(mapper, fusedBlock, prodBlock.getArgument(num - 1));
168  // Clone bodies of the producer and consumer in new evaluation order.
169  auto *acc = prodBlock.getTerminator()->getOperand(0).getDefiningOp();
170  auto *sampler = consBlock.getTerminator()->getOperand(0).getDefiningOp();
171  rewriter.setInsertionPointToStart(fusedBlock);
172  Value last;
173  for (auto &op : prodBlock.without_terminator())
174  if (&op != acc) {
175  last = op.getResult(0);
176  rewriter.clone(op, mapper);
177  }
178  mapper.map(consBlock.getArgument(other), fusedBlock->back().getResult(0));
179  mapper.map(last, rewriter.clone(*sampler, mapper)->getResult(0));
180  last = rewriter.clone(*acc, mapper)->getResult(0);
181  rewriter.create<linalg::YieldOp>(loc, last);
182  // Replace consumer with fused operation. Old producer
183  // and consumer ops will be removed by DCE.
184  rewriter.replaceOp(op, fusedOp->getResults());
185  return success();
186  }
187 
188 private:
189  // Helper to add argument and record the mapping.
190  static void addArg(BlockAndValueMapping &mapper, Block *b, BlockArgument a) {
191  mapper.map(a, b->addArgument(a.getType(), a.getLoc()));
192  }
193 };
194 } // namespace
