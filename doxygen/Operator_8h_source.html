259  // Pair representing either a index to an argument or a type constraint. Only
260  // one of these entries should have the non-default value.
261  struct ArgOrType {
262  explicit ArgOrType(int index) : index(index), constraint(None) {}
263  explicit ArgOrType(TypeConstraint constraint)
264  : index(None), constraint(constraint) {}
265  bool isArg() const {
266  assert(constraint.hasValue() ^ index.hasValue());
267  return index.hasValue();
268  }
269  bool isType() const {
270  assert(constraint.hasValue() ^ index.hasValue());
271  return constraint.hasValue();
272  }
273 
274  int getArg() const { return *index; }
275  TypeConstraint getType() const { return *constraint; }
276 
277  private:
278  Optional<int> index;
279  Optional<TypeConstraint> constraint;
280  };

286  // Pair consisting kind of argument and index into operands or attributes.
287  struct OperandOrAttribute {
288  enum class Kind { Operand, Attribute };
289  OperandOrAttribute(Kind kind, int index) {
290  packed = (index << 1) | (kind == Kind::Attribute);
291  }
292  int operandOrAttributeIndex() const { return (packed >> 1); }
293  Kind kind() { return (packed & 0x1) ? Kind::Attribute : Kind::Operand; }
294 
295  private:
296  int packed;
297  };
