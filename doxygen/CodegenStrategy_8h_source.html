186 /// Codegen strategy controls how a Linalg op is progressively lowered.
187 struct CodegenStrategy {
188  /// Append a pattern to tile the Op `opName` and fuse its producers with
189  /// tiling and fusion `options`.
190  CodegenStrategy &
191  tileAndFuse(StringRef opName, const LinalgTilingAndFusionOptions &options,
192  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
193  transformationSequence.emplace_back(
194  std::make_unique<TileAndFuse>(opName, options, f));
195  return *this;
196  }
197  /// Conditionally append a pattern to tile the Op `opName` and fuse its
198  /// producers with tiling and fusion `options`.
199  CodegenStrategy &
200  tileAndFuseIf(bool b, StringRef opName, LinalgTilingAndFusionOptions options,
201  LinalgTransformationFilter::FilterFunction f = nullptr) {
202  return b ? tileAndFuse(opName, std::move(options), std::move(f)) : *this;
203  }
204  /// Append a pattern to add a level of tiling for Op `opName` with tiling
205  /// `options`.
206  CodegenStrategy &
207  tile(StringRef opName, const linalg::LinalgTilingOptions &options,
208  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
209  transformationSequence.emplace_back(
210  std::make_unique<Tile>(opName, options, f));
211  return *this;
212  }
213  /// Conditionally append a pattern to add a level of tiling for
214  /// `LinalgOpType` with tiling `options`.
215  CodegenStrategy &
216  tileIf(bool b, StringRef opName, linalg::LinalgTilingOptions options,
217  LinalgTransformationFilter::FilterFunction f = nullptr) {
218  return b ? tile(opName, std::move(options), std::move(f)) : *this;
219  }
220  /// Append a pattern to pad and hoist the operands of Op `opName` with padding
221  /// `options`.
222  CodegenStrategy &
223  pad(StringRef opName, const linalg::LinalgPaddingOptions &options,
224  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
225  transformationSequence.emplace_back(
226  std::make_unique<Pad>(opName, options, f));
227  return *this;
228  }
229  /// Conditionally append a pattern to pad and hoist the operands of Op
230  /// `opName` with padding `options`.
231  CodegenStrategy &
232  padIf(bool b, StringRef opName, linalg::LinalgPaddingOptions options,
233  LinalgTransformationFilter::FilterFunction f = nullptr) {
234  return b ? pad(opName, std::move(options), std::move(f)) : *this;
235  }
236  /// Append a pattern to add a level of promotion for `LinalgOpType` with
237  /// promotion `options`.
238  CodegenStrategy &
239  promote(StringRef opName, const linalg::LinalgPromotionOptions &options,
240  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
241  transformationSequence.emplace_back(
242  std::make_unique<Promote>(opName, options, f));
243  return *this;
244  }
245  /// Conditionally append a pattern to add a level of promotion for
246  /// `LinalgOpType` with promotion `options`.
247  CodegenStrategy &
248  promoteIf(bool b, StringRef opName, linalg::LinalgPromotionOptions options,
249  LinalgTransformationFilter::FilterFunction f = nullptr) {
250  return b ? promote(opName, std::move(options), std::move(f)) : *this;
251  }
252  /// Append a pattern to generalize named operations.
253  CodegenStrategy &
254  generalize(StringRef opName,
255  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
256  transformationSequence.emplace_back(
257  std::make_unique<Generalize>(opName, f));
258  return *this;
259  }
260  /// Conditionally append a pattern to generalize named operations.
261  CodegenStrategy &
262  generalizeIf(bool b, StringRef opName,
263  LinalgTransformationFilter::FilterFunction f = nullptr) {
264  return b ? generalize(opName, std::move(f)) : *this;
265  }
266  /// Append a pattern to interchange iterators.
267  CodegenStrategy &
268  interchange(ArrayRef<int64_t> iteratorInterchange,
269  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
270  transformationSequence.emplace_back(
271  std::make_unique<Interchange>(iteratorInterchange, f));
272  return *this;
273  }
274  /// Conditionally append a pattern to interchange iterators.
275  CodegenStrategy &
276  interchangeIf(bool b, ArrayRef<int64_t> iteratorInterchange,
277  LinalgTransformationFilter::FilterFunction f = nullptr) {
278  return b ? interchange(iteratorInterchange, std::move(f)) : *this;
279  }
280  /// Append patterns to decompose convolutions.
281  CodegenStrategy &
282  decompose(const LinalgTransformationFilter::FilterFunction &f = nullptr) {
283  transformationSequence.emplace_back(std::make_unique<Decompose>(f));
284  return *this;
285  }
286  /// Conditionally append patterns to decompose convolutions.
287  CodegenStrategy &
288  decomposeIf(bool b, LinalgTransformationFilter::FilterFunction f = nullptr) {
289  return b ? decompose(std::move(f)) : *this;
290  }
291  /// Append a pattern to rewrite `LinalgOpType` as a vector operation.
292  CodegenStrategy &
293  vectorize(StringRef opName,
294  const LinalgTransformationFilter::FilterFunction &f = nullptr,
295  bool vectorizePadding = false) {
296  transformationSequence.emplace_back(std::make_unique<Vectorize>(
297  opName, linalg::LinalgVectorizationOptions(), f, vectorizePadding));
298  return *this;
299  }
300  /// Conditionally append a pattern to rewrite `LinalgOpType` as a vector
301  /// operation.
302  CodegenStrategy &
303  vectorizeIf(bool b, StringRef opName,
304  LinalgTransformationFilter::FilterFunction f = nullptr,
305  bool vectorizePadding = false) {
306  return b ? vectorize(opName, std::move(f), vectorizePadding) : *this;
307  }
308  /// Append a pattern to lower all vector operations.
309  CodegenStrategy &vectorLowering(LinalgVectorLoweringOptions options) {
310  transformationSequence.emplace_back(
311  std::make_unique<VectorLowering>(options));
312  return *this;
313  }
314  /// Configure the post staged-patterns global enabling passes options.
315  CodegenStrategy &
316  setVectorTransferToSCFOptions(LinalgEnablingOptions options) {
317  linalgEnablingOptions = options;
318  return *this;
319  }
320 
321  /// Apply the transformation patterns in sequence with cleanup
322  /// transformations interleaved.
323  void configurePassPipeline(OpPassManager &pm, MLIRContext *context,
324  bool addEnablePass = true) const;
325 
326 private:
327  LogicalResult postPatternTransforms(Operation *func) const;
328 
329  LinalgEnablingOptions linalgEnablingOptions;
330  SmallVector<std::unique_ptr<Transformation>, 4> transformationSequence;
331 };
