26 /// A PresburgerRelation represents a union of IntegerRelations that live in
27 /// the same PresburgerSpace with support for union, intersection, subtraction,
28 /// and complement operations, as well as sampling.
29 ///
30 /// The IntegerRelations (disjuncts) are stored in a vector, and the set
31 /// represents the union of these relations. An empty list corresponds to
32 /// the empty set.
33 ///
34 /// Note that there are no invariants guaranteed on the list of disjuncts
35 /// other than that they are all in the same PresburgerSpace. For example, the
36 /// relations may overlap with each other.
37 class PresburgerRelation {
38 public:
39  /// Return a universe set of the specified type that contains all points.
40  static PresburgerRelation getUniverse(const PresburgerSpace &space);
41 
42  /// Return an empty set of the specified type that contains no points.
43  static PresburgerRelation getEmpty(const PresburgerSpace &space);
44 
45  explicit PresburgerRelation(const IntegerRelation &disjunct);
46 
47  unsigned getNumDomainIds() const { return space.getNumDomainIds(); }
48  unsigned getNumRangeIds() const { return space.getNumRangeIds(); }
49  unsigned getNumSymbolIds() const { return space.getNumSymbolIds(); }
50  unsigned getNumLocalIds() const { return space.getNumLocalIds(); }
51  unsigned getNumIds() const { return space.getNumIds(); }
52 
53  /// Return the number of disjuncts in the union.
54  unsigned getNumDisjuncts() const;
55 
56  const PresburgerSpace &getSpace() const { return space; }
57 
58  /// Return a reference to the list of disjuncts.
59  ArrayRef<IntegerRelation> getAllDisjuncts() const;
60 
61  /// Return the disjunct at the specified index.
62  const IntegerRelation &getDisjunct(unsigned index) const;
63 
64  /// Mutate this set, turning it into the union of this set and the given
65  /// disjunct.
66  void unionInPlace(const IntegerRelation &disjunct);
67 
68  /// Mutate this set, turning it into the union of this set and the given set.
69  void unionInPlace(const PresburgerRelation &set);
70 
71  /// Return the union of this set and the given set.
72  PresburgerRelation unionSet(const PresburgerRelation &set) const;
73 
74  /// Return the intersection of this set and the given set.
75  PresburgerRelation intersect(const PresburgerRelation &set) const;
76 
77  /// Return true if the set contains the given point, and false otherwise.
78  bool containsPoint(ArrayRef<int64_t> point) const;
79 
80  /// Return the complement of this set. All local variables in the set must
81  /// correspond to floor divisions.
82  PresburgerRelation complement() const;
83 
84  /// Return the set difference of this set and the given set, i.e.,
85  /// return `this \ set`. All local variables in `set` must correspond
86  /// to floor divisions, but local variables in `this` need not correspond to
87  /// divisions.
88  PresburgerRelation subtract(const PresburgerRelation &set) const;
89 
90  /// Return true if this set is a subset of the given set, and false otherwise.
91  bool isSubsetOf(const PresburgerRelation &set) const;
92 
93  /// Return true if this set is equal to the given set, and false otherwise.
94  /// All local variables in both sets must correspond to floor divisions.
95  bool isEqual(const PresburgerRelation &set) const;
96 
97  /// Return true if all the sets in the union are known to be integer empty
98  /// false otherwise.
99  bool isIntegerEmpty() const;
100 
101  /// Find an integer sample from the given set. This should not be called if
102  /// any of the disjuncts in the union are unbounded.
103  bool findIntegerSample(SmallVectorImpl<int64_t> &sample);
104 
105  /// Compute an overapproximation of the number of integer points in the
106  /// disjunct. Symbol ids are currently not supported. If the computed
107  /// overapproximation is infinite, an empty optional is returned.
108  ///
109  /// This currently just sums up the overapproximations of the volumes of the
110  /// disjuncts, so the approximation might be far from the true volume in the
111  /// case when there is a lot of overlap between disjuncts.
112  Optional<uint64_t> computeVolume() const;
113 
114  /// Simplifies the representation of a PresburgerRelation.
115  ///
116  /// In particular, removes all disjuncts which are subsets of other
117  /// disjuncts in the union.
118  PresburgerRelation coalesce() const;
119 
120  /// Print the set's internal state.
121  void print(raw_ostream &os) const;
122  void dump() const;
123 
124 protected:
125  /// Construct an empty PresburgerRelation with the specified number of
126  /// dimension and symbols.
127  explicit PresburgerRelation(const PresburgerSpace &space) : space(space) {
128  assert(space.getNumLocalIds() == 0 &&
129  "PresburgerRelation cannot have local ids.");
130  }
131 
132  PresburgerSpace space;
133 
134  /// The list of disjuncts that this set is the union of.
135  SmallVector<IntegerRelation, 2> disjuncts;
136 
137  friend class SetCoalescer;
138 };
