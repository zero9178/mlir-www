10 // defined on a domain that is a union of IntegerPolyhedrons, and on each domain

12 // tuple being an affine expression in the ids of the IntegerPolyhedron.

25 /// This class represents a multi-affine function whose domain is given by an
26 /// IntegerPolyhedron. This can be thought of as an IntegerPolyhedron with a
27 /// tuple of integer values attached to every point in the polyhedron, with the
28 /// value of each element of the tuple given by an affine expression in the ids
29 /// of the polyhedron. For example we could have the domain
30 ///
31 /// (x, y) : (x >= 5, y >= x)
32 ///
33 /// and a tuple of three integers defined at every point in the polyhedron:
34 ///
35 /// (x, y) -> (x + 2, 2*x - 3y + 5, 2*x + y).
36 ///
37 /// In this way every point in the polyhedron has a tuple of integers associated
38 /// with it. If the integer polyhedron has local ids, then the output
39 /// expressions can use them as well. The output expressions are represented as
40 /// a matrix with one row for every element in the output vector one column for
41 /// each id, and an extra column at the end for the constant term.
42 ///
43 /// Checking equality of two such functions is supported, as well as finding the
44 /// value of the function at a specified point.
45 class MultiAffineFunction {
46 public:
47  MultiAffineFunction(const IntegerPolyhedron &domain, const Matrix &output)
48  : domainSet(domain), output(output) {}
49  MultiAffineFunction(const Matrix &output, const PresburgerSpace &space)
50  : domainSet(space), output(output) {}
51 
52  unsigned getNumInputs() const { return domainSet.getNumDimAndSymbolIds(); }
53  unsigned getNumOutputs() const { return output.getNumRows(); }
54  bool isConsistent() const {
55  return output.getNumColumns() == domainSet.getNumIds() + 1;
56  }
57  const IntegerPolyhedron &getDomain() const { return domainSet; }
58  const PresburgerSpace &getDomainSpace() const { return domainSet.getSpace(); }
59 
60  /// Insert `num` identifiers of the specified kind at position `pos`.
61  /// Positions are relative to the kind of identifier. The coefficient columns
62  /// corresponding to the added identifiers are initialized to zero. Return the
63  /// absolute column position (i.e., not relative to the kind of identifier)
64  /// of the first added identifier.
65  unsigned insertId(IdKind kind, unsigned pos, unsigned num = 1);
66 
67  /// Remove the specified range of ids.
68  void removeIdRange(IdKind kind, unsigned idStart, unsigned idLimit);
69 
70  /// Given a MAF `other`, merges local identifiers such that both funcitons
71  /// have union of local ids, without changing the set of points in domain or
72  /// the output.
73  void mergeLocalIds(MultiAffineFunction &other);
74 
75  /// Return whether the outputs of `this` and `other` agree wherever both
76  /// functions are defined, i.e., the outputs should be equal for all points in
77  /// the intersection of the domains.
78  bool isEqualWhereDomainsOverlap(MultiAffineFunction other) const;
79 
80  /// Return whether the `this` and `other` are equal. This is the case if
81  /// they lie in the same space, i.e. have the same dimensions, and their
82  /// domains are identical and their outputs are equal on their domain.
83  bool isEqual(const MultiAffineFunction &other) const;
84 
85  /// Get the value of the function at the specified point. If the point lies
86  /// outside the domain, an empty optional is returned.
87  Optional<SmallVector<int64_t, 8>> valueAt(ArrayRef<int64_t> point) const;
88 
89  /// Truncate the output dimensions to the first `count` dimensions.
90  ///
91  /// TODO: refactor so that this can be accomplished through removeIdRange.
92  void truncateOutput(unsigned count);
93 
94  void print(raw_ostream &os) const;
95  void dump() const;
96 
97 private:
98  /// The IntegerPolyhedron representing the domain over which the function is
99  /// defined.
100  IntegerPolyhedron domainSet;
101 
102  /// The function's output is a tuple of integers, with the ith element of the
103  /// tuple defined by the affine expression given by the ith row of this output
104  /// matrix.
105  Matrix output;
106 };

108 /// This class represents a piece-wise MultiAffineFunction. This can be thought
109 /// of as a list of MultiAffineFunction with disjoint domains, with each having
110 /// their own affine expressions for their output tuples. For example, we could
111 /// have a function with two input variables (x, y), defined as
112 ///
113 /// f(x, y) = (2*x + y, y - 4) if x >= 0, y >= 0
114 /// = (-2*x + y, y + 4) if x < 0, y < 0
115 /// = (4, 1) if x < 0, y >= 0
116 ///
117 /// Note that the domains all have to be *disjoint*. Otherwise, the behaviour of
118 /// this class is undefined. The domains need not cover all possible points;
119 /// this represents a partial function and so could be undefined at some points.
120 ///
121 /// As in PresburgerSets, the input ids are partitioned into dimension ids and
122 /// symbolic ids.
123 ///
124 /// Support is provided to compare equality of two such functions as well as
125 /// finding the value of the function at a point.
126 class PWMAFunction {
127 public:
128  PWMAFunction(const PresburgerSpace &space, unsigned numOutputs)
129  : space(space), numOutputs(numOutputs) {
130  assert(space.getNumDomainIds() == 0 &&
131  "Set type space should have zero domain ids.");
132  assert(space.getNumLocalIds() == 0 &&
133  "PWMAFunction cannot have local ids.");
134  assert(numOutputs >= 1 && "The function must output something!");
135  }
136 
137  const PresburgerSpace &getSpace() const { return space; }
138 
139  void addPiece(const MultiAffineFunction &piece);
140  void addPiece(const IntegerPolyhedron &domain, const Matrix &output);
141 
142  const MultiAffineFunction &getPiece(unsigned i) const { return pieces[i]; }
143  unsigned getNumPieces() const { return pieces.size(); }
144  unsigned getNumOutputs() const { return numOutputs; }
145  unsigned getNumInputs() const { return space.getNumIds(); }
146  MultiAffineFunction &getPiece(unsigned i) { return pieces[i]; }
147 
148  /// Return the domain of this piece-wise MultiAffineFunction. This is the
149  /// union of the domains of all the pieces.
150  PresburgerSet getDomain() const;
151 
152  /// Return the value at the specified point and an empty optional if the
153  /// point does not lie in the domain.
154  Optional<SmallVector<int64_t, 8>> valueAt(ArrayRef<int64_t> point) const;
155 
156  /// Return whether `this` and `other` are equal as PWMAFunctions, i.e. whether
157  /// they have the same dimensions, the same domain and they take the same
158  /// value at every point in the domain.
159  bool isEqual(const PWMAFunction &other) const;
160 
161  /// Truncate the output dimensions to the first `count` dimensions.
162  ///
163  /// TODO: refactor so that this can be accomplished through removeIdRange.
164  void truncateOutput(unsigned count);
165 
166  void print(raw_ostream &os) const;
167  void dump() const;
168 
169 private:
170  PresburgerSpace space;
171 
172  /// The list of pieces in this piece-wise MultiAffineFunction.
173  SmallVector<MultiAffineFunction, 4> pieces;
174 
175  /// The number of output ids.
176  unsigned numOutputs;
177 };
