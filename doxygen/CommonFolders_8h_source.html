24 /// Performs constant folding `calculate` with element-wise behavior on the two
25 /// attributes in `operands` and returns the result if possible.
26 template <class AttrElementT,
27  class ElementValueT = typename AttrElementT::ValueType,
28  class CalculationT =
29  function_ref<ElementValueT(ElementValueT, ElementValueT)>>
30 Attribute constFoldBinaryOp(ArrayRef<Attribute> operands,
31  const CalculationT &calculate) {
32  assert(operands.size() == 2 && "binary op takes two operands");
33  if (!operands[0] || !operands[1])
34  return {};
35  if (operands[0].getType() != operands[1].getType())
36  return {};
37 
38  if (operands[0].isa<AttrElementT>() && operands[1].isa<AttrElementT>()) {
39  auto lhs = operands[0].cast<AttrElementT>();
40  auto rhs = operands[1].cast<AttrElementT>();
41 
42  return AttrElementT::get(lhs.getType(),
43  calculate(lhs.getValue(), rhs.getValue()));
44  }
45  if (operands[0].isa<SplatElementsAttr>() &&
46  operands[1].isa<SplatElementsAttr>()) {
47  // Both operands are splats so we can avoid expanding the values out and
48  // just fold based on the splat value.
49  auto lhs = operands[0].cast<SplatElementsAttr>();
50  auto rhs = operands[1].cast<SplatElementsAttr>();
51 
52  auto elementResult = calculate(lhs.getSplatValue<ElementValueT>(),
53  rhs.getSplatValue<ElementValueT>());
54  return DenseElementsAttr::get(lhs.getType(), elementResult);
55  } else if (operands[0].isa<ElementsAttr>() &&
56  operands[1].isa<ElementsAttr>()) {
57  // Operands are ElementsAttr-derived; perform an element-wise fold by
58  // expanding the values.
59  auto lhs = operands[0].cast<ElementsAttr>();
60  auto rhs = operands[1].cast<ElementsAttr>();
61 
62  auto lhsIt = lhs.value_begin<ElementValueT>();
63  auto rhsIt = rhs.value_begin<ElementValueT>();
64  SmallVector<ElementValueT, 4> elementResults;
65  elementResults.reserve(lhs.getNumElements());
66  for (size_t i = 0, e = lhs.getNumElements(); i < e; ++i, ++lhsIt, ++rhsIt)
67  elementResults.push_back(calculate(*lhsIt, *rhsIt));
68  return DenseElementsAttr::get(lhs.getType(), elementResults);
69  }
70  return {};
71 }

73 /// Performs constant folding `calculate` with element-wise behavior on the one
74 /// attributes in `operands` and returns the result if possible.
75 template <class AttrElementT,
76  class ElementValueT = typename AttrElementT::ValueType,
77  class CalculationT = function_ref<ElementValueT(ElementValueT)>>
78 Attribute constFoldUnaryOp(ArrayRef<Attribute> operands,
79  const CalculationT &&calculate) {
80  assert(operands.size() == 1 && "unary op takes one operands");
81  if (!operands[0])
82  return {};
83 
84  if (operands[0].isa<AttrElementT>()) {
85  auto op = operands[0].cast<AttrElementT>();
86 
87  return AttrElementT::get(op.getType(), calculate(op.getValue()));
88  }
89  if (operands[0].isa<SplatElementsAttr>()) {
90  // Both operands are splats so we can avoid expanding the values out and
91  // just fold based on the splat value.
92  auto op = operands[0].cast<SplatElementsAttr>();
93 
94  auto elementResult = calculate(op.getSplatValue<ElementValueT>());
95  return DenseElementsAttr::get(op.getType(), elementResult);
96  } else if (operands[0].isa<ElementsAttr>()) {
97  // Operands are ElementsAttr-derived; perform an element-wise fold by
98  // expanding the values.
99  auto op = operands[0].cast<ElementsAttr>();
100 
101  auto opIt = op.value_begin<ElementValueT>();
102  SmallVector<ElementValueT> elementResults;
103  elementResults.reserve(op.getNumElements());
104  for (size_t i = 0, e = op.getNumElements(); i < e; ++i, ++opIt)
105  elementResults.push_back(calculate(*opIt));
106  return DenseElementsAttr::get(op.getType(), elementResults);
107  }
108  return {};
109 }

111 template <
112  class AttrElementT, class TargetAttrElementT,
113  class ElementValueT = typename AttrElementT::ValueType,
114  class TargetElementValueT = typename TargetAttrElementT::ValueType,
115  class CalculationT = function_ref<TargetElementValueT(ElementValueT, bool)>>
116 Attribute constFoldCastOp(ArrayRef<Attribute> operands, Type resType,
117  const CalculationT &calculate) {
118  assert(operands.size() == 1 && "Cast op takes one operand");
119  if (!operands[0])
120  return {};
121 
122  if (operands[0].isa<AttrElementT>()) {
123  auto op = operands[0].cast<AttrElementT>();
124  bool castStatus = true;
125  auto res = calculate(op.getValue(), castStatus);
126  if (!castStatus)
127  return {};
128  return TargetAttrElementT::get(resType, res);
129  }
130  if (operands[0].isa<SplatElementsAttr>()) {
131  // The operand is a splat so we can avoid expanding the values out and
132  // just fold based on the splat value.
133  auto op = operands[0].cast<SplatElementsAttr>();
134  bool castStatus = true;
135  auto elementResult =
136  calculate(op.getSplatValue<ElementValueT>(), castStatus);
137  if (!castStatus)
138  return {};
139  return DenseElementsAttr::get(resType, elementResult);
140  }
141  if (operands[0].isa<ElementsAttr>()) {
142  // Operand is ElementsAttr-derived; perform an element-wise fold by
143  // expanding the value.
144  auto op = operands[0].cast<ElementsAttr>();
145  bool castStatus = true;
146  auto opIt = op.value_begin<ElementValueT>();
147  SmallVector<TargetElementValueT> elementResults;
148  elementResults.reserve(op.getNumElements());
149  for (size_t i = 0, e = op.getNumElements(); i < e; ++i, ++opIt) {
150  auto elt = calculate(*opIt, castStatus);
151  if (!castStatus)
152  return {};
153  elementResults.push_back(elt);
154  }
155 
156  return DenseElementsAttr::get(resType, elementResults);
157  }
158  return {};
159 }
